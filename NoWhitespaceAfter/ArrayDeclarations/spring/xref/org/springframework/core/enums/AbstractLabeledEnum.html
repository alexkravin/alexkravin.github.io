17  package org.springframework.core.enums;
18  
19  /**
20   * Abstract base superclass for LabeledEnum implementations.
21   *
22   * @author Keith Donald
23   * @author Juergen Hoeller
24   * @author Sam Brannen
25   * @since 1.2.2
26   * @deprecated as of Spring 3.0, in favor of Java 5 enums.
27   */
28  @Deprecated
29  public abstract class AbstractLabeledEnum implements LabeledEnum {
30  
31  	/**
32  	 * Create a new AbstractLabeledEnum instance.
33  	 */
34  	protected AbstractLabeledEnum() {
35  	}
36  
37  	public Class getType() {
38  		// Could be coded as getClass().isAnonymousClass() on JDK 1.5
39  		boolean isAnonymous = (getClass().getDeclaringClass() == null && getClass().getName().indexOf('$') != -1);
40  		return (isAnonymous ? getClass().getSuperclass() : getClass());
41  	}
42  
43  	public int compareTo(Object obj) {
44  		if (!(obj instanceof LabeledEnum)) {
45  			throw new ClassCastException("You may only compare LabeledEnums");
46  		}
47  		LabeledEnum that = (LabeledEnum) obj;
48  		if (!this.getType().equals(that.getType())) {
49  			throw new ClassCastException("You may only compare LabeledEnums of the same type");
50  		}
51  		return this.getCode().compareTo(that.getCode());
52  	}
53  
54  	@Override
55  	public boolean equals(Object obj) {
56  		if (this == obj) {
57  			return true;
58  		}
59  		if (!(obj instanceof LabeledEnum)) {
60  			return false;
61  		}
62  		LabeledEnum other = (LabeledEnum) obj;
63  		return (this.getType().equals(other.getType()) && this.getCode().equals(other.getCode()));
64  	}
65  
66  	@Override
67  	public int hashCode() {
68  		return (getType().hashCode() * 29 + getCode().hashCode());
69  	}
70  
71  	@Override
72  	public String toString() {
73  		return getLabel();
74  	}
75  
76  }
