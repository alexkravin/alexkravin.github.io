56  /**
57   * Responsible for maintaining the queue of actions related to events.
58   * </p>
59   * The ActionQueue holds the DML operations queued as part of a session's
60   * transactional-write-behind semantics.  DML operations are queued here
61   * until a flush forces them to be executed against the database.
62   *
63   * @author Steve Ebersole
64   */
65  public class ActionQueue {

220 	 * Check whether the given tables/query-spaces are to be executed against
221 	 * given the currently queued actions.
222 	 *
223 	 * @param tables The table/query-spaces to check.
224 	 *
225 	 * @return True if we contain pending actions against any of the given

228 	public boolean areTablesToBeUpdated(Set tables) {
229 		return areTablesToUpdated( updates, tables ) ||
230 				areTablesToUpdated( insertions, tables ) ||
231 				areTablesToUpdated( deletions, tables ) ||
232 				areTablesToUpdated( collectionUpdates, tables ) ||
233 				areTablesToUpdated( collectionCreations, tables ) ||
234 				areTablesToUpdated( collectionRemovals, tables );
235 	}

246 	@SuppressWarnings({ "unchecked" })
247 	private static boolean areTablesToUpdated(List actions, Set tableSpaces) {
248 		for ( Executable action : (List<Executable>) actions ) {
249 			final Serializable[] spaces = action.getPropertySpaces();
250 			for ( Serializable space : spaces ) {
251 				if ( tableSpaces.contains( space ) ) {
252 					if ( log.isDebugEnabled() ) {
253 						log.debug( "changes must be flushed to space: " + space );
254 					}
255 					return true;
256 				}
257 			}
258 		}
259 		return false;
260 	}

262 	private void executeActions(List list) throws HibernateException {
263 		int size = list.size();
264 		for ( int i = 0; i < size; i++ ) {
265 			execute( ( Executable ) list.get( i ) );
266 		}
267 		list.clear();
268 		session.getBatcher().executeBatch();
269 	}

271 	public void execute(Executable executable) {
272 		try {
273 			executable.execute();
274 		}
275 		finally {
276 			registerCleanupActions( executable );
277 		}
278 	}

280 	private void registerCleanupActions(Executable executable) {
281 		beforeTransactionProcesses.register( executable.getBeforeTransactionCompletionProcess() );
282 		if ( session.getFactory().getSettings().isQueryCacheEnabled() ) {
283 			final String[] spaces = (String[]) executable.getPropertySpaces();
284 			afterTransactionProcesses.addSpacesToInvalidate( spaces );
285 			session.getFactory().getUpdateTimestampsCache().preinvalidate( spaces );
286 		}
287 		afterTransactionProcesses.register( executable.getAfterTransactionCompletionProcess() );
288 	}

359 	/**
360 	 * Order the {@link #insertions} queue such that we group inserts
361 	 * against the same entity together (without violating constraints).  The
362 	 * original order is generated by cascade order, which in turn is based on
363 	 * the directionality of foreign-keys.  So even though we will be changing
364 	 * the ordering here, we need to make absolutely certain that we do not
365 	 * circumvent this FK ordering to the extent of causing constraint
366 	 * violations
367 	 */
368 	private void sortInsertActions() {
369 		new InsertActionSorter().sort();
370 	}

378 		collectionCreations.clear();

406 	public void unScheduleDeletion(EntityEntry entry, Object rescuedEntity) {
407 		for ( int i = 0; i < deletions.size(); i++ ) {
408 			EntityDeleteAction action = (EntityDeleteAction) deletions.get( i );
409 			if ( action.getInstance() == rescuedEntity ) {
410 				deletions.remove( i );
411 				return;
412 			}
413 		}
414 		throw new AssertionFailure( "Unable to perform un-delete for instance " + entry.getEntityName() );
415 	}

417 	/**
418 	 * Used by the owning session to explicitly control serialization of the
419 	 * action queue
420 	 *
421 	 * @param oos The stream to which the action queue should get written
422 	 *
423 	 * @throws IOException Indicates an error writing to the stream
424 	 */
425 	public void serialize(ObjectOutputStream oos) throws IOException {
426 		log.trace( "serializing action-queue" );
427 
428 		int queueSize = insertions.size();
429 		log.trace( "starting serialization of [" + queueSize + "] insertions entries" );
430 		oos.writeInt( queueSize );
431 		for ( int i = 0; i < queueSize; i++ ) {
432 			oos.writeObject( insertions.get( i ) );
433 		}
434 
435 		queueSize = deletions.size();
436 		log.trace( "starting serialization of [" + queueSize + "] deletions entries" );
437 		oos.writeInt( queueSize );
438 		for ( int i = 0; i < queueSize; i++ ) {
439 			oos.writeObject( deletions.get( i ) );
440 		}
441 
442 		queueSize = updates.size();
443 		log.trace( "starting serialization of [" + queueSize + "] updates entries" );
444 		oos.writeInt( queueSize );
445 		for ( int i = 0; i < queueSize; i++ ) {
446 			oos.writeObject( updates.get( i ) );
447 		}
448 
449 		queueSize = collectionUpdates.size();
450 		log.trace( "starting serialization of [" + queueSize + "] collectionUpdates entries" );
451 		oos.writeInt( queueSize );
452 		for ( int i = 0; i < queueSize; i++ ) {
453 			oos.writeObject( collectionUpdates.get( i ) );
454 		}
455 
456 		queueSize = collectionRemovals.size();
457 		log.trace( "starting serialization of [" + queueSize + "] collectionRemovals entries" );
458 		oos.writeInt( queueSize );
459 		for ( int i = 0; i < queueSize; i++ ) {
460 			oos.writeObject( collectionRemovals.get( i ) );
461 		}
462 
463 		queueSize = collectionCreations.size();
464 		log.trace( "starting serialization of [" + queueSize + "] collectionCreations entries" );
465 		oos.writeInt( queueSize );
466 		for ( int i = 0; i < queueSize; i++ ) {
467 			oos.writeObject( collectionCreations.get( i ) );
468 		}
469 	}

472 	 * Used by the owning session to explicitly control deserialization of the
473 	 * action queue
474 	 *
475 	 * @param ois The stream from which to read the action queue
476 	 * @param session The session to which the action queue belongs
477 	 *
478 	 * @return The deserialized action queue
479 	 *
480 	 * @throws IOException indicates a problem reading from the stream
481 	 * @throws ClassNotFoundException Generally means we were unable to locate user classes.
482 	 */
483 	@SuppressWarnings({ "unchecked" })
484 	public static ActionQueue deserialize(
485 			ObjectInputStream ois,
486 			SessionImplementor session) throws IOException, ClassNotFoundException {
487 		log.trace( "deserializing action-queue" );
488 		ActionQueue rtn = new ActionQueue( session );
489 
490 		int queueSize = ois.readInt();
491 		log.trace( "starting deserialization of [" + queueSize + "] insertions entries" );
492 		rtn.insertions = new ArrayList<Executable>( queueSize );
493 		for ( int i = 0; i < queueSize; i++ ) {
494 			rtn.insertions.add( ois.readObject() );
495 		}
496 
497 		queueSize = ois.readInt();
498 		log.trace( "starting deserialization of [" + queueSize + "] deletions entries" );
499 		rtn.deletions = new ArrayList<Executable>( queueSize );
500 		for ( int i = 0; i < queueSize; i++ ) {
501 			rtn.deletions.add( ois.readObject() );
502 		}
503 
504 		queueSize = ois.readInt();
505 		log.trace( "starting deserialization of [" + queueSize + "] updates entries" );
506 		rtn.updates = new ArrayList<Executable>( queueSize );
507 		for ( int i = 0; i < queueSize; i++ ) {
508 			rtn.updates.add( ois.readObject() );
509 		}
510 
511 		queueSize = ois.readInt();
512 		log.trace( "starting deserialization of [" + queueSize + "] collectionUpdates entries" );
513 		rtn.collectionUpdates = new ArrayList<Executable>( queueSize );
514 		for ( int i = 0; i < queueSize; i++ ) {
515 			rtn.collectionUpdates.add( ois.readObject() );
516 		}
517 
518 		queueSize = ois.readInt();
519 		log.trace( "starting deserialization of [" + queueSize + "] collectionRemovals entries" );
520 		rtn.collectionRemovals = new ArrayList<Executable>( queueSize );
521 		for ( int i = 0; i < queueSize; i++ ) {
522 			rtn.collectionRemovals.add( ois.readObject() );
523 		}
524 
525 		queueSize = ois.readInt();
526 		log.trace( "starting deserialization of [" + queueSize + "] collectionCreations entries" );
527 		rtn.collectionCreations = new ArrayList<Executable>( queueSize );
528 		for ( int i = 0; i < queueSize; i++ ) {
529 			rtn.collectionCreations.add( ois.readObject() );
530 		}
531 		return rtn;
532 	}

534 	private static class BeforeTransactionCompletionProcessQueue {
535 		private SessionImplementor session;
536 		private List<BeforeTransactionCompletionProcess> processes = new ArrayList<BeforeTransactionCompletionProcess>();
537 
538 		private BeforeTransactionCompletionProcessQueue(SessionImplementor session) {
539 			this.session = session;
540 		}
541 
542 		public void register(BeforeTransactionCompletionProcess process) {
543 			if ( process == null ) {
544 				return;
545 			}
546 			processes.add( process );
547 		}
548 
549 		public void beforeTransactionCompletion() {
550 			final int size = processes.size();
551 			for ( int i = 0; i < size; i++ ) {
552 				try {
553 					BeforeTransactionCompletionProcess process = processes.get( i );
554 					process.doBeforeTransactionCompletion( session );
555 				}
556 				catch ( HibernateException he ) {
557 					throw he;
558 				}
559 				catch ( Exception e ) {
560 					throw new AssertionFailure( "Unable to perform beforeTransactionCompletion callback", e );
561 				}
562 			}
563 			processes.clear();
564 		}
565 	}

567 	private static class AfterTransactionCompletionProcessQueue {
568 		private SessionImplementor session;
569 		private Set<String> querySpacesToInvalidate = new HashSet<String>();
570 		private List<AfterTransactionCompletionProcess> processes
571 				= new ArrayList<AfterTransactionCompletionProcess>( INIT_QUEUE_LIST_SIZE * 3 );
572 
573 		private AfterTransactionCompletionProcessQueue(SessionImplementor session) {
574 			this.session = session;
575 		}
576 
577 		public void addSpacesToInvalidate(String[] spaces) {
578 			if ( spaces == null ) {
579 				return;
580 			}
581 			for ( int i = 0, max = spaces.length; i < max; i++ ) {
582 				addSpaceToInvalidate( spaces[i] );
583 			}
584 		}
585 
586 		public void addSpaceToInvalidate(String space) {
587 			querySpacesToInvalidate.add( space );
588 		}
589 
590 		public void register(AfterTransactionCompletionProcess process) {
591 			if ( process == null ) {
592 				return;
593 			}
594 			processes.add( process );
595 		}
596 
597 		public void afterTransactionCompletion(boolean success) {
598 			final int size = processes.size();
599 			for ( int i = 0; i < size; i++ ) {
600 				try {
601 					AfterTransactionCompletionProcess process = processes.get( i );
602 					process.doAfterTransactionCompletion( success, session );
603 				}
604 				catch ( CacheException ce ) {
605 					log.error( "could not release a cache lock", ce );
606 					// continue loop
607 				}
608 				catch ( Exception e ) {
609 					throw new AssertionFailure( "Exception releasing cache locks", e );
610 				}
611 			}
612 			processes.clear();
613 
614 			if ( session.getFactory().getSettings().isQueryCacheEnabled() ) {
615 				session.getFactory().getUpdateTimestampsCache().invalidate(
616 						querySpacesToInvalidate.toArray( new String[ querySpacesToInvalidate.size()] )
617 				);
618 			}
619 			querySpacesToInvalidate.clear();
620 		}
621 	}

623 	/**
624 	 * Sorts the insert actions using more hashes.
625 	 *
626 	 * @author Jay Erb
627 	 */
628 	private class InsertActionSorter {
629 		// the mapping of entity names to their latest batch numbers.
630 		private HashMap latestBatches = new HashMap();
631 		private HashMap entityBatchNumber;
632 
633 		// the map of batch numbers to EntityInsertAction lists
634 		private HashMap actionBatches = new HashMap();
635 
636 		public InsertActionSorter() {
637 			//optimize the hash size to eliminate a rehash.
638 			entityBatchNumber = new HashMap( insertions.size() + 1, 1.0f );
639 		}
640 
641 		/**
642 		 * Sort the insert actions.
643 		 */
644 		@SuppressWarnings({ "unchecked", "UnnecessaryBoxing" })
645 		public void sort() {
646 			// the list of entity names that indicate the batch number
647 			for ( EntityInsertAction action : (List<EntityInsertAction>) insertions ) {
648 				// remove the current element from insertions. It will be added back later.
649 				String entityName = action.getEntityName();
650 
651 				// the entity associated with the current action.
652 				Object currentEntity = action.getInstance();
653 
654 				Integer batchNumber;
655 				if ( latestBatches.containsKey( entityName ) ) {
656 					// There is already an existing batch for this type of entity.
657 					// Check to see if the latest batch is acceptable.
658 					batchNumber = findBatchNumber( action, entityName );
659 				}
660 				else {
661 					// add an entry for this type of entity.
662 					// we can be assured that all referenced entities have already
663 					// been processed,
664 					// so specify that this entity is with the latest batch.
665 					// doing the batch number before adding the name to the list is
666 					// a faster way to get an accurate number.
667 
668 					batchNumber = Integer.valueOf( actionBatches.size() );
669 					latestBatches.put( entityName, batchNumber );
670 				}
671 				entityBatchNumber.put( currentEntity, batchNumber );
672 				addToBatch( batchNumber, action );
673 			}
674 			insertions.clear();
675 
676 			// now rebuild the insertions list. There is a batch for each entry in the name list.
677 			for ( int i = 0; i < actionBatches.size(); i++ ) {
678 				List batch = ( List ) actionBatches.get( new Integer( i ) );
679 				for ( Object aBatch : batch ) {
680 					EntityInsertAction action = (EntityInsertAction) aBatch;
681 					insertions.add( action );
682 				}
683 			}
684 		}
685 
686 		/**
687 		 * Finds an acceptable batch for this entity to be a member as part of the {@link InsertActionSorter}
688 		 *
689 		 * @param action The action being sorted
690 		 * @param entityName The name of the entity affected by the action
691 		 *
692 		 * @return An appropriate batch number; todo document this process better
693 		 */
694 		@SuppressWarnings({ "UnnecessaryBoxing", "unchecked" })
695 		private Integer findBatchNumber(
696 				EntityInsertAction action,
697 				String entityName) {
698 			// loop through all the associated entities and make sure they have been
699 			// processed before the latest
700 			// batch associated with this entity type.
701 
702 			// the current batch number is the latest batch for this entity type.
703 			Integer latestBatchNumberForType = ( Integer ) latestBatches.get( entityName );
704 
705 			// loop through all the associations of the current entity and make sure that they are processed
706 			// before the current batch number
707 			Object[] propertyValues = action.getState();
708 			Type[] propertyTypes = action.getPersister().getClassMetadata()
709 					.getPropertyTypes();
710 
711 			for ( int i = 0; i < propertyValues.length; i++ ) {
712 				Object value = propertyValues[i];
713 				Type type = propertyTypes[i];
714 				if ( type.isEntityType() && value != null ) {
715 					// find the batch number associated with the current association, if any.
716 					Integer associationBatchNumber = ( Integer ) entityBatchNumber.get( value );
717 					if ( associationBatchNumber != null && associationBatchNumber.compareTo( latestBatchNumberForType ) > 0 ) {
718 						// create a new batch for this type. The batch number is the number of current batches.
719 						latestBatchNumberForType = Integer.valueOf( actionBatches.size() );
720 						latestBatches.put( entityName, latestBatchNumberForType );
721 						// since this entity will now be processed in the latest possible batch,
722 						// we can be assured that it will come after all other associations,
723 						// there's not need to continue checking.
724 						break;
725 					}
726 				}
727 			}
728 			return latestBatchNumberForType;
729 		}
730 
731 		@SuppressWarnings({ "unchecked" })
732 		private void addToBatch(Integer batchNumber, EntityInsertAction action) {
733 			List actions = ( List ) actionBatches.get( batchNumber );
734 
735 			if ( actions == null ) {
736 				actions = new LinkedList();
737 				actionBatches.put( batchNumber, actions );
738 			}
739 			actions.add( action );
740 		}
