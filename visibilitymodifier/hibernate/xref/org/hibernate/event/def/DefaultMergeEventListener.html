76  	/**
77  	 * Handle the given merge event.
78  	 *
79  	 * @param event The merge event to be handled.
80  	 * @throws HibernateException
81  	 */
82  	public void onMerge(MergeEvent event) throws HibernateException {
83  		EventCache copyCache = new EventCache();
84  		onMerge( event, copyCache );
85  		// TODO: iteratively get transient entities and retry merge until one of the following conditions:
86  		//       1) transientCopyCache.size() == 0
87  		//       2) transientCopyCache.size() is not decreasing and copyCache.size() is not increasing
88  		// TODO: find out if retrying can add entities to copyCache (don't think it can...)
89  		// For now, just retry once; throw TransientObjectException if there are still any transient entities
90  		Map transientCopyCache = getTransientCopyCache(event, copyCache );
91  		if ( transientCopyCache.size() > 0 ) {
92  			retryMergeTransientEntities( event, transientCopyCache, copyCache, true );
93  			// find any entities that are still transient after retry
94  			transientCopyCache = getTransientCopyCache(event, copyCache );
95  			if ( transientCopyCache.size() > 0 ) {
96  				Set transientEntityNames = new HashSet();
97  				for( Iterator it=transientCopyCache.entrySet().iterator(); it.hasNext(); ) {
98  					Object transientEntity = ( ( Map.Entry ) it.next() ).getKey();
99  					String transientEntityName = event.getSession().guessEntityName( transientEntity );
100 					transientEntityNames.add( transientEntityName );
101 					log.trace( "transient instance could not be processed by merge when checking nullability: " +
102 							transientEntityName + "[" + transientEntity + "]" );
103 				}
104 				if ( isNullabilityCheckedGlobal( event.getSession() ) ) {
105 					throw new TransientObjectException(
106 						"one or more objects is an unsaved transient instance - save transient instance(s) before merging: " +
107 						transientEntityNames );
108 				}
109 				else {
110 					log.trace( "retry saving transient instances without checking nullability" );
111 					// failures will be detected later...
112 					retryMergeTransientEntities( event, transientCopyCache, copyCache, false );
113 				}
114 			}
115 		}
116 		copyCache.clear();
117 		copyCache = null;
118 	}

120 	protected EventCache getTransientCopyCache(MergeEvent event, EventCache copyCache) {
121 		EventCache transientCopyCache = new EventCache();
122 		for ( Iterator it=copyCache.entrySet().iterator(); it.hasNext(); ) {
123 			Map.Entry mapEntry = ( Map.Entry ) it.next();
124 			Object entity = mapEntry.getKey();
125 			Object copy = mapEntry.getValue();
126 			if ( copy instanceof HibernateProxy ) {
127 				copy = ( (HibernateProxy) copy ).getHibernateLazyInitializer().getImplementation();
128 			}
129 			EntityEntry copyEntry = event.getSession().getPersistenceContext().getEntry( copy );
130 			if ( copyEntry == null ) {
131 				// entity name will not be available for non-POJO entities
132 				// TODO: cache the entity name somewhere so that it is available to this exception
133 				log.trace( "transient instance could not be processed by merge: " +
134 						event.getSession().guessEntityName( copy ) + "[" + entity + "]" );
135 				// merge did not cascade to this entity; it's in copyCache because a
136 				// different entity has a non-nullable reference to this entity;
137 				// this entity should not be put in transientCopyCache, because it was
138 				// not included in the merge;
139 				// if the global setting for checking nullability is false, the non-nullable
140 				// reference to this entity will be detected later
141 				if ( isNullabilityCheckedGlobal( event.getSession() ) ) {
142 					throw new TransientObjectException(
143 						"object is an unsaved transient instance - save the transient instance before merging: " +
144 							event.getSession().guessEntityName( copy )
145 					);
146 				}
147 			}
148 			else if ( copyEntry.getStatus() == Status.SAVING ) {
149 				transientCopyCache.put( entity, copy, copyCache.isOperatedOn( entity ) );
150 			}
151 			else if ( copyEntry.getStatus() != Status.MANAGED && copyEntry.getStatus() != Status.READ_ONLY ) {
152 				throw new AssertionFailure( "Merged entity does not have status set to MANAGED or READ_ONLY; "+copy+" status="+copyEntry.getStatus() );
153 			}
154 		}
155 		return transientCopyCache;
156 	}

158 	protected void retryMergeTransientEntities(
159 			MergeEvent event,
160 			Map transientCopyCache,
161 			EventCache copyCache,
162 			boolean isNullabilityChecked) {
163 		// TODO: The order in which entities are saved may matter (e.g., a particular transient entity
164 		//       may need to be saved before other transient entities can be saved;
165 		//       Keep retrying the batch of transient entities until either:
166 		//       1) there are no transient entities left in transientCopyCache
167 		//       or 2) no transient entities were saved in the last batch
168 		// For now, just run through the transient entities and retry the merge
169 		for ( Iterator it=transientCopyCache.entrySet().iterator(); it.hasNext(); ) {
170 			Map.Entry mapEntry = ( Map.Entry ) it.next();
171 			Object entity = mapEntry.getKey();
172 			Object copy = transientCopyCache.get( entity );
173 			EntityEntry copyEntry = event.getSession().getPersistenceContext().getEntry( copy );
174 			mergeTransientEntity(
175 					entity,
176 					copyEntry.getEntityName(),
177 					( entity == event.getEntity() ? event.getRequestedId() : copyEntry.getId() ),
178 					event.getSession(),
179 					copyCache,
180 					isNullabilityChecked
181 			);
182 		}
183 	}

185 	/** 
186 	 * Handle the given merge event.
187 	 *
188 	 * @param event The merge event to be handled.
189 	 * @throws HibernateException
190 	 */
191 	public void onMerge(MergeEvent event, Map copiedAlready) throws HibernateException {
192 
193 		final EventCache copyCache = ( EventCache ) copiedAlready;
194 		final EventSource source = event.getSession();
195 		final Object original = event.getOriginal();
196 
197 		if ( original != null ) {
198 
199 			final Object entity;
200 			if ( original instanceof HibernateProxy ) {
201 				LazyInitializer li = ( (HibernateProxy) original ).getHibernateLazyInitializer();
202 				if ( li.isUninitialized() ) {
203 					log.trace("ignoring uninitialized proxy");
204 					event.setResult( source.load( li.getEntityName(), li.getIdentifier() ) );
205 					return; //EARLY EXIT!
206 				}
207 				else {
208 					entity = li.getImplementation();
209 				}
210 			}
211 			else {
212 				entity = original;
213 			}
214 
215 			if ( copyCache.containsKey( entity ) &&
216 					( copyCache.isOperatedOn( entity ) ) ) {
217 				log.trace("already in merge process");
218 				event.setResult( entity );				
219 			}
220 			else {
221 				if ( copyCache.containsKey( entity ) ) {
222 					log.trace("already in copyCache; setting in merge process");					
223 					copyCache.setOperatedOn( entity, true );
224 				}
225 				event.setEntity( entity );
226 				int entityState = -1;
227 
228 				// Check the persistence context for an entry relating to this
229 				// entity to be merged...
230 				EntityEntry entry = source.getPersistenceContext().getEntry( entity );
231 				if ( entry == null ) {
232 					EntityPersister persister = source.getEntityPersister( event.getEntityName(), entity );
233 					Serializable id = persister.getIdentifier( entity, source );
234 					if ( id != null ) {
235 						EntityKey key = new EntityKey( id, persister, source.getEntityMode() );
236 						Object managedEntity = source.getPersistenceContext().getEntity( key );
237 						entry = source.getPersistenceContext().getEntry( managedEntity );
238 						if ( entry != null ) {
239 							// we have specialized case of a detached entity from the
240 							// perspective of the merge operation.  Specifically, we
241 							// have an incoming entity instance which has a corresponding
242 							// entry in the current persistence context, but registered
243 							// under a different entity instance
244 							entityState = DETACHED;
245 						}
246 					}
247 				}
248 
249 				if ( entityState == -1 ) {
250 					entityState = getEntityState( entity, event.getEntityName(), entry, source );
251 				}
252 				
253 				switch (entityState) {
254 					case DETACHED:
255 						entityIsDetached(event, copyCache);
256 						break;
257 					case TRANSIENT:
258 						entityIsTransient(event, copyCache);
259 						break;
260 					case PERSISTENT:
261 						entityIsPersistent(event, copyCache);
262 						break;
263 					default: //DELETED
264 						throw new ObjectDeletedException(
265 								"deleted instance passed to merge", 
266 								null,
267 								getLoggableName( event.getEntityName(), entity )
268 							);			
269 				}
270 			}
271 			
272 		}
273 		
274 	}

276 	protected void entityIsPersistent(MergeEvent event, Map copyCache) {
277 		log.trace("ignoring persistent instance");
278 		
279 		//TODO: check that entry.getIdentifier().equals(requestedId)
280 		
281 		final Object entity = event.getEntity();
282 		final EventSource source = event.getSession();
283 		final EntityPersister persister = source.getEntityPersister( event.getEntityName(), entity );
284 
285 		( ( EventCache ) copyCache ).put( entity, entity, true  );  //before cascade!
286 		
287 		cascadeOnMerge(source, persister, entity, copyCache);
288 		copyValues(persister, entity, entity, source, copyCache);
289 		
290 		event.setResult(entity);
291 	}

294 		
295 		log.trace("merging transient instance");
296 		
297 		final Object entity = event.getEntity();
298 		final EventSource source = event.getSession();
299 
300 		final EntityPersister persister = source.getEntityPersister( event.getEntityName(), entity );
301 		final String entityName = persister.getEntityName();
302 
303 		event.setResult( mergeTransientEntity( entity, entityName, event.getRequestedId(), source, copyCache, true ) );
304 	}

310 	private Object mergeTransientEntity(
311 			Object entity,
312 			String entityName,
313 			Serializable requestedId,
314 			EventSource source,
315 			Map copyCache,
316 			boolean isNullabilityChecked) {
317 
318 		log.trace("merging transient instance");
319 
320 		final EntityPersister persister = source.getEntityPersister( entityName, entity );
321 
322 		final Serializable id = persister.hasIdentifierProperty() ?
323 				persister.getIdentifier( entity, source ) :
324 		        null;
325 		if ( copyCache.containsKey( entity ) ) {
326 			persister.setIdentifier( copyCache.get( entity ), id, source );
327 		}
328 		else {
329 			( ( EventCache ) copyCache ).put( entity, source.instantiate( persister, id ), true ); //before cascade!
330 		}
331 		final Object copy = copyCache.get( entity );
332 
333 		// cascade first, so that all unsaved objects get their
334 		// copy created before we actually copy
335 		//cascadeOnMerge(event, persister, entity, copyCache, Cascades.CASCADE_BEFORE_MERGE);
336 		super.cascadeBeforeSave(source, persister, entity, copyCache);
337 		copyValues(persister, entity, copy, source, copyCache, ForeignKeyDirection.FOREIGN_KEY_FROM_PARENT);
338 
339 		try {
340 			// try saving; check for non-nullable properties that are null or transient entities before saving
341 			saveTransientEntity( copy, entityName, requestedId, source, copyCache, isNullabilityChecked );
342 		}
343 		catch (PropertyValueException ex) {
344 			String propertyName = ex.getPropertyName();
345 			Object propertyFromCopy = persister.getPropertyValue( copy, propertyName, source.getEntityMode() );
346 			Object propertyFromEntity = persister.getPropertyValue( entity, propertyName, source.getEntityMode() );
347 			Type propertyType = persister.getPropertyType( propertyName );
348 			EntityEntry copyEntry = source.getPersistenceContext().getEntry( copy );
349 			if ( propertyFromCopy == null ||
350 					propertyFromEntity == null ||
351 					! propertyType.isEntityType() ||
352 					! copyCache.containsKey( propertyFromEntity ) ) {
353 				if ( log.isTraceEnabled() ) {
354 					String fullPropertyName = "property '" + copyEntry.getEntityName() + "." + propertyName;
355 					log.trace( fullPropertyName + " in copy is " + ( propertyFromCopy == null ? "null" : propertyFromCopy ) );
356 					log.trace( fullPropertyName + " in original is " + ( propertyFromCopy == null ? "null" : propertyFromCopy ) );
357 					log.trace( fullPropertyName + ( propertyType.isEntityType() ? " is" : " is not" ) + " an entity type" );
358 					if ( propertyFromEntity != null && ! copyCache.containsKey( propertyFromEntity ) ) {
359 						log.trace( fullPropertyName + " is not in copy cache" );
360 					}
361 				}
362 				if ( isNullabilityCheckedGlobal( source ) ) {
363 					throw ex;
364 				}
365 				else {
366 					// retry save w/o checking for non-nullable properties
367 					// (the failure will be detected later)
368 					saveTransientEntity( copy, entityName, requestedId, source, copyCache, false );
369 				}
370 			}
371 			if ( log.isTraceEnabled() && propertyFromEntity != null ) {
372 				if ( ( ( EventCache ) copyCache ).isOperatedOn( propertyFromEntity ) ) {
373 					log.trace( "property '" + copyEntry.getEntityName() + "." + propertyName +
374 							"' from original entity is in copyCache and is in the process of being merged; " +
375 							propertyName + " =["+propertyFromEntity+"]");
376 				}
377 				else {
378 					log.trace( "property '" + copyEntry.getEntityName() + "." + propertyName +
379 							"' from original entity is in copyCache and is not in the process of being merged; " +
380 							propertyName + " =["+propertyFromEntity+"]");
381 				}
382 			}
383 			// continue...; we'll find out if it ends up not getting saved later
384 		}
385 
386 		// cascade first, so that all unsaved objects get their
387 		// copy created before we actually copy
388 		super.cascadeAfterSave(source, persister, entity, copyCache);
389 		copyValues(persister, entity, copy, source, copyCache, ForeignKeyDirection.FOREIGN_KEY_TO_PARENT);
390 
391 		return copy;
392 
393 	}

395 	private boolean isNullabilityCheckedGlobal(EventSource source) {
396 		return source.getFactory().getSettings().isCheckNullability();
397 	}

399 	private void saveTransientEntity(
400 			Object entity,
401 			String entityName,
402 			Serializable requestedId,
403 			EventSource source,
404 			Map copyCache,
405 			boolean isNullabilityChecked) {
406 
407 		boolean isNullabilityCheckedOrig =
408 			source.getFactory().getSettings().isCheckNullability();
409 		try {
410 			source.getFactory().getSettings().setCheckNullability( isNullabilityChecked );
411 			//this bit is only *really* absolutely necessary for handling
412 			//requestedId, but is also good if we merge multiple object
413 			//graphs, since it helps ensure uniqueness
414 			if (requestedId==null) {
415 				saveWithGeneratedId( entity, entityName, copyCache, source, false );
416 			}
417 			else {
418 				saveWithRequestedId( entity, requestedId, entityName, copyCache, source );
419 			}
420 		}
421 		finally {
422 			source.getFactory().getSettings().setCheckNullability( isNullabilityCheckedOrig );
423 		}
424 	}
425 	protected void entityIsDetached(MergeEvent event, Map copyCache) {
426 		
427 		log.trace("merging detached instance");
428 		
429 		final Object entity = event.getEntity();
430 		final EventSource source = event.getSession();
431 
432 		final EntityPersister persister = source.getEntityPersister( event.getEntityName(), entity );
433 		final String entityName = persister.getEntityName();
434 
435 		Serializable id = event.getRequestedId();
436 		if ( id == null ) {
437 			id = persister.getIdentifier( entity, source );
438 		}
439 		else {
440 			// check that entity id = requestedId
441 			Serializable entityId = persister.getIdentifier( entity, source );
442 			if ( !persister.getIdentifierType().isEqual( id, entityId, source.getEntityMode(), source.getFactory() ) ) {
443 				throw new HibernateException( "merge requested with id not matching id of passed entity" );
444 			}
445 		}
446 		
447 		String previousFetchProfile = source.getFetchProfile();
448 		source.setFetchProfile("merge");
449 		//we must clone embedded composite identifiers, or 
450 		//we will get back the same instance that we pass in
451 		final Serializable clonedIdentifier = (Serializable) persister.getIdentifierType()
452 				.deepCopy( id, source.getEntityMode(), source.getFactory() );
453 		final Object result = source.get(entityName, clonedIdentifier);
454 		source.setFetchProfile(previousFetchProfile);
455 		
456 		if ( result == null ) {
457 			//TODO: we should throw an exception if we really *know* for sure  
458 			//      that this is a detached instance, rather than just assuming
459 			//throw new StaleObjectStateException(entityName, id);
460 			
461 			// we got here because we assumed that an instance
462 			// with an assigned id was detached, when it was
463 			// really persistent
464 			entityIsTransient(event, copyCache);
465 		}
466 		else {
467 			( ( EventCache ) copyCache ).put( entity, result, true ); //before cascade!
468 	
469 			final Object target = source.getPersistenceContext().unproxy(result);
470 			if ( target == entity ) {
471 				throw new AssertionFailure("entity was not detached");
472 			}
473 			else if ( !source.getEntityName(target).equals(entityName) ) {
474 				throw new WrongClassException(
475 						"class of the given object did not match class of persistent copy",
476 						event.getRequestedId(),
477 						entityName
478 					);
479 			}
480 			else if ( isVersionChanged( entity, source, persister, target ) ) {
481 				if ( source.getFactory().getStatistics().isStatisticsEnabled() ) {
482 					source.getFactory().getStatisticsImplementor()
483 							.optimisticFailure( entityName );
484 				}
485 				throw new StaleObjectStateException( entityName, id );
486 			}
487 	
488 			// cascade first, so that all unsaved objects get their 
489 			// copy created before we actually copy
490 			cascadeOnMerge(source, persister, entity, copyCache);
491 			copyValues(persister, entity, target, source, copyCache);
492 			
493 			//copyValues works by reflection, so explicitly mark the entity instance dirty
494 			markInterceptorDirty( entity, target );
495 			
496 			event.setResult(result);
497 		}
498 
499 	}

501 	private void markInterceptorDirty(final Object entity, final Object target) {
502 		if ( FieldInterceptionHelper.isInstrumented( entity ) ) {
503 			FieldInterceptor interceptor = FieldInterceptionHelper.extractFieldInterceptor( target );
504 			if ( interceptor != null ) {
505 				interceptor.dirty();
506 			}
507 		}
508 	}

510 	private boolean isVersionChanged(Object entity, EventSource source, EntityPersister persister, Object target) {
511 		if ( ! persister.isVersioned() ) {
512 			return false;
513 		}
514 		// for merging of versioned entities, we consider the version having
515 		// been changed only when:
516 		// 1) the two version values are different;
517 		//      *AND*
518 		// 2) The target actually represents database state!
519 		//
520 		// This second condition is a special case which allows
521 		// an entity to be merged during the same transaction
522 		// (though during a seperate operation) in which it was
523 		// originally persisted/saved
524 		boolean changed = ! persister.getVersionType().isSame(
525 				persister.getVersion( target, source.getEntityMode() ),
526 				persister.getVersion( entity, source.getEntityMode() ),
527 				source.getEntityMode()
528 		);
529 
530 		// TODO : perhaps we should additionally require that the incoming entity
531 		// version be equivalent to the defined unsaved-value?
532 		return changed && existsInDatabase( target, source, persister );
533 	}

535 	private boolean existsInDatabase(Object entity, EventSource source, EntityPersister persister) {
536 		EntityEntry entry = source.getPersistenceContext().getEntry( entity );
537 		if ( entry == null ) {
538 			Serializable id = persister.getIdentifier( entity, source );
539 			if ( id != null ) {
540 				EntityKey key = new EntityKey( id, persister, source.getEntityMode() );
541 				Object managedEntity = source.getPersistenceContext().getEntity( key );
542 				entry = source.getPersistenceContext().getEntry( managedEntity );
543 			}
544 		}
545 
546 		if ( entry == null ) {
547 			// perhaps this should be an exception since it is only ever used
548 			// in the above method?
549 			return false;
550 		}
551 		else {
552 			return entry.isExistsInDatabase();
553 		}
554 	}

556 	protected void copyValues(
557 		final EntityPersister persister, 
558 		final Object entity, 
559 		final Object target, 
560 		final SessionImplementor source,
561 		final Map copyCache
562 	) {
563 		final Object[] copiedValues = TypeHelper.replace(
564 				persister.getPropertyValues( entity, source.getEntityMode() ),
565 				persister.getPropertyValues( target, source.getEntityMode() ),
566 				persister.getPropertyTypes(),
567 				source,
568 				target, 
569 				copyCache
570 			);
571 
572 		persister.setPropertyValues( target, copiedValues, source.getEntityMode() );
573 	}

575 	protected void copyValues(
576 			final EntityPersister persister,
577 			final Object entity, 
578 			final Object target, 
579 			final SessionImplementor source,
580 			final Map copyCache,
581 			final ForeignKeyDirection foreignKeyDirection) {
582 
583 		final Object[] copiedValues;
584 
585 		if ( foreignKeyDirection == ForeignKeyDirection.FOREIGN_KEY_TO_PARENT ) {
586 			// this is the second pass through on a merge op, so here we limit the
587 			// replacement to associations types (value types were already replaced
588 			// during the first pass)
589 			copiedValues = TypeHelper.replaceAssociations(
590 					persister.getPropertyValues( entity, source.getEntityMode() ),
591 					persister.getPropertyValues( target, source.getEntityMode() ),
592 					persister.getPropertyTypes(),
593 					source,
594 					target,
595 					copyCache,
596 					foreignKeyDirection
597 			);
598 		}
599 		else {
600 			copiedValues = TypeHelper.replace(
601 					persister.getPropertyValues( entity, source.getEntityMode() ),
602 					persister.getPropertyValues( target, source.getEntityMode() ),
603 					persister.getPropertyTypes(),
604 					source,
605 					target,
606 					copyCache,
607 					foreignKeyDirection
608 			);
609 		}
610 
611 		persister.setPropertyValues( target, copiedValues, source.getEntityMode() );
612 	}
613 
614 	/** 
615 	 * Perform any cascades needed as part of this copy event.
616 	 *
617 	 * @param source The merge event being processed.
618 	 * @param persister The persister of the entity being copied.
619 	 * @param entity The entity being copied.
620 	 * @param copyCache A cache of already copied instance.
621 	 */
622 	protected void cascadeOnMerge(
623 		final EventSource source,
624 		final EntityPersister persister,
625 		final Object entity,
626 		final Map copyCache
627 	) {
628 		source.getPersistenceContext().incrementCascadeLevel();
629 		try {
630 			new Cascade( getCascadeAction(), Cascade.BEFORE_MERGE, source )
631 					.cascade(persister, entity, copyCache);
632 		}
633 		finally {
634 			source.getPersistenceContext().decrementCascadeLevel();
635 		}
636 	}
637 
638 
639 	protected CascadingAction getCascadeAction() {
640 		return CascadingAction.MERGE;
641 	}
642 
643 	protected Boolean getAssumedUnsaved() {
644 		return Boolean.FALSE;
645 	}
