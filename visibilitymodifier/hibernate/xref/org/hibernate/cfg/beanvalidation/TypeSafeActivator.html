74  	public static void activateBeanValidation(EventListeners eventListeners, Properties properties) {
75  		ValidatorFactory factory = getValidatorFactory( properties );
76  		BeanValidationEventListener beanValidationEventListener = new BeanValidationEventListener(
77  				factory, properties
78  		);
79  
80  		{
81  			PreInsertEventListener[] listeners = eventListeners.getPreInsertEventListeners();
82  			int length = listeners.length + 1;
83  			PreInsertEventListener[] newListeners = new PreInsertEventListener[length];
84  			System.arraycopy( listeners, 0, newListeners, 0, length - 1 );
85  			newListeners[length - 1] = beanValidationEventListener;
86  			eventListeners.setPreInsertEventListeners( newListeners );
87  		}
88  
89  		{
90  			PreUpdateEventListener[] listeners = eventListeners.getPreUpdateEventListeners();
91  			int length = listeners.length + 1;
92  			PreUpdateEventListener[] newListeners = new PreUpdateEventListener[length];
93  			System.arraycopy( listeners, 0, newListeners, 0, length - 1 );
94  			newListeners[length - 1] = beanValidationEventListener;
95  			eventListeners.setPreUpdateEventListeners( newListeners );
96  		}
97  
98  		{
99  			PreDeleteEventListener[] listeners = eventListeners.getPreDeleteEventListeners();
100 			int length = listeners.length + 1;
101 			PreDeleteEventListener[] newListeners = new PreDeleteEventListener[length];
102 			System.arraycopy( listeners, 0, newListeners, 0, length - 1 );
103 			newListeners[length - 1] = beanValidationEventListener;
104 			eventListeners.setPreDeleteEventListeners( newListeners );
105 		}
106 	}

108 	public static void applyDDL(Collection<PersistentClass> persistentClasses, Properties properties) {
109 		ValidatorFactory factory = getValidatorFactory( properties );
110 		Class<?>[] groupsArray = new GroupsPerOperation( properties ).get( GroupsPerOperation.Operation.DDL );
111 		Set<Class<?>> groups = new HashSet<Class<?>>( Arrays.asList( groupsArray ) );
112 
113 		for ( PersistentClass persistentClass : persistentClasses ) {
114 			final String className = persistentClass.getClassName();
115 
116 			if ( className == null || className.length() == 0 ) {
117 				continue;
118 			}
119 			Class<?> clazz;
120 			try {
121 				clazz = ReflectHelper.classForName( className, TypeSafeActivator.class );
122 			}
123 			catch ( ClassNotFoundException e ) {
124 				throw new AssertionFailure( "Entity class not found", e );
125 			}
126 
127 			try {
128 				applyDDL( "", persistentClass, clazz, factory, groups, true );
129 			}
130 			catch ( Exception e ) {
131 				logger.warn( "Unable to apply constraints on DDL for " + className, e );
132 			}
133 		}
134 	}

136 	private static void applyDDL(String prefix,
137 								 PersistentClass persistentClass,
138 								 Class<?> clazz,
139 								 ValidatorFactory factory,
140 								 Set<Class<?>> groups,
141 								 boolean activateNotNull) {
142 		final BeanDescriptor descriptor = factory.getValidator().getConstraintsForClass( clazz );
143 		//no bean level constraints can be applied, go to the properties
144 
145 		for ( PropertyDescriptor propertyDesc : descriptor.getConstrainedProperties() ) {
146 			Property property = findPropertyByName( persistentClass, prefix + propertyDesc.getPropertyName() );
147 			boolean hasNotNull;
148 			if ( property != null ) {
149 				hasNotNull = applyConstraints(
150 						propertyDesc.getConstraintDescriptors(), property, propertyDesc, groups, activateNotNull
151 				);
152 				if ( property.isComposite() && propertyDesc.isCascaded() ) {
153 					Class<?> componentClass = ( (Component) property.getValue() ).getComponentClass();
154 
155 					/*
156 					 * we can apply not null if the upper component let's us activate not null
157 					 * and if the property is not null.
158 					 * Otherwise, all sub columns should be left nullable
159 					 */
160 					final boolean canSetNotNullOnColumns = activateNotNull && hasNotNull;
161 					applyDDL(
162 							prefix + propertyDesc.getPropertyName() + ".",
163 							persistentClass, componentClass, factory, groups,
164 							canSetNotNullOnColumns
165 					);
166 				}
167 				//FIXME add collection of components
168 			}
169 		}
170 	}

172 	private static boolean applyConstraints(Set<ConstraintDescriptor<?>> constraintDescriptors,
173 											Property property,
174 											PropertyDescriptor propertyDesc,
175 											Set<Class<?>> groups,
176 											boolean canApplyNotNull
177 	) {
178 		boolean hasNotNull = false;
179 		for ( ConstraintDescriptor<?> descriptor : constraintDescriptors ) {
180 			if ( groups != null && Collections.disjoint( descriptor.getGroups(), groups ) ) {
181 				continue;
182 			}
183 
184 			if ( canApplyNotNull ) {
185 				hasNotNull = hasNotNull || applyNotNull( property, descriptor );
186 			}
187 
188 			// apply bean validation specific constraints
189 			applyDigits( property, descriptor );
190 			applySize( property, descriptor, propertyDesc );
191 			applyMin( property, descriptor );
192 			applyMax( property, descriptor );
193 
194 			// apply hibernate validator specific constraints - we cannot import any HV specific classes though!
195 			// no need to check explicitly for @Range. @Range is a composed constraint using @Min and @Max which
196 			// will be taken care later
197 			applyLength( property, descriptor, propertyDesc );
198 
199 			// pass an empty set as composing constraints inherit the main constraint and thus are matching already
200 			hasNotNull = hasNotNull || applyConstraints(
201 					descriptor.getComposingConstraints(),
202 					property, propertyDesc, null,
203 					canApplyNotNull
204 			);
205 		}
206 		return hasNotNull;
207 	}

209 	private static void applyMin(Property property, ConstraintDescriptor<?> descriptor) {
210 		if ( Min.class.equals( descriptor.getAnnotation().annotationType() ) ) {
211 			@SuppressWarnings("unchecked")
212 			ConstraintDescriptor<Min> minConstraint = (ConstraintDescriptor<Min>) descriptor;
213 			long min = minConstraint.getAnnotation().value();
214 
215 			Column col = (Column) property.getColumnIterator().next();
216 			String checkConstraint = col.getName() + ">=" + min;
217 			applySQLCheck( col, checkConstraint );
218 		}
219 	}
220 
221 	private static void applyMax(Property property, ConstraintDescriptor<?> descriptor) {
222 		if ( Max.class.equals( descriptor.getAnnotation().annotationType() ) ) {
223 			@SuppressWarnings("unchecked")
224 			ConstraintDescriptor<Max> maxConstraint = (ConstraintDescriptor<Max>) descriptor;
225 			long max = maxConstraint.getAnnotation().value();
226 			Column col = (Column) property.getColumnIterator().next();
227 			String checkConstraint = col.getName() + "<=" + max;
228 			applySQLCheck( col, checkConstraint );
229 		}
230 	}

232 	private static void applySQLCheck(Column col, String checkConstraint) {
233 		String existingCheck = col.getCheckConstraint();
234 		// need to check whether the new check is already part of the existing check, because applyDDL can be called
235 		// multiple times
236 		if ( StringHelper.isNotEmpty( existingCheck ) && !existingCheck.contains( checkConstraint ) ) {
237 			checkConstraint = col.getCheckConstraint() + " AND " + checkConstraint;
238 		}
239 		col.setCheckConstraint( checkConstraint );
240 	}

242 	private static boolean applyNotNull(Property property, ConstraintDescriptor<?> descriptor) {
243 		boolean hasNotNull = false;
244 		if ( NotNull.class.equals( descriptor.getAnnotation().annotationType() ) ) {
245 			if ( !( property.getPersistentClass() instanceof SingleTableSubclass ) ) {
246 				//single table should not be forced to null
247 				if ( !property.isComposite() ) { //composite should not add not-null on all columns
248 					@SuppressWarnings("unchecked")
249 					Iterator<Column> iter = (Iterator<Column>) property.getColumnIterator();
250 					while ( iter.hasNext() ) {
251 						iter.next().setNullable( false );
252 						hasNotNull = true;
253 					}
254 				}
255 			}
256 			hasNotNull = true;
257 		}
258 		return hasNotNull;
259 	}

261 	private static void applyDigits(Property property, ConstraintDescriptor<?> descriptor) {
262 		if ( Digits.class.equals( descriptor.getAnnotation().annotationType() ) ) {
263 			@SuppressWarnings("unchecked")
264 			ConstraintDescriptor<Digits> digitsConstraint = (ConstraintDescriptor<Digits>) descriptor;
265 			int integerDigits = digitsConstraint.getAnnotation().integer();
266 			int fractionalDigits = digitsConstraint.getAnnotation().fraction();
267 			Column col = (Column) property.getColumnIterator().next();
268 			col.setPrecision( integerDigits + fractionalDigits );
269 			col.setScale( fractionalDigits );
270 		}
271 	}
272 
273 	private static void applySize(Property property, ConstraintDescriptor<?> descriptor, PropertyDescriptor propertyDescriptor) {
274 		if ( Size.class.equals( descriptor.getAnnotation().annotationType() )
275 				&& String.class.equals( propertyDescriptor.getElementClass() ) ) {
276 			@SuppressWarnings("unchecked")
277 			ConstraintDescriptor<Size> sizeConstraint = (ConstraintDescriptor<Size>) descriptor;
278 			int max = sizeConstraint.getAnnotation().max();
279 			Column col = (Column) property.getColumnIterator().next();
280 			if ( max < Integer.MAX_VALUE ) {
281 				col.setLength( max );
282 			}
283 		}
284 	}
285 
286 	private static void applyLength(Property property, ConstraintDescriptor<?> descriptor, PropertyDescriptor propertyDescriptor) {
287 		if ( "org.hibernate.validator.constraints.Length".equals(
288 				descriptor.getAnnotation().annotationType().getName()
289 		)
290 				&& String.class.equals( propertyDescriptor.getElementClass() ) ) {
291 			@SuppressWarnings("unchecked")
292 			int max = (Integer) descriptor.getAttributes().get( "max" );
293 			Column col = (Column) property.getColumnIterator().next();
294 			if ( max < Integer.MAX_VALUE ) {
295 				col.setLength( max );
296 			}
297 		}
298 	}

300 	/**
301 	 * Retrieve the property by path in a recursive way, including IndentifierProperty in the loop
302 	 * If propertyName is null or empty, the IdentifierProperty is returned
303 	 */
304 	private static Property findPropertyByName(PersistentClass associatedClass, String propertyName) {
305 		Property property = null;
306 		Property idProperty = associatedClass.getIdentifierProperty();
307 		String idName = idProperty != null ? idProperty.getName() : null;
308 		try {
309 			if ( propertyName == null
310 					|| propertyName.length() == 0
311 					|| propertyName.equals( idName ) ) {
312 				//default to id
313 				property = idProperty;
314 			}
315 			else {
316 				if ( propertyName.indexOf( idName + "." ) == 0 ) {
317 					property = idProperty;
318 					propertyName = propertyName.substring( idName.length() + 1 );
319 				}
320 				StringTokenizer st = new StringTokenizer( propertyName, ".", false );
321 				while ( st.hasMoreElements() ) {
322 					String element = (String) st.nextElement();
323 					if ( property == null ) {
324 						property = associatedClass.getProperty( element );
325 					}
326 					else {
327 						if ( !property.isComposite() ) {
328 							return null;
329 						}
330 						property = ( (Component) property.getValue() ).getProperty( element );
331 					}
332 				}
333 			}
334 		}
335 		catch ( MappingException e ) {
336 			try {
337 				//if we do not find it try to check the identifier mapper
338 				if ( associatedClass.getIdentifierMapper() == null ) {
339 					return null;
340 				}
341 				StringTokenizer st = new StringTokenizer( propertyName, ".", false );
342 				while ( st.hasMoreElements() ) {
343 					String element = (String) st.nextElement();
344 					if ( property == null ) {
345 						property = associatedClass.getIdentifierMapper().getProperty( element );
346 					}
347 					else {
348 						if ( !property.isComposite() ) {
349 							return null;
350 						}
351 						property = ( (Component) property.getValue() ).getProperty( element );
352 					}
353 				}
354 			}
355 			catch ( MappingException ee ) {
356 				return null;
357 			}
358 		}
359 		return property;
360 	}

362 	private static ValidatorFactory getValidatorFactory(Map<Object, Object> properties) {
363 		ValidatorFactory factory = null;
364 		if ( properties != null ) {
365 			Object unsafeProperty = properties.get( FACTORY_PROPERTY );
366 			if ( unsafeProperty != null ) {
367 				try {
368 					factory = ValidatorFactory.class.cast( unsafeProperty );
369 				}
370 				catch ( ClassCastException e ) {
371 					throw new HibernateException(
372 							"Property " + FACTORY_PROPERTY
373 									+ " should contain an object of type " + ValidatorFactory.class.getName()
374 					);
375 				}
376 			}
377 		}
378 		if ( factory == null ) {
379 			try {
380 				factory = Validation.buildDefaultValidatorFactory();
381 			}
382 			catch ( Exception e ) {
383 				throw new HibernateException( "Unable to build the default ValidatorFactory", e );
384 			}
385 		}
386 		return factory;
387 	}
