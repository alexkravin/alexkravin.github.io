179 /**
180  * An instance of <tt>Configuration</tt> allows the application
181  * to specify properties and mapping documents to be used when
182  * creating a <tt>SessionFactory</tt>. Usually an application will create
183  * a single <tt>Configuration</tt>, build a single instance of
184  * <tt>SessionFactory</tt> and then instantiate <tt>Session</tt>s in
185  * threads servicing client requests. The <tt>Configuration</tt> is meant

187  * immutable and do not retain any association back to the
188  * <tt>Configuration</tt>.<br>
189  * <br>
190  * A new <tt>Configuration</tt> will use the properties specified in

518 	public void add(XmlDocument metadataXml) {
519 		if ( inSecondPass || !isOrmXml( metadataXml ) ) {
520 			metadataSourceQueue.add( metadataXml );
521 		}
522 		else {
523 			final MetadataProvider metadataProvider = ( (MetadataProviderInjector) reflectionManager ).getMetadataProvider();
524 			JPAMetadataProvider jpaMetadataProvider = ( JPAMetadataProvider ) metadataProvider;
525 			List<String> classNames = jpaMetadataProvider.getXMLContext().addDocument( metadataXml.getDocumentTree() );
526 			for ( String className : classNames ) {
527 				try {
528 					metadataSourceQueue.add( reflectionManager.classForName( className, this.getClass() ) );
529 				}
530 				catch ( ClassNotFoundException e ) {
531 					throw new AnnotationException( "Unable to load class defined in XML: " + className, e );
532 				}
533 			}
534 		}
535 	}

541 	/**
542 	 * Add a cached mapping file.  A cached file is a serialized representation
543 	 * of the DOM structure of a particular mapping.  It is saved from a previous
544 	 * call as a file with the name <tt>xmlFile + ".bin"</tt> where xmlFile is
545 	 * the name of the original mapping file.
546 	 * </p>
547 	 * If a cached <tt>xmlFile + ".bin"</tt> exists and is newer than
548 	 * <tt>xmlFile</tt> the <tt>".bin"</tt> file will be read directly. Otherwise
549 	 * xmlFile is read and then serialized to <tt>xmlFile + ".bin"</tt> for use
550 	 * the next time.
551 	 *
552 	 * @param xmlFile The cacheable mapping file to be added.
553 	 * @return this (for method chaining purposes)
554 	 * @throws MappingException Indicates problems reading the cached file or processing
555 	 * the non-cached file.
556 	 */
557 	public Configuration addCacheableFile(File xmlFile) throws MappingException {
558 		File cachedFile = determineCachedDomFile( xmlFile );
559 
560 		try {
561 			return addCacheableFileStrictly( xmlFile );
562 		}
563 		catch ( SerializationException e ) {
564 			log.warn( "Could not deserialize cache file: " + cachedFile.getPath() + " : " + e );
565 		}
566 		catch ( FileNotFoundException e ) {
567 			log.warn( "I/O reported cached file could not be found : " + cachedFile.getPath() + " : " + e );
568 		}
569 
570 		final String name = xmlFile.getAbsolutePath();
571 		final FileInputStream fileInputStream;
572 		try {
573 			fileInputStream = new FileInputStream( xmlFile );
574 		}
575 		catch ( FileNotFoundException e ) {
576 			throw new MappingNotFoundException( "file", xmlFile.toString() );
577 		}
578 		final XmlDocument metadataXml;
579 		try {
580 			final InputSource inputSource = new InputSource( fileInputStream );
581 			log.info( "Reading mappings from file: " + xmlFile );
582 			metadataXml = add( inputSource, "file", name );
583 		}
584 		finally {
585 			try {
586 				fileInputStream.close();
587 			}
588 			catch ( IOException e ) {
589 				log.warn( "I/O exception while closing mapping file : " + cachedFile.getPath() + " : " + e );
590 			}
591 		}
592 
593 		FileOutputStream fileOutputStream = null;
594 		try {
595 			fileOutputStream = new FileOutputStream( cachedFile );
596 		}
597 		catch ( FileNotFoundException e ) {
598 			log.warn( "I/O reported error writing cached file : " + cachedFile.getPath(), e );
599 		}
600 		if ( fileInputStream != null ) {
601 			try {
602 				if ( log.isDebugEnabled() ) {
603 					log.debug( "Writing cache file for: " + xmlFile + " to: " + cachedFile );
604 				}
605 				SerializationHelper.serialize( ( Serializable ) metadataXml.getDocumentTree(), fileOutputStream );
606 			}
607 			catch ( SerializationException e ) {
608 				log.warn( "Could not write cached file: " + cachedFile, e );
609 			}
610 			finally {
611 				try {
612 					fileInputStream.close();
613 				}
614 				catch ( IOException e ) {
615 					log.warn( "I/O exception while closing cache for mapping file : " + cachedFile + " : " + e );
616 				}
617 			}
618 		}
619 
620 		return this;
621 	}

1565 	private void processEndOfQueue(List<FkSecondPass> endOfQueueFkSecondPasses) {
1566 		/*
1567 		 * If a second pass raises a recoverableException, queue it for next round
1568 		 * stop of no pass has to be processed or if the number of pass to processes
1569 		 * does not diminish between two rounds.
1570 		 * If some failing pass remain, raise the original exception
1571 		 */
1572 		boolean stopProcess = false;
1573 		RuntimeException originalException = null;
1574 		while ( !stopProcess ) {
1575 			List<FkSecondPass> failingSecondPasses = new ArrayList<FkSecondPass>();
1576 			Iterator<FkSecondPass> it = endOfQueueFkSecondPasses.listIterator();
1577 			while ( it.hasNext() ) {
1578 				final FkSecondPass pass = it.next();
1579 				try {
1580 					pass.doSecondPass( classes );
1581 				}
1582 				catch ( RecoverableException e ) {

1585 						originalException = ( RuntimeException ) e.getCause();
1586 					}
1587 				}
1588 			}
1589 			stopProcess = failingSecondPasses.size() == 0 || failingSecondPasses.size() == endOfQueueFkSecondPasses.size();
1590 			endOfQueueFkSecondPasses = failingSecondPasses;
1591 		}
1592 		if ( endOfQueueFkSecondPasses.size() > 0 ) {
1593 			throw originalException;
1594 		}
1595 	}
