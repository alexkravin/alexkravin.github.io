46  /**
47   * Provides the basic templating of how instrumentation should occur.
48   *
49   * @author Steve Ebersole
50   */
51  public abstract class AbstractInstrumenter implements Instrumenter {
52  	private static final int ZIP_MAGIC = 0x504B0304;
53  	private static final int CLASS_MAGIC = 0xCAFEBABE;
54  	
55  	protected final Logger logger;
56  	protected final Options options;
57  
58  	/**
59  	 * Creates the basic instrumentation strategy.
60  	 *
61  	 * @param logger The bridge to the environment's logging system.
62  	 * @param options User-supplied options.
63  	 */
64  	public AbstractInstrumenter(Logger logger, Options options) {
65  		this.logger = logger;
66  		this.options = options;
67  	}
68  
69  	/**
70  	 * Given the bytecode of a java class, retrieve the descriptor for that class.
71  	 *
72  	 * @param byecode The class bytecode.
73  	 *
74  	 * @return The class's descriptor
75  	 *
76  	 * @throws Exception Indicates problems access the bytecode.
77  	 */
78  	protected abstract ClassDescriptor getClassDescriptor(byte[] byecode) throws Exception;
79  
80  	/**
81  	 * Create class transformer for the class.
82  	 *
83  	 * @param descriptor The descriptor of the class to be instrumented.
84  	 * @param classNames The names of all classes to be instrumented; the "pipeline" if you will.
85  	 *
86  	 * @return The transformer for the given class; may return null to indicate that transformation should
87  	 * be skipped (ala already instrumented).
88  	 */
89  	protected abstract ClassTransformer getClassTransformer(ClassDescriptor descriptor, Set classNames);
90  
91  	/**
92  	 * The main instrumentation entry point.  Given a set of files, perform instrumentation on each discovered class
93  	 * file.
94  	 *
95  	 * @param files The files.
96  	 */
97  	public void execute(Set files) {
98  		Set classNames = new HashSet();
99  
100 		if ( options.performExtendedInstrumentation() ) {
101 			logger.debug( "collecting class names for extended instrumentation determination" );
102 			try {
103 				Iterator itr = files.iterator();
104 				while ( itr.hasNext() ) {
105 					final File file = ( File ) itr.next();
106 					collectClassNames( file, classNames );
107 				}
108 			}
109 			catch ( ExecutionException ee ) {
110 				throw ee;
111 			}
112 			catch ( Exception e ) {
113 				throw new ExecutionException( e );
114 			}
115 		}
116 
117 		logger.info( "starting instrumentation" );
118 		try {
119 			Iterator itr = files.iterator();
120 			while ( itr.hasNext() ) {
121 				final File file = ( File ) itr.next();
122 				processFile( file, classNames );
123 			}
124 		}
125 		catch ( ExecutionException ee ) {
126 			throw ee;
127 		}
128 		catch ( Exception e ) {
129 			throw new ExecutionException( e );
130 		}
131 	}
132 
133 	/**
134 	 * Extract the names of classes from file, adding them to the classNames collection.
135 	 * <p/>
136 	 * IMPL NOTE : file here may be either a class file or a jar.  If a jar, all entries in the jar file are
137 	 * processed.
138 	 *
139 	 * @param file The file from which to extract class metadata (descriptor).
140 	 * @param classNames The collected class name collection.
141 	 *
142 	 * @throws Exception indicates problems accessing the file or its contents.
143 	 */
144 	private void collectClassNames(File file, final Set classNames) throws Exception {
145 	    if ( isClassFile( file ) ) {
146 			byte[] bytes = ByteCodeHelper.readByteCode( file );
147 			ClassDescriptor descriptor = getClassDescriptor( bytes );
148 		    classNames.add( descriptor.getName() );
149 	    }
150 	    else if ( isJarFile( file ) ) {
151 		    ZipEntryHandler collector = new ZipEntryHandler() {
152 			    public void handleEntry(ZipEntry entry, byte[] byteCode) throws Exception {
153 					if ( !entry.isDirectory() ) {
154 						// see if the entry represents a class file
155 						DataInputStream din = new DataInputStream( new ByteArrayInputStream( byteCode ) );
156 						if ( din.readInt() == CLASS_MAGIC ) {
157 				            classNames.add( getClassDescriptor( byteCode ).getName() );
158 						}
159 					}
160 			    }
161 		    };
162 			ZipFileProcessor processor = new ZipFileProcessor( collector );
163 		    processor.process( file );
164 	    }
165 	}
166 
167 	/**
168 	 * Does this file represent a compiled class?
169 	 *
170 	 * @param file The file to check.
171 	 *
172 	 * @return True if the file is a class; false otherwise.
173 	 *
174 	 * @throws IOException Indicates problem access the file.
175 	 */
176 	protected final boolean isClassFile(File file) throws IOException {
177         return checkMagic( file, CLASS_MAGIC );
178     }
179 
180 	/**
181 	 * Does this file represent a zip file of some format?
182 	 *
183 	 * @param file The file to check.
184 	 *
185 	 * @return True if the file is n archive; false otherwise.
186 	 *
187 	 * @throws IOException Indicates problem access the file.
188 	 */
189     protected final boolean isJarFile(File file) throws IOException {
190         return checkMagic(file, ZIP_MAGIC);
191     }
192 
193 	protected final boolean checkMagic(File file, long magic) throws IOException {
194         DataInputStream in = new DataInputStream( new FileInputStream( file ) );
195         try {
196             int m = in.readInt();
197             return magic == m;
198         }
199         finally {
200             in.close();
201         }
202     }
203 
204 	/**
205 	 * Actually process the file by applying instrumentation transformations to any classes it contains.
206 	 * <p/>
207 	 * Again, just like with {@link #collectClassNames} this method can handle both class and archive files.
208 	 *
209 	 * @param file The file to process.
210 	 * @param classNames The 'pipeline' of classes to be processed.  Only actually populated when the user
211 	 * specifies to perform {@link Options#performExtendedInstrumentation() extended} instrumentation.
212 	 *
213 	 * @throws Exception Indicates an issue either access files or applying the transformations.
214 	 */
215 	protected void processFile(File file, Set classNames) throws Exception {
216 	    if ( isClassFile( file ) ) {
217 			logger.debug( "processing class file : " + file.getAbsolutePath() );
218 	        processClassFile( file, classNames );
219 	    }
220 	    else if ( isJarFile( file ) ) {
221 			logger.debug( "processing jar file : " + file.getAbsolutePath() );
222 	        processJarFile( file, classNames );
223 	    }
224 	    else {
225 		    logger.debug( "ignoring file : " + file.getAbsolutePath() );
226 	    }
227 	}
228 
229 	/**
230 	 * Process a class file.  Delegated to from {@link #processFile} in the case of a class file.
231 	 *
232 	 * @param file The class file to process.
233 	 * @param classNames The 'pipeline' of classes to be processed.  Only actually populated when the user
234 	 * specifies to perform {@link Options#performExtendedInstrumentation() extended} instrumentation.
235 	 *
236 	 * @throws Exception Indicates an issue either access files or applying the transformations.
237 	 */
238 	protected void processClassFile(File file, Set classNames) throws Exception {
239 		byte[] bytes = ByteCodeHelper.readByteCode( file );
240 		ClassDescriptor descriptor = getClassDescriptor( bytes );
241 		ClassTransformer transformer = getClassTransformer( descriptor, classNames );
242 		if ( transformer == null ) {
243 			logger.debug( "no trasformer for class file : " + file.getAbsolutePath() );
244 			return;
245 		}
246 
247 		logger.info( "processing class : " + descriptor.getName() + ";  file = " + file.getAbsolutePath() );
248 		byte[] transformedBytes = transformer.transform(
249 				getClass().getClassLoader(),
250 				descriptor.getName(),
251 				null,
252 				null,
253 				descriptor.getBytes()
254 		);
255 
256 		OutputStream out = new FileOutputStream( file );
257 		try {
258 			out.write( transformedBytes );
259 			out.flush();
260 		}
261 		finally {
262 			try {
263 				out.close();
264 			}
265 			catch ( IOException ignore) {
266 				// intentionally empty
267 			}
268 		}
269 	}

271 	/**
272 	 * Process an archive file.  Delegated to from {@link #processFile} in the case of an archive file.
273 	 *
274 	 * @param file The archive file to process.
275 	 * @param classNames The 'pipeline' of classes to be processed.  Only actually populated when the user
276 	 * specifies to perform {@link Options#performExtendedInstrumentation() extended} instrumentation.
277 	 *
278 	 * @throws Exception Indicates an issue either access files or applying the transformations.
279 	 */
280 	protected void processJarFile(final File file, final Set classNames) throws Exception {
281         File tempFile = File.createTempFile(
282 		        file.getName(),
283 		        null,
284 		        new File( file.getAbsoluteFile().getParent() )
285         );
286 
287         try {
288 			FileOutputStream fout = new FileOutputStream( tempFile, false );
289 			try {
290 				final ZipOutputStream out = new ZipOutputStream( fout );
291 				ZipEntryHandler transformer = new ZipEntryHandler() {
292 					public void handleEntry(ZipEntry entry, byte[] byteCode) throws Exception {
293 								logger.debug( "starting zip entry : " + entry.toString() );
294 								if ( !entry.isDirectory() ) {
295 									// see if the entry represents a class file
296 									DataInputStream din = new DataInputStream( new ByteArrayInputStream( byteCode ) );
297 									if ( din.readInt() == CLASS_MAGIC ) {
298 										ClassDescriptor descriptor = getClassDescriptor( byteCode );
299 										ClassTransformer transformer = getClassTransformer( descriptor, classNames );
300 										if ( transformer == null ) {
301 											logger.debug( "no transformer for zip entry :  " + entry.toString() );
302 										}
303 										else {
304 											logger.info( "processing class : " + descriptor.getName() + ";  entry = " + file.getAbsolutePath() );
305 											byteCode = transformer.transform(
306 													getClass().getClassLoader(),
307 													descriptor.getName(),
308 													null,
309 													null,
310 													descriptor.getBytes()
311 											);
312 										}
313 									}
314 									else {
315 										logger.debug( "ignoring zip entry : " + entry.toString() );
316 									}
317 								}
318 
319 								ZipEntry outEntry = new ZipEntry( entry.getName() );
320 								outEntry.setMethod( entry.getMethod() );
321 								outEntry.setComment( entry.getComment() );
322 								outEntry.setSize( byteCode.length );
323 
324 								if ( outEntry.getMethod() == ZipEntry.STORED ){
325 									CRC32 crc = new CRC32();
326 									crc.update( byteCode );
327 									outEntry.setCrc( crc.getValue() );
328 									outEntry.setCompressedSize( byteCode.length );
329 								}
330 								out.putNextEntry( outEntry );
331 								out.write( byteCode );
332 								out.closeEntry();
333 					}
334 				};
335 				ZipFileProcessor processor = new ZipFileProcessor( transformer );
336 				processor.process( file );
337 				out.close();
338 			}
339 			finally{
340 				fout.close();
341 			}
342 
343             if ( file.delete() ) {
344 	            File newFile = new File( tempFile.getAbsolutePath() );
345                 if( !newFile.renameTo( file ) ) {
346 	                throw new IOException( "can not rename " + tempFile + " to " + file );
347                 }
348             }
349             else {
350 	            throw new IOException( "can not delete " + file );
351             }
352         }
353         finally {
354 	        if ( ! tempFile.delete() ) {
355 				logger.info( "Unable to cleanup temporary jar file : " + tempFile.getAbsolutePath() );
356 			}
357         }
358 	}

360 	/**
361 	 * Allows control over what exacctly to transform.
362 	 */
363 	protected class CustomFieldFilter implements FieldFilter {
364 		private final ClassDescriptor descriptor;
365 		private final Set classNames;
366 
367 		public CustomFieldFilter(ClassDescriptor descriptor, Set classNames) {
368 			this.descriptor = descriptor;
369 			this.classNames = classNames;
370 		}
371 
372 		public boolean shouldInstrumentField(String className, String fieldName) {
373 			if ( descriptor.getName().equals( className ) ) {
374 				logger.trace( "accepting transformation of field [" + className + "." + fieldName + "]" );
375 				return true;
376 			}
377 			else {
378 				logger.trace( "rejecting transformation of field [" + className + "." + fieldName + "]" );
379 				return false;
380 			}
381 		}
382 
383 		public boolean shouldTransformFieldAccess(
384 				String transformingClassName,
385 				String fieldOwnerClassName,
386 				String fieldName) {
387 			if ( descriptor.getName().equals( fieldOwnerClassName ) ) {
388 				logger.trace( "accepting transformation of field access [" + fieldOwnerClassName + "." + fieldName + "]" );
389 				return true;
390 			}
391 			else if ( options.performExtendedInstrumentation() && classNames.contains( fieldOwnerClassName ) ) {
392 				logger.trace( "accepting extended transformation of field access [" + fieldOwnerClassName + "." + fieldName + "]" );
393 				return true;
394 			}
395 			else {
396 				logger.trace( "rejecting transformation of field access [" + fieldOwnerClassName + "." + fieldName + "]; caller = " + transformingClassName  );
397 				return false;
398 			}
399 		}
400 	}

402 	/**
403 	 * General strategy contract for handling entries in an archive file.
404 	 */
405 	private static interface ZipEntryHandler {
406 		/**
407 		 * Apply strategy to the given archive entry.
408 		 *
409 		 * @param entry The archive file entry.
410 		 * @param byteCode
411 		 *
412 		 * @throws Exception
413 		 */
414 		public void handleEntry(ZipEntry entry, byte[] byteCode) throws Exception;
415 	}
416 
417 	/**
418 	 * Applies {@link ZipEntryHandler} strategies to the entries of an archive file.
419 	 */
420 	private static class ZipFileProcessor {
421 		private final ZipEntryHandler entryHandler;
422 
423 		public ZipFileProcessor(ZipEntryHandler entryHandler) {
424 			this.entryHandler = entryHandler;
425 		}
426 
427 		public void process(File file) throws Exception {
428 			ZipInputStream zip = new ZipInputStream( new FileInputStream( file ) );
429 
430 			try {
431 				ZipEntry entry;
432 				while ( (entry = zip.getNextEntry()) != null ) {
433 					byte bytes[] = ByteCodeHelper.readByteCode( zip );
434 					entryHandler.handleEntry( entry, bytes );
435 					zip.closeEntry();
436 				}
437             }
438             finally {
439 	            zip.close();
440             }
441 		}
442 	}
