163 		FromReferenceNode lhs = ( FromReferenceNode ) getFirstChild();
164 		SqlNode property = ( SqlNode ) lhs.getNextSibling();
165 
166 		// Set the attributes of the property reference expression.
167 		String propName = property.getText();
168 		propertyName = propName;
169 		// If the uresolved property path isn't set yet, just use the property name.
170 		if ( propertyPath == null ) {
171 			propertyPath = propName;
172 		}
173 		// Resolve the LHS fully, generate implicit joins.  Pass in the property name so that the resolver can
174 		// discover foreign key (id) properties.
175 		lhs.resolve( true, true, null, this );
176 		setFromElement( lhs.getFromElement() );			// The 'from element' that the property is in.
177 
178 		checkSubclassOrSuperclassPropertyReference( lhs, propName );
179 	}

181 	public void resolveInFunctionCall(boolean generateJoin, boolean implicitJoin) throws SemanticException {
182 		if ( isResolved() ) {
183 			return;
184 		}
185 		Type propertyType = prepareLhs();			// Prepare the left hand side and get the data type.
186 		if ( propertyType!=null && propertyType.isCollectionType() ) {
187 			resolveIndex(null);
188 		}
189 		else {
190 			resolveFirstChild();
191 			super.resolve(generateJoin, implicitJoin);
192 		}
193 	}

204 	public void resolve(boolean generateJoin, boolean implicitJoin, String classAlias, AST parent)
205 	throws SemanticException {
206 		// If this dot has already been resolved, stop now.
207 		if ( isResolved() ) {
208 			return;
209 		}
210 		Type propertyType = prepareLhs(); // Prepare the left hand side and get the data type.
211 
212 		// If there is no data type for this node, and we're at the end of the path (top most dot node), then
213 		// this might be a Java constant.
214 		if ( propertyType == null ) {
215 			if ( parent == null ) {
216 				getWalker().getLiteralProcessor().lookupConstant( this );
217 			}
218 			// If the propertyType is null and there isn't a parent, just
219 			// stop now... there was a problem resolving the node anyway.
220 			return;
221 		}
222 
223 		if ( propertyType.isComponentType() ) {
224 			// The property is a component...
225 			checkLhsIsNotCollection();
226 			dereferenceComponent( parent );
227 			initText();
228 		}
229 		else if ( propertyType.isEntityType() ) {
230 			// The property is another class..
231 			checkLhsIsNotCollection();
232 			dereferenceEntity( ( EntityType ) propertyType, implicitJoin, classAlias, generateJoin, parent );
233 			initText();
234 		}
235 		else if ( propertyType.isCollectionType() ) {
236 			// The property is a collection...
237 			checkLhsIsNotCollection();
238 			dereferenceCollection( ( CollectionType ) propertyType, implicitJoin, false, classAlias, parent );
239 		}
240 		else {
241 			// Otherwise, this is a primitive type.
242 			if ( ! CollectionProperties.isAnyCollectionProperty( propertyName ) ) {
243 				checkLhsIsNotCollection();
244 			}
245 			dereferenceType = DEREF_PRIMITIVE;
246 			initText();
247 		}
248 		setResolved();
249 	}

251 	private void initText() {
252 		String[] cols = getColumns();
253 		String text = StringHelper.join( ", ", cols );
254 		if ( cols.length > 1 && getWalker().isComparativeExpressionClause() ) {
255 			text = "(" + text + ")";
256 		}
257 		setText( text );
258 	}

266 	private void dereferenceCollection(CollectionType collectionType, boolean implicitJoin, boolean indexed, String classAlias, AST parent)
267 	throws SemanticException {
268 
269 		dereferenceType = DEREF_COLLECTION;
270 		String role = collectionType.getRole();
271 
272 		//foo.bars.size (also handles deprecated stuff like foo.bars.maxelement for backwardness)
273 		boolean isSizeProperty = getNextSibling()!=null &&
274 			CollectionProperties.isAnyCollectionProperty( getNextSibling().getText() );
275 
276 		if ( isSizeProperty ) indexed = true; //yuck!
277 
278 		QueryableCollection queryableCollection = getSessionFactoryHelper().requireQueryableCollection( role );
279 		String propName = getPath();
280 		FromClause currentFromClause = getWalker().getCurrentFromClause();
281 
282 		if ( getWalker().getStatementType() != SqlTokenTypes.SELECT && indexed && classAlias == null ) {
283 			// should indicate that we are processing an INSERT/UPDATE/DELETE
284 			// query with a subquery implied via a collection property
285 			// function. Here, we need to use the table name itself as the
286 			// qualification alias.
287 			// TODO : verify this works for all databases...
288 			// TODO : is this also the case in non-"indexed" scenarios?
289 			String alias = getLhs().getFromElement().getQueryable().getTableName();
290 			columns = getFromElement().toColumns( alias, propertyPath, false, true );
291 		}
292 
293 		//We do not look for an existing join on the same path, because
294 		//it makes sense to join twice on the same collection role
295 		FromElementFactory factory = new FromElementFactory(
296 		        currentFromClause,
297 		        getLhs().getFromElement(),
298 		        propName,
299 				classAlias,
300 		        getColumns(),
301 		        implicitJoin
302 		);
303 		FromElement elem = factory.createCollection( queryableCollection, role, joinType, fetch, indexed );
304 
305 		if ( log.isDebugEnabled() ) {
306 			log.debug( "dereferenceCollection() : Created new FROM element for " + propName + " : " + elem );
307 		}
308 
309 		setImpliedJoin( elem );
310 		setFromElement( elem );	// This 'dot' expression now refers to the resulting from element.
311 
312 		if ( isSizeProperty ) {
313 			elem.setText("");
314 			elem.setUseWhereFragment(false);
315 		}
316 
317 		if ( !implicitJoin ) {
318 			EntityPersister entityPersister = elem.getEntityPersister();
319 			if ( entityPersister != null ) {
320 				getWalker().addQuerySpaces( entityPersister.getQuerySpaces() );
321 			}
322 		}
323 		getWalker().addQuerySpaces( queryableCollection.getCollectionSpaces() );	// Always add the collection's query spaces.
324 	}

326 	private void dereferenceEntity(EntityType entityType, boolean implicitJoin, String classAlias, boolean generateJoin, AST parent) throws SemanticException {
327 		checkForCorrelatedSubquery( "dereferenceEntity" );
328 		// three general cases we check here as to whether to render a physical SQL join:
329 		// 1) is our parent a DotNode as well?  If so, our property reference is
330 		// 		being further de-referenced...
331 		// 2) is this a DML statement
332 		// 3) we were asked to generate any needed joins (generateJoins==true) *OR*
333 		//		we are currently processing a select or from clause
334 		// (an additional check is the REGRESSION_STYLE_JOIN_SUPPRESSION check solely intended for the test suite)
335 		//
336 		// The REGRESSION_STYLE_JOIN_SUPPRESSION is an additional check
337 		// intended solely for use within the test suite.  This forces the
338 		// implicit join resolution to behave more like the classic parser.
339 		// The underlying issue is that classic translator is simply wrong
340 		// about its decisions on whether or not to render an implicit join
341 		// into a physical SQL join in a lot of cases.  The piece it generally
342 		// tends to miss is that INNER joins effect the results by further
343 		// restricting the data set!  A particular manifestation of this is
344 		// the fact that the classic translator will skip the physical join
345 		// for ToOne implicit joins *if the query is shallow*; the result
346 		// being that Query.list() and Query.iterate() could return
347 		// different number of results!
348 		DotNode parentAsDotNode = null;
349 		String property = propertyName;
350 		final boolean joinIsNeeded;
351 
352 		if ( isDotNode( parent ) ) {
353 			// our parent is another dot node, meaning we are being further dereferenced.
354 			// thus we need to generate a join unless the parent refers to the associated
355 			// entity's PK (because 'our' table would know the FK).
356 			parentAsDotNode = ( DotNode ) parent;
357 			property = parentAsDotNode.propertyName;
358 			joinIsNeeded = generateJoin && !isReferenceToPrimaryKey( parentAsDotNode.propertyName, entityType );
359 		}
360 		else if ( ! getWalker().isSelectStatement() ) {
361 			// in non-select queries, the only time we should need to join is if we are in a subquery from clause
362 			joinIsNeeded = getWalker().getCurrentStatementType() == SqlTokenTypes.SELECT && getWalker().isInFrom();
363 		}
364 		else if ( REGRESSION_STYLE_JOIN_SUPPRESSION ) {
365 			// this is the regression style determination which matches the logic of the classic translator
366 			joinIsNeeded = generateJoin && ( !getWalker().isInSelect() || !getWalker().isShallowQuery() );
367 		}
368 		else {
369 			joinIsNeeded = generateJoin || ( getWalker().isInSelect() || getWalker().isInFrom() );
370 		}
371 
372 		if ( joinIsNeeded ) {
373 			dereferenceEntityJoin( classAlias, entityType, implicitJoin, parent );
374 		}
375 		else {
376 			dereferenceEntityIdentifier( property, parentAsDotNode );
377 		}
378 
379 	}

385 	private void dereferenceEntityJoin(String classAlias, EntityType propertyType, boolean impliedJoin, AST parent)
386 	throws SemanticException {
387 		dereferenceType = DEREF_ENTITY;
388 		if ( log.isDebugEnabled() ) {
389 			log.debug( "dereferenceEntityJoin() : generating join for " + propertyName + " in "
390 					+ getFromElement().getClassName() + " "
391 					+ ( ( classAlias == null ) ? "{no alias}" : "(" + classAlias + ")" )
392 					+ " parent = " + ASTUtil.getDebugString( parent )
393 			);
394 		}
395 		// Create a new FROM node for the referenced class.
396 		String associatedEntityName = propertyType.getAssociatedEntityName();
397 		String tableAlias = getAliasGenerator().createName( associatedEntityName );
398 
399 		String[] joinColumns = getColumns();
400 		String joinPath = getPath();
401 
402 		if ( impliedJoin && getWalker().isInFrom() ) {
403 			joinType = getWalker().getImpliedJoinType();
404 		}
405 
406 		FromClause currentFromClause = getWalker().getCurrentFromClause();
407 		FromElement elem = currentFromClause.findJoinByPath( joinPath );
408 
409 ///////////////////////////////////////////////////////////////////////////////
410 //
411 // This is the piece which recognizes the condition where an implicit join path
412 // resolved earlier in a correlated subquery is now being referenced in the
413 // outer query.  For 3.0final, we just let this generate a second join (which
414 // is exactly how the old parser handles this).  Eventually we need to add this
415 // logic back in and complete the logic in FromClause.promoteJoin; however,
416 // FromClause.promoteJoin has its own difficulties (see the comments in
417 // FromClause.promoteJoin).
418 //
419 //		if ( elem == null ) {
420 //			// see if this joinPath has been used in a "child" FromClause, and if so
421 //			// promote that element to the outer query
422 //			FromClause currentNodeOwner = getFromElement().getFromClause();
423 //			FromClause currentJoinOwner = currentNodeOwner.locateChildFromClauseWithJoinByPath( joinPath );
424 //			if ( currentJoinOwner != null && currentNodeOwner != currentJoinOwner ) {
425 //				elem = currentJoinOwner.findJoinByPathLocal( joinPath );
426 //				if ( elem != null ) {
427 //					currentFromClause.promoteJoin( elem );
428 //					// EARLY EXIT!!!
429 //					return;
430 //				}
431 //			}
432 //		}
433 //
434 ///////////////////////////////////////////////////////////////////////////////
435 
436 		boolean found = elem != null;
437 		// even though we might find a pre-existing element by join path, for FromElements originating in a from-clause
438 		// we should only ever use the found element if the aliases match (null != null here).  Implied joins are
439 		// always (?) ok to reuse.
440 		boolean useFoundFromElement = found && ( elem.isImplied() || areSame( classAlias, elem.getClassAlias() ) );
441 
442 		if ( ! useFoundFromElement ) {
443 			// If this is an implied join in a from element, then use the impled join type which is part of the
444 			// tree parser's state (set by the gramamar actions).
445 			JoinSequence joinSequence = getSessionFactoryHelper()
446 				.createJoinSequence( impliedJoin, propertyType, tableAlias, joinType, joinColumns );
447 
448 			// If the lhs of the join is a "component join", we need to go back to the
449 			// first non-component-join as the origin to properly link aliases and
450 			// join columns
451 			FromElement lhsFromElement = getLhs().getFromElement();
452 			while ( lhsFromElement != null &&  ComponentJoin.class.isInstance( lhsFromElement ) ) {
453 				lhsFromElement = lhsFromElement.getOrigin();
454 			}
455 			if ( lhsFromElement == null ) {
456 				throw new QueryException( "Unable to locate appropriate lhs" );
457 			}
458 
459 			FromElementFactory factory = new FromElementFactory(
460 			        currentFromClause,
461 					lhsFromElement,
462 					joinPath,
463 					classAlias,
464 					joinColumns,
465 					impliedJoin
466 			);
467 			elem = factory.createEntityJoin(
468 					associatedEntityName,
469 					tableAlias,
470 					joinSequence,
471 					fetch,
472 					getWalker().isInFrom(),
473 					propertyType
474 			);
475 		}
476 		else {
477 			// NOTE : addDuplicateAlias() already performs nullness checks on the alias.
478 			currentFromClause.addDuplicateAlias( classAlias, elem );
479 		}
480 		setImpliedJoin( elem );
481 		getWalker().addQuerySpaces( elem.getEntityPersister().getQuerySpaces() );
482 		setFromElement( elem );	// This 'dot' expression now refers to the resulting from element.
483 	}

504 	/**
505 	 * Is the given property name a reference to the primary key of the associated
506 	 * entity construed by the given entity type?
507 	 * <p/>
508 	 * For example, consider a fragment like order.customer.id
509 	 * (where order is a from-element alias).  Here, we'd have:
510 	 * propertyName = "id" AND
511 	 * owningType = ManyToOneType(Customer)
512 	 * and are being asked to determine whether "customer.id" is a reference
513 	 * to customer's PK...
514 	 *
515 	 * @param propertyName The name of the property to check.
516 	 * @param owningType The type represeting the entity "owning" the property
517 	 * @return True if propertyName references the entity's (owningType->associatedEntity)
518 	 * primary key; false otherwise.
519 	 */
520 	private boolean isReferenceToPrimaryKey(String propertyName, EntityType owningType) {
521 		EntityPersister persister = getSessionFactoryHelper()
522 				.getFactory()
523 				.getEntityPersister( owningType.getAssociatedEntityName() );
524 		if ( persister.getEntityMetamodel().hasNonIdentifierPropertyNamedId() ) {
525 			// only the identifier property field name can be a reference to the associated entity's PK...
526 			return propertyName.equals( persister.getIdentifierPropertyName() ) && owningType.isReferenceToPrimaryKey();
527 		}
528 		else {
529 			// here, we have two possibilities:
530 			// 		1) the property-name matches the explicitly identifier property name
531 			//		2) the property-name matches the implicit 'id' property name
532 			if ( EntityPersister.ENTITY_ID.equals( propertyName ) ) {
533 				// the referenced node text is the special 'id'
534 				return owningType.isReferenceToPrimaryKey();
535 			}
536 			else {
537 				String keyPropertyName = getSessionFactoryHelper().getIdentifierOrUniqueKeyPropertyName( owningType );
538 				return keyPropertyName != null && keyPropertyName.equals( propertyName ) && owningType.isReferenceToPrimaryKey();
539 			}
540 		}
541 	}

566 	private void dereferenceEntityIdentifier(String propertyName, DotNode dotParent) {
567 		// special shortcut for id properties, skip the join!
568 		// this must only occur at the _end_ of a path expression
569 		if ( log.isDebugEnabled() ) {
570 			log.debug( "dereferenceShortcut() : property " +
571 				propertyName + " in " + getFromElement().getClassName() +
572 				" does not require a join." );
573 		}
574 
575 		initText();
576 		setPropertyNameAndPath( dotParent ); // Set the unresolved path in this node and the parent.
577 		// Set the text for the parent.
578 		if ( dotParent != null ) {
579 			dotParent.dereferenceType = DEREF_IDENTIFIER;
580 			dotParent.setText( getText() );
581 			dotParent.columns = getColumns();
582 		}
583 	}

585 	private void setPropertyNameAndPath(AST parent) {
586 		if ( isDotNode( parent ) ) {
587 			DotNode dotNode = ( DotNode ) parent;
588 			AST lhs = dotNode.getFirstChild();
589 			AST rhs = lhs.getNextSibling();
590 			propertyName = rhs.getText();
591 			propertyPath = propertyPath + "." + propertyName; // Append the new property name onto the unresolved path.
592 			dotNode.propertyPath = propertyPath;
593 			if ( log.isDebugEnabled() ) {
594 				log.debug( "Unresolved property path is now '" + dotNode.propertyPath + "'" );
595 			}
596 		}
597 		else {
598 			if ( log.isDebugEnabled() ) {
599 				log.debug( "terminal propertyPath = [" + propertyPath + "]" );
600 			}
601 		}
602 	}

604 	public Type getDataType() {
605 		if ( super.getDataType() == null ) {
606 			FromElement fromElement = getLhs().getFromElement();
607 			if ( fromElement == null ) {
608 				return null;
609 			}
610 			// If the lhs is a collection, use CollectionPropertyMapping
611 			Type propertyType = fromElement.getPropertyType( propertyName, propertyPath );
612 			if ( log.isDebugEnabled() ) {
613 				log.debug( "getDataType() : " + propertyPath + " -> " + propertyType );
614 			}
615 			super.setDataType( propertyType );
616 		}
617 		return super.getDataType();
618 	}

664 	/**
665 	 * Special method to resolve expressions in the SELECT list.
666 	 *
667 	 * @throws SemanticException if this cannot be resolved.
668 	 */
669 	public void resolveSelectExpression() throws SemanticException {
670 		if ( getWalker().isShallowQuery() || getWalker().getCurrentFromClause().isSubQuery() ) {
671 			resolve(false, true);
672 		}
673 		else {
674 			resolve(true, false);
675 			Type type = getDataType();
676 			if ( type.isEntityType() ) {
677 				FromElement fromElement = getFromElement();
678 				fromElement.setIncludeSubclasses( true ); // Tell the destination fromElement to 'includeSubclasses'.
679 				if ( useThetaStyleImplicitJoins ) {
680 					fromElement.getJoinSequence().setUseThetaStyle( true );	// Use theta style (for regression)
681 					// Move the node up, after the origin node.
682 					FromElement origin = fromElement.getOrigin();
683 					if ( origin != null ) {
684 						ASTUtil.makeSiblingOfParent( origin, fromElement );
685 					}
686 				}
687 			}
688 		}
689 
690 		FromReferenceNode lhs = getLhs();
691 		while ( lhs != null ) {
692 			checkSubclassOrSuperclassPropertyReference( lhs, lhs.getNextSibling().getText() );
693 			lhs = ( FromReferenceNode ) lhs.getFirstChild();
694 		}
695 	}

703 	private boolean checkSubclassOrSuperclassPropertyReference(FromReferenceNode lhs, String propertyName) {
704 		if ( lhs != null && !( lhs instanceof IndexNode ) ) {
705 			final FromElement source = lhs.getFromElement();
706 			if ( source != null ) {
707 				source.handlePropertyBeingDereferenced( lhs.getDataType(), propertyName );
708 			}
709 		}
710 		return false;
711 	}
