131 /**
132  * Concrete implementation of the <tt>SessionFactory</tt> interface. Has the following
133  * responsibilities
134  * <ul>
135  * <li>caches configuration settings (immutably)
136  * <li>caches "compiled" mappings ie. <tt>EntityPersister</tt>s and
137  *     <tt>CollectionPersister</tt>s (immutable)
138  * <li>caches "compiled" queries (memory sensitive cache)
139  * <li>manages <tt>PreparedStatement</tt>s
140  * <li> delegates JDBC <tt>Connection</tt> management to the <tt>ConnectionProvider</tt>
141  * <li>factory for instances of <tt>SessionImpl</tt>
142  * </ul>
143  * This class must appear immutable to clients, even if it does all kinds of caching
144  * and pooling under the covers. It is crucial that the class is not only thread
145  * safe, but also highly concurrent. Synchronization must be used extremely sparingly.
146  *
147  * @see org.hibernate.connection.ConnectionProvider
148  * @see org.hibernate.classic.Session
149  * @see org.hibernate.hql.QueryTranslator
150  * @see org.hibernate.persister.entity.EntityPersister
151  * @see org.hibernate.persister.collection.CollectionPersister
152  * @author Gavin King
153  */
154 public final class SessionFactoryImpl implements SessionFactory, SessionFactoryImplementor {

196 	public SessionFactoryImpl(
197 			Configuration cfg,
198 	        Mapping mapping,
199 	        Settings settings,
200 	        EventListeners listeners,
201 			SessionFactoryObserver observer) throws HibernateException {
202 		log.info("building session factory");
203 
204 		this.statistics = new ConcurrentStatisticsImpl( this );
205 		getStatistics().setStatisticsEnabled( settings.isStatisticsEnabled() );
206 		log.debug( "Statistics initialized [enabled={}]}", settings.isStatisticsEnabled() );
207 
208 		this.properties = new Properties();
209 		this.properties.putAll( cfg.getProperties() );
210 		this.interceptor = cfg.getInterceptor();
211 		this.settings = settings;
212 		this.sqlFunctionRegistry = new SQLFunctionRegistry(settings.getDialect(), cfg.getSqlFunctions());
213         this.eventListeners = listeners;
214 		this.observer = observer != null ? observer : new SessionFactoryObserver() {
215 			public void sessionFactoryCreated(SessionFactory factory) {
216 			}
217 			public void sessionFactoryClosed(SessionFactory factory) {
218 			}
219 		};
220 
221 		this.typeResolver = cfg.getTypeResolver().scope( this );
222 		this.typeHelper = new TypeLocatorImpl( typeResolver );
223 
224 		this.filters = new HashMap();
225 		this.filters.putAll( cfg.getFilterDefinitions() );
226 
227 		if ( log.isDebugEnabled() ) {
228 			log.debug("Session factory constructed with filter configurations : " + filters);
229 		}
230 
231 		if ( log.isDebugEnabled() ) {
232 			log.debug(
233 					"instantiating session factory with properties: " + properties
234 			);
235 		}
236 
237 		// Caches
238 		settings.getRegionFactory().start( settings, properties );
239 		this.queryPlanCache = new QueryPlanCache( this );
240 
241 		//Generators:
242 
243 		identifierGenerators = new HashMap();
244 		Iterator classes = cfg.getClassMappings();
245 		while ( classes.hasNext() ) {
246 			PersistentClass model = (PersistentClass) classes.next();
247 			if ( !model.isInherited() ) {
248 				IdentifierGenerator generator = model.getIdentifier().createIdentifierGenerator(
249 						cfg.getIdentifierGeneratorFactory(),
250 						settings.getDialect(),
251 				        settings.getDefaultCatalogName(),
252 				        settings.getDefaultSchemaName(),
253 				        (RootClass) model
254 				);
255 				identifierGenerators.put( model.getEntityName(), generator );
256 			}
257 		}
258 
259 
260 		///////////////////////////////////////////////////////////////////////
261 		// Prepare persisters and link them up with their cache
262 		// region/access-strategy
263 
264 		final String cacheRegionPrefix = settings.getCacheRegionPrefix() == null ? "" : settings.getCacheRegionPrefix() + ".";
265 
266 		entityPersisters = new HashMap();
267 		Map entityAccessStrategies = new HashMap();
268 		Map<String,ClassMetadata> classMeta = new HashMap<String,ClassMetadata>();
269 		classes = cfg.getClassMappings();
270 		while ( classes.hasNext() ) {
271 			final PersistentClass model = (PersistentClass) classes.next();
272 			model.prepareTemporaryTables( mapping, settings.getDialect() );
273 			final String cacheRegionName = cacheRegionPrefix + model.getRootClass().getCacheRegionName();
274 			// cache region is defined by the root-class in the hierarchy...
275 			EntityRegionAccessStrategy accessStrategy = ( EntityRegionAccessStrategy ) entityAccessStrategies.get( cacheRegionName );
276 			if ( accessStrategy == null && settings.isSecondLevelCacheEnabled() ) {
277 				final AccessType accessType = AccessType.parse( model.getCacheConcurrencyStrategy() );
278 				if ( accessType != null ) {
279 					log.trace( "Building cache for entity data [" + model.getEntityName() + "]" );
280 					EntityRegion entityRegion = settings.getRegionFactory().buildEntityRegion( cacheRegionName, properties, CacheDataDescriptionImpl.decode( model ) );
281 					accessStrategy = entityRegion.buildAccessStrategy( accessType );
282 					entityAccessStrategies.put( cacheRegionName, accessStrategy );
283 					allCacheRegions.put( cacheRegionName, entityRegion );
284 				}
285 			}
286 			EntityPersister cp = PersisterFactory.createClassPersister( model, accessStrategy, this, mapping );
287 			entityPersisters.put( model.getEntityName(), cp );
288 			classMeta.put( model.getEntityName(), cp.getClassMetadata() );
289 		}
290 		this.classMetadata = Collections.unmodifiableMap(classMeta);
291 
292 		Map<String,Set<String>> tmpEntityToCollectionRoleMap = new HashMap<String,Set<String>>();
293 		collectionPersisters = new HashMap();
294 		Iterator collections = cfg.getCollectionMappings();
295 		while ( collections.hasNext() ) {
296 			Collection model = (Collection) collections.next();
297 			final String cacheRegionName = cacheRegionPrefix + model.getCacheRegionName();
298 			final AccessType accessType = AccessType.parse( model.getCacheConcurrencyStrategy() );
299 			CollectionRegionAccessStrategy accessStrategy = null;
300 			if ( accessType != null && settings.isSecondLevelCacheEnabled() ) {
301 				log.trace( "Building cache for collection data [" + model.getRole() + "]" );
302 				CollectionRegion collectionRegion = settings.getRegionFactory().buildCollectionRegion( cacheRegionName, properties, CacheDataDescriptionImpl.decode( model ) );
303 				accessStrategy = collectionRegion.buildAccessStrategy( accessType );
304 				entityAccessStrategies.put( cacheRegionName, accessStrategy );
305 				allCacheRegions.put( cacheRegionName, collectionRegion );
306 			}
307 			CollectionPersister persister = PersisterFactory.createCollectionPersister( cfg, model, accessStrategy, this) ;
308 			collectionPersisters.put( model.getRole(), persister.getCollectionMetadata() );
309 			Type indexType = persister.getIndexType();
310 			if ( indexType != null && indexType.isAssociationType() && !indexType.isAnyType() ) {
311 				String entityName = ( ( AssociationType ) indexType ).getAssociatedEntityName( this );
312 				Set roles = ( Set ) tmpEntityToCollectionRoleMap.get( entityName );
313 				if ( roles == null ) {
314 					roles = new HashSet();
315 					tmpEntityToCollectionRoleMap.put( entityName, roles );
316 				}
317 				roles.add( persister.getRole() );
318 			}
319 			Type elementType = persister.getElementType();
320 			if ( elementType.isAssociationType() && !elementType.isAnyType() ) {
321 				String entityName = ( ( AssociationType ) elementType ).getAssociatedEntityName( this );
322 				Set roles = ( Set ) tmpEntityToCollectionRoleMap.get( entityName );
323 				if ( roles == null ) {
324 					roles = new HashSet();
325 					tmpEntityToCollectionRoleMap.put( entityName, roles );
326 				}
327 				roles.add( persister.getRole() );
328 			}
329 		}
330 		collectionMetadata = Collections.unmodifiableMap(collectionPersisters);
331 		Iterator itr = tmpEntityToCollectionRoleMap.entrySet().iterator();
332 		while ( itr.hasNext() ) {
333 			final Map.Entry entry = ( Map.Entry ) itr.next();
334 			entry.setValue( Collections.unmodifiableSet( ( Set ) entry.getValue() ) );
335 		}
336 		collectionRolesByEntityParticipant = Collections.unmodifiableMap( tmpEntityToCollectionRoleMap );
337 
338 		//Named Queries:
339 		namedQueries = new HashMap( cfg.getNamedQueries() );
340 		namedSqlQueries = new HashMap( cfg.getNamedSQLQueries() );
341 		sqlResultSetMappings = new HashMap( cfg.getSqlResultSetMappings() );
342 		imports = new HashMap( cfg.getImports() );
343 
344 		// after *all* persisters and named queries are registered
345 		Iterator iter = entityPersisters.values().iterator();
346 		while ( iter.hasNext() ) {
347 			final EntityPersister persister = ( ( EntityPersister ) iter.next() );
348 			persister.postInstantiate();
349 			registerEntityNameResolvers( persister );
350 
351 		}
352 		iter = collectionPersisters.values().iterator();
353 		while ( iter.hasNext() ) {
354 			final CollectionPersister persister = ( ( CollectionPersister ) iter.next() );
355 			persister.postInstantiate();
356 		}
357 
358 		//JNDI + Serialization:
359 
360 		name = settings.getSessionFactoryName();
361 		try {
362 			uuid = (String) UUID_GENERATOR.generate(null, null);
363 		}
364 		catch (Exception e) {
365 			throw new AssertionFailure("Could not generate UUID");
366 		}
367 		SessionFactoryObjectFactory.addInstance(uuid, name, this, properties);
368 
369 		log.debug("instantiated session factory");
370 
371 		if ( settings.isAutoCreateSchema() ) {
372 			new SchemaExport( cfg, settings ).create( false, true );
373 		}
374 		if ( settings.isAutoUpdateSchema() ) {
375 			new SchemaUpdate( cfg, settings ).execute( false, true );
376 		}
377 		if ( settings.isAutoValidateSchema() ) {
378 			new SchemaValidator( cfg, settings ).validate();
379 		}
380 		if ( settings.isAutoDropSchema() ) {
381 			schemaExport = new SchemaExport( cfg, settings );
382 		}
383 
384 		if ( settings.getTransactionManagerLookup()!=null ) {
385 			log.debug("obtaining JTA TransactionManager");
386 			transactionManager = settings.getTransactionManagerLookup().getTransactionManager(properties);
387 		}
388 		else {
389 			if ( settings.getTransactionFactory().isTransactionManagerRequired() ) {
390 				throw new HibernateException("The chosen transaction strategy requires access to the JTA TransactionManager");
391 			}
392 			transactionManager = null;
393 		}
394 
395 		currentSessionContext = buildCurrentSessionContext();
396 
397 		if ( settings.isQueryCacheEnabled() ) {
398 			updateTimestampsCache = new UpdateTimestampsCache(settings, properties);
399 			queryCache = settings.getQueryCacheFactory()
400 			        .getQueryCache(null, updateTimestampsCache, settings, properties);
401 			queryCaches = new HashMap<String,QueryCache>();
402 			allCacheRegions.put( updateTimestampsCache.getRegion().getName(), updateTimestampsCache.getRegion() );
403 			allCacheRegions.put( queryCache.getRegion().getName(), queryCache.getRegion() );
404 		}
405 		else {
406 			updateTimestampsCache = null;
407 			queryCache = null;
408 			queryCaches = null;
409 		}
410 
411 		//checking for named queries
412 		if ( settings.isNamedQueryStartupCheckingEnabled() ) {
413 			Map errors = checkNamedQueries();
414 			if ( !errors.isEmpty() ) {
415 				Set keys = errors.keySet();
416 				StringBuffer failingQueries = new StringBuffer( "Errors in named queries: " );
417 				for ( Iterator iterator = keys.iterator() ; iterator.hasNext() ; ) {
418 					String queryName = ( String ) iterator.next();
419 					HibernateException e = ( HibernateException ) errors.get( queryName );
420 					failingQueries.append( queryName );
421 					if ( iterator.hasNext() ) {
422 						failingQueries.append( ", " );
423 					}
424 					log.error( "Error in named query: " + queryName, e );
425 				}
426 				throw new HibernateException( failingQueries.toString() );
427 			}
428 		}
429 
430 		// EntityNotFoundDelegate
431 		EntityNotFoundDelegate entityNotFoundDelegate = cfg.getEntityNotFoundDelegate();
432 		if ( entityNotFoundDelegate == null ) {
433 			entityNotFoundDelegate = new EntityNotFoundDelegate() {
434 				public void handleEntityNotFound(String entityName, Serializable id) {
435 					throw new ObjectNotFoundException( id, entityName );
436 				}
437 			};
438 		}
439 		this.entityNotFoundDelegate = entityNotFoundDelegate;
440 
441 		// this needs to happen after persisters are all ready to go...
442 		this.fetchProfiles = new HashMap();
443 		itr = cfg.iterateFetchProfiles();
444 		while ( itr.hasNext() ) {
445 			final org.hibernate.mapping.FetchProfile mappingProfile =
446 					( org.hibernate.mapping.FetchProfile ) itr.next();
447 			final FetchProfile fetchProfile = new FetchProfile( mappingProfile.getName() );
448 			Iterator fetches = mappingProfile.getFetches().iterator();
449 			while ( fetches.hasNext() ) {
450 				final org.hibernate.mapping.FetchProfile.Fetch mappingFetch =
451 						( org.hibernate.mapping.FetchProfile.Fetch ) fetches.next();
452 				// resolve the persister owning the fetch
453 				final String entityName = getImportedClassName( mappingFetch.getEntity() );
454 				final EntityPersister owner = ( EntityPersister ) ( entityName == null ? null : entityPersisters.get( entityName ) );
455 				if ( owner == null ) {
456 					throw new HibernateException(
457 							"Unable to resolve entity reference [" + mappingFetch.getEntity()
458 									+ "] in fetch profile [" + fetchProfile.getName() + "]"
459 					);
460 				}
461 
462 				// validate the specified association fetch
463 				Type associationType = owner.getPropertyType( mappingFetch.getAssociation() );
464 				if ( associationType == null || !associationType.isAssociationType() ) {
465 					throw new HibernateException( "Fetch profile [" + fetchProfile.getName() + "] specified an invalid association" );
466 				}
467 
468 				// resolve the style
469 				final Fetch.Style fetchStyle = Fetch.Style.parse( mappingFetch.getStyle() );
470 
471 				// then construct the fetch instance...
472 				fetchProfile.addFetch( new Association( owner, mappingFetch.getAssociation() ), fetchStyle );
473 				( ( Loadable ) owner ).registerAffectingFetchProfile( fetchProfile.getName() );
474 			}
475 			fetchProfiles.put( fetchProfile.getName(), fetchProfile );
476 		}
477 
478 		this.observer.sessionFactoryCreated( this );
479 	}

493 	private void registerEntityNameResolvers(EntityPersister persister) {
494 		if ( persister.getEntityMetamodel() == null || persister.getEntityMetamodel().getTuplizerMapping() == null ) {
495 			return;
496 		}
497 		Iterator itr = persister.getEntityMetamodel().getTuplizerMapping().iterateTuplizers();
498 		while ( itr.hasNext() ) {
499 			final EntityTuplizer tuplizer = ( EntityTuplizer ) itr.next();
500 			registerEntityNameResolvers( tuplizer );
501 		}
502 	}
503 
504 	private void registerEntityNameResolvers(EntityTuplizer tuplizer) {
505 		EntityNameResolver[] resolvers = tuplizer.getEntityNameResolvers();
506 		if ( resolvers == null ) {
507 			return;
508 		}
509 
510 		for ( int i = 0; i < resolvers.length; i++ ) {
511 			registerEntityNameResolver( resolvers[i], tuplizer.getEntityMode() );
512 		}
513 	}
514 
515 	public void registerEntityNameResolver(EntityNameResolver resolver, EntityMode entityMode) {
516 		LinkedHashSet resolversForMode = ( LinkedHashSet ) entityNameResolvers.get( entityMode );
517 		if ( resolversForMode == null ) {
518 			resolversForMode = new LinkedHashSet();
519 			entityNameResolvers.put( entityMode, resolversForMode );
520 		}
521 		resolversForMode.add( resolver );
522 	}
523 
524 	public Iterator iterateEntityNameResolvers(EntityMode entityMode) {
525 		Set actualEntityNameResolvers = ( Set ) entityNameResolvers.get( entityMode );
526 		return actualEntityNameResolvers == null
527 				? EmptyIterator.INSTANCE
528 				: actualEntityNameResolvers.iterator();
529 	}

535 	private Map checkNamedQueries() throws HibernateException {
536 		Map errors = new HashMap();
537 
538 		// Check named HQL queries
539 		log.debug("Checking " + namedQueries.size() + " named HQL queries");
540 		Iterator itr = namedQueries.entrySet().iterator();
541 		while ( itr.hasNext() ) {
542 			final Map.Entry entry = ( Map.Entry ) itr.next();
543 			final String queryName = ( String ) entry.getKey();
544 			final NamedQueryDefinition qd = ( NamedQueryDefinition ) entry.getValue();
545 			// this will throw an error if there's something wrong.
546 			try {
547 				log.debug("Checking named query: " + queryName);
548 				//TODO: BUG! this currently fails for named queries for non-POJO entities
549 				queryPlanCache.getHQLQueryPlan( qd.getQueryString(), false, CollectionHelper.EMPTY_MAP );
550 			}
551 			catch ( QueryException e ) {
552 				errors.put( queryName, e );
553 			}
554 			catch ( MappingException e ) {
555 				errors.put( queryName, e );
556 			}
557 		}
558 
559 		log.debug("Checking " + namedSqlQueries.size() + " named SQL queries");
560 		itr = namedSqlQueries.entrySet().iterator();
561 		while ( itr.hasNext() ) {
562 			final Map.Entry entry = ( Map.Entry ) itr.next();
563 			final String queryName = ( String ) entry.getKey();
564 			final NamedSQLQueryDefinition qd = ( NamedSQLQueryDefinition ) entry.getValue();
565 			// this will throw an error if there's something wrong.
566 			try {
567 				log.debug("Checking named SQL query: " + queryName);
568 				// TODO : would be really nice to cache the spec on the query-def so as to not have to re-calc the hash;
569 				// currently not doable though because of the resultset-ref stuff...
570 				NativeSQLQuerySpecification spec;
571 				if ( qd.getResultSetRef() != null ) {
572 					ResultSetMappingDefinition definition = ( ResultSetMappingDefinition ) sqlResultSetMappings.get( qd.getResultSetRef() );
573 					if ( definition == null ) {
574 						throw new MappingException( "Unable to find resultset-ref definition: " + qd.getResultSetRef() );
575 					}
576 					spec = new NativeSQLQuerySpecification(
577 							qd.getQueryString(),
578 					        definition.getQueryReturns(),
579 					        qd.getQuerySpaces()
580 					);
581 				}
582 				else {
583 					spec =  new NativeSQLQuerySpecification(
584 							qd.getQueryString(),
585 					        qd.getQueryReturns(),
586 					        qd.getQuerySpaces()
587 					);
588 				}
589 				queryPlanCache.getNativeSQLQueryPlan( spec );
590 			}
591 			catch ( QueryException e ) {
592 				errors.put( queryName, e );
593 			}
594 			catch ( MappingException e ) {
595 				errors.put( queryName, e );
596 			}
597 		}
598 
599 		return errors;
600 	}

818 	/**
819 	 * Return the names of all persistent (mapped) classes that extend or implement the
820 	 * given class or interface, accounting for implicit/explicit polymorphism settings
821 	 * and excluding mapped subclasses/joined-subclasses of other classes in the result.
822 	 */
823 	public String[] getImplementors(String className) throws MappingException {
824 
825 		final Class clazz;
826 		try {
827 			clazz = ReflectHelper.classForName(className);
828 		}
829 		catch (ClassNotFoundException cnfe) {
830 			return new String[] { className }; //for a dynamic-class
831 		}
832 
833 		ArrayList results = new ArrayList();
834 		Iterator iter = entityPersisters.values().iterator();
835 		while ( iter.hasNext() ) {
836 			//test this entity to see if we must query it
837 			EntityPersister testPersister = (EntityPersister) iter.next();
838 			if ( testPersister instanceof Queryable ) {
839 				Queryable testQueryable = (Queryable) testPersister;
840 				String testClassName = testQueryable.getEntityName();
841 				boolean isMappedClass = className.equals(testClassName);
842 				if ( testQueryable.isExplicitPolymorphism() ) {
843 					if ( isMappedClass ) {
844 						return new String[] {className}; //NOTE EARLY EXIT
845 					}
846 				}
847 				else {
848 					if (isMappedClass) {
849 						results.add(testClassName);
850 					}
851 					else {
852 						final Class mappedClass = testQueryable.getMappedClass( EntityMode.POJO );
853 						if ( mappedClass!=null && clazz.isAssignableFrom( mappedClass ) ) {
854 							final boolean assignableSuperclass;
855 							if ( testQueryable.isInherited() ) {
856 								Class mappedSuperclass = getEntityPersister( testQueryable.getMappedSuperclass() ).getMappedClass( EntityMode.POJO);
857 								assignableSuperclass = clazz.isAssignableFrom(mappedSuperclass);
858 							}
859 							else {
860 								assignableSuperclass = false;
861 							}
862 							if ( !assignableSuperclass ) {
863 								results.add( testClassName );
864 							}
865 						}
866 					}
867 				}
868 			}
869 		}
870 		return (String[]) results.toArray( new String[ results.size() ] );
871 	}

906 	/**
907 	 * Closes the session factory, releasing all held resources.
908 	 *
909 	 * <ol>
910 	 * <li>cleans up used cache regions and "stops" the cache provider.
911 	 * <li>close the JDBC connection
912 	 * <li>remove the JNDI binding
913 	 * </ol>
914 	 *
915 	 * Note: Be aware that the sessionfactory instance still can
916 	 * be a "heavy" object memory wise after close() has been called.  Thus
917 	 * it is important to not keep referencing the instance to let the garbage
918 	 * collector release the memory.
919 	 */
920 	public void close() throws HibernateException {
921 
922 		if ( isClosed ) {
923 			log.trace( "already closed" );
924 			return;
925 		}
926 
927 		log.info("closing");
928 
929 		isClosed = true;
930 
931 		Iterator iter = entityPersisters.values().iterator();
932 		while ( iter.hasNext() ) {
933 			EntityPersister p = (EntityPersister) iter.next();
934 			if ( p.hasCache() ) {
935 				p.getCacheAccessStrategy().getRegion().destroy();
936 			}
937 		}
938 
939 		iter = collectionPersisters.values().iterator();
940 		while ( iter.hasNext() ) {
941 			CollectionPersister p = (CollectionPersister) iter.next();
942 			if ( p.hasCache() ) {
943 				p.getCacheAccessStrategy().getRegion().destroy();
944 			}
945 		}
946 
947 		if ( settings.isQueryCacheEnabled() )  {
948 			queryCache.destroy();
949 
950 			iter = queryCaches.values().iterator();
951 			while ( iter.hasNext() ) {
952 				QueryCache cache = (QueryCache) iter.next();
953 				cache.destroy();
954 			}
955 			updateTimestampsCache.destroy();
956 		}
957 
958 		settings.getRegionFactory().stop();
959 
960 		if ( settings.isAutoDropSchema() ) {
961 			schemaExport.drop( false, true );
962 		}
963 
964 		try {
965 			settings.getConnectionProvider().close();
966 		}
967 		finally {
968 			SessionFactoryObjectFactory.removeInstance(uuid, name, properties);
969 		}
970 
971 		observer.sessionFactoryClosed( this );
972 		eventListeners.destroyListeners();
973 	}

1225 	private CurrentSessionContext buildCurrentSessionContext() {
1226 		String impl = properties.getProperty( Environment.CURRENT_SESSION_CONTEXT_CLASS );
1227 		// for backward-compatability
1228 		if ( impl == null && transactionManager != null ) {
1229 			impl = "jta";
1230 		}
1231 
1232 		if ( impl == null ) {
1233 			return null;
1234 		}
1235 		else if ( "jta".equals( impl ) ) {
1236 			if ( settings.getTransactionFactory().areCallbacksLocalToHibernateTransactions() ) {
1237 				log.warn( "JTASessionContext being used with JDBCTransactionFactory; auto-flush will not operate correctly with getCurrentSession()" );
1238 			}
1239 			return new JTASessionContext( this );
1240 		}
1241 		else if ( "thread".equals( impl ) ) {
1242 			return new ThreadLocalSessionContext( this );
1243 		}
1244 		else if ( "managed".equals( impl ) ) {
1245 			return new ManagedSessionContext( this );
1246 		}
1247 		else {
1248 			try {
1249 				Class implClass = ReflectHelper.classForName( impl );
1250 				return ( CurrentSessionContext ) implClass
1251 						.getConstructor( new Class[] { SessionFactoryImplementor.class } )
1252 						.newInstance( new Object[] { this } );
1253 			}
1254 			catch( Throwable t ) {
1255 				log.error( "Unable to construct current session context [" + impl + "]", t );
1256 				return null;
1257 			}
1258 		}
1259 	}

1306 	 * @throws ClassNotFoundException indicates problems reading back serial data stream
1307 	 */
1308 	static SessionFactoryImpl deserialize(ObjectInputStream ois) throws IOException, ClassNotFoundException {
1309 		String uuid = ois.readUTF();
1310 		boolean isNamed = ois.readBoolean();
1311 		String name = null;
1312 		if ( isNamed ) {
1313 			name = ois.readUTF();
1314 		}
1315 		Object result = SessionFactoryObjectFactory.getInstance( uuid );
1316 		if ( result == null ) {
1317 			log.trace( "could not locate session factory by uuid [" + uuid + "] during session deserialization; trying name" );
1318 			if ( isNamed ) {
1319 				result = SessionFactoryObjectFactory.getNamedInstance( name );
1320 			}
1321 			if ( result == null ) {
1322 				throw new InvalidObjectException( "could not resolve session factory during session deserialization [uuid=" + uuid + ", name=" + name + "]" );
1323 			}
1324 		}
