253 	public String getSelectClauseNullString(int sqlType) {
254 		String typeName = getTypeName(sqlType, 1, 1, 0);
255 		//trim off the length/precision/scale
256 		int loc = typeName.indexOf('(');
257 		if (loc>-1) {
258 			typeName = typeName.substring(0, loc);
259 		}
260 		return "null::" + typeName;
261 	}

309 	/**
310 	 * Constraint-name extractor for Postgres contraint violation exceptions.
311 	 * Orginally contributed by Denny Bartelt.
312 	 */
313 	private static ViolatedConstraintNameExtracter EXTRACTER = new TemplatedViolatedConstraintNameExtracter() {
314 		public String extractConstraintName(SQLException sqle) {
315 			try {
316 				int sqlState = Integer.valueOf( JDBCExceptionHelper.extractSqlState(sqle)).intValue();
317 				switch (sqlState) {
318 					// CHECK VIOLATION
319 					case 23514: return extractUsingTemplate("violates check constraint \"","\"", sqle.getMessage());
320 					// UNIQUE VIOLATION
321 					case 23505: return extractUsingTemplate("violates unique constraint \"","\"", sqle.getMessage());
322 					// FOREIGN KEY VIOLATION
323 					case 23503: return extractUsingTemplate("violates foreign key constraint \"","\"", sqle.getMessage());
324 					// NOT NULL VIOLATION
325 					case 23502: return extractUsingTemplate("null value in column \"","\" violates not-null constraint", sqle.getMessage());
326 					// TODO: RESTRICT VIOLATION
327 					case 23001: return null;
328 					// ALL OTHER
329 					default: return null;
330 				}
331 			} catch (NumberFormatException nfe) {
332 				return null;
333 			}
334 		}
335 	};

337 	public int registerResultSetOutParameter(CallableStatement statement, int col) throws SQLException {
338 		// Register the type of the out param - PostgreSQL uses Types.OTHER
339 		statement.registerOutParameter(col, Types.OTHER);
340 		col++;
341 		return col;
342 	}
