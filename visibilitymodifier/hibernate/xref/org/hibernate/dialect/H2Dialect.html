58  			final Class constants = ReflectHelper.classForName( "org.h2.engine.Constants" );
59  			final int majorVersion = ( Integer ) constants.getDeclaredField( "VERSION_MAJOR" ).get( null );
60  			final int minorVersion = ( Integer ) constants.getDeclaredField( "VERSION_MINOR" ).get( null );
61  			final int buildId = ( Integer ) constants.getDeclaredField( "BUILD_ID" ).get( null );
62  			if ( buildId < 32 ) {
63  				querySequenceString = "select name from information_schema.sequences";
64  			}
65  			if ( !( majorVersion > 1 || minorVersion > 2 || buildId >= 139 ) ) {
66  				log.warn(

68  								"current transaction; multi-table, bulk hql/jpaql will not work properly",
69  						( majorVersion + "." + minorVersion + "." + buildId )

272 	private static ViolatedConstraintNameExtracter EXTRACTER = new TemplatedViolatedConstraintNameExtracter() {
273 		/**
274 		 * Extract the name of the violated constraint from the given SQLException.
275 		 *
276 		 * @param sqle The exception that was the result of the constraint violation.
277 		 * @return The extracted constraint name.
278 		 */
279 		public String extractConstraintName(SQLException sqle) {
280 			String constraintName = null;
281 			// 23000: Check constraint violation: {0}
282 			// 23001: Unique index or primary key violation: {0}
283 			if ( sqle.getSQLState().startsWith( "23" ) ) {
284 				final String message = sqle.getMessage();
285 				int idx = message.indexOf( "violation: " );
286 				if ( idx > 0 ) {
287 					constraintName = message.substring( idx + "violation: ".length() );
288 				}
289 			}
290 			return constraintName;
291 		}
292 	};
