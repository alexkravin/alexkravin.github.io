59  /**
60   * The default implementation of the <tt>EntityPersister</tt> interface.
61   * Implements the "table-per-class-hierarchy" or "roll-up" mapping strategy
62   * for an entity class and its inheritence hierarchy.  This is implemented
63   * as a single table holding all classes in the hierarchy with a discrimator
64   * column used to determine which concrete class is referenced.
65   *
66   * @author Gavin King
67   */
68  public class SingleTableEntityPersister extends AbstractEntityPersister {

127 	public SingleTableEntityPersister(
128 			final PersistentClass persistentClass, 
129 			final EntityRegionAccessStrategy cacheAccessStrategy,
130 			final SessionFactoryImplementor factory,
131 			final Mapping mapping) throws HibernateException {
132 
133 		super( persistentClass, cacheAccessStrategy, factory );
134 
135 		// CLASS + TABLE
136 
137 		joinSpan = persistentClass.getJoinClosureSpan()+1;
138 		qualifiedTableNames = new String[joinSpan];
139 		isInverseTable = new boolean[joinSpan];
140 		isNullableTable = new boolean[joinSpan];
141 		keyColumnNames = new String[joinSpan][];
142 		final Table table = persistentClass.getRootTable();
143 		qualifiedTableNames[0] = table.getQualifiedName( 
144 				factory.getDialect(), 
145 				factory.getSettings().getDefaultCatalogName(), 
146 				factory.getSettings().getDefaultSchemaName() 
147 		);
148 		isInverseTable[0] = false;
149 		isNullableTable[0] = false;
150 		keyColumnNames[0] = getIdentifierColumnNames();
151 		cascadeDeleteEnabled = new boolean[joinSpan];
152 
153 		// Custom sql
154 		customSQLInsert = new String[joinSpan];
155 		customSQLUpdate = new String[joinSpan];
156 		customSQLDelete = new String[joinSpan];
157 		insertCallable = new boolean[joinSpan];
158 		updateCallable = new boolean[joinSpan];
159 		deleteCallable = new boolean[joinSpan];
160 		insertResultCheckStyles = new ExecuteUpdateResultCheckStyle[joinSpan];
161 		updateResultCheckStyles = new ExecuteUpdateResultCheckStyle[joinSpan];
162 		deleteResultCheckStyles = new ExecuteUpdateResultCheckStyle[joinSpan];
163 
164 		customSQLInsert[0] = persistentClass.getCustomSQLInsert();
165 		insertCallable[0] = customSQLInsert[0] != null && persistentClass.isCustomInsertCallable();
166 		insertResultCheckStyles[0] = persistentClass.getCustomSQLInsertCheckStyle() == null
167 									  ? ExecuteUpdateResultCheckStyle.determineDefault( customSQLInsert[0], insertCallable[0] )
168 									  : persistentClass.getCustomSQLInsertCheckStyle();
169 		customSQLUpdate[0] = persistentClass.getCustomSQLUpdate();
170 		updateCallable[0] = customSQLUpdate[0] != null && persistentClass.isCustomUpdateCallable();
171 		updateResultCheckStyles[0] = persistentClass.getCustomSQLUpdateCheckStyle() == null
172 									  ? ExecuteUpdateResultCheckStyle.determineDefault( customSQLUpdate[0], updateCallable[0] )
173 									  : persistentClass.getCustomSQLUpdateCheckStyle();
174 		customSQLDelete[0] = persistentClass.getCustomSQLDelete();
175 		deleteCallable[0] = customSQLDelete[0] != null && persistentClass.isCustomDeleteCallable();
176 		deleteResultCheckStyles[0] = persistentClass.getCustomSQLDeleteCheckStyle() == null
177 									  ? ExecuteUpdateResultCheckStyle.determineDefault( customSQLDelete[0], deleteCallable[0] )
178 									  : persistentClass.getCustomSQLDeleteCheckStyle();
179 
180 		// JOINS
181 
182 		Iterator joinIter = persistentClass.getJoinClosureIterator();
183 		int j = 1;
184 		while ( joinIter.hasNext() ) {
185 			Join join = (Join) joinIter.next();
186 			qualifiedTableNames[j] = join.getTable().getQualifiedName( 
187 					factory.getDialect(), 
188 					factory.getSettings().getDefaultCatalogName(), 
189 					factory.getSettings().getDefaultSchemaName() 
190 			);
191 			isInverseTable[j] = join.isInverse();
192 			isNullableTable[j] = join.isOptional();
193 			cascadeDeleteEnabled[j] = join.getKey().isCascadeDeleteEnabled() && 
194 				factory.getDialect().supportsCascadeDelete();
195 
196 			customSQLInsert[j] = join.getCustomSQLInsert();
197 			insertCallable[j] = customSQLInsert[j] != null && join.isCustomInsertCallable();
198 			insertResultCheckStyles[j] = join.getCustomSQLInsertCheckStyle() == null
199 			                              ? ExecuteUpdateResultCheckStyle.determineDefault( customSQLInsert[j], insertCallable[j] )
200 		                                  : join.getCustomSQLInsertCheckStyle();
201 			customSQLUpdate[j] = join.getCustomSQLUpdate();
202 			updateCallable[j] = customSQLUpdate[j] != null && join.isCustomUpdateCallable();
203 			updateResultCheckStyles[j] = join.getCustomSQLUpdateCheckStyle() == null
204 			                              ? ExecuteUpdateResultCheckStyle.determineDefault( customSQLUpdate[j], updateCallable[j] )
205 		                                  : join.getCustomSQLUpdateCheckStyle();
206 			customSQLDelete[j] = join.getCustomSQLDelete();
207 			deleteCallable[j] = customSQLDelete[j] != null && join.isCustomDeleteCallable();
208 			deleteResultCheckStyles[j] = join.getCustomSQLDeleteCheckStyle() == null
209 			                              ? ExecuteUpdateResultCheckStyle.determineDefault( customSQLDelete[j], deleteCallable[j] )
210 		                                  : join.getCustomSQLDeleteCheckStyle();
211 
212 			Iterator iter = join.getKey().getColumnIterator();
213 			keyColumnNames[j] = new String[ join.getKey().getColumnSpan() ];
214 			int i = 0;
215 			while ( iter.hasNext() ) {
216 				Column col = (Column) iter.next();
217 				keyColumnNames[j][i++] = col.getQuotedName( factory.getDialect() );
218 			}
219 
220 			j++;
221 		}
222 
223 		constraintOrderedTableNames = new String[qualifiedTableNames.length];
224 		constraintOrderedKeyColumnNames = new String[qualifiedTableNames.length][];
225 		for ( int i = qualifiedTableNames.length - 1, position = 0; i >= 0; i--, position++ ) {
226 			constraintOrderedTableNames[position] = qualifiedTableNames[i];
227 			constraintOrderedKeyColumnNames[position] = keyColumnNames[i];
228 		}
229 
230 		spaces = ArrayHelper.join(
231 				qualifiedTableNames, 
232 				ArrayHelper.toStringArray( persistentClass.getSynchronizedTables() )
233 		);
234 		
235 		final boolean lazyAvailable = isInstrumented(EntityMode.POJO);
236 
237 		boolean hasDeferred = false;
238 		ArrayList subclassTables = new ArrayList();
239 		ArrayList joinKeyColumns = new ArrayList();
240 		ArrayList isConcretes = new ArrayList();
241 		ArrayList isDeferreds = new ArrayList();
242 		ArrayList isInverses = new ArrayList();
243 		ArrayList isNullables = new ArrayList();
244 		ArrayList isLazies = new ArrayList();
245 		subclassTables.add( qualifiedTableNames[0] );
246 		joinKeyColumns.add( getIdentifierColumnNames() );
247 		isConcretes.add(Boolean.TRUE);
248 		isDeferreds.add(Boolean.FALSE);
249 		isInverses.add(Boolean.FALSE);
250 		isNullables.add(Boolean.FALSE);
251 		isLazies.add(Boolean.FALSE);
252 		joinIter = persistentClass.getSubclassJoinClosureIterator();
253 		while ( joinIter.hasNext() ) {
254 			Join join = (Join) joinIter.next();
255 			isConcretes.add( new Boolean( persistentClass.isClassOrSuperclassJoin(join) ) );
256 			isDeferreds.add( new Boolean( join.isSequentialSelect() ) );
257 			isInverses.add( new Boolean( join.isInverse() ) );
258 			isNullables.add( new Boolean( join.isOptional() ) );
259 			isLazies.add( new Boolean( lazyAvailable && join.isLazy() ) );
260 			if ( join.isSequentialSelect() && !persistentClass.isClassOrSuperclassJoin(join) ) hasDeferred = true;
261 			subclassTables.add( join.getTable().getQualifiedName( 
262 					factory.getDialect(), 
263 					factory.getSettings().getDefaultCatalogName(), 
264 					factory.getSettings().getDefaultSchemaName() 
265 			) );
266 			Iterator iter = join.getKey().getColumnIterator();
267 			String[] keyCols = new String[ join.getKey().getColumnSpan() ];
268 			int i = 0;
269 			while ( iter.hasNext() ) {
270 				Column col = (Column) iter.next();
271 				keyCols[i++] = col.getQuotedName( factory.getDialect() );
272 			}
273 			joinKeyColumns.add(keyCols);
274 		}
275 		
276 		subclassTableSequentialSelect = ArrayHelper.toBooleanArray(isDeferreds);
277 		subclassTableNameClosure = ArrayHelper.toStringArray(subclassTables);
278 		subclassTableIsLazyClosure = ArrayHelper.toBooleanArray(isLazies);
279 		subclassTableKeyColumnClosure = ArrayHelper.to2DStringArray(joinKeyColumns);
280 		isClassOrSuperclassTable = ArrayHelper.toBooleanArray(isConcretes);
281 		isInverseSubclassTable = ArrayHelper.toBooleanArray(isInverses);
282 		isNullableSubclassTable = ArrayHelper.toBooleanArray(isNullables);
283 		hasSequentialSelects = hasDeferred;
284 
285 		// DISCRIMINATOR
286 
287 		final Object discriminatorValue;
288 		if ( persistentClass.isPolymorphic() ) {
289 			Value discrimValue = persistentClass.getDiscriminator();
290 			if (discrimValue==null) {
291 				throw new MappingException("discriminator mapping required for single table polymorphic persistence");
292 			}
293 			forceDiscriminator = persistentClass.isForceDiscriminator();
294 			Selectable selectable = (Selectable) discrimValue.getColumnIterator().next();
295 			if ( discrimValue.hasFormula() ) {
296 				Formula formula = (Formula) selectable;
297 				discriminatorFormula = formula.getFormula();
298 				discriminatorFormulaTemplate = formula.getTemplate( factory.getDialect(), factory.getSqlFunctionRegistry() );
299 				discriminatorColumnName = null;
300 				discriminatorColumnReaders = null;
301 				discriminatorColumnReaderTemplate = null;
302 				discriminatorAlias = "clazz_";
303 			}
304 			else {
305 				Column column = (Column) selectable;
306 				discriminatorColumnName = column.getQuotedName( factory.getDialect() );
307 				discriminatorColumnReaders = column.getReadExpr( factory.getDialect() );
308 				discriminatorColumnReaderTemplate = column.getTemplate( factory.getDialect(), factory.getSqlFunctionRegistry() );
309 				discriminatorAlias = column.getAlias( factory.getDialect(), persistentClass.getRootTable() );
310 				discriminatorFormula = null;
311 				discriminatorFormulaTemplate = null;
312 			}
313 			discriminatorType = persistentClass.getDiscriminator().getType();
314 			if ( persistentClass.isDiscriminatorValueNull() ) {
315 				discriminatorValue = NULL_DISCRIMINATOR;
316 				discriminatorSQLValue = InFragment.NULL;
317 				discriminatorInsertable = false;
318 			}
319 			else if ( persistentClass.isDiscriminatorValueNotNull() ) {
320 				discriminatorValue = NOT_NULL_DISCRIMINATOR;
321 				discriminatorSQLValue = InFragment.NOT_NULL;
322 				discriminatorInsertable = false;
323 			}
324 			else {
325 				discriminatorInsertable = persistentClass.isDiscriminatorInsertable() && !discrimValue.hasFormula();
326 				try {
327 					DiscriminatorType dtype = (DiscriminatorType) discriminatorType;
328 					discriminatorValue = dtype.stringToObject( persistentClass.getDiscriminatorValue() );
329 					discriminatorSQLValue = dtype.objectToSQLString( discriminatorValue, factory.getDialect() );
330 				}
331 				catch (ClassCastException cce) {
332 					throw new MappingException("Illegal discriminator type: " + discriminatorType.getName() );
333 				}
334 				catch (Exception e) {
335 					throw new MappingException("Could not format discriminator value to SQL string", e);
336 				}
337 			}
338 		}
339 		else {
340 			forceDiscriminator = false;
341 			discriminatorInsertable = false;
342 			discriminatorColumnName = null;
343 			discriminatorColumnReaders = null;
344 			discriminatorColumnReaderTemplate = null;
345 			discriminatorAlias = null;
346 			discriminatorType = null;
347 			discriminatorValue = null;
348 			discriminatorSQLValue = null;
349 			discriminatorFormula = null;
350 			discriminatorFormulaTemplate = null;
351 		}
352 
353 		// PROPERTIES
354 
355 		propertyTableNumbers = new int[ getPropertySpan() ];
356 		Iterator iter = persistentClass.getPropertyClosureIterator();
357 		int i=0;
358 		while( iter.hasNext() ) {
359 			Property prop = (Property) iter.next();
360 			propertyTableNumbers[i++] = persistentClass.getJoinNumber(prop);
361 
362 		}
363 
364 		//TODO: code duplication with JoinedSubclassEntityPersister
365 		
366 		ArrayList columnJoinNumbers = new ArrayList();
367 		ArrayList formulaJoinedNumbers = new ArrayList();
368 		ArrayList propertyJoinNumbers = new ArrayList();
369 		
370 		iter = persistentClass.getSubclassPropertyClosureIterator();
371 		while ( iter.hasNext() ) {
372 			Property prop = (Property) iter.next();
373 			Integer join = new Integer( persistentClass.getJoinNumber(prop) );
374 			propertyJoinNumbers.add(join);
375 
376 			//propertyTableNumbersByName.put( prop.getName(), join );
377 			propertyTableNumbersByNameAndSubclass.put( 
378 					prop.getPersistentClass().getEntityName() + '.' + prop.getName(), 
379 					join 
380 			);
381 
382 			Iterator citer = prop.getColumnIterator();
383 			while ( citer.hasNext() ) {
384 				Selectable thing = (Selectable) citer.next();
385 				if ( thing.isFormula() ) {
386 					formulaJoinedNumbers.add(join);
387 				}
388 				else {
389 					columnJoinNumbers.add(join);
390 				}
391 			}
392 		}
393 		subclassColumnTableNumberClosure = ArrayHelper.toIntArray(columnJoinNumbers);
394 		subclassFormulaTableNumberClosure = ArrayHelper.toIntArray(formulaJoinedNumbers);
395 		subclassPropertyTableNumberClosure = ArrayHelper.toIntArray(propertyJoinNumbers);
396 
397 		int subclassSpan = persistentClass.getSubclassSpan() + 1;
398 		subclassClosure = new String[subclassSpan];
399 		subclassClosure[0] = getEntityName();
400 		if ( persistentClass.isPolymorphic() ) {
401 			subclassesByDiscriminatorValue.put( discriminatorValue, getEntityName() );
402 		}
403 
404 		// SUBCLASSES
405 		if ( persistentClass.isPolymorphic() ) {
406 			iter = persistentClass.getSubclassIterator();
407 			int k=1;
408 			while ( iter.hasNext() ) {
409 				Subclass sc = (Subclass) iter.next();
410 				subclassClosure[k++] = sc.getEntityName();
411 				if ( sc.isDiscriminatorValueNull() ) {
412 					subclassesByDiscriminatorValue.put( NULL_DISCRIMINATOR, sc.getEntityName() );
413 				}
414 				else if ( sc.isDiscriminatorValueNotNull() ) {
415 					subclassesByDiscriminatorValue.put( NOT_NULL_DISCRIMINATOR, sc.getEntityName() );
416 				}
417 				else {
418 					try {
419 						DiscriminatorType dtype = (DiscriminatorType) discriminatorType;
420 						subclassesByDiscriminatorValue.put(
421 							dtype.stringToObject( sc.getDiscriminatorValue() ),
422 							sc.getEntityName()
423 						);
424 					}
425 					catch (ClassCastException cce) {
426 						throw new MappingException("Illegal discriminator type: " + discriminatorType.getName() );
427 					}
428 					catch (Exception e) {
429 						throw new MappingException("Error parsing discriminator value", e);
430 					}
431 				}
432 			}
433 		}
434 
435 		initLockers();
436 
437 		initSubclassPropertyAliasesMap(persistentClass);
438 		
439 		postConstruct(mapping);
440 
441 	}

550 	private String discriminatorFilterFragment(String alias) throws MappingException {
551 		if ( needsDiscriminator() ) {
552 			InFragment frag = new InFragment();
553 
554 			if ( isDiscriminatorFormula() ) {
555 				frag.setFormula( alias, getDiscriminatorFormulaTemplate() );
556 			}
557 			else {
558 				frag.setColumn( alias, getDiscriminatorColumnName() );
559 			}
560 
561 			String[] subclasses = getSubclassClosure();
562 			for ( int i=0; i<subclasses.length; i++ ) {
563 				final Queryable queryable = (Queryable) getFactory().getEntityPersister( subclasses[i] );
564 				if ( !queryable.isAbstract() ) frag.addValue( queryable.getDiscriminatorSQLValue() );
565 			}
566 
567 			StringBuffer buf = new StringBuffer(50)
568 				.append(" and ")
569 				.append( frag.toFragmentString() );
570 
571 			return buf.toString();
572 		}
573 		else {
574 			return "";
575 		}
576 	}

636 	private int getSubclassPropertyTableNumber(String propertyName, String entityName) {
637 		Type type = propertyMapping.toType(propertyName);
638 		if ( type.isAssociationType() && ( (AssociationType) type ).useLHSPrimaryKey() ) return 0;
639 		final Integer tabnum = (Integer) propertyTableNumbersByNameAndSubclass.get(entityName + '.' + propertyName);
640 		return tabnum==null ? 0 : tabnum.intValue();
641 	}

647 	private String generateSequentialSelect(Loadable persister) {
648 		//if ( this==persister || !hasSequentialSelects ) return null;
649 
650 		//note that this method could easily be moved up to BasicEntityPersister,
651 		//if we ever needed to reuse it from other subclasses
652 		
653 		//figure out which tables need to be fetched
654 		AbstractEntityPersister subclassPersister = (AbstractEntityPersister) persister;
655 		HashSet tableNumbers = new HashSet();
656 		String[] props = subclassPersister.getPropertyNames();
657 		String[] classes = subclassPersister.getPropertySubclassNames();
658 		for ( int i=0; i<props.length; i++ ) {
659 			int propTableNumber = getSubclassPropertyTableNumber( props[i], classes[i] );
660 			if ( isSubclassTableSequentialSelect(propTableNumber) && !isSubclassTableLazy(propTableNumber) ) {
661 				tableNumbers.add( new Integer(propTableNumber) );
662 			}
663 		}
664 		if ( tableNumbers.isEmpty() ) return null;
665 		
666 		//figure out which columns are needed
667 		ArrayList columnNumbers = new ArrayList();
668 		final int[] columnTableNumbers = getSubclassColumnTableNumberClosure();
669 		for ( int i=0; i<getSubclassColumnClosure().length; i++ ) {
670 			if ( tableNumbers.contains( new Integer( columnTableNumbers[i] ) ) ) {
671 				columnNumbers.add( new Integer(i) );
672 			}
673 		}
674 		
675 		//figure out which formulas are needed
676 		ArrayList formulaNumbers = new ArrayList();
677 		final int[] formulaTableNumbers = getSubclassColumnTableNumberClosure();
678 		for ( int i=0; i<getSubclassFormulaTemplateClosure().length; i++ ) {
679 			if ( tableNumbers.contains( new Integer( formulaTableNumbers[i] ) ) ) {
680 				formulaNumbers.add( new Integer(i) );
681 			}
682 		}
683 		
684 		//render the SQL
685 		return renderSelect( 
686 			ArrayHelper.toIntArray(tableNumbers),
687 			ArrayHelper.toIntArray(columnNumbers),
688 			ArrayHelper.toIntArray(formulaNumbers)
689 		);
690 	}

727 	public void postInstantiate() {
728 		super.postInstantiate();
729 		if (hasSequentialSelects) {
730 			String[] entityNames = getSubclassClosure();
731 			for ( int i=1; i<entityNames.length; i++ ) {
732 				Loadable loadable = (Loadable) getFactory().getEntityPersister( entityNames[i] );
733 				if ( !loadable.isAbstract() ) { //perhaps not really necessary...
734 					String sequentialSelect = generateSequentialSelect(loadable);
735 					sequentialSelectStringsByEntityName.put( entityNames[i], sequentialSelect );
736 				}
737 			}
738 		}
