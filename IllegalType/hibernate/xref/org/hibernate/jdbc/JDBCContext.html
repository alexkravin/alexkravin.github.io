70  	public static interface Context extends TransactionFactory.Context {

75  		public void afterTransactionBegin(Transaction tx);
76  		public void beforeTransactionCompletion(Transaction tx);
77  		public void afterTransactionCompletion(boolean success, Transaction tx);
78  		public ConnectionReleaseMode getConnectionReleaseMode();
79  		public boolean isAutoCloseSessionEnabled();

89  	public JDBCContext(Context owner, Connection connection, Interceptor interceptor) {
90  		this.owner = owner;
91  		this.connectionManager = new ConnectionManager(
92  		        owner.getFactory(),
93  		        this,
94  		        owner.getConnectionReleaseMode(),
95  		        connection,
96  		        interceptor
97  			);
98  
99  		final boolean registerSynchronization = owner.isAutoCloseSessionEnabled()
100 		        || owner.isFlushBeforeCompletionEnabled()
101 		        || owner.getConnectionReleaseMode() == ConnectionReleaseMode.AFTER_TRANSACTION;
102 		if ( registerSynchronization ) {
103 			registerSynchronizationIfPossible();
104 		}
105 	}

174 	public boolean registerSynchronizationIfPossible() {
175 		if ( isTransactionCallbackRegistered ) {
176 			// we already have a callback registered; either a local
177 			// (org.hibernate.Transaction) transaction has accepted
178 			// callback responsibilities, or we have previously
179 			// registered a transaction synch.
180 			return true;
181 		}
182 		boolean localCallbacksOnly = owner.getFactory().getSettings()
183 				.getTransactionFactory()
184 				.areCallbacksLocalToHibernateTransactions();
185 		if ( localCallbacksOnly ) {
186 			// the configured transaction-factory says it only supports
187 			// local callback mode, so no sense attempting to register a
188 			// JTA Synchronization
189 			return false;
190 		}
191 		TransactionManager tm = owner.getFactory().getTransactionManager();
192 		if ( tm == null ) {
193 			// if there is no TM configured, we will not be able to access
194 			// the javax.transaction.Transaction object in order to
195 			// register a synch anyway.
196 			return false;
197 		}
198 		else {
199 			try {
200 				if ( !isTransactionInProgress() ) {
201 					log.trace( "TransactionFactory reported no active transaction; Synchronization not registered" );
202 					return false;
203 				}
204 				else {
205 					javax.transaction.Transaction tx = tm.getTransaction();
206 					if ( JTAHelper.isMarkedForRollback( tx ) ) {
207 						// transactions marked for rollback-only cause some TM impls to throw exceptions
208 						log.debug( "Transaction is marked for rollback; skipping Synchronization registration" );
209 						return false;
210 					}
211 					else {
212 						if ( hibernateTransaction == null ) {
213 							hibernateTransaction = owner.getFactory().getSettings().getTransactionFactory().createTransaction( this, owner );
214 						}
215 						tx.registerSynchronization(
216 								new HibernateSynchronizationImpl( getJtaSynchronizationCallbackCoordinator( tx ) )
217 						);
218 //						tx.registerSynchronization( new CacheSynchronization(owner, this, tx, hibernateTransaction) );
219 						isTransactionCallbackRegistered = true;
220 						log.debug("successfully registered Synchronization");
221 						return true;
222 					}
223 				}
224 			}
225 			catch( HibernateException e ) {
226 				throw e;
227 			}
228 			catch (Exception e) {
229 				throw new TransactionException( "could not register synchronization with JTA TransactionManager", e );
230 			}
231 		}
232 	}

239 	public Transaction getTransaction() throws HibernateException {
240 		if (hibernateTransaction==null) {
241 			hibernateTransaction = owner.getFactory().getSettings()
242 					.getTransactionFactory()
243 					.createTransaction( this, owner );
244 		}
245 		return hibernateTransaction;

262 	public void afterTransactionCompletion(boolean success, Transaction tx) {
263 		log.trace( "after transaction completion" );
264 
265 		if ( getFactory().getStatistics().isStatisticsEnabled() ) {
266 			getFactory().getStatisticsImplementor().endTransaction(success);
267 		}
268 
269 		connectionManager.afterTransaction();
270 
271 		isTransactionCallbackRegistered = false;
272 		hibernateTransaction = null;
273 		owner.afterTransactionCompletion(success, tx);
274 	}
275 	
276 	/**
277 	 * Called after executing a query outside the scope of
278 	 * a Hibernate or JTA transaction
279 	 */
280 	public void afterNontransactionalQuery(boolean success) {
281 		log.trace( "after autocommit" );
282 		try {
283 			// check to see if the connection is in auto-commit 
284 			// mode (no connection means aggressive connection
285 			// release outside a JTA transaction context, so MUST
286 			// be autocommit mode)
287 			boolean isAutocommit = connectionManager.isAutoCommit();
288 
289 			connectionManager.afterTransaction();
290 			
291 			if ( isAutocommit ) {
292 				owner.afterTransactionCompletion(success, null);
293 			}
294 		}
295 		catch (SQLException sqle) {
296 			throw JDBCExceptionHelper.convert( 
297 					owner.getFactory().getSQLExceptionConverter(),
298 					sqle,
299 					"could not inspect JDBC autocommit mode"
300 				);
301 		}
302 	}

307 	private void writeObject(ObjectOutputStream oos) throws IOException {
308 		// isTransactionCallbackRegistered denotes whether any Hibernate
309 		// Transaction has registered as a callback against this
310 		// JDBCContext; only one such callback is allowed.  Directly
311 		// serializing this value causes problems with JDBCTransaction,
312 		// or really any Transaction impl where the callback is local
313 		// to the Transaction instance itself, since that Transaction
314 		// is not serialized along with the JDBCContext.  Thus we
315 		// handle that fact here explicitly...
316 		oos.defaultWriteObject();
317 		boolean deserHasCallbackRegistered = isTransactionCallbackRegistered
318 				&& ! owner.getFactory().getSettings().getTransactionFactory()
319 				.areCallbacksLocalToHibernateTransactions();
320 		oos.writeBoolean( deserHasCallbackRegistered );
321 	}
322 
323 	private void readObject(ObjectInputStream ois) throws IOException, ClassNotFoundException {
324 		ois.defaultReadObject();
325 		isTransactionCallbackRegistered = ois.readBoolean();
326 	}

339 	/**
340 	 * Custom deserialization routine used during deserialization of a
341 	 * Session/PersistenceContext for increased performance.
342 	 *
343 	 * @param ois The stream from which to read the entry.
344 	 * @throws IOException
345 	 */
346 	public static JDBCContext deserialize(
347 			ObjectInputStream ois,
348 	        Context context,
349 	        Interceptor interceptor) throws IOException {
350 		JDBCContext jdbcContext = new JDBCContext();
351 		jdbcContext.owner = context;
352 		jdbcContext.connectionManager = ConnectionManager.deserialize(
353 				ois,
354 				context.getFactory(),
355 		        interceptor,
356 		        context.getConnectionReleaseMode(),
357 		        jdbcContext
358 		);
359 		return jdbcContext;
360 	}
