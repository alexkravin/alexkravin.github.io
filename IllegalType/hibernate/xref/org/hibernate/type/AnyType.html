100 	public Object nullSafeGet(ResultSet rs,	String[] names,	SessionImplementor session,	Object owner)
101 	throws HibernateException, SQLException {
102 		return resolveAny(
103 				(String) metaType.nullSafeGet(rs, names[0], session, owner),
104 				(Serializable) identifierType.nullSafeGet(rs, names[1], session, owner),
105 				session
106 			);
107 	}

109 	public Object hydrate(ResultSet rs,	String[] names,	SessionImplementor session,	Object owner)
110 	throws HibernateException, SQLException {
111 		String entityName = (String) metaType.nullSafeGet(rs, names[0], session, owner);
112 		Serializable id = (Serializable) identifierType.nullSafeGet(rs, names[1], session, owner);
113 		return new ObjectTypeCacheEntry(entityName, id);
114 	}

127 	private Object resolveAny(String entityName, Serializable id, SessionImplementor session)
128 	throws HibernateException {
129 		return entityName==null || id==null ?
130 				null : session.internalLoad( entityName, id, false, false );
131 	}

138 	public void nullSafeSet(PreparedStatement st, Object value,	int index, boolean[] settable, SessionImplementor session)
139 	throws HibernateException, SQLException {
140 
141 		Serializable id;
142 		String entityName;
143 		if (value==null) {
144 			id=null;
145 			entityName=null;
146 		}
147 		else {
148 			entityName = session.bestGuessEntityName(value);
149 			id = ForeignKeys.getEntityIdentifierIfNotUnsaved(entityName, value, session);
150 		}
151 		
152 		// metaType is assumed to be single-column type
153 		if ( settable==null || settable[0] ) {
154 			metaType.nullSafeSet(st, entityName, index, session);
155 		}
156 		if (settable==null) {
157 			identifierType.nullSafeSet(st, id, index+1, session);
158 		}
159 		else {
160 			boolean[] idsettable = new boolean[ settable.length-1 ];
161 			System.arraycopy(settable, 1, idsettable, 0, idsettable.length);
162 			identifierType.nullSafeSet(st, id, index+1, idsettable, session);
163 		}
164 	}

213 	public Serializable disassemble(Object value, SessionImplementor session, Object owner)
214 	throws HibernateException {
215 		return value==null ?
216 			null :
217 			new ObjectTypeCacheEntry(
218 						session.bestGuessEntityName(value),
219 						ForeignKeys.getEntityIdentifierIfNotUnsaved( 
220 								session.bestGuessEntityName(value), value, session 
221 							)
222 					);
223 	}

229 	public Object replace(
230 			Object original, 
231 			Object target,
232 			SessionImplementor session, 
233 			Object owner, 
234 			Map copyCache)
235 	throws HibernateException {
236 		if (original==null) {
237 			return null;
238 		}
239 		else {
240 			String entityName = session.bestGuessEntityName(original);
241 			Serializable id = ForeignKeys.getEntityIdentifierIfNotUnsaved( 
242 					entityName, 
243 					original, 
244 					session 
245 				);
246 			return session.internalLoad( 
247 					entityName, 
248 					id, 
249 					false, 
250 					false
251 				);
252 		}
253 	}

283 		try {
284 			return ForeignKeys.getEntityIdentifierIfNotUnsaved( session.bestGuessEntityName(value), value, session );

327 	public boolean isModified(Object old, Object current, boolean[] checkable, SessionImplementor session)
328 	throws HibernateException {
329 		if (current==null) return old!=null;
330 		if (old==null) return current!=null;
331 		ObjectTypeCacheEntry holder = (ObjectTypeCacheEntry) old;
332 		boolean[] idcheckable = new boolean[checkable.length-1];
333 		System.arraycopy(checkable, 1, idcheckable, 0, idcheckable.length);
334 		return ( checkable[0] && !holder.entityName.equals( session.bestGuessEntityName(current) ) ) ||
335 				identifierType.isModified(holder.id, getIdentifier(current, session), idcheckable, session);
336 	}
