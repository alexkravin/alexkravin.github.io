86  	public void addColumn(Column column) {

89  		column.setTypeIndex( columns.size()-1 );
90  	}

96  	public boolean hasFormula() {
97  		Iterator iter = getColumnIterator();
98  		while ( iter.hasNext() ) {
99  			Object o = iter.next();
100 			if (o instanceof Formula) return true;
101 		}
102 		return false;
103 	}

126 	public void createForeignKeyOfEntity(String entityName) {
127 		if ( !hasFormula() && !"none".equals(getForeignKeyName())) {
128 			ForeignKey fk = table.createForeignKey( getForeignKeyName(), getConstraintColumns(), entityName );
129 			fk.setCascadeDeleteEnabled(cascadeDeleteEnabled);
130 		}

133 	public IdentifierGenerator createIdentifierGenerator(
134 			IdentifierGeneratorFactory identifierGeneratorFactory,
135 			Dialect dialect, 
136 			String defaultCatalog, 
137 			String defaultSchema, 
138 			RootClass rootClass) throws MappingException {
139 		
140 		Properties params = new Properties();
141 		
142 		//if the hibernate-mapping did not specify a schema/catalog, use the defaults
143 		//specified by properties - but note that if the schema/catalog were specified
144 		//in hibernate-mapping, or as params, they will already be initialized and
145 		//will override the values set here (they are in identifierGeneratorProperties)
146 		if ( defaultSchema!=null ) {
147 			params.setProperty(PersistentIdentifierGenerator.SCHEMA, defaultSchema);
148 		}
149 		if ( defaultCatalog!=null ) {
150 			params.setProperty(PersistentIdentifierGenerator.CATALOG, defaultCatalog);
151 		}
152 		
153 		//pass the entity-name, if not a collection-id
154 		if (rootClass!=null) {
155 			params.setProperty( IdentifierGenerator.ENTITY_NAME, rootClass.getEntityName() );
156 		}
157 		
158 		//init the table here instead of earlier, so that we can get a quoted table name
159 		//TODO: would it be better to simply pass the qualified table name, instead of
160 		//      splitting it up into schema/catalog/table names
161 		String tableName = getTable().getQuotedName(dialect);
162 		params.setProperty( PersistentIdentifierGenerator.TABLE, tableName );
163 		
164 		//pass the column name (a generated id almost always has a single column)
165 		String columnName = ( (Column) getColumnIterator().next() ).getQuotedName(dialect);
166 		params.setProperty( PersistentIdentifierGenerator.PK, columnName );
167 		
168 		if (rootClass!=null) {
169 			StringBuffer tables = new StringBuffer();
170 			Iterator iter = rootClass.getIdentityTables().iterator();
171 			while ( iter.hasNext() ) {
172 				Table table= (Table) iter.next();
173 				tables.append( table.getQuotedName(dialect) );
174 				if ( iter.hasNext() ) tables.append(", ");
175 			}
176 			params.setProperty( PersistentIdentifierGenerator.TABLES, tables.toString() );
177 		}
178 		else {
179 			params.setProperty( PersistentIdentifierGenerator.TABLES, tableName );
180 		}
181 
182 		if (identifierGeneratorProperties!=null) {
183 			params.putAll(identifierGeneratorProperties);
184 		}
185 
186 		// TODO : we should pass along all settings once "config lifecycle" is hashed out...
187 		params.put(
188 				Environment.PREFER_POOLED_VALUES_LO,
189 				mappings.getConfigurationProperties().getProperty( Environment.PREFER_POOLED_VALUES_LO, "false" )
190 		);
191 
192 		identifierGeneratorFactory.setDialect( dialect );
193 		return identifierGeneratorFactory.createIdentifierGenerator( identifierGeneratorStrategy, getType(), params );
194 		
195 	}

272 	public boolean isNullable() {
273 		if ( hasFormula() ) return true;
274 		boolean nullable = true;
275 		Iterator iter = getColumnIterator();
276 		while ( iter.hasNext() ) {
277 			if ( !( (Column) iter.next() ).isNullable() ) {
278 				nullable = false;
279 				return nullable; //shortcut
280 			}
281 		}
282 		return nullable;
283 	}

293 	public Type getType() throws MappingException {
294 		if (typeName==null) {
295 			throw new MappingException("No type name");
296 		}
297 		Type result = mappings.getTypeResolver().heuristicType(typeName, typeParameters);
298 		if (result==null) {
299 			String msg = "Could not determine type for: " + typeName;
300 			if(table != null){
301 				msg += ", at table: " + table.getName();
302 			}
303 			if(columns!=null && columns.size()>0) {
304 				msg += ", for columns: " + columns;
305 			}
306 			throw new MappingException(msg);
307 		}
308 		return result;
309 	}

311 	public void setTypeUsingReflection(String className, String propertyName) throws MappingException {
312 		if (typeName==null) {
313 			if (className==null) {
314 				throw new MappingException("you must specify types for a dynamic entity: " + propertyName);
315 			}
316 			typeName = ReflectHelper.reflectedPropertyClass(className, propertyName).getName();
317 		}
318 	}

340 	public boolean[] getColumnInsertability() {
341 		boolean[] result = new boolean[ getColumnSpan() ];
342 		int i = 0;
343 		Iterator iter = getColumnIterator();
344 		while ( iter.hasNext() ) {
345 			Selectable s = (Selectable) iter.next();
346 			result[i++] = !s.isFormula();
347 		}
348 		return result;
349 	}

351 	public boolean[] getColumnUpdateability() {
352 		return getColumnInsertability();
