58  /**
59   * Defines the internal contract between the <tt>SessionFactory</tt> and other parts of
60   * Hibernate such as implementors of <tt>Type</tt>.
61   *
62   * @see org.hibernate.SessionFactory
63   * @see org.hibernate.impl.SessionFactoryImpl
64   * @author Gavin King
65   */
66  public interface SessionFactoryImplementor extends Mapping, SessionFactory {
67  	/**
68  	 * Retrieve the {@link Type} resolver associated with this factory.
69  	 *
70  	 * @return The type resolver
71  	 */
72  	public TypeResolver getTypeResolver();
73  
74  	/**
75  	 * Get a copy of the Properties used to configure this session factory.
76  	 *
77  	 * @return The properties.
78  	 */
79  	public Properties getProperties();
80  
81  	/**
82  	 * Get the persister for the named entity
83  	 *
84  	 * @param entityName The name of the entity for which to retrieve the persister.
85  	 * @return The persister
86  	 * @throws MappingException Indicates persister could not be found with that name.
87  	 */
88  	public EntityPersister getEntityPersister(String entityName) throws MappingException;
89  
90  	/**
91  	 * Get the persister object for a collection role.
92  	 *
93  	 * @param role The role (name) of the collection for which to retrieve the
94  	 * persister.
95  	 * @return The persister
96  	 * @throws MappingException Indicates persister could not be found with that role.
97  	 */
98  	public CollectionPersister getCollectionPersister(String role) throws MappingException;
99  
100 	/**
101 	 * Get the SQL dialect.
102 	 * <p/>
103 	 * Shorthand for {@link #getSettings()}.{@link Settings#getDialect()}
104 	 *
105 	 * @return The dialect
106 	 */
107 	public Dialect getDialect();
108 
109 	/**
110 	 * Get the factory scoped interceptor for this factory.
111 	 *
112 	 * @return The factory scope interceptor, or null if none.
113 	 */
114 	public Interceptor getInterceptor();
115 
116 	public QueryPlanCache getQueryPlanCache();
117 
118 	/**
119 	 * Get the return types of a query
120 	 */
121 	public Type[] getReturnTypes(String queryString) throws HibernateException;
122 
123 	/**
124 	 * Get the return aliases of a query
125 	 */
126 	public String[] getReturnAliases(String queryString) throws HibernateException;
127 
128 	/**
129 	 * Get the connection provider
130 	 */
131 	public ConnectionProvider getConnectionProvider();
132 	/**
133 	 * Get the names of all persistent classes that implement/extend the given interface/class
134 	 */
135 	public String[] getImplementors(String className) throws MappingException;
136 	/**
137 	 * Get a class name, using query language imports
138 	 */
139 	public String getImportedClassName(String name);
140 
141 
142 	/**
143 	 * Get the JTA transaction manager
144 	 */
145 	public TransactionManager getTransactionManager();
146 
147 
148 	/**
149 	 * Get the default query cache
150 	 */
151 	public QueryCache getQueryCache();
152 	/**
153 	 * Get a particular named query cache, or the default cache
154 	 * @param regionName the name of the cache region, or null for the default query cache
155 	 * @return the existing cache, or a newly created cache if none by that region name
156 	 */
157 	public QueryCache getQueryCache(String regionName) throws HibernateException;
158 	
159 	/**
160 	 * Get the cache of table update timestamps
161 	 */
162 	public UpdateTimestampsCache getUpdateTimestampsCache();
163 	/**
164 	 * Statistics SPI
165 	 */
166 	public StatisticsImplementor getStatisticsImplementor();
167 	
168 	public NamedQueryDefinition getNamedQuery(String queryName);
169 	public NamedSQLQueryDefinition getNamedSQLQuery(String queryName);
170 	public ResultSetMappingDefinition getResultSetMapping(String name);
171 
172 	/**
173 	 * Get the identifier generator for the hierarchy
174 	 */
175 	public IdentifierGenerator getIdentifierGenerator(String rootEntityName);
176 	
177 	/**
178 	 * Get a named second-level cache region
179 	 *
180 	 * @param regionName The name of the region to retrieve.
181 	 * @return The region
182 	 */
183 	public Region getSecondLevelCacheRegion(String regionName);
184 
185 	/**
186 	 * Get a map of all the second level cache regions currently maintained in
187 	 * this session factory.  The map is structured with the region name as the
188 	 * key and the {@link Region} instances as the values.
189 	 *
190 	 * @return The map of regions
191 	 */
192 	public Map getAllSecondLevelCacheRegions();
193 	
194 	/**
195 	 * Retrieves the SQLExceptionConverter in effect for this SessionFactory.
196 	 *
197 	 * @return The SQLExceptionConverter for this SessionFactory.
198 	 */
199 	public SQLExceptionConverter getSQLExceptionConverter();
200 
201 	public Settings getSettings();
202 
203 	/**
204 	 * Get a nontransactional "current" session for Hibernate EntityManager
205 	 */
206 	public org.hibernate.classic.Session openTemporarySession() throws HibernateException;
207 
208 	/**
209 	 * Open a session conforming to the given parameters.  Used mainly by
210 	 * {@link org.hibernate.context.JTASessionContext} for current session processing.
211 	 *
212 	 * @param connection The external jdbc connection to use, if one (i.e., optional).
213 	 * @param flushBeforeCompletionEnabled Should the session be auto-flushed
214 	 * prior to transaction completion?
215 	 * @param autoCloseSessionEnabled Should the session be auto-closed after
216 	 * transaction completion?
217 	 * @param connectionReleaseMode The release mode for managed jdbc connections.
218 	 * @return An appropriate session.
219 	 * @throws HibernateException
220 	 */
221 	public org.hibernate.classic.Session openSession(
222 			final Connection connection,
223 			final boolean flushBeforeCompletionEnabled,
224 			final boolean autoCloseSessionEnabled,
225 			final ConnectionReleaseMode connectionReleaseMode) throws HibernateException;
226 
227 	/**
228 	 * Retrieves a set of all the collection roles in which the given entity
229 	 * is a participant, as either an index or an element.
230 	 *
231 	 * @param entityName The entity name for which to get the collection roles.
232 	 * @return set of all the collection roles in which the given entityName participates.
233 	 */
234 	public Set<String> getCollectionRolesByEntityParticipant(String entityName);
235 
236 	public EntityNotFoundDelegate getEntityNotFoundDelegate();
237 
238 	public SQLFunctionRegistry getSqlFunctionRegistry();
239 
240 	/**
241 	 * Retrieve fetch profile by name.
242 	 *
243 	 * @param name The name of the profile to retrieve.
244 	 * @return The profile definition
245 	 */
246 	public FetchProfile getFetchProfile(String name);
247 
248 	/**
249 	 * Return the SessionFactoryObserver attached to the SessionFactory if any
250 	 * @return
251 	 */
252 	public SessionFactoryObserver getFactoryObserver();
253 
254 }
