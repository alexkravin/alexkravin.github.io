34  /**
35   * A contract for defining the aspects of cascading various persistence actions.
36   *
37   * @see CascadingAction
38   *
39   * @author Gavin King
40   */
41  public abstract class CascadeStyle implements Serializable {
42  
43  	/**
44  	 * For this style, should the given action be cascaded?
45  	 *
46  	 * @param action The action to be checked for cascade-ability.
47  	 * @return True if the action should be cascaded under this style; false otherwise.
48  	 */
49  	public abstract boolean doCascade(CascadingAction action);
50  
51  	/**
52  	 * Probably more aptly named something like doCascadeToCollectionElements(); it is
53  	 * however used from both the collection and to-one logic branches...
54  	 * <p/>
55  	 * For this style, should the given action really be cascaded?  The default
56  	 * implementation is simply to return {@link #doCascade}; for certain
57  	 * styles (currently only delete-orphan), however, we need to be able to
58  	 * control this seperately.
59  	 *
60  	 * @param action The action to be checked for cascade-ability.
61  	 * @return True if the action should be really cascaded under this style;
62  	 * false otherwise.
63  	 */
64  	public boolean reallyDoCascade(CascadingAction action) {
65  		return doCascade(action);
66  	}
67  
68  	/**
69  	 * Do we need to delete orphaned collection elements?
70  	 *
71  	 * @return True if this style need to account for orphan delete
72  	 * operations; false othwerwise.
73  	 */
74  	public boolean hasOrphanDelete() {
75  		return false;
76  	}
77  
78  	public static final class MultipleCascadeStyle extends CascadeStyle {
79  		private final CascadeStyle[] styles;
80  		public MultipleCascadeStyle(CascadeStyle[] styles) {
81  			this.styles = styles;
82  		}
83  		public boolean doCascade(CascadingAction action) {
84  			for (int i=0; i<styles.length; i++) {
85  				if ( styles[i].doCascade(action) ) return true;
86  			}
87  			return false;
88  		}
89  		public boolean reallyDoCascade(CascadingAction action) {
90  			for (int i=0; i<styles.length; i++) {
91  				if ( styles[i].reallyDoCascade(action) ) return true;
92  			}
93  			return false;
94  		}
95  		public boolean hasOrphanDelete() {
96  			for (int i=0; i<styles.length; i++) {
97  				if ( styles[i].hasOrphanDelete() ) return true;
98  			}
99  			return false;
100 		}
101 		public String toString() {
102 			return ArrayHelper.toString(styles);
103 		}
104 	}

276 		STYLES.put( "evict", EVICT );
277 		STYLES.put( "delete", DELETE );
278 		STYLES.put( "remove", DELETE ); // adds remove as a sort-of alias for delete...
279 		STYLES.put( "delete-orphan", DELETE_ORPHAN );
280 		STYLES.put( "none", NONE );
281 	}
282 
283 	/**
284 	 * Factory method for obtaining named cascade styles
285 	 *
286 	 * @param cascade The named cascade style name.
287 	 * @return The appropriate CascadeStyle
288 	 */
289 	public static CascadeStyle getCascadeStyle(String cascade) {
290 		CascadeStyle style = (CascadeStyle) STYLES.get(cascade);
291 		if (style==null) {
292 			throw new MappingException("Unsupported cascade style: " + cascade);
293 		}
294 		else {
295 			return style;
296 		}
297 	}
