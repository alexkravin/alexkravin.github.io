66  /**
67   * Reads configuration properties and builds a {@link Settings} instance.
68   *
69   * @author Gavin King
70   */
71  public class SettingsFactory implements Serializable {
72  	private static final Logger log = LoggerFactory.getLogger( SettingsFactory.class );
73  	private static final long serialVersionUID = -1194386144994524825L;
74  
75  	public static final String DEF_CACHE_REG_FACTORY = NoCachingRegionFactory.class.getName();

77  	protected SettingsFactory() {
78  	}

80  	public Settings buildSettings(Properties props) {
81  		Settings settings = new Settings();
82  
83  		//SessionFactory name:
84  
85  		String sessionFactoryName = props.getProperty(Environment.SESSION_FACTORY_NAME);
86  		settings.setSessionFactoryName(sessionFactoryName);
87  
88  		//JDBC and connection settings:
89  
90  		ConnectionProvider connections = createConnectionProvider(props);
91  		settings.setConnectionProvider(connections);
92  
93  		//Interrogate JDBC metadata
94  
95  		boolean metaSupportsScrollable = false;
96  		boolean metaSupportsGetGeneratedKeys = false;
97  		boolean metaSupportsBatchUpdates = false;
98  		boolean metaReportsDDLCausesTxnCommit = false;
99  		boolean metaReportsDDLInTxnSupported = true;
100 		Dialect dialect = null;
101 		JdbcSupport jdbcSupport = null;
102 
103 		// 'hibernate.temp.use_jdbc_metadata_defaults' is a temporary magic value.
104 		// The need for it is intended to be alleviated with future development, thus it is
105 		// not defined as an Environment constant...
106 		//
107 		// it is used to control whether we should consult the JDBC metadata to determine
108 		// certain Settings default values; it is useful to *not* do this when the database
109 		// may not be available (mainly in tools usage).
110 		boolean useJdbcMetadata = PropertiesHelper.getBoolean( "hibernate.temp.use_jdbc_metadata_defaults", props, true );
111 		if ( useJdbcMetadata ) {
112 			try {
113 				Connection conn = connections.getConnection();
114 				try {
115 					DatabaseMetaData meta = conn.getMetaData();
116 
117 					dialect = DialectFactory.buildDialect( props, conn );
118 					jdbcSupport = JdbcSupportLoader.loadJdbcSupport( conn );
119 
120 					metaSupportsScrollable = meta.supportsResultSetType( ResultSet.TYPE_SCROLL_INSENSITIVE );
121 					metaSupportsBatchUpdates = meta.supportsBatchUpdates();
122 					metaReportsDDLCausesTxnCommit = meta.dataDefinitionCausesTransactionCommit();
123 					metaReportsDDLInTxnSupported = !meta.dataDefinitionIgnoredInTransactions();
124 					metaSupportsGetGeneratedKeys = meta.supportsGetGeneratedKeys();
125 					
126 					log.info( "Database ->\n" +
127 							"       name : " + meta.getDatabaseProductName() + '\n' +
128 							"    version : " +  meta.getDatabaseProductVersion() + '\n' +
129 							"      major : " + meta.getDatabaseMajorVersion() + '\n' +
130 							"      minor : " + meta.getDatabaseMinorVersion()
131 					);
132 					log.info( "Driver ->\n" +
133 							"       name : " + meta.getDriverName() + '\n' +
134 							"    version : " + meta.getDriverVersion() + '\n' +
135 							"      major : " + meta.getDriverMajorVersion() + '\n' +
136 							"      minor : " + meta.getDriverMinorVersion()
137 					);
138 				}
139 				catch ( SQLException sqle ) {
140 					log.warn( "Could not obtain connection metadata", sqle );
141 				}
142 				finally {
143 					connections.closeConnection( conn );
144 				}
145 			}
146 			catch ( SQLException sqle ) {
147 				log.warn( "Could not obtain connection to query metadata", sqle );
148 				dialect = DialectFactory.buildDialect( props );
149 			}
150 			catch ( UnsupportedOperationException uoe ) {
151 				// user supplied JDBC connections
152 				dialect = DialectFactory.buildDialect( props );
153 			}
154 		}
155 		else {
156 			dialect = DialectFactory.buildDialect( props );
157 		}
158 
159 		settings.setDataDefinitionImplicitCommit( metaReportsDDLCausesTxnCommit );
160 		settings.setDataDefinitionInTransactionSupported( metaReportsDDLInTxnSupported );
161 		settings.setDialect( dialect );
162 		if ( jdbcSupport == null ) {
163 			jdbcSupport = JdbcSupportLoader.loadJdbcSupport( null );
164 		}
165 		settings.setJdbcSupport( jdbcSupport );
166 
167 		//use dialect default properties
168 		final Properties properties = new Properties();
169 		properties.putAll( dialect.getDefaultProperties() );
170 		properties.putAll( props );
171 
172 		// Transaction settings:
173 
174 		TransactionFactory transactionFactory = createTransactionFactory(properties);
175 		settings.setTransactionFactory(transactionFactory);
176 		settings.setTransactionManagerLookup( createTransactionManagerLookup(properties) );
177 
178 		boolean flushBeforeCompletion = PropertiesHelper.getBoolean(Environment.FLUSH_BEFORE_COMPLETION, properties);
179 		log.info("Automatic flush during beforeCompletion(): " + enabledDisabled(flushBeforeCompletion) );
180 		settings.setFlushBeforeCompletionEnabled(flushBeforeCompletion);
181 
182 		boolean autoCloseSession = PropertiesHelper.getBoolean(Environment.AUTO_CLOSE_SESSION, properties);
183 		log.info("Automatic session close at end of transaction: " + enabledDisabled(autoCloseSession) );
184 		settings.setAutoCloseSessionEnabled(autoCloseSession);
185 
186 		//JDBC and connection settings:
187 
188 		int batchSize = PropertiesHelper.getInt(Environment.STATEMENT_BATCH_SIZE, properties, 0);
189 		if ( !metaSupportsBatchUpdates ) batchSize = 0;
190 		if (batchSize>0) log.info("JDBC batch size: " + batchSize);
191 		settings.setJdbcBatchSize(batchSize);
192 		boolean jdbcBatchVersionedData = PropertiesHelper.getBoolean(Environment.BATCH_VERSIONED_DATA, properties, false);
193 		if (batchSize>0) log.info("JDBC batch updates for versioned data: " + enabledDisabled(jdbcBatchVersionedData) );
194 		settings.setJdbcBatchVersionedData(jdbcBatchVersionedData);
195 		settings.setBatcherFactory( createBatcherFactory(properties, batchSize) );
196 
197 		boolean useScrollableResultSets = PropertiesHelper.getBoolean(Environment.USE_SCROLLABLE_RESULTSET, properties, metaSupportsScrollable);
198 		log.info("Scrollable result sets: " + enabledDisabled(useScrollableResultSets) );
199 		settings.setScrollableResultSetsEnabled(useScrollableResultSets);
200 
201 		boolean wrapResultSets = PropertiesHelper.getBoolean(Environment.WRAP_RESULT_SETS, properties, false);
202 		log.debug( "Wrap result sets: " + enabledDisabled(wrapResultSets) );
203 		settings.setWrapResultSetsEnabled(wrapResultSets);
204 
205 		boolean useGetGeneratedKeys = PropertiesHelper.getBoolean(Environment.USE_GET_GENERATED_KEYS, properties, metaSupportsGetGeneratedKeys);
206 		log.info("JDBC3 getGeneratedKeys(): " + enabledDisabled(useGetGeneratedKeys) );
207 		settings.setGetGeneratedKeysEnabled(useGetGeneratedKeys);
208 
209 		Integer statementFetchSize = PropertiesHelper.getInteger(Environment.STATEMENT_FETCH_SIZE, properties);
210 		if (statementFetchSize!=null) log.info("JDBC result set fetch size: " + statementFetchSize);
211 		settings.setJdbcFetchSize(statementFetchSize);
212 
213 		String releaseModeName = PropertiesHelper.getString( Environment.RELEASE_CONNECTIONS, properties, "auto" );
214 		log.info( "Connection release mode: " + releaseModeName );
215 		ConnectionReleaseMode releaseMode;
216 		if ( "auto".equals(releaseModeName) ) {
217 			releaseMode = transactionFactory.getDefaultReleaseMode();
218 		}
219 		else {
220 			releaseMode = ConnectionReleaseMode.parse( releaseModeName );
221 			if ( releaseMode == ConnectionReleaseMode.AFTER_STATEMENT && !connections.supportsAggressiveRelease() ) {
222 				log.warn( "Overriding release mode as connection provider does not support 'after_statement'" );
223 				releaseMode = ConnectionReleaseMode.AFTER_TRANSACTION;
224 			}
225 		}
226 		settings.setConnectionReleaseMode( releaseMode );
227 
228 		//SQL Generation settings:
229 
230 		String defaultSchema = properties.getProperty(Environment.DEFAULT_SCHEMA);
231 		String defaultCatalog = properties.getProperty(Environment.DEFAULT_CATALOG);
232 		if (defaultSchema!=null) log.info("Default schema: " + defaultSchema);
233 		if (defaultCatalog!=null) log.info("Default catalog: " + defaultCatalog);
234 		settings.setDefaultSchemaName(defaultSchema);
235 		settings.setDefaultCatalogName(defaultCatalog);
236 
237 		Integer maxFetchDepth = PropertiesHelper.getInteger(Environment.MAX_FETCH_DEPTH, properties);
238 		if (maxFetchDepth!=null) log.info("Maximum outer join fetch depth: " + maxFetchDepth);
239 		settings.setMaximumFetchDepth(maxFetchDepth);
240 		int batchFetchSize = PropertiesHelper.getInt(Environment.DEFAULT_BATCH_FETCH_SIZE, properties, 1);
241 		log.info("Default batch fetch size: " + batchFetchSize);
242 		settings.setDefaultBatchFetchSize(batchFetchSize);
243 
244 		boolean comments = PropertiesHelper.getBoolean(Environment.USE_SQL_COMMENTS, properties);
245 		log.info( "Generate SQL with comments: " + enabledDisabled(comments) );
246 		settings.setCommentsEnabled(comments);
247 
248 		boolean orderUpdates = PropertiesHelper.getBoolean(Environment.ORDER_UPDATES, properties);
249 		log.info( "Order SQL updates by primary key: " + enabledDisabled(orderUpdates) );
250 		settings.setOrderUpdatesEnabled(orderUpdates);
251 
252 		boolean orderInserts = PropertiesHelper.getBoolean(Environment.ORDER_INSERTS, properties);
253 		log.info( "Order SQL inserts for batching: " + enabledDisabled( orderInserts ) );
254 		settings.setOrderInsertsEnabled( orderInserts );
255 
256 		//Query parser settings:
257 
258 		settings.setQueryTranslatorFactory( createQueryTranslatorFactory(properties) );
259 
260 		Map querySubstitutions = PropertiesHelper.toMap(Environment.QUERY_SUBSTITUTIONS, " ,=;:\n\t\r\f", properties);
261 		log.info("Query language substitutions: " + querySubstitutions);
262 		settings.setQuerySubstitutions(querySubstitutions);
263 
264 		boolean jpaqlCompliance = PropertiesHelper.getBoolean( Environment.JPAQL_STRICT_COMPLIANCE, properties, false );
265 		settings.setStrictJPAQLCompliance( jpaqlCompliance );
266 		log.info( "JPA-QL strict compliance: " + enabledDisabled( jpaqlCompliance ) );
267 
268 		// Second-level / query cache:
269 
270 		boolean useSecondLevelCache = PropertiesHelper.getBoolean(Environment.USE_SECOND_LEVEL_CACHE, properties, true);
271 		log.info( "Second-level cache: " + enabledDisabled(useSecondLevelCache) );
272 		settings.setSecondLevelCacheEnabled(useSecondLevelCache);
273 
274 		boolean useQueryCache = PropertiesHelper.getBoolean(Environment.USE_QUERY_CACHE, properties);
275 		log.info( "Query cache: " + enabledDisabled(useQueryCache) );
276 		settings.setQueryCacheEnabled(useQueryCache);
277 
278 		// The cache provider is needed when we either have second-level cache enabled
279 		// or query cache enabled.  Note that useSecondLevelCache is enabled by default
280 		settings.setRegionFactory( createRegionFactory( properties, ( useSecondLevelCache || useQueryCache ) ) );
281 
282 		boolean useMinimalPuts = PropertiesHelper.getBoolean(
283 				Environment.USE_MINIMAL_PUTS, properties, settings.getRegionFactory().isMinimalPutsEnabledByDefault()
284 		);
285 		log.info( "Optimize cache for minimal puts: " + enabledDisabled(useMinimalPuts) );
286 		settings.setMinimalPutsEnabled(useMinimalPuts);
287 
288 		String prefix = properties.getProperty(Environment.CACHE_REGION_PREFIX);
289 		if ( StringHelper.isEmpty(prefix) ) prefix=null;
290 		if (prefix!=null) log.info("Cache region prefix: "+ prefix);
291 		settings.setCacheRegionPrefix(prefix);
292 
293 		boolean useStructuredCacheEntries = PropertiesHelper.getBoolean(Environment.USE_STRUCTURED_CACHE, properties, false);
294 		log.info( "Structured second-level cache entries: " + enabledDisabled(useStructuredCacheEntries) );
295 		settings.setStructuredCacheEntriesEnabled(useStructuredCacheEntries);
296 
297 		if (useQueryCache) settings.setQueryCacheFactory( createQueryCacheFactory(properties) );
298 
299 		//SQL Exception converter:
300 
301 		SQLExceptionConverter sqlExceptionConverter;
302 		try {
303 			sqlExceptionConverter = SQLExceptionConverterFactory.buildSQLExceptionConverter( dialect, properties );
304 		}
305 		catch(HibernateException e) {
306 			log.warn("Error building SQLExceptionConverter; using minimal converter");
307 			sqlExceptionConverter = SQLExceptionConverterFactory.buildMinimalSQLExceptionConverter();
308 		}
309 		settings.setSQLExceptionConverter(sqlExceptionConverter);
310 
311 		//Statistics and logging:
312 
313 		boolean showSql = PropertiesHelper.getBoolean(Environment.SHOW_SQL, properties);
314 		if (showSql) log.info("Echoing all SQL to stdout");
315 //		settings.setShowSqlEnabled(showSql);
316 
317 		boolean formatSql = PropertiesHelper.getBoolean(Environment.FORMAT_SQL, properties);
318 //		settings.setFormatSqlEnabled(formatSql);
319 
320 		settings.setSqlStatementLogger( new SQLStatementLogger( showSql, formatSql ) );
321 
322 		boolean useStatistics = PropertiesHelper.getBoolean(Environment.GENERATE_STATISTICS, properties);
323 		log.info( "Statistics: " + enabledDisabled(useStatistics) );
324 		settings.setStatisticsEnabled(useStatistics);
325 
326 		boolean useIdentifierRollback = PropertiesHelper.getBoolean(Environment.USE_IDENTIFIER_ROLLBACK, properties);
327 		log.info( "Deleted entity synthetic identifier rollback: " + enabledDisabled(useIdentifierRollback) );
328 		settings.setIdentifierRollbackEnabled(useIdentifierRollback);
329 
330 		//Schema export:
331 
332 		String autoSchemaExport = properties.getProperty(Environment.HBM2DDL_AUTO);
333 		if ( "validate".equals(autoSchemaExport) ) settings.setAutoValidateSchema(true);
334 		if ( "update".equals(autoSchemaExport) ) settings.setAutoUpdateSchema(true);
335 		if ( "create".equals(autoSchemaExport) ) settings.setAutoCreateSchema(true);
336 		if ( "create-drop".equals(autoSchemaExport) ) {
337 			settings.setAutoCreateSchema(true);
338 			settings.setAutoDropSchema(true);
339 		}
340 		settings.setImportFiles( properties.getProperty( Environment.HBM2DDL_IMPORT_FILES ) );
341 
342 		EntityMode defaultEntityMode = EntityMode.parse( properties.getProperty( Environment.DEFAULT_ENTITY_MODE ) );
343 		log.info( "Default entity-mode: " + defaultEntityMode );
344 		settings.setDefaultEntityMode( defaultEntityMode );
345 
346 		boolean namedQueryChecking = PropertiesHelper.getBoolean( Environment.QUERY_STARTUP_CHECKING, properties, true );
347 		log.info( "Named query checking : " + enabledDisabled( namedQueryChecking ) );
348 		settings.setNamedQueryStartupCheckingEnabled( namedQueryChecking );
349 
350 		boolean checkNullability = PropertiesHelper.getBoolean(Environment.CHECK_NULLABILITY, properties, true);
351 		log.info( "Check Nullability in Core (should be disabled when Bean Validation is on): " + enabledDisabled(checkNullability) );
352 		settings.setCheckNullability(checkNullability);
353 
354 
355 //		String provider = properties.getProperty( Environment.BYTECODE_PROVIDER );
356 //		log.info( "Bytecode provider name : " + provider );
357 //		BytecodeProvider bytecodeProvider = buildBytecodeProvider( provider );
358 //		settings.setBytecodeProvider( bytecodeProvider );
359 
360 		return settings;
361 
362 	}

364 	protected BytecodeProvider buildBytecodeProvider(String providerName) {
365 		if ( "javassist".equals( providerName ) ) {
366 			return new org.hibernate.bytecode.javassist.BytecodeProviderImpl();
367 		}
368 		else if ( "cglib".equals( providerName ) ) {
369 			return new org.hibernate.bytecode.cglib.BytecodeProviderImpl();
370 		}
371 		else {
372 			log.debug( "using javassist as bytecode provider by default" );
373 			return new org.hibernate.bytecode.javassist.BytecodeProviderImpl();
374 		}
375 	}

381 	protected QueryCacheFactory createQueryCacheFactory(Properties properties) {
382 		String queryCacheFactoryClassName = PropertiesHelper.getString(
383 				Environment.QUERY_CACHE_FACTORY, properties, "org.hibernate.cache.StandardQueryCacheFactory"
384 		);
385 		log.info("Query cache factory: " + queryCacheFactoryClassName);
386 		try {
387 			return (QueryCacheFactory) ReflectHelper.classForName(queryCacheFactoryClassName).newInstance();
388 		}
389 		catch (Exception cnfe) {
390 			throw new HibernateException("could not instantiate QueryCacheFactory: " + queryCacheFactoryClassName, cnfe);
391 		}
392 	}

394 	public static RegionFactory createRegionFactory(Properties properties, boolean cachingEnabled) {
395 		String regionFactoryClassName = PropertiesHelper.getString( Environment.CACHE_REGION_FACTORY, properties, null );
396 		if ( regionFactoryClassName == null && cachingEnabled ) {
397 			String providerClassName = PropertiesHelper.getString( Environment.CACHE_PROVIDER, properties, null );
398 			if ( providerClassName != null ) {
399 				// legacy behavior, apply the bridge...
400 				regionFactoryClassName = RegionFactoryCacheProviderBridge.class.getName();
401 			}
402 		}
403 		if ( regionFactoryClassName == null ) {
404 			regionFactoryClassName = DEF_CACHE_REG_FACTORY;
405 		}
406 		log.info( "Cache region factory : " + regionFactoryClassName );
407 		try {
408 			try {
409 				return (RegionFactory) ReflectHelper.classForName( regionFactoryClassName )
410 						.getConstructor( Properties.class )
411 						.newInstance( properties );
412 			}
413 			catch ( NoSuchMethodException nsme ) {
414 				// no constructor accepting Properties found, try no arg constructor
415 				log.debug(
416 						regionFactoryClassName + " did not provide constructor accepting java.util.Properties; " +
417 								"attempting no-arg constructor."
418 				);
419 				return (RegionFactory) ReflectHelper.classForName( regionFactoryClassName ).newInstance();
420 			}
421 		}
422 		catch ( Exception e ) {
423 			throw new HibernateException( "could not instantiate RegionFactory [" + regionFactoryClassName + "]", e );
424 		}
425 	}

440 	protected BatcherFactory createBatcherFactory(Properties properties, int batchSize) {
441 		String batcherClass = properties.getProperty(Environment.BATCH_STRATEGY);
442 		if (batcherClass==null) {
443 			return batchSize == 0
444 					? new NonBatchingBatcherFactory()
445 					: new BatchingBatcherFactory();
446 		}
447 		else {
448 			log.info("Batcher factory: " + batcherClass);
449 			try {
450 				return (BatcherFactory) ReflectHelper.classForName(batcherClass).newInstance();
451 			}
452 			catch (Exception cnfe) {
453 				throw new HibernateException("could not instantiate BatcherFactory: " + batcherClass, cnfe);
454 			}
455 		}
456 	}
