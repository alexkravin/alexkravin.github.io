76  	/**
77  	 * Handle the given load event.
78  	 *
79  	 * @param event The load event to be handled.
80  	 * @throws HibernateException
81  	 */
82  	public void onLoad(LoadEvent event, LoadEventListener.LoadType loadType) throws HibernateException {
83  
84  		final SessionImplementor source = event.getSession();
85  
86  		EntityPersister persister;
87  		if ( event.getInstanceToLoad() != null ) {
88  			persister = source.getEntityPersister( null, event.getInstanceToLoad() ); //the load() which takes an entity does not pass an entityName
89  			event.setEntityClassName( event.getInstanceToLoad().getClass().getName() );
90  		}
91  		else {
92  			persister = source.getFactory().getEntityPersister( event.getEntityClassName() );
93  		}
94  
95  		if ( persister == null ) {
96  			throw new HibernateException(
97  					"Unable to locate persister: " +
98  					event.getEntityClassName()
99  				);
100 		}
101 
102 		final Class idClass = persister.getIdentifierType().getReturnedClass();
103 		if ( persister.getIdentifierType().isComponentType() && EntityMode.DOM4J == event.getSession().getEntityMode() ) {
104 			// skip this check for composite-ids relating to dom4j entity-mode;
105 			// alternatively, we could add a check to make sure the incoming id value is
106 			// an instance of Element...
107 		}
108 		else {
109 			if ( idClass != null && ! idClass.isInstance( event.getEntityId() ) ) {
110 				// we may have the kooky jpa requirement of allowing find-by-id where
111 				// "id" is the "simple pk value" of a dependent objects parent.  This
112 				// is part of its generally goofy "derived identity" "feature"
113 				if ( persister.getEntityMetamodel().getIdentifierProperty().isEmbedded() ) {
114 					final EmbeddedComponentType dependentIdType =
115 							(EmbeddedComponentType) persister.getEntityMetamodel().getIdentifierProperty().getType();
116 					if ( dependentIdType.getSubtypes().length == 1 ) {
117 						final Type singleSubType = dependentIdType.getSubtypes()[0];
118 						if ( singleSubType.isEntityType() ) {
119 							final EntityType dependentParentType = (EntityType) singleSubType;
120 							final Type dependentParentIdType = dependentParentType.getIdentifierOrUniqueKeyType( source.getFactory() );
121 							if ( dependentParentIdType.getReturnedClass().isInstance( event.getEntityId() ) ) {
122 								// yep that's what we have...
123 								loadByDerivedIdentitySimplePkValue(
124 										event,
125 										loadType,
126 										persister,
127 										dependentIdType,
128 										source.getFactory().getEntityPersister( dependentParentType.getAssociatedEntityName() )
129 								);
130 								return;
131 							}
132 						}
133 					}
134 				}
135 				throw new TypeMismatchException(
136 						"Provided id of the wrong type for class " + persister.getEntityName() + ". Expected: " + idClass + ", got " + event.getEntityId().getClass()
137 				);
138 			}
139 		}
140 
141 		EntityKey keyToLoad = new EntityKey( event.getEntityId(), persister, source.getEntityMode()  );
142 
143 		try {
144 			if ( loadType.isNakedEntityReturned() ) {
145 				//do not return a proxy!
146 				//(this option indicates we are initializing a proxy)
147 				event.setResult( load(event, persister, keyToLoad, loadType) );
148 			}
149 			else {
150 				//return a proxy if appropriate
151 				if ( event.getLockMode() == LockMode.NONE ) {
152 					event.setResult( proxyOrLoad(event, persister, keyToLoad, loadType) );
153 				}
154 				else {
155 					event.setResult( lockAndLoad(event, persister, keyToLoad, loadType, source) );
156 				}
157 			}
158 		}
159 		catch(HibernateException e) {
160 			log.info("Error performing load command", e);
161 			throw e;
162 		}
163 	}

165 	private void loadByDerivedIdentitySimplePkValue(
166 			LoadEvent event,
167 			LoadEventListener.LoadType options,
168 			EntityPersister dependentPersister,
169 			EmbeddedComponentType dependentIdType,
170 			EntityPersister parentPersister) {
171 		final EntityKey parentEntityKey = new EntityKey(
172 				event.getEntityId(),
173 				parentPersister,
174 				event.getSession().getEntityMode()
175 		);
176 		final Object parent = doLoad(
177 				event,
178 				parentPersister,
179 				parentEntityKey,
180 				options
181 		);
182 
183 		Serializable dependent = (Serializable) dependentIdType.instantiate( parent, event.getSession() );
184 		dependentIdType.setPropertyValues( dependent, new Object[] {parent}, event.getSession().getEntityMode() );
185 		final EntityKey dependentEntityKey = new EntityKey(
186 				dependent,
187 				dependentPersister,
188 				event.getSession().getEntityMode()
189 		);
190 		event.setEntityId( dependent );
191 		dependent = (Serializable) doLoad(
192 				event,
193 				dependentPersister,
194 				dependentEntityKey,
195 				options
196 		);
197 
198 		event.setResult( dependent );
199 	}

201 	/**
202 	 * Perfoms the load of an entity.
203 	 *
204 	 * @param event The initiating load request event
205 	 * @param persister The persister corresponding to the entity to be loaded
206 	 * @param keyToLoad The key of the entity to be loaded
207 	 * @param options The defined load options
208 	 * @return The loaded entity.
209 	 * @throws HibernateException
210 	 */
211 	protected Object load(
212 		final LoadEvent event,
213 		final EntityPersister persister,
214 		final EntityKey keyToLoad,
215 		final LoadEventListener.LoadType options) {
216 
217 		if ( event.getInstanceToLoad() != null ) {
218 			if ( event.getSession().getPersistenceContext().getEntry( event.getInstanceToLoad() ) != null ) {
219 				throw new PersistentObjectException(
220 						"attempted to load into an instance that was already associated with the session: " +
221 						MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() )
222 					);
223 			}
224 			persister.setIdentifier( event.getInstanceToLoad(), event.getEntityId(), event.getSession() );
225 		}
226 
227 		Object entity = doLoad(event, persister, keyToLoad, options);
228 
229 		boolean isOptionalInstance = event.getInstanceToLoad() != null;
230 
231 		if ( !options.isAllowNulls() || isOptionalInstance ) {
232 			if ( entity == null ) {
233 				event.getSession().getFactory().getEntityNotFoundDelegate().handleEntityNotFound( event.getEntityClassName(), event.getEntityId() );
234 			}
235 		}
236 
237 		if ( isOptionalInstance && entity != event.getInstanceToLoad() ) {
238 			throw new NonUniqueObjectException( event.getEntityId(), event.getEntityClassName() );
239 		}
240 
241 		return entity;
242 	}

244 	/**
245 	 * Based on configured options, will either return a pre-existing proxy,
246 	 * generate a new proxy, or perform an actual load.
247 	 *
248 	 * @param event The initiating load request event
249 	 * @param persister The persister corresponding to the entity to be loaded
250 	 * @param keyToLoad The key of the entity to be loaded
251 	 * @param options The defined load options
252 	 * @return The result of the proxy/load operation.
253 	 */
254 	protected Object proxyOrLoad(
255 		final LoadEvent event,
256 		final EntityPersister persister,
257 		final EntityKey keyToLoad,
258 		final LoadEventListener.LoadType options) {
259 
260 		if ( log.isTraceEnabled() ) {
261 			log.trace(
262 					"loading entity: " +
263 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() )
264 				);
265 		}
266 
267 		if ( !persister.hasProxy() ) {
268 			// this class has no proxies (so do a shortcut)
269 			return load(event, persister, keyToLoad, options);
270 		}
271 		else {
272 			final PersistenceContext persistenceContext = event.getSession().getPersistenceContext();
273 
274 			// look for a proxy
275 			Object proxy = persistenceContext.getProxy(keyToLoad);
276 			if ( proxy != null ) {
277 				return returnNarrowedProxy( event, persister, keyToLoad, options, persistenceContext, proxy );
278 			}
279 			else {
280 				if ( options.isAllowProxyCreation() ) {
281 					return createProxyIfNecessary( event, persister, keyToLoad, options, persistenceContext );
282 				}
283 				else {
284 					// return a newly loaded object
285 					return load(event, persister, keyToLoad, options);
286 				}
287 			}
288 
289 		}
290 	}

292 	/**
293 	 * Given a proxy, initialize it and/or narrow it provided either
294 	 * is necessary.
295 	 *
296 	 * @param event The initiating load request event
297 	 * @param persister The persister corresponding to the entity to be loaded
298 	 * @param keyToLoad The key of the entity to be loaded
299 	 * @param options The defined load options
300 	 * @param persistenceContext The originating session
301 	 * @param proxy The proxy to narrow
302 	 * @return The created/existing proxy
303 	 */
304 	private Object returnNarrowedProxy(
305 			final LoadEvent event,
306 			final EntityPersister persister,
307 			final EntityKey keyToLoad,
308 			final LoadEventListener.LoadType options,
309 			final PersistenceContext persistenceContext,
310 			final Object proxy) {
311 		log.trace("entity proxy found in session cache");
312 		LazyInitializer li = ( (HibernateProxy) proxy ).getHibernateLazyInitializer();
313 		if ( li.isUnwrap() ) {
314 			return li.getImplementation();
315 		}
316 		Object impl = null;
317 		if ( !options.isAllowProxyCreation() ) {
318 			impl = load( event, persister, keyToLoad, options );
319 			if ( impl == null ) {
320 				event.getSession().getFactory().getEntityNotFoundDelegate().handleEntityNotFound( persister.getEntityName(), keyToLoad.getIdentifier());
321 			}
322 		}
323 		return persistenceContext.narrowProxy( proxy, persister, keyToLoad, impl );
324 	}

326 	/**
327 	 * If there is already a corresponding proxy associated with the
328 	 * persistence context, return it; otherwise create a proxy, associate it
329 	 * with the persistence context, and return the just-created proxy.
330 	 *
331 	 * @param event The initiating load request event
332 	 * @param persister The persister corresponding to the entity to be loaded
333 	 * @param keyToLoad The key of the entity to be loaded
334 	 * @param options The defined load options
335 	 * @param persistenceContext The originating session
336 	 * @return The created/existing proxy
337 	 */
338 	private Object createProxyIfNecessary(
339 			final LoadEvent event,
340 			final EntityPersister persister,
341 			final EntityKey keyToLoad,
342 			final LoadEventListener.LoadType options,
343 			final PersistenceContext persistenceContext) {
344 		Object existing = persistenceContext.getEntity( keyToLoad );
345 		if ( existing != null ) {
346 			// return existing object or initialized proxy (unless deleted)
347 			log.trace( "entity found in session cache" );
348 			if ( options.isCheckDeleted() ) {
349 				EntityEntry entry = persistenceContext.getEntry( existing );
350 				Status status = entry.getStatus();
351 				if ( status == Status.DELETED || status == Status.GONE ) {
352 					return null;
353 				}
354 			}
355 			return existing;
356 		}
357 		else {
358 			log.trace( "creating new proxy for entity" );
359 			// return new uninitialized proxy
360 			Object proxy = persister.createProxy( event.getEntityId(), event.getSession() );
361 			persistenceContext.getBatchFetchQueue().addBatchLoadableEntityKey(keyToLoad);
362 			persistenceContext.addProxy(keyToLoad, proxy);
363 			return proxy;
364 		}
365 	}

367 	/**
368 	 * If the class to be loaded has been configured with a cache, then lock
369 	 * given id in that cache and then perform the load.
370 	 *
371 	 * @param event The initiating load request event
372 	 * @param persister The persister corresponding to the entity to be loaded
373 	 * @param keyToLoad The key of the entity to be loaded
374 	 * @param options The defined load options
375 	 * @param source The originating session
376 	 * @return The loaded entity
377 	 * @throws HibernateException
378 	 */
379 	protected Object lockAndLoad(
380 			final LoadEvent event,
381 			final EntityPersister persister,
382 			final EntityKey keyToLoad,
383 			final LoadEventListener.LoadType options,
384 			final SessionImplementor source) {
385 		SoftLock lock = null;
386 		final CacheKey ck;
387 		if ( persister.hasCache() ) {
388 			ck = new CacheKey(
389 					event.getEntityId(),
390 					persister.getIdentifierType(),
391 					persister.getRootEntityName(),
392 					source.getEntityMode(),
393 					source.getFactory()
394 			);
395 			lock = persister.getCacheAccessStrategy().lockItem( ck, null );
396 		}
397 		else {
398 			ck = null;
399 		}
400 
401 		Object entity;
402 		try {
403 			entity = load(event, persister, keyToLoad, options);
404 		}
405 		finally {
406 			if ( persister.hasCache() ) {
407 				persister.getCacheAccessStrategy().unlockItem( ck, lock );
408 			}
409 		}
410 
411 		return event.getSession().getPersistenceContext().proxyFor( persister, keyToLoad, entity );
412 	}

415 	/**
416 	 * Coordinates the efforts to load a given entity.  First, an attempt is
417 	 * made to load the entity from the session-level cache.  If not found there,
418 	 * an attempt is made to locate it in second-level cache.  Lastly, an
419 	 * attempt is made to load it directly from the datasource.
420 	 *
421 	 * @param event The load event
422 	 * @param persister The persister for the entity being requested for load
423 	 * @param keyToLoad The EntityKey representing the entity to be loaded.
424 	 * @param options The load options.
425 	 * @return The loaded entity, or null.
426 	 */
427 	protected Object doLoad(
428 			final LoadEvent event,
429 			final EntityPersister persister,
430 			final EntityKey keyToLoad,
431 			final LoadEventListener.LoadType options) {
432 
433 		if ( log.isTraceEnabled() ) {
434 			log.trace(
435 					"attempting to resolve: " +
436 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() )
437 				);
438 		}
439 
440 		Object entity = loadFromSessionCache( event, keyToLoad, options );
441 		if ( entity == REMOVED_ENTITY_MARKER ) {
442 			log.debug( "load request found matching entity in context, but it is scheduled for removal; returning null" );
443 			return null;
444 		}
445 		if ( entity == INCONSISTENT_RTN_CLASS_MARKER ) {
446 			log.debug( "load request found matching entity in context, but the matched entity was of an inconsistent return type; returning null" );
447 			return null;
448 		}
449 		if ( entity != null ) {
450 			if ( log.isTraceEnabled() ) {
451 				log.trace(
452 						"resolved object in session cache: " +
453 						MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory()  )
454 					);
455 			}
456 			return entity;
457 		}
458 
459 		entity = loadFromSecondLevelCache(event, persister, options);
460 		if ( entity != null ) {
461 			if ( log.isTraceEnabled() ) {
462 				log.trace(
463 						"resolved object in second-level cache: " +
464 						MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() )
465 					);
466 			}
467 			return entity;
468 		}
469 
470 		if ( log.isTraceEnabled() ) {
471 			log.trace(
472 					"object not resolved in any cache: " +
473 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() )
474 				);
475 		}
476 
477 		return loadFromDatasource(event, persister, keyToLoad, options);
478 	}

480 	/**
481 	 * Performs the process of loading an entity from the configured
482 	 * underlying datasource.
483 	 *
484 	 * @param event The load event
485 	 * @param persister The persister for the entity being requested for load
486 	 * @param keyToLoad The EntityKey representing the entity to be loaded.
487 	 * @param options The load options.
488 	 * @return The object loaded from the datasource, or null if not found.
489 	 */
490 	protected Object loadFromDatasource(
491 			final LoadEvent event,
492 			final EntityPersister persister,
493 			final EntityKey keyToLoad,
494 			final LoadEventListener.LoadType options) {
495 		final SessionImplementor source = event.getSession();
496 		Object entity = persister.load(
497 				event.getEntityId(),
498 				event.getInstanceToLoad(),
499 				event.getLockOptions(),
500 				source
501 		);
502 
503 		if ( event.isAssociationFetch() && source.getFactory().getStatistics().isStatisticsEnabled() ) {
504 			source.getFactory().getStatisticsImplementor().fetchEntity( event.getEntityClassName() );
505 		}
506 
507 		return entity;
508 	}

510 	/**
511 	 * Attempts to locate the entity in the session-level cache.
512 	 * <p/>
513 	 * If allowed to return nulls, then if the entity happens to be found in
514 	 * the session cache, we check the entity type for proper handling
515 	 * of entity hierarchies.
516 	 * <p/>
517 	 * If checkDeleted was set to true, then if the entity is found in the
518 	 * session-level cache, it's current status within the session cache
519 	 * is checked to see if it has previously been scheduled for deletion.
520 	 *
521 	 * @param event The load event
522 	 * @param keyToLoad The EntityKey representing the entity to be loaded.
523 	 * @param options The load options.
524 	 * @return The entity from the session-level cache, or null.
525 	 * @throws HibernateException Generally indicates problems applying a lock-mode.
526 	 */
527 	protected Object loadFromSessionCache(
528 			final LoadEvent event,
529 			final EntityKey keyToLoad,
530 			final LoadEventListener.LoadType options) throws HibernateException {
531 
532 		SessionImplementor session = event.getSession();
533 		Object old = session.getEntityUsingInterceptor( keyToLoad );
534 
535 		if ( old != null ) {
536 			// this object was already loaded
537 			EntityEntry oldEntry = session.getPersistenceContext().getEntry( old );
538 			if ( options.isCheckDeleted() ) {
539 				Status status = oldEntry.getStatus();
540 				if ( status == Status.DELETED || status == Status.GONE ) {
541 					return REMOVED_ENTITY_MARKER;
542 				}
543 			}
544 			if ( options.isAllowNulls() ) {
545 //				EntityPersister persister = event.getSession().getFactory().getEntityPersister( event.getEntityClassName() );
546 				EntityPersister persister = event.getSession().getFactory().getEntityPersister( keyToLoad.getEntityName() );
547 				if ( ! persister.isInstance( old, event.getSession().getEntityMode() ) ) {
548 					return INCONSISTENT_RTN_CLASS_MARKER;
549 				}
550 			}
551 			upgradeLock( old, oldEntry, event.getLockOptions(), event.getSession() );
552 		}
553 
554 		return old;
555 	}

557 	/**
558 	 * Attempts to load the entity from the second-level cache.
559 	 *
560 	 * @param event The load event
561 	 * @param persister The persister for the entity being requested for load
562 	 * @param options The load options.
563 	 * @return The entity from the second-level cache, or null.
564 	 */
565 	protected Object loadFromSecondLevelCache(
566 			final LoadEvent event,
567 			final EntityPersister persister,
568 			final LoadEventListener.LoadType options) {
569 
570 		final SessionImplementor source = event.getSession();
571 
572 		final boolean useCache = persister.hasCache()
573 				&& source.getCacheMode().isGetEnabled()
574 				&& event.getLockMode().lessThan(LockMode.READ);
575 
576 		if ( useCache ) {
577 
578 			final SessionFactoryImplementor factory = source.getFactory();
579 
580 			final CacheKey ck = new CacheKey(
581 					event.getEntityId(),
582 					persister.getIdentifierType(),
583 					persister.getRootEntityName(),
584 					source.getEntityMode(),
585 					source.getFactory()
586 			);
587 			Object ce = persister.getCacheAccessStrategy().get( ck, source.getTimestamp() );
588 			if ( factory.getStatistics().isStatisticsEnabled() ) {
589 				if ( ce == null ) {
590 					factory.getStatisticsImplementor().secondLevelCacheMiss(
591 							persister.getCacheAccessStrategy().getRegion().getName()
592 					);
593 				}
594 				else {
595 					factory.getStatisticsImplementor().secondLevelCacheHit(
596 							persister.getCacheAccessStrategy().getRegion().getName()
597 					);
598 				}
599 			}
600 
601 			if ( ce != null ) {
602 				CacheEntry entry = (CacheEntry) persister.getCacheEntryStructure().destructure( ce, factory );
603 
604 				// Entity was found in second-level cache...
605 				return assembleCacheEntry(
606 						entry,
607 						event.getEntityId(),
608 						persister,
609 						event
610 				);
611 			}
612 		}
613 
614 		return null;
615 	}

617 	private Object assembleCacheEntry(
618 			final CacheEntry entry,
619 			final Serializable id,
620 			final EntityPersister persister,
621 			final LoadEvent event) throws HibernateException {
622 
623 		final Object optionalObject = event.getInstanceToLoad();
624 		final EventSource session = event.getSession();
625 		final SessionFactoryImplementor factory = session.getFactory();
626 
627 		if ( log.isTraceEnabled() ) {
628 			log.trace(
629 					"assembling entity from second-level cache: " +
630 					MessageHelper.infoString( persister, id, factory )
631 				);
632 		}
633 
634 		EntityPersister subclassPersister = factory.getEntityPersister( entry.getSubclass() );
635 		Object result = optionalObject == null ?
636 				session.instantiate( subclassPersister, id ) : optionalObject;
637 
638 		// make it circular-reference safe
639 		EntityKey entityKey = new EntityKey( id, subclassPersister, session.getEntityMode() );
640 		TwoPhaseLoad.addUninitializedCachedEntity(
641 				entityKey,
642 				result,
643 				subclassPersister,
644 				LockMode.NONE,
645 				entry.areLazyPropertiesUnfetched(),
646 				entry.getVersion(),
647 				session
648 			);
649 
650 		Type[] types = subclassPersister.getPropertyTypes();
651 		Object[] values = entry.assemble( result, id, subclassPersister, session.getInterceptor(), session ); // intializes result by side-effect
652 		TypeHelper.deepCopy(
653 				values,
654 				types,
655 				subclassPersister.getPropertyUpdateability(),
656 				values,
657 				session
658 		);
659 
660 		Object version = Versioning.getVersion( values, subclassPersister );
661 		if ( log.isTraceEnabled() ) log.trace( "Cached Version: " + version );
662 
663 		final PersistenceContext persistenceContext = session.getPersistenceContext();
664 		boolean isReadOnly = session.isDefaultReadOnly();
665 		if ( persister.isMutable() ) {
666 			Object proxy = persistenceContext.getProxy( entityKey );
667 			if ( proxy != null ) {
668 				// there is already a proxy for this impl
669 				// only set the status to read-only if the proxy is read-only
670 				isReadOnly = ( ( HibernateProxy ) proxy ).getHibernateLazyInitializer().isReadOnly();
671 			}
672 		}
673 		else {
674 			isReadOnly = true;
675 		}
676 		persistenceContext.addEntry(
677 				result,
678 				( isReadOnly ? Status.READ_ONLY : Status.MANAGED ),
679 				values,
680 				null,
681 				id,
682 				version,
683 				LockMode.NONE,
684 				true,
685 				subclassPersister,
686 				false,
687 				entry.areLazyPropertiesUnfetched()
688 			);
689 		subclassPersister.afterInitialize( result, entry.areLazyPropertiesUnfetched(), session );
690 		persistenceContext.initializeNonLazyCollections();
691 		// upgrade the lock if necessary:
692 		//lock(result, lockMode);
693 
694 		//PostLoad is needed for EJB3
695 		//TODO: reuse the PostLoadEvent...
696 		PostLoadEvent postLoadEvent = new PostLoadEvent(session).setEntity(result)
697 				.setId(id).setPersister(persister);
698 		PostLoadEventListener[] listeners = session.getListeners().getPostLoadEventListeners();
699 		for ( int i = 0; i < listeners.length; i++ ) {
700 			listeners[i].onPostLoad(postLoadEvent);
701 		}
702 
703 		return result;
704 	}
