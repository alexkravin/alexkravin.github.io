62  	/**
63  	 * make sure user didn't mangle the id
64  	 */
65  	public void checkId(
66  			Object object,
67  			EntityPersister persister,
68  			Serializable id,
69  			EntityMode entityMode,
70  			SessionImplementor session) throws HibernateException {
71  
72  		if ( id != null && id instanceof DelayedPostInsertIdentifier ) {
73  			// this is a situation where the entity id is assigned by a post-insert generator
74  			// and was saved outside the transaction forcing it to be delayed
75  			return;
76  		}
77  
78  		if ( persister.canExtractIdOutOfEntity() ) {
79  
80  			Serializable oid = persister.getIdentifier( object, session );
81  			if (id==null) {
82  				throw new AssertionFailure("null id in " + persister.getEntityName() + " entry (don't flush the Session after an exception occurs)");
83  			}
84  			if ( !persister.getIdentifierType().isEqual( id, oid, entityMode, session.getFactory() ) ) {
85  				throw new HibernateException(
86  						"identifier of an instance of " +
87  						persister.getEntityName() +
88  						" was altered from " + id +
89  						" to " + oid
90  					);
91  			}
92  		}
93  
94  	}

96  	private void checkNaturalId(
97  			EntityPersister persister,
98  	        EntityEntry entry,
99  	        Object[] current,
100 	        Object[] loaded,
101 	        EntityMode entityMode,
102 	        SessionImplementor session) {
103 		if ( persister.hasNaturalIdentifier() && entry.getStatus() != Status.READ_ONLY ) {
104  			Object[] snapshot = null;			
105 			Type[] types = persister.getPropertyTypes();
106 			int[] props = persister.getNaturalIdentifierProperties();
107 			boolean[] updateable = persister.getPropertyUpdateability();
108 			for ( int i=0; i<props.length; i++ ) {
109 				int prop = props[i];
110 				if ( !updateable[prop] ) {
111  					Object loadedVal;
112  					if ( loaded == null ) {
113  						if ( snapshot == null) {
114  							snapshot = session.getPersistenceContext().getNaturalIdSnapshot( entry.getId(), persister );
115  						}
116  						loadedVal = snapshot[i];
117  					} else {
118  						loadedVal = loaded[prop];
119  					}
120  					if ( !types[prop].isEqual( current[prop], loadedVal, entityMode ) ) {						
121 						throw new HibernateException(
122 								"immutable natural identifier of an instance of " +
123 								persister.getEntityName() +
124 								" was altered"
125 							);
126 					}
127 				}
128 			}
129 		}
130 	}

132 	/**
133 	 * Flushes a single entity's state to the database, by scheduling
134 	 * an update action, if necessary
135 	 */
136 	public void onFlushEntity(FlushEntityEvent event) throws HibernateException {
137 		final Object entity = event.getEntity();
138 		final EntityEntry entry = event.getEntityEntry();
139 		final EventSource session = event.getSession();
140 		final EntityPersister persister = entry.getPersister();
141 		final Status status = entry.getStatus();
142 		final EntityMode entityMode = session.getEntityMode();
143 		final Type[] types = persister.getPropertyTypes();
144 
145 		final boolean mightBeDirty = entry.requiresDirtyCheck(entity);
146 
147 		final Object[] values = getValues( entity, entry, entityMode, mightBeDirty, session );
148 
149 		event.setPropertyValues(values);
150 
151 		//TODO: avoid this for non-new instances where mightBeDirty==false
152 		boolean substitute = wrapCollections( session, persister, types, values);
153 
154 		if ( isUpdateNecessary( event, mightBeDirty ) ) {
155 			substitute = scheduleUpdate( event ) || substitute;
156 		}
157 
158 		if ( status != Status.DELETED ) {
159 			// now update the object .. has to be outside the main if block above (because of collections)
160 			if (substitute) persister.setPropertyValues( entity, values, entityMode );
161 
162 			// Search for collections by reachability, updating their role.
163 			// We don't want to touch collections reachable from a deleted object
164 			if ( persister.hasCollections() ) {
165 				new FlushVisitor(session, entity).processEntityPropertyValues(values, types);
166 			}
167 		}
168 
169 	}

171 	private Object[] getValues(
172 			Object entity,
173 			EntityEntry entry,
174 			EntityMode entityMode,
175 			boolean mightBeDirty,
176 	        SessionImplementor session) {
177 		final Object[] loadedState = entry.getLoadedState();
178 		final Status status = entry.getStatus();
179 		final EntityPersister persister = entry.getPersister();
180 
181 		final Object[] values;
182 		if ( status == Status.DELETED ) {
183 			//grab its state saved at deletion
184 			values = entry.getDeletedState();
185 		}
186 		else if ( !mightBeDirty && loadedState!=null ) {
187 			values = loadedState;
188 		}
189 		else {
190 			checkId( entity, persister, entry.getId(), entityMode, session );
191 
192 			// grab its current state
193 			values = persister.getPropertyValues( entity, entityMode );
194 
195 			checkNaturalId( persister, entry, values, loadedState, entityMode, session );
196 		}
197 		return values;
198 	}

200 	private boolean wrapCollections(
201 			EventSource session,
202 			EntityPersister persister,
203 			Type[] types,
204 			Object[] values
205 	) {
206 		if ( persister.hasCollections() ) {
207 
208 			// wrap up any new collections directly referenced by the object
209 			// or its components
210 
211 			// NOTE: we need to do the wrap here even if its not "dirty",
212 			// because collections need wrapping but changes to _them_
213 			// don't dirty the container. Also, for versioned data, we
214 			// need to wrap before calling searchForDirtyCollections
215 
216 			WrapVisitor visitor = new WrapVisitor(session);
217 			// substitutes into values by side-effect
218 			visitor.processEntityPropertyValues(values, types);
219 			return visitor.isSubstitutionRequired();
220 		}
221 		else {
222 			return false;
223 		}
224 	}

226 	private boolean isUpdateNecessary(final FlushEntityEvent event, final boolean mightBeDirty) {
227 		final Status status = event.getEntityEntry().getStatus();
228 		if ( mightBeDirty || status==Status.DELETED ) {
229 			// compare to cached state (ignoring collections unless versioned)
230 			dirtyCheck(event);
231 			if ( isUpdateNecessary(event) ) {
232 				return true;
233 			}
234 			else {
235 				FieldInterceptionHelper.clearDirty( event.getEntity() );
236 				return false;
237 			}
238 		}
239 		else {
240 			return hasDirtyCollections( event, event.getEntityEntry().getPersister(), status );
241 		}
242 	}

244 	private boolean scheduleUpdate(final FlushEntityEvent event) {
245 		
246 		final EntityEntry entry = event.getEntityEntry();
247 		final EventSource session = event.getSession();
248 		final Object entity = event.getEntity();
249 		final Status status = entry.getStatus();
250 		final EntityMode entityMode = session.getEntityMode();
251 		final EntityPersister persister = entry.getPersister();
252 		final Object[] values = event.getPropertyValues();
253 		
254 		if ( log.isTraceEnabled() ) {
255 			if ( status == Status.DELETED ) {
256 				if ( ! persister.isMutable() ) {
257 					log.trace(
258 						"Updating immutable, deleted entity: " +
259 						MessageHelper.infoString( persister, entry.getId(), session.getFactory() )
260 					);
261 				}
262 				else if ( ! entry.isModifiableEntity() ) {
263 					log.trace(
264 						"Updating non-modifiable, deleted entity: " +
265 						MessageHelper.infoString( persister, entry.getId(), session.getFactory() )
266 					);
267 				}
268 				else {
269 					log.trace(
270 						"Updating deleted entity: " +
271 						MessageHelper.infoString( persister, entry.getId(), session.getFactory() )
272 					);
273 				}
274 			}
275 			else {
276 				log.trace(
277 						"Updating entity: " +
278 						MessageHelper.infoString( persister, entry.getId(), session.getFactory()  )
279 					);
280 			}
281 		}
282 
283 		final boolean intercepted;
284 		if ( !entry.isBeingReplicated() ) {
285 			// give the Interceptor a chance to process property values, if the properties
286 			// were modified by the Interceptor, we need to set them back to the object
287 			intercepted = handleInterception( event );
288 		}
289 		else {
290 			intercepted = false;
291 		}
292 
293 		validate( entity, persister, status, entityMode );
294 
295 		// increment the version number (if necessary)
296 		final Object nextVersion = getNextVersion(event);
297 
298 		// if it was dirtied by a collection only
299 		int[] dirtyProperties = event.getDirtyProperties();
300 		if ( event.isDirtyCheckPossible() && dirtyProperties == null ) {
301 			if ( ! intercepted && !event.hasDirtyCollection() ) {
302 				throw new AssertionFailure( "dirty, but no dirty properties" );
303 			}
304 			dirtyProperties = ArrayHelper.EMPTY_INT_ARRAY;
305 		}
306 
307 		// check nullability but do not doAfterTransactionCompletion command execute
308 		// we'll use scheduled updates for that.
309 		new Nullability(session).checkNullability( values, persister, true );
310 
311 		// schedule the update
312 		// note that we intentionally do _not_ pass in currentPersistentState!
313 		session.getActionQueue().addAction(
314 				new EntityUpdateAction(
315 						entry.getId(),
316 						values,
317 						dirtyProperties,
318 						event.hasDirtyCollection(),
319 						( status == Status.DELETED && ! entry.isModifiableEntity() ?
320 								persister.getPropertyValues( entity, entityMode ) :
321 								entry.getLoadedState() ),
322 						entry.getVersion(),
323 						nextVersion,
324 						entity,
325 						entry.getRowId(),
326 						persister,
327 						session
328 					)
329 			);
330 		
331 		return intercepted;
332 	}

334 	protected void validate(Object entity, EntityPersister persister, Status status, EntityMode entityMode) {
335 		// validate() instances of Validatable
336 		if ( status == Status.MANAGED && persister.implementsValidatable( entityMode ) ) {
337 			( (Validatable) entity ).validate();
338 		}
339 	}

341 	protected boolean handleInterception(FlushEntityEvent event) {
342 		SessionImplementor session = event.getSession();
343 		EntityEntry entry = event.getEntityEntry();
344 		EntityPersister persister = entry.getPersister();
345 		Object entity = event.getEntity();
346 		
347 		//give the Interceptor a chance to modify property values
348 		final Object[] values = event.getPropertyValues();
349 		final boolean intercepted = invokeInterceptor( session, entity, entry, values, persister );
350 
351 		//now we might need to recalculate the dirtyProperties array
352 		if ( intercepted && event.isDirtyCheckPossible() && !event.isDirtyCheckHandledByInterceptor() ) {
353 			int[] dirtyProperties;
354 			if ( event.hasDatabaseSnapshot() ) {
355 				dirtyProperties = persister.findModified( event.getDatabaseSnapshot(), values, entity, session );
356 			}
357 			else {
358 				dirtyProperties = persister.findDirty( values, entry.getLoadedState(), entity, session );
359 			}
360 			event.setDirtyProperties(dirtyProperties);
361 		}
362 		
363 		return intercepted;
364 	}

366 	protected boolean invokeInterceptor(
367 			SessionImplementor session,
368 			Object entity,
369 			EntityEntry entry,
370 			final Object[] values,
371 			EntityPersister persister) {
372 		return session.getInterceptor().onFlushDirty(
373 				entity,
374 				entry.getId(),
375 				values,
376 				entry.getLoadedState(),
377 				persister.getPropertyNames(),
378 				persister.getPropertyTypes()
379 		);
380 	}

382 	/**
383 	 * Convience method to retreive an entities next version value
384 	 */
385 	private Object getNextVersion(FlushEntityEvent event) throws HibernateException {
386 		
387 		EntityEntry entry = event.getEntityEntry();
388 		EntityPersister persister = entry.getPersister();
389 		if ( persister.isVersioned() ) {
390 
391 			Object[] values = event.getPropertyValues();
392 		    
393 			if ( entry.isBeingReplicated() ) {
394 				return Versioning.getVersion(values, persister);
395 			}
396 			else {
397 				int[] dirtyProperties = event.getDirtyProperties();
398 				
399 				final boolean isVersionIncrementRequired = isVersionIncrementRequired( 
400 						event, 
401 						entry, 
402 						persister, 
403 						dirtyProperties 
404 					);
405 				
406 				final Object nextVersion = isVersionIncrementRequired ?
407 						Versioning.increment( entry.getVersion(), persister.getVersionType(), event.getSession() ) :
408 						entry.getVersion(); //use the current version
409 						
410 				Versioning.setVersion(values, nextVersion, persister);
411 				
412 				return nextVersion;
413 			}
414 		}
415 		else {
416 			return null;
417 		}
418 		
419 	}

421 	private boolean isVersionIncrementRequired(
422 			FlushEntityEvent event, 
423 			EntityEntry entry, 
424 			EntityPersister persister, 
425 			int[] dirtyProperties
426 	) {
427 		final boolean isVersionIncrementRequired = entry.getStatus()!=Status.DELETED && ( 
428 				dirtyProperties==null || 
429 				Versioning.isVersionIncrementRequired( 
430 						dirtyProperties, 
431 						event.hasDirtyCollection(),
432 						persister.getPropertyVersionability()
433 					) 
434 			);
435 		return isVersionIncrementRequired;
436 	}

438 	/**
439 	 * Performs all necessary checking to determine if an entity needs an SQL update
440 	 * to synchronize its state to the database. Modifies the event by side-effect!
441 	 * Note: this method is quite slow, avoid calling if possible!
442 	 */
443 	protected final boolean isUpdateNecessary(FlushEntityEvent event) throws HibernateException {
444 
445 		EntityPersister persister = event.getEntityEntry().getPersister();
446 		Status status = event.getEntityEntry().getStatus();
447 		
448 		if ( !event.isDirtyCheckPossible() ) {
449 			return true;
450 		}
451 		else {
452 			
453 			int[] dirtyProperties = event.getDirtyProperties();
454 			if ( dirtyProperties!=null && dirtyProperties.length!=0 ) {
455 				return true; //TODO: suck into event class
456 			}
457 			else {
458 				return hasDirtyCollections( event, persister, status );
459 			}
460 			
461 		}
462 	}

464 	private boolean hasDirtyCollections(FlushEntityEvent event, EntityPersister persister, Status status) {
465 		if ( isCollectionDirtyCheckNecessary(persister, status ) ) {
466 			DirtyCollectionSearchVisitor visitor = new DirtyCollectionSearchVisitor( 
467 					event.getSession(),
468 					persister.getPropertyVersionability()
469 				);
470 			visitor.processEntityPropertyValues( event.getPropertyValues(), persister.getPropertyTypes() );
471 			boolean hasDirtyCollections = visitor.wasDirtyCollectionFound();
472 			event.setHasDirtyCollection(hasDirtyCollections);
473 			return hasDirtyCollections;
474 		}
475 		else {
476 			return false;
477 		}
478 	}

480 	private boolean isCollectionDirtyCheckNecessary(EntityPersister persister, Status status) {
481 		return ( status == Status.MANAGED || status == Status.READ_ONLY ) &&
482 				persister.isVersioned() &&
483 				persister.hasCollections();
484 	}

486 	/**
487 	 * Perform a dirty check, and attach the results to the event
488 	 */
489 	protected void dirtyCheck(FlushEntityEvent event) throws HibernateException {
490 		
491 		final Object entity = event.getEntity();
492 		final Object[] values = event.getPropertyValues();
493 		final SessionImplementor session = event.getSession();
494 		final EntityEntry entry = event.getEntityEntry();
495 		final EntityPersister persister = entry.getPersister();
496 		final Serializable id = entry.getId();
497 		final Object[] loadedState = entry.getLoadedState();
498 
499 		int[] dirtyProperties = session.getInterceptor().findDirty( 
500 				entity, 
501 				id, 
502 				values, 
503 				loadedState, 
504 				persister.getPropertyNames(), 
505 				persister.getPropertyTypes() 
506 			);
507 		
508 		event.setDatabaseSnapshot(null);
509 
510 		final boolean interceptorHandledDirtyCheck;
511 		boolean cannotDirtyCheck;
512 		
513 		if ( dirtyProperties==null ) {
514 			// Interceptor returned null, so do the dirtycheck ourself, if possible
515 			interceptorHandledDirtyCheck = false;
516 			
517 			cannotDirtyCheck = loadedState==null; // object loaded by update()
518 			if ( !cannotDirtyCheck ) {
519 				// dirty check against the usual snapshot of the entity
520 				dirtyProperties = persister.findDirty( values, loadedState, entity, session );
521 			}
522 			else if ( entry.getStatus() == Status.DELETED && ! event.getEntityEntry().isModifiableEntity() ) {
523 				// A non-modifiable (e.g., read-only or immutable) entity needs to be have
524 				// references to transient entities set to null before being deleted. No other
525 				// fields should be updated.
526 				if ( values != entry.getDeletedState() ) {
527 					throw new IllegalStateException(
528 							"Entity has status Status.DELETED but values != entry.getDeletedState"
529 					);
530 				}
531 				// Even if loadedState == null, we can dirty-check by comparing currentState and
532 				// entry.getDeletedState() because the only fields to be updated are those that
533 				// refer to transient entities that are being set to null.
534 				// - currentState contains the entity's current property values.
535 				// - entry.getDeletedState() contains the entity's current property values with
536 				//   references to transient entities set to null.
537 				// - dirtyProperties will only contain properties that refer to transient entities
538 				final Object[] currentState =
539 						persister.getPropertyValues( event.getEntity(), event.getSession().getEntityMode() );
540 				dirtyProperties = persister.findDirty( entry.getDeletedState(), currentState, entity, session );
541 				cannotDirtyCheck = false;
542 			}
543 			else {
544 				// dirty check against the database snapshot, if possible/necessary
545 				final Object[] databaseSnapshot = getDatabaseSnapshot(session, persister, id);
546 				if ( databaseSnapshot != null ) {
547 					dirtyProperties = persister.findModified(databaseSnapshot, values, entity, session);
548 					cannotDirtyCheck = false;
549 					event.setDatabaseSnapshot(databaseSnapshot);
550 				}
551 			}
552 		}
553 		else {
554 			// the Interceptor handled the dirty checking
555 			cannotDirtyCheck = false;
556 			interceptorHandledDirtyCheck = true;
557 		}
558 
559 		logDirtyProperties( id, dirtyProperties, persister );
560 
561 		event.setDirtyProperties(dirtyProperties);
562 		event.setDirtyCheckHandledByInterceptor(interceptorHandledDirtyCheck);
563 		event.setDirtyCheckPossible(!cannotDirtyCheck);
564 		
565 	}

567 	private void logDirtyProperties(Serializable id, int[] dirtyProperties, EntityPersister persister) {
568 		if ( log.isTraceEnabled() && dirtyProperties != null && dirtyProperties.length > 0 ) {
569 			final String[] allPropertyNames = persister.getPropertyNames();
570 			final String[] dirtyPropertyNames = new String[ dirtyProperties.length ];
571 			for ( int i = 0; i < dirtyProperties.length; i++ ) {
572 				dirtyPropertyNames[i] = allPropertyNames[ dirtyProperties[i]];
573 			}
574 			log.trace(
575 					"Found dirty properties [{}] : {}",
576 					MessageHelper.infoString( persister.getEntityName(), id ),
577 					dirtyPropertyNames
578 			);
579 		}
580 	}

582 	private Object[] getDatabaseSnapshot(SessionImplementor session, EntityPersister persister, Serializable id) {
583 		if ( persister.isSelectBeforeUpdateRequired() ) {
584 			Object[] snapshot = session.getPersistenceContext()
585 					.getDatabaseSnapshot(id, persister);
586 			if (snapshot==null) {
587 				//do we even really need this? the update will fail anyway....
588 				if ( session.getFactory().getStatistics().isStatisticsEnabled() ) {
589 					session.getFactory().getStatisticsImplementor()
590 							.optimisticFailure( persister.getEntityName() );
591 				}
592 				throw new StaleObjectStateException( persister.getEntityName(), id );
593 			}
594 			else {
595 				return snapshot;
596 			}
597 		}
598 		else {
599 			//TODO: optimize away this lookup for entities w/o unsaved-value="undefined"
600 			EntityKey entityKey = new EntityKey( id, persister, session.getEntityMode() );
601 			return session.getPersistenceContext()
602 					.getCachedDatabaseSnapshot( entityKey ); 
603 		}
604 	}
