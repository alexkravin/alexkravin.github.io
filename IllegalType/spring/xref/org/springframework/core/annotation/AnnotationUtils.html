29  /**
30   * General utility methods for working with annotations, handling bridge methods (which the compiler
31   * generates for generic declarations) as well as super methods (for optional &quot;annotation inheritance&quot;).
32   * Note that none of this is provided by the JDK's introspection facilities themselves.
33   *
34   * <p>As a general rule for runtime-retained annotations (e.g. for transaction control, authorization or service
35   * exposure), always use the lookup methods on this class (e.g., {@link #findAnnotation(Method, Class)}, {@link
36   * #getAnnotation(Method, Class)}, and {@link #getAnnotations(Method)}) instead of the plain annotation lookup
37   * methods in the JDK. You can still explicitly choose between lookup on the given class level only ({@link
38   * #getAnnotation(Method, Class)}) and lookup in the entire inheritance hierarchy of the given method ({@link
39   * #findAnnotation(Method, Class)}).
40   *
41   * @author Rob Harrop
42   * @author Juergen Hoeller
43   * @author Sam Brannen
44   * @author Mark Fisher
45   * @author Chris Beams
46   * @since 2.0
47   * @see java.lang.reflect.Method#getAnnotations()
48   * @see java.lang.reflect.Method#getAnnotation(Class)
49   */
50  public abstract class AnnotationUtils {

55  	private static final Map<Class<?>, Boolean> annotatedInterfaceCache = new WeakHashMap<Class<?>, Boolean>();

58  	/**
59  	 * Get a single {@link Annotation} of {@code annotationType} from the supplied
60  	 * Method, Constructor or Field. Meta-annotations will be searched if the annotation
61  	 * is not declared locally on the supplied element.
62  	 * @param ae the Method, Constructor or Field from which to get the annotation
63  	 * @param annotationType the annotation class to look for, both locally and as a meta-annotation
64  	 * @return the matching annotation or {@code null} if not found
65  	 * @since 3.1
66  	 */
67  	public static <T extends Annotation> T getAnnotation(AnnotatedElement ae, Class<T> annotationType) {
68  		T ann = ae.getAnnotation(annotationType);
69  		if (ann == null) {
70  			for (Annotation metaAnn : ae.getAnnotations()) {
71  				ann = metaAnn.annotationType().getAnnotation(annotationType);
72  				if (ann != null) {
73  					break;
74  				}
75  			}
76  		}
77  		return ann;
78  	}

91  	/**
92  	 * Get a single {@link Annotation} of <code>annotationType</code> from the supplied {@link Method}.
93  	 * <p>Correctly handles bridge {@link Method Methods} generated by the compiler.
94  	 * @param method the method to look for annotations on
95  	 * @param annotationType the annotation class to look for
96  	 * @return the annotations found
97  	 * @see org.springframework.core.BridgeMethodResolver#findBridgedMethod(Method)
98  	 */
99  	public static <A extends Annotation> A getAnnotation(Method method, Class<A> annotationType) {
100 		Method resolvedMethod = BridgeMethodResolver.findBridgedMethod(method);
101 		A ann = resolvedMethod.getAnnotation(annotationType);
102 		if (ann == null) {
103 			for (Annotation metaAnn : resolvedMethod.getAnnotations()) {
104 				ann = metaAnn.annotationType().getAnnotation(annotationType);
105 				if (ann != null) {
106 					break;
107 				}
108 			}
109 		}
110 		return ann;
111 	}

113 	/**
114 	 * Get a single {@link Annotation} of <code>annotationType</code> from the supplied {@link Method},
115 	 * traversing its super methods if no annotation can be found on the given method itself.
116 	 * <p>Annotations on methods are not inherited by default, so we need to handle this explicitly.
117 	 * @param method the method to look for annotations on
118 	 * @param annotationType the annotation class to look for
119 	 * @return the annotation found, or <code>null</code> if none found
120 	 */
121 	public static <A extends Annotation> A findAnnotation(Method method, Class<A> annotationType) {
122 		A annotation = getAnnotation(method, annotationType);
123 		Class<?> cl = method.getDeclaringClass();
124 		if (annotation == null) {
125 			annotation = searchOnInterfaces(method, annotationType, cl.getInterfaces());
126 		}
127 		while (annotation == null) {
128 			cl = cl.getSuperclass();
129 			if (cl == null || cl == Object.class) {
130 				break;
131 			}
132 			try {
133 				Method equivalentMethod = cl.getDeclaredMethod(method.getName(), method.getParameterTypes());
134 				annotation = getAnnotation(equivalentMethod, annotationType);
135 				if (annotation == null) {
136 					annotation = searchOnInterfaces(method, annotationType, cl.getInterfaces());
137 				}
138 			}
139 			catch (NoSuchMethodException ex) {
140 				// We're done...
141 			}
142 		}
143 		return annotation;
144 	}
145 
146 	private static <A extends Annotation> A searchOnInterfaces(Method method, Class<A> annotationType, Class<?>[] ifcs) {
147 		A annotation = null;
148 		for (Class<?> iface : ifcs) {
149 			if (isInterfaceWithAnnotatedMethods(iface)) {
150 				try {
151 					Method equivalentMethod = iface.getMethod(method.getName(), method.getParameterTypes());
152 					annotation = getAnnotation(equivalentMethod, annotationType);
153 				}
154 				catch (NoSuchMethodException ex) {
155 					// Skip this interface - it doesn't have the method...
156 				}
157 				if (annotation != null) {
158 					break;
159 				}
160 			}
161 		}
162 		return annotation;
163 	}
164 
165 	private static boolean isInterfaceWithAnnotatedMethods(Class<?> iface) {
166 		synchronized (annotatedInterfaceCache) {
167 			Boolean flag = annotatedInterfaceCache.get(iface);
168 			if (flag != null) {
169 				return flag;
170 			}
171 			boolean found = false;
172 			for (Method ifcMethod : iface.getMethods()) {
173 				if (ifcMethod.getAnnotations().length > 0) {
174 					found = true;
175 					break;
176 				}
177 			}
178 			annotatedInterfaceCache.put(iface, found);
179 			return found;
180 		}
181 	}

183 	/**
184 	 * Find a single {@link Annotation} of <code>annotationType</code> from the supplied {@link Class},
185 	 * traversing its interfaces and superclasses if no annotation can be found on the given class itself.
186 	 * <p>This method explicitly handles class-level annotations which are not declared as
187 	 * {@link java.lang.annotation.Inherited inherited} <i>as well as annotations on interfaces</i>.
188 	 * <p>The algorithm operates as follows: Searches for an annotation on the given class and returns
189 	 * it if found. Else searches all interfaces that the given class declares, returning the annotation
190 	 * from the first matching candidate, if any. Else proceeds with introspection of the superclass
191 	 * of the given class, checking the superclass itself; if no annotation found there, proceeds
192 	 * with the interfaces that the superclass declares. Recursing up through the entire superclass
193 	 * hierarchy if no match is found.
194 	 * @param clazz the class to look for annotations on
195 	 * @param annotationType the annotation class to look for
196 	 * @return the annotation found, or <code>null</code> if none found
197 	 */
198 	public static <A extends Annotation> A findAnnotation(Class<?> clazz, Class<A> annotationType) {
199 		Assert.notNull(clazz, "Class must not be null");
200 		A annotation = clazz.getAnnotation(annotationType);
201 		if (annotation != null) {
202 			return annotation;
203 		}
204 		for (Class<?> ifc : clazz.getInterfaces()) {
205 			annotation = findAnnotation(ifc, annotationType);
206 			if (annotation != null) {
207 				return annotation;
208 			}
209 		}
210 		if (!Annotation.class.isAssignableFrom(clazz)) {
211 			for (Annotation ann : clazz.getAnnotations()) {
212 				annotation = findAnnotation(ann.annotationType(), annotationType);
213 				if (annotation != null) {
214 					return annotation;
215 				}
216 			}
217 		}
218 		Class<?> superClass = clazz.getSuperclass();
219 		if (superClass == null || superClass == Object.class) {
220 			return null;
221 		}
222 		return findAnnotation(superClass, annotationType);
223 	}

225 	/**
226 	 * Find the first {@link Class} in the inheritance hierarchy of the specified <code>clazz</code>
227 	 * (including the specified <code>clazz</code> itself) which declares an annotation for the
228 	 * specified <code>annotationType</code>, or <code>null</code> if not found. If the supplied
229 	 * <code>clazz</code> is <code>null</code>, <code>null</code> will be returned.
230 	 * <p>If the supplied <code>clazz</code> is an interface, only the interface itself will be checked;
231 	 * the inheritance hierarchy for interfaces will not be traversed.
232 	 * <p>The standard {@link Class} API does not provide a mechanism for determining which class
233 	 * in an inheritance hierarchy actually declares an {@link Annotation}, so we need to handle
234 	 * this explicitly.
235 	 * @param annotationType the Class object corresponding to the annotation type
236 	 * @param clazz the Class object corresponding to the class on which to check for the annotation,
237 	 * or <code>null</code>
238 	 * @return the first {@link Class} in the inheritance hierarchy of the specified <code>clazz</code>
239 	 * which declares an annotation for the specified <code>annotationType</code>, or <code>null</code>
240 	 * if not found
241 	 * @see Class#isAnnotationPresent(Class)
242 	 * @see Class#getDeclaredAnnotations()
243 	 */
244 	public static Class<?> findAnnotationDeclaringClass(Class<? extends Annotation> annotationType, Class<?> clazz) {
245 		Assert.notNull(annotationType, "Annotation type must not be null");
246 		if (clazz == null || clazz.equals(Object.class)) {
247 			return null;
248 		}
249 		return (isAnnotationDeclaredLocally(annotationType, clazz)) ? clazz :
250 				findAnnotationDeclaringClass(annotationType, clazz.getSuperclass());
251 	}

253 	/**
254 	 * Determine whether an annotation for the specified <code>annotationType</code> is
255 	 * declared locally on the supplied <code>clazz</code>. The supplied {@link Class}
256 	 * may represent any type.
257 	 * <p>Note: This method does <strong>not</strong> determine if the annotation is
258 	 * {@link java.lang.annotation.Inherited inherited}. For greater clarity regarding inherited
259 	 * annotations, consider using {@link #isAnnotationInherited(Class, Class)} instead.
260 	 * @param annotationType the Class object corresponding to the annotation type
261 	 * @param clazz the Class object corresponding to the class on which to check for the annotation
262 	 * @return <code>true</code> if an annotation for the specified <code>annotationType</code>
263 	 * is declared locally on the supplied <code>clazz</code>
264 	 * @see Class#getDeclaredAnnotations()
265 	 * @see #isAnnotationInherited(Class, Class)
266 	 */
267 	public static boolean isAnnotationDeclaredLocally(Class<? extends Annotation> annotationType, Class<?> clazz) {
268 		Assert.notNull(annotationType, "Annotation type must not be null");
269 		Assert.notNull(clazz, "Class must not be null");
270 		boolean declaredLocally = false;
271 		for (Annotation annotation : Arrays.asList(clazz.getDeclaredAnnotations())) {
272 			if (annotation.annotationType().equals(annotationType)) {
273 				declaredLocally = true;
274 				break;
275 			}
276 		}
277 		return declaredLocally;
278 	}

280 	/**
281 	 * Determine whether an annotation for the specified <code>annotationType</code> is present
282 	 * on the supplied <code>clazz</code> and is {@link java.lang.annotation.Inherited inherited}
283 	 * i.e., not declared locally for the class).
284 	 * <p>If the supplied <code>clazz</code> is an interface, only the interface itself will be checked.
285 	 * In accordance with standard meta-annotation semantics, the inheritance hierarchy for interfaces
286 	 * will not be traversed. See the {@link java.lang.annotation.Inherited JavaDoc} for the
287 	 * &#064;Inherited meta-annotation for further details regarding annotation inheritance.
288 	 * @param annotationType the Class object corresponding to the annotation type
289 	 * @param clazz the Class object corresponding to the class on which to check for the annotation
290 	 * @return <code>true</code> if an annotation for the specified <code>annotationType</code> is present
291 	 * on the supplied <code>clazz</code> and is {@link java.lang.annotation.Inherited inherited}
292 	 * @see Class#isAnnotationPresent(Class)
293 	 * @see #isAnnotationDeclaredLocally(Class, Class)
294 	 */
295 	public static boolean isAnnotationInherited(Class<? extends Annotation> annotationType, Class<?> clazz) {
296 		Assert.notNull(annotationType, "Annotation type must not be null");
297 		Assert.notNull(clazz, "Class must not be null");
298 		return (clazz.isAnnotationPresent(annotationType) && !isAnnotationDeclaredLocally(annotationType, clazz));
299 	}

333 	/**
334 	 * Retrieve the given annotation's attributes as an {@link AnnotationAttributes}
335 	 * map structure. Implemented in Spring 3.1.1 to provide fully recursive annotation
336 	 * reading capabilities on par with that of the reflection-based
337 	 * {@link org.springframework.core.type.StandardAnnotationMetadata}.
338 	 * @param annotation the annotation to retrieve the attributes for
339 	 * @param classValuesAsString whether to turn Class references into Strings (for
340 	 * compatibility with {@link org.springframework.core.type.AnnotationMetadata} or to
341 	 * preserve them as Class references
342 	 * @param nestedAnnotationsAsMap whether to turn nested Annotation instances into
343 	 * {@link AnnotationAttributes} maps (for compatibility with
344 	 * {@link org.springframework.core.type.AnnotationMetadata} or to preserve them as
345 	 * Annotation instances
346 	 * @return the annotation attributes (a specialized Map) with attribute names as keys
347 	 * and corresponding attribute values as values
348 	 * @since 3.1.1
349 	 */
350 	public static AnnotationAttributes getAnnotationAttributes(
351 			Annotation annotation, boolean classValuesAsString, boolean nestedAnnotationsAsMap) {
352 
353 		AnnotationAttributes attrs = new AnnotationAttributes();
354 		Method[] methods = annotation.annotationType().getDeclaredMethods();
355 		for (Method method : methods) {
356 			if (method.getParameterTypes().length == 0 && method.getReturnType() != void.class) {
357 				try {
358 					Object value = method.invoke(annotation);
359 					if (classValuesAsString) {
360 						if (value instanceof Class) {
361 							value = ((Class<?>) value).getName();
362 						}
363 						else if (value instanceof Class[]) {
364 							Class<?>[] clazzArray = (Class[]) value;
365 							String[] newValue = new String[clazzArray.length];
366 							for (int i = 0; i < clazzArray.length; i++) {
367 								newValue[i] = clazzArray[i].getName();
368 							}
369 							value = newValue;
370 						}
371 					}
372 					if (nestedAnnotationsAsMap && value instanceof Annotation) {
373 						attrs.put(method.getName(), getAnnotationAttributes(
374 								(Annotation)value, classValuesAsString, nestedAnnotationsAsMap));
375 					}
376 					else if (nestedAnnotationsAsMap && value instanceof Annotation[]) {
377 						Annotation[] realAnnotations = (Annotation[])value;
378 						AnnotationAttributes[] mappedAnnotations = new AnnotationAttributes[realAnnotations.length];
379 						for (int i = 0; i < realAnnotations.length; i++) {
380 							mappedAnnotations[i] = getAnnotationAttributes(
381 									realAnnotations[i], classValuesAsString, nestedAnnotationsAsMap);
382 						}
383 						attrs.put(method.getName(), mappedAnnotations);
384 					}
385 					else {
386 						attrs.put(method.getName(), value);
387 					}
388 				}
389 				catch (Exception ex) {
390 					throw new IllegalStateException("Could not obtain annotation attribute values", ex);
391 				}
392 			}
393 		}
394 		return attrs;
395 	}

411 	 * @param attributeName the name of the attribute value to retrieve
412 	 * @return the attribute value, or <code>null</code> if not found

415 	public static Object getValue(Annotation annotation, String attributeName) {
416 		try {
417 			Method method = annotation.annotationType().getDeclaredMethod(attributeName, new Class[0]);
418 			return method.invoke(annotation);
419 		}
420 		catch (Exception ex) {
421 			return null;
