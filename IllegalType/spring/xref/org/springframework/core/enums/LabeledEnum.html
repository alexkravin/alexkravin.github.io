17  package org.springframework.core.enums;
18  
19  import java.io.Serializable;
20  import java.util.Comparator;
21  
22  import org.springframework.util.comparator.CompoundComparator;
23  import org.springframework.util.comparator.NullSafeComparator;
24  
25  /**
26   * An interface for objects that represent a labeled enumeration.
27   * Each such enum instance has the following characteristics:
28   *
29   * <ul>
30   * <li>A type that identifies the enum's class.
31   * For example: <code>com.mycompany.util.FileFormat</code>.</li>
32   *
33   * <li>A code that uniquely identifies the enum within the context of its type.
34   * For example: &quot;CSV&quot;. Different classes of codes are possible
35   * (e.g., Character, Integer, String).</li>
36   *
37   * <li>A descriptive label. For example: "the CSV File Format".</li>
38   * </ul>
39   *
40   * @author Keith Donald
41   * @since 1.2.2
42   * @deprecated as of Spring 3.0, in favor of Java 5 enums.
43   */
44  @Deprecated
45  public interface LabeledEnum extends Comparable, Serializable {
46  
47  	/**
48  	 * Return this enumeration's type.
49  	 */
50  	Class getType();
51  
52  	/**
53  	 * Return this enumeration's code.
54  	 * <p>Each code should be unique within enumerations of the same type.
55  	 */
56  	Comparable getCode();
57  
58  	/**
59  	 * Return a descriptive, optional label.
60  	 */
61  	String getLabel();
62  
63  
64  	// Constants for standard enum ordering (Comparator implementations)
65  
66  	/**
67  	 * Shared Comparator instance that sorts enumerations by <code>CODE_ORDER</code>.
68  	 */
69  	Comparator CODE_ORDER = new Comparator() {
70  		public int compare(Object o1, Object o2) {
71  			Comparable c1 = ((LabeledEnum) o1).getCode();
72  			Comparable c2 = ((LabeledEnum) o2).getCode();
73  			return c1.compareTo(c2);
74  		}
75  	};
76  
77  	/**
78  	 * Shared Comparator instance that sorts enumerations by <code>LABEL_ORDER</code>.
79  	 */
80  	Comparator LABEL_ORDER = new Comparator() {
81  		public int compare(Object o1, Object o2) {
82  			LabeledEnum e1 = (LabeledEnum) o1;
83  			LabeledEnum e2 = (LabeledEnum) o2;
84  			Comparator comp = new NullSafeComparator(String.CASE_INSENSITIVE_ORDER, true);
85  			return comp.compare(e1.getLabel(), e2.getLabel());
86  		}
87  	};
88  
89  	/**
90  	 * Shared Comparator instance that sorts enumerations by <code>LABEL_ORDER</code>,
91  	 * then <code>CODE_ORDER</code>.
92  	 */
93  	Comparator DEFAULT_ORDER =
94  			new CompoundComparator(new Comparator[] { LABEL_ORDER, CODE_ORDER });
95  
96  }
