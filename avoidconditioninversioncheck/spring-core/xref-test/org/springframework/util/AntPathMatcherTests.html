308 		assertEquals(Collections.singletonMap("hotel", "1"), result);
309 
310 		result = pathMatcher.extractUriTemplateVariables("/h?tels/{hotel}", "/hotels/1");
311 		assertEquals(Collections.singletonMap("hotel", "1"), result);
312 
313 		result = pathMatcher.extractUriTemplateVariables("/hotels/{hotel}/bookings/{booking}", "/hotels/1/bookings/2");
314 		Map<String, String> expected = new LinkedHashMap<String, String>();
315 		expected.put("hotel", "1");
316 		expected.put("booking", "2");
317 		assertEquals(expected, result);
318 
319 		result = pathMatcher.extractUriTemplateVariables("/**/hotels/**/{hotel}", "/foo/hotels/bar/1");
320 		assertEquals(Collections.singletonMap("hotel", "1"), result);
321 
322 		result = pathMatcher.extractUriTemplateVariables("/{page}.html", "/42.html");
323 		assertEquals(Collections.singletonMap("page", "42"), result);
324 
325 		result = pathMatcher.extractUriTemplateVariables("/{page}.*", "/42.html");
326 		assertEquals(Collections.singletonMap("page", "42"), result);
327 
328 		result = pathMatcher.extractUriTemplateVariables("/A-{B}-C", "/A-b-C");
329 		assertEquals(Collections.singletonMap("B", "b"), result);
330 
331 		result = pathMatcher.extractUriTemplateVariables("/{name}.{extension}", "/test.html");
332 		expected = new LinkedHashMap<String, String>();
333 		expected.put("name", "test");
334 		expected.put("extension", "html");
335 		assertEquals(expected, result);
336 	}
