202 	 * Return the type of the method/constructor parameter.
203 	 * @return the parameter type (never <code>null</code>)
204 	 */
205 	public Class<?> getParameterType() {
206 		if (this.parameterType == null) {
207 			if (this.parameterIndex < 0) {
208 				this.parameterType = (this.method != null ? this.method.getReturnType() : null);
209 			}
210 			else {
211 				this.parameterType = (this.method != null ?
212 					this.method.getParameterTypes()[this.parameterIndex] :
213 					this.constructor.getParameterTypes()[this.parameterIndex]);
214 			}
215 		}
216 		return this.parameterType;
217 	}

237 	public Class<?> getNestedParameterType() {
238 		if (this.nestingLevel > 1) {
239 			Type type = getGenericParameterType();
240 			if (type instanceof ParameterizedType) {
241 				Integer index = getTypeIndexForCurrentLevel();
242 				Type arg = ((ParameterizedType) type).getActualTypeArguments()[index != null ? index : 0];
243 				if (arg instanceof Class) {
244 					return (Class) arg;
245 				}
246 				else if (arg instanceof ParameterizedType) {
247 					arg = ((ParameterizedType) arg).getRawType();
248 					if (arg instanceof Class) {
249 						return (Class) arg;
250 					}
251 				}
252 			}
253 			return Object.class;
254 		}
255 		else {
256 			return getParameterType();
257 		}
258 	}

277 	/**
278 	 * Return the annotations associated with the specific method/constructor parameter.
279 	 */
280 	public Annotation[] getParameterAnnotations() {
281 		if (this.parameterAnnotations == null) {
282 			Annotation[][] annotationArray = (this.method != null ?
283 					this.method.getParameterAnnotations() : this.constructor.getParameterAnnotations());
284 			if (this.parameterIndex >= 0 && this.parameterIndex < annotationArray.length) {
285 				this.parameterAnnotations = annotationArray[this.parameterIndex];
286 			}
287 			else {
288 				this.parameterAnnotations = new Annotation[0];
289 			}
290 		}
291 		return this.parameterAnnotations;
292 	}

295 	 * Return the parameter annotation of the given type, if available.
296 	 * @param annotationType the annotation type to look for
297 	 * @return the annotation object, or <code>null</code> if not found
298 	 */
299 	@SuppressWarnings("unchecked")
300 	public <T extends Annotation> T getParameterAnnotation(Class<T> annotationType) {
301 		Annotation[] anns = getParameterAnnotations();
302 		for (Annotation ann : anns) {
303 			if (annotationType.isInstance(ann)) {
304 				return (T) ann;
305 			}
306 		}
307 		return null;
308 	}

336 	 * @return the parameter name (may be <code>null</code> if no
337 	 * parameter name metadata is contained in the class file or no
338 	 * {@link #initParameterNameDiscovery ParameterNameDiscoverer}

341 	public String getParameterName() {
342 		if (this.parameterNameDiscoverer != null) {
343 			String[] parameterNames = (this.method != null ?
344 					this.parameterNameDiscoverer.getParameterNames(this.method) :
345 					this.parameterNameDiscoverer.getParameterNames(this.constructor));
346 			if (parameterNames != null) {
347 				this.parameterName = parameterNames[this.parameterIndex];
348 			}
349 			this.parameterNameDiscoverer = null;
350 		}
351 		return this.parameterName;

421 	/**
422 	 * Create a new MethodParameter for the given method or constructor.
423 	 * <p>This is a convenience constructor for scenarios where a
424 	 * Method or Constructor reference is treated in a generic fashion.
425 	 * @param methodOrConstructor the Method or Constructor to specify a parameter for
426 	 * @param parameterIndex the index of the parameter
427 	 * @return the corresponding MethodParameter instance
428 	 */
429 	public static MethodParameter forMethodOrConstructor(Object methodOrConstructor, int parameterIndex) {
430 		if (methodOrConstructor instanceof Method) {
431 			return new MethodParameter((Method) methodOrConstructor, parameterIndex);
432 		}
433 		else if (methodOrConstructor instanceof Constructor) {
434 			return new MethodParameter((Constructor) methodOrConstructor, parameterIndex);
435 		}
436 		else {
437 			throw new IllegalArgumentException(
438 					"Given object [" + methodOrConstructor + "] is neither a Method nor a Constructor");
439 		}
440 	}
