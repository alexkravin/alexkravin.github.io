35  /**
36   * Helper class for resolving generic types against type variables.
37   *
38   * <p>Mainly intended for usage within the framework, resolving method
39   * parameter types even when they are declared generically.
40   *
41   * @author Juergen Hoeller
42   * @author Rob Harrop
43   * @since 2.5.2
44   * @see GenericCollectionTypeResolver
45   */

53  	/**
54  	 * Determine the target type for the given parameter specification.
55  	 * @param methodParam the method parameter specification
56  	 * @return the corresponding generic parameter type
57  	 */
58  	public static Type getTargetType(MethodParameter methodParam) {
59  		Assert.notNull(methodParam, "MethodParameter must not be null");
60  		if (methodParam.getConstructor() != null) {
61  			return methodParam.getConstructor().getGenericParameterTypes()[methodParam.getParameterIndex()];
62  		}
63  		else {
64  			if (methodParam.getParameterIndex() >= 0) {
65  				return methodParam.getMethod().getGenericParameterTypes()[methodParam.getParameterIndex()];
66  			}
67  			else {
68  				return methodParam.getMethod().getGenericReturnType();
69  			}
70  		}
71  	}

73  	/**
74  	 * Determine the target type for the given generic parameter type.
75  	 * @param methodParam the method parameter specification
76  	 * @param clazz the class to resolve type variables against
77  	 * @return the corresponding generic parameter or return type
78  	 */
79  	public static Class<?> resolveParameterType(MethodParameter methodParam, Class clazz) {
80  		Type genericType = getTargetType(methodParam);
81  		Assert.notNull(clazz, "Class must not be null");
82  		Map<TypeVariable, Type> typeVariableMap = getTypeVariableMap(clazz);
83  		Type rawType = getRawType(genericType, typeVariableMap);
84  		Class result = (rawType instanceof Class ? (Class) rawType : methodParam.getParameterType());
85  		methodParam.setParameterType(result);
86  		methodParam.typeVariableMap = typeVariableMap;
87  		return result;
88  	}

91  	 * Determine the target type for the generic return type of the given method.
92  	 * @param method the method to introspect
93  	 * @param clazz the class to resolve type variables against
94  	 * @return the corresponding generic parameter or return type
95  	 */
96  	public static Class<?> resolveReturnType(Method method, Class clazz) {
97  		Assert.notNull(method, "Method must not be null");
98  		Type genericType = method.getGenericReturnType();
99  		Assert.notNull(clazz, "Class must not be null");
100 		Map<TypeVariable, Type> typeVariableMap = getTypeVariableMap(clazz);
101 		Type rawType = getRawType(genericType, typeVariableMap);
102 		return (rawType instanceof Class ? (Class) rawType : method.getReturnType());
103 	}

105 	/**
106 	 * Resolve the single type argument of the given generic interface against the given
107 	 * target method which is assumed to return the given interface or an implementation
108 	 * of it.
109 	 * @param method the target method to check the return type of
110 	 * @param genericIfc the generic interface or superclass to resolve the type argument from
111 	 * @return the resolved parameter type of the method return type, or <code>null</code>
112 	 * if not resolvable or if the single argument is of type {@link WildcardType}.
113 	 */
114 	public static Class<?> resolveReturnTypeArgument(Method method, Class<?> genericIfc) {
115 		Assert.notNull(method, "method must not be null");
116 		Type returnType = method.getReturnType();
117 		Type genericReturnType = method.getGenericReturnType();
118 		if (returnType.equals(genericIfc)) {
119 			if (genericReturnType instanceof ParameterizedType) {
120 				ParameterizedType targetType = (ParameterizedType) genericReturnType;
121 				Type[] actualTypeArguments = targetType.getActualTypeArguments();
122 				Type typeArg = actualTypeArguments[0];
123 				if (!(typeArg instanceof WildcardType)) {
124 					return (Class<?>) typeArg;
125 				}
126 			}
127 			else {
128 				return null;
129 			}
130 		}
131 		return GenericTypeResolver.resolveTypeArgument((Class<?>) returnType, genericIfc);
132 	}

134 	/**
135 	 * Resolve the single type argument of the given generic interface against
136 	 * the given target class which is assumed to implement the generic interface
137 	 * and possibly declare a concrete type for its type variable.
138 	 * @param clazz the target class to check against
139 	 * @param genericIfc the generic interface or superclass to resolve the type argument from
140 	 * @return the resolved type of the argument, or <code>null</code> if not resolvable
141 	 */
142 	public static Class<?> resolveTypeArgument(Class clazz, Class genericIfc) {
143 		Class[] typeArgs = resolveTypeArguments(clazz, genericIfc);
144 		if (typeArgs == null) {
145 			return null;
146 		}
147 		if (typeArgs.length != 1) {
148 			throw new IllegalArgumentException("Expected 1 type argument on generic interface [" +
149 					genericIfc.getName() + "] but found " + typeArgs.length);
150 		}
151 		return typeArgs[0];
152 	}

154 	/**
155 	 * Resolve the type arguments of the given generic interface against the given
156 	 * target class which is assumed to implement the generic interface and possibly
157 	 * declare concrete types for its type variables.
158 	 * @param clazz the target class to check against
159 	 * @param genericIfc the generic interface or superclass to resolve the type argument from
160 	 * @return the resolved type of each argument, with the array size matching the
161 	 * number of actual type arguments, or <code>null</code> if not resolvable
162 	 */
163 	public static Class[] resolveTypeArguments(Class clazz, Class genericIfc) {
164 		return doResolveTypeArguments(clazz, clazz, genericIfc);
165 	}

167 	private static Class[] doResolveTypeArguments(Class ownerClass, Class classToIntrospect, Class genericIfc) {
168 		while (classToIntrospect != null) {
169 			if (genericIfc.isInterface()) {
170 				Type[] ifcs = classToIntrospect.getGenericInterfaces();
171 				for (Type ifc : ifcs) {
172 					Class[] result = doResolveTypeArguments(ownerClass, ifc, genericIfc);
173 					if (result != null) {
174 						return result;
175 					}
176 				}
177 			}
178 			else {
179 				Class[] result = doResolveTypeArguments(
180 						ownerClass, classToIntrospect.getGenericSuperclass(), genericIfc);
181 				if (result != null) {
182 					return result;
183 				}
184 			}
185 			classToIntrospect = classToIntrospect.getSuperclass();
186 		}
187 		return null;
188 	}

190 	private static Class[] doResolveTypeArguments(Class ownerClass, Type ifc, Class genericIfc) {
191 		if (ifc instanceof ParameterizedType) {
192 			ParameterizedType paramIfc = (ParameterizedType) ifc;
193 			Type rawType = paramIfc.getRawType();
194 			if (genericIfc.equals(rawType)) {
195 				Type[] typeArgs = paramIfc.getActualTypeArguments();
196 				Class[] result = new Class[typeArgs.length];
197 				for (int i = 0; i < typeArgs.length; i++) {
198 					Type arg = typeArgs[i];
199 					result[i] = extractClass(ownerClass, arg);
200 				}
201 				return result;
202 			}
203 			else if (genericIfc.isAssignableFrom((Class) rawType)) {
204 				return doResolveTypeArguments(ownerClass, (Class) rawType, genericIfc);
205 			}
206 		}
207 		else if (ifc != null && genericIfc.isAssignableFrom((Class) ifc)) {
208 			return doResolveTypeArguments(ownerClass, (Class) ifc, genericIfc);
209 		}
210 		return null;
211 	}

213 	/**
214 	 * Extract a class instance from given Type.
215 	 */
216 	private static Class extractClass(Class ownerClass, Type arg) {
217 		if (arg instanceof ParameterizedType) {
218 			return extractClass(ownerClass, ((ParameterizedType) arg).getRawType());
219 		}
220 		else if (arg instanceof GenericArrayType) {
221 			GenericArrayType gat = (GenericArrayType) arg;
222 			Type gt = gat.getGenericComponentType();
223 			Class<?> componentClass = extractClass(ownerClass, gt);
224 			return Array.newInstance(componentClass, 0).getClass();
225 		}
226 		else if (arg instanceof TypeVariable) {
227 			TypeVariable tv = (TypeVariable) arg;
228 			arg = getTypeVariableMap(ownerClass).get(tv);
229 			if (arg == null) {
230 				arg = extractBoundForTypeVariable(tv);
231 			}
232 			else {
233 				arg = extractClass(ownerClass, arg);
234 			}
235 		}
236 		return (arg instanceof Class ? (Class) arg : Object.class);
237 	}

241 	 * Resolve the specified generic type against the given TypeVariable map.
242 	 * @param genericType the generic type to resolve
243 	 * @param typeVariableMap the TypeVariable Map to resolved against
244 	 * @return the type if it resolves to a Class, or <code>Object.class</code> otherwise
245 	 */
246 	public static Class<?> resolveType(Type genericType, Map<TypeVariable, Type> typeVariableMap) {
247 		Type rawType = getRawType(genericType, typeVariableMap);
248 		return (rawType instanceof Class ? (Class) rawType : Object.class);
249 	}

251 	/**
252 	 * Determine the raw type for the given generic parameter type.
253 	 * @param genericType the generic type to resolve
254 	 * @param typeVariableMap the TypeVariable Map to resolved against
255 	 * @return the resolved raw type
256 	 */
257 	static Type getRawType(Type genericType, Map<TypeVariable, Type> typeVariableMap) {
258 		Type resolvedType = genericType;
259 		if (genericType instanceof TypeVariable) {
260 			TypeVariable tv = (TypeVariable) genericType;
261 			resolvedType = typeVariableMap.get(tv);
262 			if (resolvedType == null) {
263 				resolvedType = extractBoundForTypeVariable(tv);
264 			}
265 		}
266 		if (resolvedType instanceof ParameterizedType) {
267 			return ((ParameterizedType) resolvedType).getRawType();
268 		}
269 		else {
270 			return resolvedType;
271 		}
272 	}

274 	/**
275 	 * Build a mapping of {@link TypeVariable#getName TypeVariable names} to concrete
276 	 * {@link Class} for the specified {@link Class}. Searches all super types,
277 	 * enclosing types and interfaces.
278 	 */
279 	public static Map<TypeVariable, Type> getTypeVariableMap(Class clazz) {
280 		Reference<Map<TypeVariable, Type>> ref = typeVariableCache.get(clazz);
281 		Map<TypeVariable, Type> typeVariableMap = (ref != null ? ref.get() : null);
282 
283 		if (typeVariableMap == null) {
284 			typeVariableMap = new HashMap<TypeVariable, Type>();
285 
286 			// interfaces
287 			extractTypeVariablesFromGenericInterfaces(clazz.getGenericInterfaces(), typeVariableMap);
288 
289 			// super class
290 			Type genericType = clazz.getGenericSuperclass();
291 			Class type = clazz.getSuperclass();
292 			while (type != null && !Object.class.equals(type)) {
293 				if (genericType instanceof ParameterizedType) {
294 					ParameterizedType pt = (ParameterizedType) genericType;
295 					populateTypeMapFromParameterizedType(pt, typeVariableMap);
296 				}
297 				extractTypeVariablesFromGenericInterfaces(type.getGenericInterfaces(), typeVariableMap);
298 				genericType = type.getGenericSuperclass();
299 				type = type.getSuperclass();
300 			}
301 
302 			// enclosing class
303 			type = clazz;
304 			while (type.isMemberClass()) {
305 				genericType = type.getGenericSuperclass();
306 				if (genericType instanceof ParameterizedType) {
307 					ParameterizedType pt = (ParameterizedType) genericType;
308 					populateTypeMapFromParameterizedType(pt, typeVariableMap);
309 				}
310 				type = type.getEnclosingClass();
311 			}
312 
313 			typeVariableCache.put(clazz, new WeakReference<Map<TypeVariable, Type>>(typeVariableMap));
314 		}
315 
316 		return typeVariableMap;
317 	}

319 	/**
320 	 * Extracts the bound <code>Type</code> for a given {@link TypeVariable}.
321 	 */
322 	static Type extractBoundForTypeVariable(TypeVariable typeVariable) {
323 		Type[] bounds = typeVariable.getBounds();
324 		if (bounds.length == 0) {
325 			return Object.class;
326 		}
327 		Type bound = bounds[0];
328 		if (bound instanceof TypeVariable) {
329 			bound = extractBoundForTypeVariable((TypeVariable) bound);
330 		}
331 		return bound;
332 	}

334 	private static void extractTypeVariablesFromGenericInterfaces(Type[] genericInterfaces, Map<TypeVariable, Type> typeVariableMap) {
335 		for (Type genericInterface : genericInterfaces) {
336 			if (genericInterface instanceof ParameterizedType) {
337 				ParameterizedType pt = (ParameterizedType) genericInterface;
338 				populateTypeMapFromParameterizedType(pt, typeVariableMap);
339 				if (pt.getRawType() instanceof Class) {
340 					extractTypeVariablesFromGenericInterfaces(
341 							((Class) pt.getRawType()).getGenericInterfaces(), typeVariableMap);
342 				}
343 			}
344 			else if (genericInterface instanceof Class) {
345 				extractTypeVariablesFromGenericInterfaces(
346 						((Class) genericInterface).getGenericInterfaces(), typeVariableMap);
347 			}
348 		}
349 	}

351 	/**
352 	 * Read the {@link TypeVariable TypeVariables} from the supplied {@link ParameterizedType}
353 	 * and add mappings corresponding to the {@link TypeVariable#getName TypeVariable name} ->
354 	 * concrete type to the supplied {@link Map}.
355 	 * <p>Consider this case:
356 	 * <pre class="code>
357 	 * public interface Foo<S, T> {
358 	 *  ..
359 	 * }
360 	 *
361 	 * public class FooImpl implements Foo<String, Integer> {
362 	 *  ..
363 	 * }</pre>
364 	 * For '<code>FooImpl</code>' the following mappings would be added to the {@link Map}:
365 	 * {S=java.lang.String, T=java.lang.Integer}.
366 	 */
367 	private static void populateTypeMapFromParameterizedType(ParameterizedType type, Map<TypeVariable, Type> typeVariableMap) {
368 		if (type.getRawType() instanceof Class) {
369 			Type[] actualTypeArguments = type.getActualTypeArguments();
370 			TypeVariable[] typeVariables = ((Class) type.getRawType()).getTypeParameters();
371 			for (int i = 0; i < actualTypeArguments.length; i++) {
372 				Type actualTypeArgument = actualTypeArguments[i];
373 				TypeVariable variable = typeVariables[i];
374 				if (actualTypeArgument instanceof Class) {
375 					typeVariableMap.put(variable, actualTypeArgument);
376 				}
377 				else if (actualTypeArgument instanceof GenericArrayType) {
378 					typeVariableMap.put(variable, actualTypeArgument);
379 				}
380 				else if (actualTypeArgument instanceof ParameterizedType) {
381 					typeVariableMap.put(variable, actualTypeArgument);
382 				}
383 				else if (actualTypeArgument instanceof TypeVariable) {
384 					// We have a type that is parameterized at instantiation time
385 					// the nearest match on the bridge method will be the bounded type.
386 					TypeVariable typeVariableArgument = (TypeVariable) actualTypeArgument;
387 					Type resolvedType = typeVariableMap.get(typeVariableArgument);
388 					if (resolvedType == null) {
389 						resolvedType = extractBoundForTypeVariable(typeVariableArgument);
390 					}
391 					typeVariableMap.put(variable, resolvedType);
392 				}
393 			}
394 		}
