31  /**
32   * <p>
33   * The Check validate abbreviations(consecutive capital letters) length in
34   * identifier name, it also allows to enforce camel case naming. Please read more at
35   * <a href="http://google-styleguide.googlecode.com/svn/trunk/javaguide.html#s5.3-camel-case">
36   * Google Style Guide</a> to get to know how to avoid long abbreviations in names.
37   * </p>
38   * <p>
39   * Option <code>allowedAbbreviationLength</code> indicates on the allowed amount of capital
40   * letters in abbreviations in the classes, interfaces,
41   * variables and methods names. Default value is '3'.
42   * </p>
43   * <p>
44   * Option <code>allowedAbbreviations</code> - list of abbreviations that
45   * must be skipped for checking. Abbreviations should be separated by comma,
46   * no spaces are allowed.
47   * </p>
48   * <p>
49   * Option <code>ignoreFinal</code> allow to skip variables with <code>final</code> modifier.
50   * Default value is <code>true</code>.
51   * </p>
52   * <p>
53   * Option <code>ignoreStatic</code> allow to skip variables with <code>static</code> modifier.
54   * Default value is <code>true</code>.
55   * </p>
56   * <p>
57   * Option <code>ignoreOverriddenMethod</code> - Allows to
58   * ignore methods tagged with <code>@Override</code> annotation
59   * (that usually mean inherited name). Default value is <code>true</code>.
60   * </p>
61   * Default configuration
62   * <pre>
63   * &lt;module name="AbbreviationAsWordInName" /&gt;
64   * </pre>
65   * <p>
66   * To configure to check variables and classes identifiers, do not ignore
67   * variables with static modifier
68   * and allow no abbreviations (enforce camel case phrase) but allow XML and URL abbreviations.
69   * </p>
70   * <pre>
71   * &lt;module name="AbbreviationAsWordInName"&gt;
72   *     &lt;property name="tokens" value="VARIABLE_DEF,CLASS_DEF"/&gt;
73   *     &lt;property name="ignoreStatic" value="false"/&gt;
74   *     &lt;property name="allowedAbbreviationLength" value="1"/&gt;
75   *     &lt;property name="allowedAbbreviations" value="XML,URL"/&gt;
76   * &lt;/module&gt;
77   * </pre>
78   *
79   * @author Roman Ivanov, Daniil Yaroslvtsev, Baratali Izmailov
80   */
81  public class AbbreviationAsWordInNameCheck extends Check

89      /**

92      private static final int DEFAULT_ALLOWED_ABBREVIATIONS_LENGTH = 3;

94      /**
95       * Variable indicates on the allowed amount of capital letters in
96       * abbreviations in the classes, interfaces, variables and methods names.
97       */
98      private int mAllowedAbbreviationLength =
99              DEFAULT_ALLOWED_ABBREVIATIONS_LENGTH;

101     /**
102      * Set of allowed abbreviation to ignore in check.
103      */
104     private Set<String> mAllowedAbbreviations = new HashSet<String>();

106     /** Allows to ignore variables with 'final' modifier. */
107     private boolean mIgnoreFinal = true;

155     /**
156      * Set a list of abbreviations that must be skipped for checking.

170     @Override
171     public int[] getDefaultTokens()
172     {
173         return new int[] {
174             TokenTypes.CLASS_DEF,
175             TokenTypes.INTERFACE_DEF,
176             TokenTypes.ENUM_DEF,
177             TokenTypes.ANNOTATION_DEF,
178             TokenTypes.ANNOTATION_FIELD_DEF,
179             TokenTypes.PARAMETER_DEF,
180             TokenTypes.VARIABLE_DEF,
181             TokenTypes.METHOD_DEF,
182         };

185     @Override
186     public void visitToken(DetailAST aAst)
187     {
188 
189         if (!isIgnoreSituation(aAst)) {
190 
191             final DetailAST nameAst = aAst.findFirstToken(TokenTypes.IDENT);
192             final String typeName = nameAst.getText();
193 
194             final String abbr = getDisallowedAbbreviation(typeName);
195             if (abbr != null) {
196                 log(nameAst.getLineNo(), MSG_KEY, mAllowedAbbreviationLength);
197             }
198         }
199     }

201     /**
202      * Checks if it is an ignore situation.
203      * @param aAst input DetailAST node.
204      * @return true if it is an ignore situation found for given input DetailAST
205      *         node.
206      */
207     private boolean isIgnoreSituation(DetailAST aAst)
208     {
209         final DetailAST modifiers = aAst.getFirstChild();
210 
211         boolean result = false;
212         if (aAst.getType() == TokenTypes.VARIABLE_DEF) {
213             if ((mIgnoreFinal || mIgnoreStatic)
214                     && isInterfaceDeclaration(aAst))
215             {
216                 // field declarations in interface are static/final
217                 result = true;
218             }
219             else {
220                 result = (mIgnoreFinal
221                           && modifiers.branchContains(TokenTypes.FINAL))
222                     || (mIgnoreStatic
223                         && modifiers.branchContains(TokenTypes.LITERAL_STATIC));
224             }
225         }
226         else if (aAst.getType() == TokenTypes.METHOD_DEF) {
227             result = mIgnoreOverriddenMethods
228                     && hasOverrideAnnotation(modifiers);
229         }
230         return result;
231     }

233     /**
234      * Check that variable definition in interface definition.
235      * @param aVariableDefAst variable definition.
236      * @return true if variable definition(aVaribaleDefAst) is in interface
237      * definition.
238      */
239     private static boolean isInterfaceDeclaration(DetailAST aVariableDefAst)
240     {
241         boolean result = false;
242         final DetailAST astBlock = aVariableDefAst.getParent();
243         if (astBlock != null) {
244             final DetailAST astParent2 = astBlock.getParent();
245             if (astParent2 != null
246                     && astParent2.getType() == TokenTypes.INTERFACE_DEF)
247             {
248                 result = true;
249             }
250         }
251         return result;
252     }

254     /**
255      * Checks that the method has "@Override" annotation.
256      * @param aMethodModifiersAST
257      *        A DetailAST nod is related to the given method modifiers
258      *        (MODIFIERS type).
259      * @return true if method has "@Override" annotation.
260      */
261     private static boolean hasOverrideAnnotation(DetailAST aMethodModifiersAST)
262     {
263         boolean result = false;
264         for (DetailAST child : getChildren(aMethodModifiersAST)) {
265             if (child.getType() == TokenTypes.ANNOTATION) {
266                 final DetailAST annotationIdent = child.findFirstToken(TokenTypes.IDENT);
267                 if (annotationIdent != null && "Override".equals(annotationIdent.getText())) {
268                     result = true;
269                     break;
270                 }
271             }
272         }
273         return result;
274     }

276     /**
277      * Gets the disallowed abbreviation contained in given String.
278      * @param aString
279      *        the given String.
280      * @return the disallowed abbreviation contained in given String as a
281      *         separate String.
282      */
283     private String getDisallowedAbbreviation(String aString)
284     {
285         int beginIndex = 0;
286         boolean abbrStarted = false;
287         String result = null;
288 
289         for (int index = 0; index < aString.length(); index++) {
290             final char symbol = aString.charAt(index);
291 
292             if (Character.isUpperCase(symbol)) {
293                 if (!abbrStarted) {
294                     abbrStarted = true;
295                     beginIndex = index;
296                 }
297             }
298             else {
299                 if (abbrStarted) {
300                     abbrStarted = false;
301 
302                     // -1 as a first capital is usually beginning of next word
303                     final int endIndex = index - 1;
304                     final int abbrLength = endIndex - beginIndex;
305                     if (abbrLength > mAllowedAbbreviationLength) {
306                         result = aString.substring(beginIndex, endIndex);
307                         if (!mAllowedAbbreviations.contains(result)) {
308                             break;
309                         }
310                         else {
311                             result = null;
312                         }
313                     }
314                     beginIndex = -1;
315                 }
316             }
317         }
318         if (abbrStarted) {
319             final int endIndex = aString.length();
320             final int abbrLength = endIndex - beginIndex;
321             if (abbrLength > 1 && abbrLength > mAllowedAbbreviationLength) {
322                 result = aString.substring(beginIndex, endIndex);
323                 if (mAllowedAbbreviations.contains(result)) {
324                     result = null;
325                 }
326             }
327         }
328         return result;
329     }

331     /**
332      * Gets all the children which are one level below on the current DetailAST
333      * parent node.
334      * @param aNode
335      *        Current parent node.
336      * @return The list of children one level below on the current parent node.
337      */
338     private static List<DetailAST> getChildren(final DetailAST aNode)
339     {
340         final List<DetailAST> result = new LinkedList<DetailAST>();
341         DetailAST curNode = aNode.getFirstChild();
342         while (curNode != null) {
343             result.add(curNode);
344             curNode = curNode.getNextSibling();
345         }
346         return result;
347     }
