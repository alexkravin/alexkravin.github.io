26  /**
27   * This check calculates the Non Commenting Source Statements (NCSS) metric for
28   * java source files and methods. The check adheres to the <a
29   * href="http://www.kclee.com/clemens/java/javancss/">JavaNCSS specification
30   * </a> and gives the same results as the JavaNCSS tool.
31   *
32   * The NCSS-metric tries to determine complexity of methods, classes and files
33   * by counting the non commenting lines. Roughly said this is (nearly)
34   * equivalent to counting the semicolons and opening curly braces.
35   *
36   * @author Lars KÃ¶dderitzsch
37   */
38  public class JavaNCSSCheck extends Check

131     @Override
132     public void beginTree(DetailAST aRootAST)
133     {
134         mCounters = new FastStack<Counter>();
135 
136         //add a counter for the file
137         mCounters.push(new Counter());
138     }

140     @Override
141     public void visitToken(DetailAST aAST)
142     {
143         final int tokenType = aAST.getType();
144 
145         if ((TokenTypes.CLASS_DEF == tokenType)
146             || (TokenTypes.METHOD_DEF == tokenType)
147             || (TokenTypes.CTOR_DEF == tokenType)
148             || (TokenTypes.STATIC_INIT == tokenType)
149             || (TokenTypes.INSTANCE_INIT == tokenType))
150         {
151             //add a counter for this class/method
152             mCounters.push(new Counter());
153         }
154 
155         //check if token is countable
156         if (isCountable(aAST)) {
157             //increment the stacked counters
158             for (final Counter c : mCounters) {
159                 c.increment();
160             }
161         }
162     }
163 
164     @Override
165     public void leaveToken(DetailAST aAST)
166     {
167         final int tokenType = aAST.getType();
168         if ((TokenTypes.METHOD_DEF == tokenType)
169             || (TokenTypes.CTOR_DEF == tokenType)
170             || (TokenTypes.STATIC_INIT == tokenType)
171             || (TokenTypes.INSTANCE_INIT == tokenType))
172         {
173             //pop counter from the stack
174             final Counter counter = mCounters.pop();
175 
176             final int count = counter.getCount();
177             if (count > mMethodMax) {
178                 log(aAST.getLineNo(), aAST.getColumnNo(), "ncss.method",
179                         count, mMethodMax);
180             }
181         }
182         else if (TokenTypes.CLASS_DEF == tokenType) {
183             //pop counter from the stack
184             final Counter counter = mCounters.pop();
185 
186             final int count = counter.getCount();
187             if (count > mClassMax) {
188                 log(aAST.getLineNo(), aAST.getColumnNo(), "ncss.class",
189                         count, mClassMax);
190             }
191         }
192     }
193 
194     @Override
195     public void finishTree(DetailAST aRootAST)
196     {
197         //pop counter from the stack
198         final Counter counter = mCounters.pop();
199 
200         final int count = counter.getCount();
201         if (count > mFileMax) {
202             log(aRootAST.getLineNo(), aRootAST.getColumnNo(), "ncss.file",
203                     count, mFileMax);
204         }
205     }

240     /**
241      * Checks if a token is countable for the ncss metric
242      *
243      * @param aAST
244      *            the AST
245      * @return true if the token is countable
246      */
247     private boolean isCountable(DetailAST aAST)
248     {
249         boolean countable = true;
250 
251         final int tokenType = aAST.getType();
252 
253         //check if an expression is countable
254         if (TokenTypes.EXPR == tokenType) {
255             countable = isExpressionCountable(aAST);
256         }
257         //check if an variable definition is countable
258         else if (TokenTypes.VARIABLE_DEF == tokenType) {
259             countable = isVariableDefCountable(aAST);
260         }
261         return countable;
262     }
263 
264     /**
265      * Checks if a variable definition is countable.
266      *
267      * @param aAST the AST
268      * @return true if the variable definition is countable, false otherwise
269      */
270     private boolean isVariableDefCountable(DetailAST aAST)
271     {
272         boolean countable = false;
273 
274         //count variable defs only if they are direct child to a slist or
275         // object block
276         final int parentType = aAST.getParent().getType();
277 
278         if ((TokenTypes.SLIST == parentType)
279             || (TokenTypes.OBJBLOCK == parentType))
280         {
281             final DetailAST prevSibling = aAST.getPreviousSibling();
282 
283             //is countable if no previous sibling is found or
284             //the sibling is no COMMA.
285             //This is done because multiple assignment on one line are countes
286             // as 1
287             countable = (prevSibling == null)
288                     || (TokenTypes.COMMA != prevSibling.getType());
289         }
290 
291         return countable;
292     }

294     /**
295      * Checks if an expression is countable for the ncss metric.
296      *
297      * @param aAST the AST
298      * @return true if the expression is countable, false otherwise
299      */
300     private boolean isExpressionCountable(DetailAST aAST)
301     {
302         boolean countable = true;
303 
304         //count expressions only if they are direct child to a slist (method
305         // body, for loop...)
306         //or direct child of label,if,else,do,while,for
307         final int parentType = aAST.getParent().getType();
308         switch (parentType) {
309         case TokenTypes.SLIST :
310         case TokenTypes.LABELED_STAT :
311         case TokenTypes.LITERAL_FOR :
312         case TokenTypes.LITERAL_DO :
313         case TokenTypes.LITERAL_WHILE :
314         case TokenTypes.LITERAL_IF :
315         case TokenTypes.LITERAL_ELSE :
316             //don't count if or loop conditions
317             final DetailAST prevSibling = aAST.getPreviousSibling();
318             countable = (prevSibling == null)
319                 || (TokenTypes.LPAREN != prevSibling.getType());
320             break;
321         default :
322             countable = false;
323             break;
324         }
325         return countable;
326     }
