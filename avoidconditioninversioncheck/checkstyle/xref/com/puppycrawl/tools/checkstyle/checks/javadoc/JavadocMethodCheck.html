55      /** compiled regexp to match Javadoc tags that take an argument * */
56      private static final Pattern MATCH_JAVADOC_ARG =
57          Utils.createPattern("@(throws|exception|param)\\s+(\\S+)\\s+\\S*");

60      private static final Pattern MATCH_JAVADOC_ARG_MULTILINE_START =
61          Utils.createPattern("@(throws|exception|param)\\s+(\\S+)\\s*$");
62  
63      /** compiled regexp to look for a continuation of the comment * */
64      private static final Pattern MATCH_JAVADOC_MULTILINE_CONT =

305     @Override
306     protected final void processAST(DetailAST aAST)
307     {
308         if ((aAST.getType() == TokenTypes.METHOD_DEF || aAST.getType() == TokenTypes.CTOR_DEF)
309             && (getMethodsNumberOfLine(aAST) <= mMinLineCount)
310             || hasAllowedAnnotations(aAST))
311         {
312             return;
313         }
314         final Scope theScope = calculateScope(aAST);
315         if (shouldCheck(aAST, theScope)) {
316             final FileContents contents = getFileContents();
317             final TextBlock cmt = contents.getJavadocBefore(aAST.getLineNo());
318 
319             if (cmt == null) {
320                 if (!isMissingJavadocAllowed(aAST)) {
321                     log(aAST, "javadoc.missing");
322                 }
323             }
324             else {
325                 checkComment(aAST, cmt);
326             }
327         }
328     }

330     /**
331      * Some javadoc.
332      * @param aMethodDef Some javadoc.
333      * @return Some javadoc.
334      */
335     private boolean hasAllowedAnnotations(DetailAST aMethodDef)
336     {
337         final DetailAST modifiersNode = aMethodDef.findFirstToken(TokenTypes.MODIFIERS);
338         DetailAST annotationNode = modifiersNode.findFirstToken(TokenTypes.ANNOTATION);
339         while (annotationNode != null && annotationNode.getType() == TokenTypes.ANNOTATION) {
340             DetailAST identNode = annotationNode.findFirstToken(TokenTypes.IDENT);
341             if (identNode == null) {
342                 identNode = annotationNode.findFirstToken(TokenTypes.DOT)
343                     .findFirstToken(TokenTypes.IDENT);
344             }
345             if (mAllowedAnnotations.contains(identNode.getText())) {
346                 return true;
347             }
348             annotationNode = annotationNode.getNextSibling();
349         }
350         return false;
351     }

353     /**
354      * Some javadoc.
355      * @param aMethodDef Some javadoc.
356      * @return Some javadoc.
357      */
358     private int getMethodsNumberOfLine(DetailAST aMethodDef)
359     {
360         int numberOfLines;
361         final DetailAST lcurly = aMethodDef.getLastChild();
362         final DetailAST rcurly = lcurly.getLastChild();
363 
364         if (lcurly.getFirstChild() == rcurly) {
365             numberOfLines = 1;
366         }
367         else {
368             numberOfLines = rcurly.getLineNo() - lcurly.getLineNo() - 1;
369         }
370         return numberOfLines;
371     }

381     /**
382      * The JavadocMethodCheck is about to report a missing Javadoc.
383      * This hook can be used by derived classes to allow a missing javadoc
384      * in some situations.  The default implementation checks
385      * <code>allowMissingJavadoc</code> and
386      * <code>allowMissingPropertyJavadoc</code> properties, do not forget
387      * to call <code>super.isMissingJavadocAllowed(aAST)</code> in case
388      * you want to keep this logic.
389      * @param aAST the tree node for the method or constructor.
390      * @return True if this method or constructor doesn't need Javadoc.
391      */
392     protected boolean isMissingJavadocAllowed(final DetailAST aAST)
393     {
394         return mAllowMissingJavadoc
395             || (mAllowMissingPropertyJavadoc
396                 && (isSetterMethod(aAST) || isGetterMethod(aAST)));
397     }

399     /**
400      * Whether we should check this node.
401      *
402      * @param aAST a given node.
403      * @param aScope the scope of the node.
404      * @return whether we should check a given node.
405      */
406     private boolean shouldCheck(final DetailAST aAST, final Scope aScope)
407     {
408         final Scope surroundingScope = ScopeUtils.getSurroundingScope(aAST);
409 
410         return aScope.isIn(mScope)
411                 && surroundingScope.isIn(mScope)
412                 && ((mExcludeScope == null) || !aScope.isIn(mExcludeScope)
413                     || !surroundingScope.isIn(mExcludeScope));
414     }

416     /**
417      * Checks the Javadoc for a method.
418      *
419      * @param aAST the token for the method
420      * @param aComment the Javadoc comment
421      */
422     private void checkComment(DetailAST aAST, TextBlock aComment)
423     {
424         final List<JavadocTag> tags = getMethodTags(aComment);
425 
426         if (hasShortCircuitTag(aAST, tags)) {
427             return;
428         }
429 
430         Iterator<JavadocTag> it = tags.iterator();
431         if (aAST.getType() != TokenTypes.ANNOTATION_FIELD_DEF) {
432             // Check for inheritDoc
433             boolean hasInheritDocTag = false;
434             while (it.hasNext() && !hasInheritDocTag) {
435                 hasInheritDocTag |= (it.next()).isInheritDocTag();
436             }
437 
438             checkParamTags(tags, aAST, !hasInheritDocTag);
439             checkThrowsTags(tags, getThrows(aAST), !hasInheritDocTag);
440             if (isFunction(aAST)) {
441                 checkReturnTag(tags, aAST.getLineNo(), !hasInheritDocTag);
442             }
443         }
444 
445         // Dump out all unused tags
446         it = tags.iterator();
447         while (it.hasNext()) {
448             final JavadocTag jt = it.next();
449             if (!jt.isSeeOrInheritDocTag()) {
450                 log(jt.getLineNo(), "javadoc.unusedTagGeneral");
451             }
452         }
453     }

455     /**
456      * Validates whether the Javadoc has a short circuit tag. Currently this is
457      * the inheritTag. Any errors are logged.
458      *
459      * @param aAST the construct being checked
460      * @param aTags the list of Javadoc tags associated with the construct
461      * @return true if the construct has a short circuit tag.
462      */
463     private boolean hasShortCircuitTag(final DetailAST aAST,
464             final List<JavadocTag> aTags)
465     {
466         // Check if it contains {@inheritDoc} tag
467         if ((aTags.size() != 1)
468                 || !(aTags.get(0)).isInheritDocTag())
469         {
470             return false;
471         }
472 
473         // Invalid if private, a constructor, or a static method
474         if (!JavadocTagInfo.INHERIT_DOC.isValidOn(aAST)) {
475             log(aAST, "javadoc.invalidInheritDoc");
476         }
477 
478         return true;
479     }

481     /**
482      * Returns the scope for the method/constructor at the specified AST. If
483      * the method is in an interface or annotation block, the scope is assumed
484      * to be public.
485      *
486      * @param aAST the token of the method/constructor
487      * @return the scope of the method/constructor
488      */
489     private Scope calculateScope(final DetailAST aAST)
490     {
491         final DetailAST mods = aAST.findFirstToken(TokenTypes.MODIFIERS);
492         final Scope declaredScope = ScopeUtils.getScopeFromMods(mods);
493         return ScopeUtils.inInterfaceOrAnnotationBlock(aAST) ? Scope.PUBLIC
494                 : declaredScope;
495     }

497     /**
498      * Returns the tags in a javadoc comment. Only finds throws, exception,
499      * param, return and see tags.
500      *
501      * @return the tags found
502      * @param aComment the Javadoc comment
503      */
504     private List<JavadocTag> getMethodTags(TextBlock aComment)
505     {
506         final String[] lines = aComment.getText();
507         final List<JavadocTag> tags = Lists.newArrayList();
508         int currentLine = aComment.getStartLineNo() - 1;
509 
510         for (int i = 0; i < lines.length; i++) {
511             currentLine++;
512             final Matcher javadocArgMatcher =
513                 MATCH_JAVADOC_ARG.matcher(lines[i]);
514             final Matcher javadocNoargMatcher =
515                 MATCH_JAVADOC_NOARG.matcher(lines[i]);
516             final Matcher noargCurlyMatcher =
517                 MATCH_JAVADOC_NOARG_CURLY.matcher(lines[i]);
518             final Matcher argMultilineStart =
519                 MATCH_JAVADOC_ARG_MULTILINE_START.matcher(lines[i]);
520             final Matcher noargMultilineStart =
521                 MATCH_JAVADOC_NOARG_MULTILINE_START.matcher(lines[i]);
522 
523             if (javadocArgMatcher.find()) {
524                 int col = javadocArgMatcher.start(1) - 1;
525                 if (i == 0) {
526                     col += aComment.getStartColNo();
527                 }
528                 tags.add(new JavadocTag(currentLine, col, javadocArgMatcher
529                         .group(1), javadocArgMatcher.group(2)));
530             }
531             else if (javadocNoargMatcher.find()) {
532                 int col = javadocNoargMatcher.start(1) - 1;
533                 if (i == 0) {
534                     col += aComment.getStartColNo();
535                 }
536                 tags.add(new JavadocTag(currentLine, col, javadocNoargMatcher
537                         .group(1)));
538             }
539             else if (noargCurlyMatcher.find()) {
540                 int col = noargCurlyMatcher.start(1) - 1;
541                 if (i == 0) {
542                     col += aComment.getStartColNo();
543                 }
544                 tags.add(new JavadocTag(currentLine, col, noargCurlyMatcher
545                         .group(1)));
546             }
547             else if (argMultilineStart.find()) {
548                 final String p1 = argMultilineStart.group(1);
549                 final String p2 = argMultilineStart.group(2);
550                 int col = argMultilineStart.start(1) - 1;
551                 if (i == 0) {
552                     col += aComment.getStartColNo();
553                 }
554 
555                 // Look for the rest of the comment if all we saw was
556                 // the tag and the name. Stop when we see '*/' (end of
557                 // Javadoc), '@' (start of next tag), or anything that's
558                 // not whitespace or '*' characters.
559                 int remIndex = i + 1;
560                 while (remIndex < lines.length) {
561                     final Matcher multilineCont = MATCH_JAVADOC_MULTILINE_CONT
562                             .matcher(lines[remIndex]);
563                     if (multilineCont.find()) {
564                         remIndex = lines.length;
565                         final String lFin = multilineCont.group(1);
566                         if (!lFin.equals(NEXT_TAG)
567                             && !lFin.equals(END_JAVADOC))
568                         {
569                             tags.add(new JavadocTag(currentLine, col, p1, p2));
570                         }
571                     }
572                     remIndex++;
573                 }
574             }
575             else if (noargMultilineStart.find()) {
576                 final String p1 = noargMultilineStart.group(1);
577                 int col = noargMultilineStart.start(1) - 1;
578                 if (i == 0) {
579                     col += aComment.getStartColNo();
580                 }
581 
582                 // Look for the rest of the comment if all we saw was
583                 // the tag and the name. Stop when we see '*/' (end of
584                 // Javadoc), '@' (start of next tag), or anything that's
585                 // not whitespace or '*' characters.
586                 int remIndex = i + 1;
587                 while (remIndex < lines.length) {
588                     final Matcher multilineCont = MATCH_JAVADOC_MULTILINE_CONT
589                             .matcher(lines[remIndex]);
590                     if (multilineCont.find()) {
591                         remIndex = lines.length;
592                         final String lFin = multilineCont.group(1);
593                         if (!lFin.equals(NEXT_TAG)
594                             && !lFin.equals(END_JAVADOC))
595                         {
596                             tags.add(new JavadocTag(currentLine, col, p1));
597                         }
598                     }
599                     remIndex++;
600                 }
601             }
602         }
603         return tags;
604     }

606     /**
607      * Computes the parameter nodes for a method.
608      *
609      * @param aAST the method node.
610      * @return the list of parameter nodes for aAST.
611      */
612     private List<DetailAST> getParameters(DetailAST aAST)
613     {
614         final DetailAST params = aAST.findFirstToken(TokenTypes.PARAMETERS);
615         final List<DetailAST> retVal = Lists.newArrayList();
616 
617         DetailAST child = params.getFirstChild();
618         while (child != null) {
619             if (child.getType() == TokenTypes.PARAMETER_DEF) {
620                 final DetailAST ident = child.findFirstToken(TokenTypes.IDENT);
621                 retVal.add(ident);
622             }
623             child = child.getNextSibling();
624         }
625         return retVal;
626     }

628     /**
629      * Computes the exception nodes for a method.
630      *
631      * @param aAST the method node.
632      * @return the list of exception nodes for aAST.
633      */
634     private List<ExceptionInfo> getThrows(DetailAST aAST)
635     {
636         final List<ExceptionInfo> retVal = Lists.newArrayList();
637         final DetailAST throwsAST = aAST
638                 .findFirstToken(TokenTypes.LITERAL_THROWS);
639         if (throwsAST != null) {
640             DetailAST child = throwsAST.getFirstChild();
641             while (child != null) {
642                 if ((child.getType() == TokenTypes.IDENT)
643                         || (child.getType() == TokenTypes.DOT))
644                 {
645                     final FullIdent fi = FullIdent.createFullIdent(child);
646                     final ExceptionInfo ei = new ExceptionInfo(new Token(fi),
647                             getCurrentClassName());
648                     retVal.add(ei);
649                 }
650                 child = child.getNextSibling();
651             }
652         }
653         return retVal;
654     }

656     /**
657      * Checks a set of tags for matching parameters.
658      *
659      * @param aTags the tags to check
660      * @param aParent the node which takes the parameters
661      * @param aReportExpectedTags whether we should report if do not find
662      *            expected tag
663      */
664     private void checkParamTags(final List<JavadocTag> aTags,
665             final DetailAST aParent, boolean aReportExpectedTags)
666     {
667         final List<DetailAST> params = getParameters(aParent);
668         final List<DetailAST> typeParams = CheckUtils
669                 .getTypeParameters(aParent);
670 
671         // Loop over the tags, checking to see they exist in the params.
672         final ListIterator<JavadocTag> tagIt = aTags.listIterator();
673         while (tagIt.hasNext()) {
674             final JavadocTag tag = tagIt.next();
675 
676             if (!tag.isParamTag()) {
677                 continue;
678             }
679 
680             tagIt.remove();
681 
682             boolean found = false;
683 
684             // Loop looking for matching param
685             final Iterator<DetailAST> paramIt = params.iterator();
686             while (paramIt.hasNext()) {
687                 final DetailAST param = paramIt.next();
688                 if (param.getText().equals(tag.getArg1())) {
689                     found = true;
690                     paramIt.remove();
691                     break;
692                 }
693             }
694 
695             if (tag.getArg1().startsWith("<") && tag.getArg1().endsWith(">")) {
696                 // Loop looking for matching type param
697                 final Iterator<DetailAST> typeParamsIt = typeParams.iterator();
698                 while (typeParamsIt.hasNext()) {
699                     final DetailAST typeParam = typeParamsIt.next();
700                     if (typeParam.findFirstToken(TokenTypes.IDENT).getText()
701                             .equals(
702                                     tag.getArg1().substring(1,
703                                             tag.getArg1().length() - 1)))
704                     {
705                         found = true;
706                         typeParamsIt.remove();
707                         break;
708                     }
709                 }
710 
711             }
712 
713             // Handle extra JavadocTag
714             if (!found) {
715                 log(tag.getLineNo(), tag.getColumnNo(), "javadoc.unusedTag",
716                         "@param", tag.getArg1());
717             }
718         }
719 
720         // Now dump out all type parameters/parameters without tags :- unless
721         // the user has chosen to suppress these problems
722         if (!mAllowMissingParamTags && aReportExpectedTags) {
723             for (DetailAST param : params) {
724                 log(param, "javadoc.expectedTag",
725                     JavadocTagInfo.PARAM.getText(), param.getText());
726             }
727 
728             for (DetailAST typeParam : typeParams) {
729                 log(typeParam, "javadoc.expectedTag",
730                     JavadocTagInfo.PARAM.getText(),
731                     "<" + typeParam.findFirstToken(TokenTypes.IDENT).getText()
732                     + ">");
733             }
734         }

793     /**
794      * Checks a set of tags for matching throws.
795      *
796      * @param aTags the tags to check
797      * @param aThrows the throws to check
798      * @param aReportExpectedTags whether we should report if do not find
799      *            expected tag
800      */
801     private void checkThrowsTags(List<JavadocTag> aTags,
802             List<ExceptionInfo> aThrows, boolean aReportExpectedTags)
803     {
804         // Loop over the tags, checking to see they exist in the throws.
805         // The foundThrows used for performance only
806         final Set<String> foundThrows = Sets.newHashSet();
807         final ListIterator<JavadocTag> tagIt = aTags.listIterator();
808         while (tagIt.hasNext()) {
809             final JavadocTag tag = tagIt.next();
810 
811             if (!tag.isThrowsTag()) {
812                 continue;
813             }
814 
815             tagIt.remove();
816 
817             // Loop looking for matching throw
818             final String documentedEx = tag.getArg1();
819             final Token token = new Token(tag.getArg1(), tag.getLineNo(), tag
820                     .getColumnNo());
821             final ClassInfo documentedCI = createClassInfo(token,
822                     getCurrentClassName());
823             boolean found = foundThrows.contains(documentedEx);
824 
825             // First look for matches on the exception name
826             ListIterator<ExceptionInfo> throwIt = aThrows.listIterator();
827             while (!found && throwIt.hasNext()) {
828                 final ExceptionInfo ei = throwIt.next();
829 
830                 if (ei.getName().getText().equals(
831                         documentedCI.getName().getText()))
832                 {
833                     found = true;
834                     ei.setFound();
835                     foundThrows.add(documentedEx);
836                 }
837             }
838 
839             // Now match on the exception type
840             throwIt = aThrows.listIterator();
841             while (!found && throwIt.hasNext()) {
842                 final ExceptionInfo ei = throwIt.next();
843 
844                 if (documentedCI.getClazz() == ei.getClazz()) {
845                     found = true;
846                     ei.setFound();
847                     foundThrows.add(documentedEx);
848                 }
849                 else if (mAllowThrowsTagsForSubclasses) {
850                     found = isSubclass(documentedCI.getClazz(), ei.getClazz());
851                 }
852             }
853 
854             // Handle extra JavadocTag.
855             if (!found) {
856                 boolean reqd = true;
857                 if (mAllowUndeclaredRTE) {
858                     reqd = !isUnchecked(documentedCI.getClazz());
859                 }
860 
861                 if (reqd && mValidateThrows) {
862                     log(tag.getLineNo(), tag.getColumnNo(),
863                         "javadoc.unusedTag",
864                         JavadocTagInfo.THROWS.getText(), tag.getArg1());
865 
866                 }
867             }
868         }
869 
870         // Now dump out all throws without tags :- unless
871         // the user has chosen to suppress these problems
872         if (!mAllowMissingThrowsTags && aReportExpectedTags) {
873             for (ExceptionInfo ei : aThrows) {
874                 if (!ei.isFound()) {
875                     final Token fi = ei.getName();
876                     log(fi.getLineNo(), fi.getColumnNo(),
877                         "javadoc.expectedTag",
878                         JavadocTagInfo.THROWS.getText(), fi.getText());
879                 }
880             }
881         }
882     }

884     /**
885      * Returns whether an AST represents a setter method.
886      * @param aAST the AST to check with
887      * @return whether the AST represents a setter method
888      */
889     private boolean isSetterMethod(final DetailAST aAST)
890     {
891         // Check have a method with exactly 7 children which are all that
892         // is allowed in a proper setter method which does not throw any
893         // exceptions.
894         if ((aAST.getType() != TokenTypes.METHOD_DEF)
895                 || (aAST.getChildCount() != MAX_CHILDREN))
896         {
897             return false;
898         }
899 
900         // Should I handle only being in a class????
901 
902         // Check the name matches format setX...
903         final DetailAST type = aAST.findFirstToken(TokenTypes.TYPE);
904         final String name = type.getNextSibling().getText();
905         if (!name.matches("^set[A-Z].*")) { // Depends on JDK 1.4
906             return false;
907         }
908 
909         // Check the return type is void
910         if (type.getChildCount(TokenTypes.LITERAL_VOID) == 0) {
911             return false;
912         }
913 
914         // Check that is had only one parameter
915         final DetailAST params = aAST.findFirstToken(TokenTypes.PARAMETERS);
916         if ((params == null)
917                 || (params.getChildCount(TokenTypes.PARAMETER_DEF) != 1))
918         {
919             return false;
920         }
921 
922         // Now verify that the body consists of:
923         // SLIST -> EXPR -> ASSIGN
924         // SEMI
925         // RCURLY
926         final DetailAST slist = aAST.findFirstToken(TokenTypes.SLIST);
927         if ((slist == null) || (slist.getChildCount() != BODY_SIZE)) {
928             return false;
929         }
930 
931         final AST expr = slist.getFirstChild();
932         if ((expr.getType() != TokenTypes.EXPR)
933                 || (expr.getFirstChild().getType() != TokenTypes.ASSIGN))
934         {
935             return false;
936         }
937 
938         return true;
939     }

941     /**
942      * Returns whether an AST represents a getter method.
943      * @param aAST the AST to check with
944      * @return whether the AST represents a getter method
945      */
946     private boolean isGetterMethod(final DetailAST aAST)
947     {
948         // Check have a method with exactly 7 children which are all that
949         // is allowed in a proper getter method which does not throw any
950         // exceptions.
951         if ((aAST.getType() != TokenTypes.METHOD_DEF)
952                 || (aAST.getChildCount() != MAX_CHILDREN))
953         {
954             return false;
955         }
956 
957         // Check the name matches format of getX or isX. Technically I should
958         // check that the format isX is only used with a boolean type.
959         final DetailAST type = aAST.findFirstToken(TokenTypes.TYPE);
960         final String name = type.getNextSibling().getText();
961         if (!name.matches("^(is|get)[A-Z].*")) { // Depends on JDK 1.4
962             return false;
963         }
964 
965         // Check the return type is void
966         if (type.getChildCount(TokenTypes.LITERAL_VOID) > 0) {
967             return false;
968         }
969 
970         // Check that is had only one parameter
971         final DetailAST params = aAST.findFirstToken(TokenTypes.PARAMETERS);
972         if ((params == null)
973                 || (params.getChildCount(TokenTypes.PARAMETER_DEF) > 0))
974         {
975             return false;
976         }
977 
978         // Now verify that the body consists of:
979         // SLIST -> RETURN
980         // RCURLY
981         final DetailAST slist = aAST.findFirstToken(TokenTypes.SLIST);
982         if ((slist == null) || (slist.getChildCount() != 2)) {
983             return false;
984         }
985 
986         final AST expr = slist.getFirstChild();
987         if ((expr.getType() != TokenTypes.LITERAL_RETURN)
988                 || (expr.getFirstChild().getType() != TokenTypes.EXPR))
989         {
990             return false;
991         }
992 
993         return true;
994     }
