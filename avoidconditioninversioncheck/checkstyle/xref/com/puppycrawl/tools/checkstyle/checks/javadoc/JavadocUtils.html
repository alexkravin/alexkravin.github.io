42      /** maps from a token name to value */
43      private static final ImmutableMap<String, Integer> TOKEN_NAME_TO_VALUE;
44      /** maps from a token value to name */
45      private static final String[] TOKEN_VALUE_TO_NAME;
46  
47      // Using reflection gets all token names and values from JavadocTokenTypes class
48      // and saves to TOKEN_NAME_TO_VALUE and TOKEN_VALUE_TO_NAME collections.
49      static {
50          final ImmutableMap.Builder<String, Integer> builder = ImmutableMap.builder();
51  
52          final Field[] fields = JavadocTokenTypes.class.getDeclaredFields();
53  
54          String[] tempTokenValueToName = new String[0];
55  
56          for (final Field f : fields) {
57  
58              // Only process public int fields.
59              if (!Modifier.isPublic(f.getModifiers())
60                      || f.getType() != Integer.TYPE)
61              {
62                  continue;
63              }
64  
65              final String name = f.getName();
66  
67              try {
68                  final int tokenValue = f.getInt(name);
69                  builder.put(name, tokenValue);
70                  if (tokenValue > tempTokenValueToName.length - 1) {
71                      final String[] temp = new String[tokenValue + 1];
72                      System.arraycopy(tempTokenValueToName, 0, temp, 0, tempTokenValueToName.length);
73                      tempTokenValueToName = temp;
74                  }
75                  if (tokenValue == -1) {
76                      tempTokenValueToName[0] = name;
77                  }
78                  else {
79                      tempTokenValueToName[tokenValue] = name;
80                  }
81              }
82              catch (Exception e) {
83                  throw new IllegalStateException("Failed to instantiate collection of Javadoc tokens"
84                          , e);
85              }
86          }
87  
88          TOKEN_NAME_TO_VALUE = builder.build();
89          TOKEN_VALUE_TO_NAME = tempTokenValueToName;
90      }

100     /**
101      * Gets validTags from a given piece of Javadoc.
102      * @param aCmt
103      *        the Javadoc comment to process.
104      * @param aTagType
105      *        the type of validTags we're interested in
106      * @return all standalone validTags from the given javadoc.
107      */
108     public static JavadocTags getJavadocTags(TextBlock aCmt,
109             JavadocTagType aTagType)
110     {
111         final String[] text = aCmt.getText();
112         final List<JavadocTag> tags = Lists.newArrayList();
113         final List<InvalidJavadocTag> invalidTags = Lists.newArrayList();
114         Pattern blockTagPattern =
115                 Utils.getPattern("/\\*{2,}\\s*@(\\p{Alpha}+)\\s");
116         for (int i = 0; i < text.length; i++) {
117             final String s = text[i];
118             final Matcher blockTagMatcher = blockTagPattern.matcher(s);
119             if ((aTagType.equals(JavadocTagType.ALL) || aTagType
120                     .equals(JavadocTagType.BLOCK)) && blockTagMatcher.find())
121             {
122                 final String tagName = blockTagMatcher.group(1);
123                 String content = s.substring(blockTagMatcher.end(1));
124                 if (content.endsWith("*/")) {
125                     content = content.substring(0, content.length() - 2);
126                 }
127                 final int line = aCmt.getStartLineNo() + i;
128                 int col = blockTagMatcher.start(1) - 1;
129                 if (i == 0) {
130                     col += aCmt.getStartColNo();
131                 }
132                 if (JavadocTagInfo.isValidName(tagName)) {
133                     tags.add(
134                             new JavadocTag(line, col, tagName, content.trim()));
135                 }
136                 else {
137                     invalidTags.add(new InvalidJavadocTag(line, col, tagName));
138                 }
139             }
140             // No block tag, so look for inline validTags
141             else if (aTagType.equals(JavadocTagType.ALL)
142                     || aTagType.equals(JavadocTagType.INLINE))
143             {
144                 // Match JavaDoc text after comment characters
145                 final Pattern commentPattern =
146                         Utils.getPattern("^\\s*(?:/\\*{2,}|\\*+)\\s*(.*)");
147                 final Matcher commentMatcher = commentPattern.matcher(s);
148                 final String commentContents;
149                 final int commentOffset; // offset including comment characters
150                 if (!commentMatcher.find()) {
151                     commentContents = s; // No leading asterisks, still valid
152                     commentOffset = 0;
153                 }
154                 else {
155                     commentContents = commentMatcher.group(1);
156                     commentOffset = commentMatcher.start(1) - 1;
157                 }
158                 final Pattern tagPattern =
159                         Utils.getPattern(".*?\\{@(\\p{Alpha}+)\\s+(.*?)\\}");
160                 final Matcher tagMatcher = tagPattern.matcher(commentContents);
161                 while (tagMatcher.find()) {
162                     if (tagMatcher.groupCount() == 2) {
163                         final String tagName = tagMatcher.group(1);
164                         final String tagValue = tagMatcher.group(2).trim();
165                         final int line = aCmt.getStartLineNo() + i;
166                         int col = commentOffset + (tagMatcher.start(1) - 1);
167                         if (i == 0) {
168                             col += aCmt.getStartColNo();
169                         }
170                         if (JavadocTagInfo.isValidName(tagName)) {
171                             tags.add(new JavadocTag(line, col, tagName,
172                                     tagValue));
173                         }
174                         else {
175                             invalidTags.add(new InvalidJavadocTag(line, col,
176                                     tagName));
177                         }
178                     }
179                     // else Error: Unexpected match count for inline JavaDoc
180                     // tag!
181                 }
182             }
183             blockTagPattern =
184                     Utils.getPattern("^\\s*\\**\\s*@(\\p{Alpha}+)\\s");
185         }
186         return new JavadocTags(tags, invalidTags);
187     }

202     /**
203      * Checks that aCommentContent starts with '*' javadoc comment identifier.
204      * @param aCommentContent
205      *        content of block comment
206      * @return true if aCommentContent starts with '*' javadoc comment
207      *         identifier.
208      */
209     public static boolean isJavadocComment(String aCommentContent)
210     {
211         boolean result = false;
212 
213         if (!aCommentContent.isEmpty()) {
214             final char docCommentIdentificator = aCommentContent.charAt(0);
215             result = docCommentIdentificator == '*';
216         }
217 
218         return result;
219     }

258     /**
259      * Returns the first child token that has a specified type.
260      * @param aNode
261      *        Javadoc AST node
262      * @param aType
263      *        the token type to match
264      * @return the matching token, or null if no match
265      */
266     public static DetailNode findFirstToken(DetailNode aNode, int aType)
267     {
268         DetailNode retVal = null;
269         for (DetailNode i = getFirstChild(aNode); i != null; i = getNextSibling(i)) {
270             if (i.getType() == aType) {
271                 retVal = i;
272                 break;
273             }
274         }
275         return retVal;
276     }

289     /**
290      * Checks whether aNode contains any node of specified type among children on any deep level.
291      *
292      * @param aNode DetailNode
293      * @param aType token type
294      * @return true if aNode contains any node of aType type among children on any deep level.
295      */
296     public static boolean branchContains(DetailNode aNode, int aType)
297     {
298         DetailNode curNode = aNode;
299         while (curNode != null) {
300 
301             if (aType == curNode.getType()) {
302                 return true;
303             }
304 
305             DetailNode toVisit = getFirstChild(curNode);
306             while ((curNode != null) && (toVisit == null)) {
307                 toVisit = getNextSibling(curNode);
308                 if (toVisit == null) {
309                     curNode = curNode.getParent();
310                 }
311             }
312 
313             if (curNode == toVisit) {
314                 break;
315             }
316 
317             curNode = toVisit;
318         }
319 
320         return false;
321     }

323     /**
324      * Gets next sibling of specified node.
325      *
326      * @param aNode DetailNode
327      * @return next sibling.
328      */
329     public static DetailNode getNextSibling(DetailNode aNode)
330     {
331         final DetailNode parent = aNode.getParent();
332         if (parent != null) {
333             final int nextSiblingIndex = aNode.getIndex() + 1;
334             final DetailNode[] children = parent.getChildren();
335             if (nextSiblingIndex <= children.length - 1) {
336                 return children[nextSiblingIndex];
337             }
338         }
339         return null;
340     }

342     /**
343      * Gets previous sibling of specified node.
344      * @param aNode DetailNode
345      * @return previous sibling
346      */
347     public static DetailNode getPreviousSibling(DetailNode aNode)
348     {
349         final DetailNode parent = aNode.getParent();
350         if (parent != null) {
351             final int previousSiblingIndex = aNode.getIndex() - 1;
352             final DetailNode[] children = parent.getChildren();
353             if (previousSiblingIndex >= 0) {
354                 return children[previousSiblingIndex];
355             }
356         }
357         return null;

360     /**
361      * Returns the name of a token for a given ID.
362      * @param aID
363      *        the ID of the token name to get
364      * @return a token name
365      */
366     public static String getTokenName(int aID)
367     {
368         if (aID == JavadocTokenTypes.EOF) {
369             return "EOF";
370         }
371         if (aID > TOKEN_VALUE_TO_NAME.length - 1) {
372             throw new IllegalArgumentException("Unknown javadoc token id. Given id: " + aID);
373         }
374         final String name = TOKEN_VALUE_TO_NAME[aID];
375         if (name == null) {
376             throw new IllegalArgumentException("Unknown javadoc token id. Given id: " + aID);
377         }
378         return name;
379     }

383      * @param aName

387     public static int getTokenId(String aName)
388     {
389         final Integer id = TOKEN_NAME_TO_VALUE.get(aName);

391             throw new IllegalArgumentException("Unknown javadoc token name. Given name " + aName);
392         }
