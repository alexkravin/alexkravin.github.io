30  /**
31   * <p>
32   * This Check warns on propagation of inner private types to outer classes:<br>
33   * - Externally accessible method if it returns private inner type.<br>
34   * - Externally accessible field if it's type is a private inner type.<br>
35   * These types could be <a href=
36   * 'http://docs.oracle.com/javase/tutorial/java/javaOO/nested.html'>private
37   * inner classes</a>, interfaces or enumerations.<br>
38   * <br>
39   * Examples: <code>
40   * <pre> 
41   * class OuterClass {
42   *  public InnerClass innerFromMain = new InnerClass(); //WARNING
43   *  private class InnerClass { ... } 
44   *  public InnerClass  getValue() { //WARNING
45   *      return new InnerClass(); 
46   *  }
47   *  <br> 
48   *  private interface InnerInterface { ... }
49   *  public Set&ltInnerInterface&gt getValue() { //WARNING
50   *      return new TreeSet&ltInnerInterface&gt;;
51   *  }
52   *  <br>
53   *  private Enum Fruit {Apple, Pear}
54   *  public Fruit getValue() { //WARNING
55   *      return Fruit.Apple;
56   *  }
57   *  <br>
58   *  public someMethod(InnerClass innerClass) { ... }  //WARNING
59   *  <br>
60   * }
61   * </pre>
62   * </code>
63   * <b>Rationale:</b> it is possible to return<br>
64   * private inner type or use it as the parameter of non-private method, but it
65   * is impossible<br>
66   * to use it in other classes (besides inner classes)<br>
67   * unless it extends or implements at least one <u>non-private</u> class or
68   * interface.<br>
69   * Such situation usually happens after bulk refactoring and usually means
70   * dead/useless code<br>
71   * </p>
72   * <br>
73   * @author <a href="mailto:nesterenko-aleksey@list.ru">Aleksey Nesterenko</a>
74   */
75  public class PublicReferenceToPrivateTypeCheck extends Check
76  {
77      /**
78       * Check message key for private types
79       */
80      public final static String MSG_KEY = "public.reference.to.private.type";
81  
82      /**
83       * List containing names of private types (classes, interfaces or enums).
84       */
85      private final Set<DetailAST> privateTypes = new HashSet<DetailAST>();
86  
87      /**
88       * List containing the names of types returned by public methods or fields.
89       */
90      private final Set<DetailAST> externallyReferencedTypes = new HashSet<DetailAST>();
91  
92      @Override
93      public int[] getDefaultTokens()
94      {
95          return new int[] { TokenTypes.CLASS_DEF, TokenTypes.METHOD_DEF,
96                  TokenTypes.INTERFACE_DEF, TokenTypes.ENUM_DEF,
97                  TokenTypes.VARIABLE_DEF };
98      }
99  
100     @Override
101     public void beginTree(DetailAST rootAST)
102     {
103         privateTypes.clear();
104         externallyReferencedTypes.clear();
105     }
106 
107     @Override
108     public void visitToken(DetailAST defAst)
109     {
110         switch (defAst.getType()) {
111         case TokenTypes.CLASS_DEF:
112         case TokenTypes.INTERFACE_DEF:
113         case TokenTypes.ENUM_DEF:
114             if (hasModifier(TokenTypes.LITERAL_PRIVATE, defAst)) {
115                 addPrivateTypes(defAst);
116             }
117             break;
118         case TokenTypes.METHOD_DEF:
119             if (isDefinedInTopLevelClass(defAst)
120                     && !hasModifier(TokenTypes.LITERAL_PRIVATE, defAst)) {
121                 addExternallyAccessibleMethodTypes(defAst);
122             }
123             break;
124         case TokenTypes.VARIABLE_DEF:
125             if (isDefinedInTopLevelClass(defAst)
126                     && !hasModifier(TokenTypes.LITERAL_PRIVATE, defAst)) {
127                 addExternallyAccessibleFieldTypes(defAst);
128             }
129             break;
130         default:
131             throw new IllegalArgumentException("Unsupported token type: "
132                     + defAst.getType());
133         }
134     }
135 
136     @Override
137     public void finishTree(DetailAST rootAst)
138     {
139         if (!privateTypes.isEmpty()) {
140             for (DetailAST privateType : privateTypes) {
141                 for (DetailAST outReturnedType : externallyReferencedTypes) {
142                     if (privateType.getText().equals(
143                             outReturnedType.getText())
144                             && !isExtendsOrImplementsSmth(privateType
145                                     .getParent())) {
146                         log(outReturnedType.getLineNo(), MSG_KEY,
147                                 outReturnedType.getText());
148                     }
149                 }
150             }
151         }
152     }
153 
154     /**
155      * Adds type to the list of private types.
156      * @param classOrInterfaceOrEnumDefAst
157      *        AST subtree that represent inner private type definition.
158      */
159     private void addPrivateTypes(DetailAST classOrInterfaceOrEnumDefAst)
160     {
161         final DetailAST definitionAst = classOrInterfaceOrEnumDefAst
162                 .findFirstToken(TokenTypes.IDENT);
163         privateTypes.add(definitionAst);
164     }
165 
166     /**
167      * Appends non-private, defined in top-level class method's returned or
168      * parameter type, or field's type to general list of out referenced types.
169      * @param methodDefAst
170      *        AST subtree that represent method definition.
171      */
172     private void addExternallyAccessibleMethodTypes(DetailAST methodDefAst)
173     {
174         DetailAST typeDefAst = methodDefAst
175                 .findFirstToken(TokenTypes.TYPE);
176         DetailAST parametersDefAst = methodDefAst
177                 .findFirstToken(TokenTypes.PARAMETERS);
178         externallyReferencedTypes.addAll(getMethodOrFieldReferencedTypes(typeDefAst));
179         externallyReferencedTypes.addAll(getMethodParameterTypes(parametersDefAst));
180     }
181 
182     /**
183      * Appends non-private, defined in top-level class field's type to general
184      * list of out referenced types.
185      * @param fieldDefAst
186      *        AST subtree that represent field definition.
187      */
188     private void addExternallyAccessibleFieldTypes(DetailAST fieldDefAst)
189     {
190         DetailAST typeDefAst = fieldDefAst.findFirstToken(TokenTypes.TYPE);
191         externallyReferencedTypes.addAll(getMethodOrFieldReferencedTypes(typeDefAst));
192     }
193 
194     /**
195      * Gets the return type of method or field type.
196      * @param typeAst
197      *        AST subtree to process.
198      */
199     private static List<DetailAST>
200             getMethodOrFieldReferencedTypes(DetailAST typeAst)
201     {
202 
203         DetailAST returnedType = null;
204         List<DetailAST> returnedTypes = new ArrayList<DetailAST>();
205         DetailAST currentNode = typeAst;
206         while (currentNode != null) {
207             if (currentNode.getType() == TokenTypes.IDENT) {
208                 returnedType = currentNode;
209                 returnedTypes.add(returnedType);
210             }
211             currentNode = getNextSubTreeNode(currentNode, typeAst);
212         }
213         return returnedTypes;
214     }
215 
216     /**
217      * Gets method's parameters types
218      * @param parametersDefAst
219      */
220     private static List<DetailAST>
221             getMethodParameterTypes(DetailAST parametersDefAst)
222     {
223         DetailAST parameterType = null;
224         List<DetailAST> parameterTypes = new ArrayList<DetailAST>();
225 
226         if (parametersDefAst.getFirstChild() != null) {
227             DetailAST currentNode = parametersDefAst;
228 
229             while (currentNode != null) {
230                 if (currentNode.getType() == TokenTypes.PARAMETER_DEF) {
231                     parameterType = currentNode;
232 
233                     while (parameterType != null) {
234                         parameterType = getNextSubTreeNode(parameterType,
235                                 currentNode);
236                         if (parameterType != null
237                                 && parameterType.getType() == TokenTypes.IDENT) {
238                             parameterTypes.add(parameterType);
239                         }
240                     }
241 
242                 }
243                 currentNode = getNextSubTreeNode(currentNode, parametersDefAst);
244             }
245 
246         }
247 
248         return parameterTypes;
249     }

251     /**
252      * Checks if defined type or interface extends or implements any
253      * <u>non-private type</u>.
254      * @param classOrInterfaceDefAst
255      * @return Method returns true if class extends or implements something.
256      */
257     private boolean isExtendsOrImplementsSmth(DetailAST classOrInterfaceDefAst)
258     {
259         return (classOrInterfaceDefAst
260                 .branchContains(TokenTypes.EXTENDS_CLAUSE)
261                 || classOrInterfaceDefAst
262                 .branchContains(TokenTypes.IMPLEMENTS_CLAUSE))
263                 && !isExtendsOrImplementsPrivate(classOrInterfaceDefAst);
264     }
265 
266     /**
267      * Checks if inner class or interface extends or implements <u>inner private
268      * type</u>
269      * @param classOrInterfaceDefAst
270      * @return true if extending or implementing type is in collection of inner
271      *         private types
272      */
273     private boolean
274             isExtendsOrImplementsPrivate(DetailAST classOrInterfaceDefAst)
275     {
276         boolean result = false;
277 
278         Set<String> inheritedTypesNamesSet = new HashSet<String>();
279         DetailAST currentNode = classOrInterfaceDefAst;
280 
281         while (currentNode != null) {
282             if (currentNode.getType() == TokenTypes.EXTENDS_CLAUSE
283                     || currentNode.getType() == TokenTypes.IMPLEMENTS_CLAUSE) {
284                 DetailAST implementingOrExtendingAst = currentNode;
285 
286                 while (implementingOrExtendingAst != null) {
287                     implementingOrExtendingAst = getNextSubTreeNode(
288                             implementingOrExtendingAst, currentNode);
289                     if (implementingOrExtendingAst != null
290                             && implementingOrExtendingAst.getType() == TokenTypes.IDENT) {
291                         inheritedTypesNamesSet
292                                 .add(implementingOrExtendingAst.getText());
293                     }
294                 }
295 
296             }
297             currentNode = getNextSubTreeNode(currentNode,
298                     classOrInterfaceDefAst);
299         }
300 
301         Set<String> existingPrivateTypes = new HashSet<String>();
302         for (DetailAST privateType : privateTypes) {
303             existingPrivateTypes.add(privateType.getText());
304         }
305         if (existingPrivateTypes.containsAll(inheritedTypesNamesSet)) {
306             result = true;
307         }
308 
309         return result;
310     }

312     /**
313      * Checks, class, interface, enumeration, method or field has "private"
314      * access modifier.
315      * @param modifierType
316      * @param defAst
317      */
318     public static boolean
319             hasModifier(int modifierType, DetailAST defAst)
320     {
321         DetailAST modifiersToken = defAst.getFirstChild();
322         return modifiersToken.findFirstToken(modifierType) != null;
323     }
324 
325     /**
326      * Checks if method or field is defined in top-level class
327      * @param methodOrFieldDefAst
328      * @return true if method is defined in top-level class
329      */
330     private static boolean
331             isDefinedInTopLevelClass(DetailAST methodOrFieldDefAst)
332     {
333         return methodOrFieldDefAst.getParent().getParent().getParent() == null;
334     }
335 
336     /**
337      * Gets the next node of a syntactical tree (child of a current node or
338      * sibling of a current node, or sibling of a parent of a current node)
339      * @param currentNode
340      *        Current node in considering
341      * @return Current node after bypassing
342      */
343     private static DetailAST
344             getNextSubTreeNode(DetailAST currentNode, DetailAST subTreeRoot)
345     {
346         DetailAST toVisit = currentNode.getFirstChild();
347         while (toVisit == null) {
348             toVisit = currentNode.getNextSibling();
349             if (toVisit == null) {
350                 if (currentNode.getParent().equals(subTreeRoot)) {
351                     break;
352                 }
353                 currentNode = currentNode.getParent();
354             }
355         }
356         currentNode = toVisit;
357         return currentNode;
358     }
