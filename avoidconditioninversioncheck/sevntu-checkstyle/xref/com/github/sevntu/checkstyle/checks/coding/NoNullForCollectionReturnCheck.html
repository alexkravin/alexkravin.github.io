29  /**
30   * <p>
31   * This check report you, when method, that must return array or collection,
32   * return null value instead of empty collection or empty array.
33   * </p>
34   * <p>
35   * Returning Null is error prone as developers forget to review method
36   * implementation or Java doc for nuances of Null meaning.
37   * </p>
38   * <p>
39   * List of collections set by default:
40   * AbstractCollection, AbstractList, AbstractQueue, AbstractSequentialList, AbstractSet,
41   * ArrayBlockingQueue, ArrayDeque, ArrayList, AttributeList, BeanContextServicesSupport, 
42   * BeanContextSupport, ConcurrentLinkedDeque, ConcurrentLinkedQueue, ConcurrentSkipListSet,
43   * CopyOnWriteArrayList, CopyOnWriteArraySet, DelayQueue, EnumSet, HashSet, JobStateReasons,
44   * LinkedBlockingDeque, LinkedBlockingQueue, LinkedHashSet, LinkedList, LinkedTransferQueue,
45   * PriorityBlockingQueue, PriorityQueue, RoleList, RoleUnresolvedList, Stack, SynchronousQueue,
46   * TreeSet, Vector, Collection, List, Map, Set.
47   * </p>
48   * @author <a href="mailto:IliaDubinin91@gmail.com">Ilja Dubinin</a>
49   */
50  public class NoNullForCollectionReturnCheck extends Check
51  {
52  	/**
53  	 * Warning message key.
54  	 */
55  	public static final String MSG_KEY = "no.null.for.collections";
56  	
57      /**
58       * <p>
59       * Default list of collection implementing classes.
60       * <p>
61       */
62      private static final String DEFAULT_COLLECTIONS = "AbstractCollection AbstractList AbstractQueue " +
63              "AbstractSequentialList AbstractSet ArrayBlockingQueue ArrayDeque ArrayList " +
64              "AttributeList BeanContextServicesSupport BeanContextSupport ConcurrentLinkedDeque " +
65              "ConcurrentLinkedQueue ConcurrentSkipListSet CopyOnWriteArrayList CopyOnWriteArraySet " +
66              "DelayQueue EnumSet HashSet JobStateReasons LinkedBlockingDeque LinkedBlockingQueue " +
67              "LinkedHashSet LinkedList LinkedTransferQueue PriorityBlockingQueue PriorityQueue " +
68              "RoleList RoleUnresolvedList Stack SynchronousQueue TreeSet Vector Collection List Map Set";
69      /**
70       * <p>
71       * List of collection, that will be check.
72       * </p>
73       */
74      private HashSet<String> mCollectionList = new HashSet<String>();
75  
76      /**
77       * <p>
78       * Search not only in return line.
79       * </p>
80       */
81      private boolean mSearchThroughMethodBody = false;
82  
83      /**
84       * <p>
85       * List of the method definition tokens, that returns collection.
86       * </p>
87       */
88      private LinkedList<DetailAST> mMethodDefs = new LinkedList<DetailAST>();
89  
90      public NoNullForCollectionReturnCheck()
91      {
92          super();
93          setCollectionList(DEFAULT_COLLECTIONS);
94      }
95      
96      /**
97       * <p>
98       * Setter for list of known collections.
99       * </p>
100      * @param aCollectionList
101      *        - line contains all collection names.
102      */
103     public void setCollectionList(String aCollectionList)
104     {
105         mCollectionList.clear();
106         for (String currentCollection : aCollectionList.split("\\s+"))
107         {
108             mCollectionList.add(currentCollection);
109         }
110     }
111 
112     /**
113      * <p>
114      * Setter for searching through body of the method.
115      * </p>
116      * @param aSearchThroughMethodBody
117      *        - deep search value.
118      */
119     public void setSearchThroughMethodBody(boolean aSearchThroughMethodBody)
120     {
121         this.mSearchThroughMethodBody = aSearchThroughMethodBody;
122     }

124     @Override
125     public int[] getDefaultTokens()
126     {
127         return new int[] { TokenTypes.METHOD_DEF, TokenTypes.LITERAL_RETURN };
128     }
129     
130     @Override
131     public void beginTree(DetailAST aRootAST)
132     {
133         mMethodDefs.clear();
134     }
135 
136     @Override
137     public void visitToken(DetailAST aDetailAST)
138     {
139         switch (aDetailAST.getType())
140         {
141         case TokenTypes.METHOD_DEF:
142         {
143             if (isReturnCollection(aDetailAST))
144             {
145                 mMethodDefs.push(aDetailAST);
146             }
147             break;
148         }
149         case TokenTypes.LITERAL_RETURN:
150         {
151             if (!mMethodDefs.isEmpty())
152             {
153                 DetailAST currentMethodDef = getMethodDef(aDetailAST);
154                 if (mMethodDefs.contains(currentMethodDef)
155                         && (hasNullLiteralInReturn(aDetailAST)
156                                 || (mSearchThroughMethodBody
157                                         && isReturnedValueBeNull(aDetailAST))))
158                 {
159                     log(aDetailAST.getLineNo(), MSG_KEY);
160                 }
161             }
162             break;
163         }
164         }
165     }

167     /**
168      * <p>
169      * Returns true, when method type is a collection or an array.
170      * </p>
171      * @param aMethodDef
172      *        - DetailAST contains method definition node.
173      * @return true, when method return collection.
174      */
175     private boolean isReturnCollection(DetailAST aMethodDef)
176     {
177         DetailAST methodType = aMethodDef.findFirstToken(TokenTypes.TYPE);
178         methodType = methodType.getFirstChild();
179         return methodType.getType() == TokenTypes.ARRAY_DECLARATOR
180                 || mCollectionList.contains(methodType.getText());
181     }

183     /**
184      * <p>
185      * Returns true when null literal has in return expression.
186      * </p>
187      * @param aReturnLit
188      *        - DetailAST contains return literal
189      * @return true when null literal has in return expression.
190      */
191     private static boolean hasNullLiteralInReturn(DetailAST aReturnLit)
192     {
193         DetailAST returnExpression = aReturnLit.findFirstToken(TokenTypes.EXPR);
194         boolean result = false; 
195         if (returnExpression != null)
196         {
197             DetailAST ternary = returnExpression.getFirstChild(); 
198             if (TokenTypes.QUESTION == ternary.getType())
199             {
200                 returnExpression = ternary;
201             }
202             result = returnExpression.getChildCount(TokenTypes.LITERAL_NULL) > 0;
203         }
204         return result;
205     }

207     /**
208      * <p>
209      * Returns true, when variable in return may be null.
210      * </p>
211      * @param aReturnLit
212      *        - DetailAST contains LITERAL_RETURN
213      * @return true, when variable may be null.
214      */
215     private static boolean isReturnedValueBeNull(DetailAST aReturnLit)
216     {
217         boolean result = false;
218         DetailAST returnedExpression = aReturnLit.getFirstChild();
219         if (returnedExpression.getType() != TokenTypes.SEMI)
220         {
221             DetailAST variable = returnedExpression.findFirstToken(TokenTypes.IDENT);
222             if (variable != null)
223             {
224                 String variableName = variable.getText();
225                 DetailAST methodDef = getMethodDef(aReturnLit);
226                 LinkedList<DetailAST> subblocks = getAllSubblocks(methodDef);
227                 subblocks.addFirst(methodDef);
228 
229                 result = hasNullInDefinition(subblocks, variableName);
230 
231                 //searching for not a null value into variable assignment
232                 if (result)
233                 {
234                     for (DetailAST subblock : subblocks)
235                     {
236                         List<DetailAST> expressions =
237                                 getChildren(getBlockBody(subblock), TokenTypes.EXPR);
238                         for (DetailAST expression : expressions)
239                         {
240                             DetailAST assign = expression.findFirstToken(TokenTypes.ASSIGN);
241                             if (assign != null
242                                     && variableName.equals(assign.findFirstToken(TokenTypes.IDENT).getText()) 
243                                         && !assign.branchContains(TokenTypes.LITERAL_NULL))
244                             {
245                                 result = false;
246                                 break;
247                             }
248                         }
249                         if (!result)
250                         {
251                             break;
252                         }
253                     }
254                 }
255             }
256         }
257         return result;
258     }

260     /**
261      * <p>
262      * Return all the nested subblocks in block.
263      * </p>
264      * @param aBlockDef
265      *        - node of the block.
266      * @return all the nested subblocks in block.
267      */
268     private static LinkedList<DetailAST> getAllSubblocks(DetailAST aBlockDef)
269     {
270         DetailAST blockBody = getBlockBody(aBlockDef);
271         LinkedList<DetailAST> subblocks = new LinkedList<DetailAST>();
272         subblocks.addAll(getChildren(blockBody, TokenTypes.LITERAL_IF));
273         LinkedList<DetailAST> elseBlocks = new LinkedList<DetailAST>();
274         for (DetailAST currentIf : subblocks)
275         {
276             if (currentIf.getChildCount(TokenTypes.LITERAL_ELSE) > 0)
277             {
278                 elseBlocks.add(currentIf.findFirstToken(TokenTypes.LITERAL_ELSE));
279             }
280         }
281         if (!elseBlocks.isEmpty())
282         {
283             subblocks.addAll(elseBlocks);
284         }
285         subblocks.addAll(getChildren(blockBody, TokenTypes.LITERAL_WHILE));
286         subblocks.addAll(getChildren(blockBody, TokenTypes.LITERAL_DO));
287         subblocks.addAll(getChildren(blockBody, TokenTypes.LITERAL_FOR));
288         subblocks.addAll(getChildren(blockBody, TokenTypes.LITERAL_TRY));
289         LinkedList<DetailAST> nestedSubblocks = new LinkedList<DetailAST>();
290         for (DetailAST currentSubblock : subblocks)
291         {
292             if (currentSubblock.branchContains(TokenTypes.SLIST))
293             {
294                 nestedSubblocks.addAll(getAllSubblocks(currentSubblock));
295             }
296         }
297         subblocks.addAll(nestedSubblocks);
298         return subblocks;
299     }

301     /**
302      * <p>
303      * Return true when variable is null into the variable definition.
304      * </p>
305      * @param aSubblocks
306      *        - list contains subblocks.
307      * @param aVariableName
308      *        - name of returned variable.
309      * @return true when variable is null into the variable definition.
310      */
311     private static boolean hasNullInDefinition(List<DetailAST> aSubblocks, String aVariableName)
312     {
313         boolean result = false;
314         for (DetailAST subblock : aSubblocks)
315         {
316             List<DetailAST> variableDefs =
317                     getChildren(getBlockBody(subblock), TokenTypes.VARIABLE_DEF);
318             boolean isFinded = false;
319             for (DetailAST currentDef : variableDefs)
320             {
321                 DetailAST variable = currentDef.findFirstToken(TokenTypes.IDENT);
322 
323                 if (aVariableName.equals(variable.getText()))
324                 {
325                     DetailAST variableDef = currentDef;
326                     DetailAST variableValue = variableDef.findFirstToken(TokenTypes.ASSIGN);
327                     if (variableValue != null)
328                     {
329                         variableValue = variableValue.findFirstToken(TokenTypes.EXPR);
330                         result = variableValue.getFirstChild().getType() == TokenTypes.LITERAL_NULL;
331                     }
332                     else
333                     {
334                         result = true;
335                     }
336                     isFinded = true;
337                     break;
338                 }
339             }
340             if (isFinded)
341             {
342                 break;
343             }
344         }
345         return result;
346     }

348     /**
349      * <p>
350      * Returns all children of that have the specified type.
351      * </p>
352      * @param aRoot
353      *        - root token of a block
354      * @param aType
355      *        - type of children
356      * @return all children of that have the specified type.
357      */
358     private static List<DetailAST> getChildren(DetailAST aRoot, int aType)
359     {
360         List<DetailAST> children = new LinkedList<DetailAST>();
361         DetailAST currentChild = aRoot.findFirstToken(aType);
362         if (currentChild != null)
363         {
364             children.add(currentChild);
365         }
366         while (children.size() < aRoot.getChildCount(aType))
367         {
368             currentChild = currentChild.getNextSibling();
369             if (currentChild.getType() == aType)
370             {
371                 children.add(currentChild);
372             }
373         }
374         return children;
375     }

377     /**
378      * <p>
379      * Return DetailAST that contained method definition.
380      * </p>
381      * @param aReturnLit
382      *        - DetailAST contains LITERAL_RETURN.
383      * @return DetailAST contains METHOD_DEF
384      */
385     private static DetailAST getMethodDef(DetailAST aReturnLit)
386     {
387         DetailAST methodDef = aReturnLit;
388         while (methodDef.getType() != TokenTypes.METHOD_DEF)
389         {
390             methodDef = methodDef.getParent();
391         }
392         return methodDef;
393     }

396      * <p>
397      * Return body of the block.
398      * </p>
399      * @param aBlockDef
400      *        - block definition node
401      * @return body of the block.
402      */
403     private static DetailAST getBlockBody(DetailAST aBlockDef)
404     {
405         DetailAST blockBody = aBlockDef.findFirstToken(TokenTypes.SLIST);
406         if (blockBody == null)
407         {
408             blockBody = aBlockDef;
409         }
410         return blockBody;
