35  /**
36   * Generated tests for {@link Monitor}.
37   *
38   * <p>This test class generates all of its own test cases in the {@link #suite()} method. Every
39   * {@code enterXxx}, {@code tryEnterXxx}, and {@code waitForXxx} method of the {@code Monitor} class
40   * is analyzed reflectively to determine appropriate test cases based on its signature. Additional
41   * ad hoc test cases can be found in {@link SupplementalMonitorTest}.
42   *
43   * @author Justin T. Sampson
44   */
45  
46  public class GeneratedMonitorTest extends TestCase {
47  
48    public static TestSuite suite() {
49      TestSuite suite = new TestSuite();
50  
51      Method[] methods = Monitor.class.getMethods();
52      sortMethods(methods);
53      for (Method method : methods) {
54        if (isAnyEnter(method) || isWaitFor(method)) {
55          validateMethod(method);
56          addTests(suite, method);
57        }
58      }
59  
60      assertEquals(548, suite.testCount());
61  
62      return suite;
63    }

109

113     SMALL(SMALL_TIMEOUT_MILLIS, SMALL_TIMEOUT_MILLIS + "ms"),
114     LARGE(UNEXPECTED_HANG_DELAY_MILLIS * 2, (2 * UNEXPECTED_HANG_DELAY_MILLIS) + "ms"),
115     MAX(Long.MAX_VALUE, "+oo");

121       this.millis = millis;
122       this.label = label;
123     }

126     public String toString() {
127       return label;
128     }

184   /**
185    * Identifies all enterXxx and tryEnterXxx methods.
186    */
187   private static boolean isAnyEnter(Method method) {
188     return method.getName().startsWith("enter") || method.getName().startsWith("tryEnter");
189   }
190 
191   /**
192    * Identifies just tryEnterXxx methods (a subset of {@link #isAnyEnter}), which never block.
193    */
194   private static boolean isTryEnter(Method method) {
195     return method.getName().startsWith("tryEnter");
196   }
197 
198   /**
199    * Identifies just enterIfXxx methods (a subset of {@link #isAnyEnter}), which are mostly like the
200    * enterXxx methods but behave like tryEnterXxx in some scenarios.
201    */
202   private static boolean isEnterIf(Method method) {
203     return method.getName().startsWith("enterIf");
204   }
205 
206   /**
207    * Identifies all waitForXxx methods, which must be called while occupying the monitor.
208    */
209   private static boolean isWaitFor(Method method) {
210     return method.getName().startsWith("waitFor");
211   }

213   /**
214    * Determines whether the given method takes a Guard as its first parameter.
215    */
216   private static boolean isGuarded(Method method) {
217     Class<?>[] parameterTypes = method.getParameterTypes();
218     return parameterTypes.length >= 1 && parameterTypes[0] == Monitor.Guard.class;
219   }
220 
221   /**
222    * Determines whether the given method takes a time and unit as its last two parameters.
223    */
224   private static boolean isTimed(Method method) {
225     Class<?>[] parameterTypes = method.getParameterTypes();
226     return parameterTypes.length >= 2
227         && parameterTypes[parameterTypes.length - 2] == long.class
228         && parameterTypes[parameterTypes.length - 1] == TimeUnit.class;
229   }

231   /**
232    * Determines whether the given method returns a boolean value.
233    */
234   private static boolean isBoolean(Method method) {
235     return method.getReturnType() == boolean.class;
236   }

238   /**
239    * Determines whether the given method can throw InterruptedException.
240    */
241   private static boolean isInterruptible(Method method) {
242     return Arrays.asList(method.getExceptionTypes()).contains(InterruptedException.class);
243   }

245   /**
246    * Sorts the given methods primarily by name and secondarily by number of parameters.
247    */
248   private static void sortMethods(Method[] methods) {
249     Arrays.sort(methods, new Comparator<Method>() {
250       @Override public int compare(Method m1, Method m2) {
251         int nameComparison = m1.getName().compareTo(m2.getName());
252         if (nameComparison != 0) {
253           return nameComparison;
254         } else {
255           return Ints.compare(m1.getParameterTypes().length, m2.getParameterTypes().length);
256         }
257       }
258     });
259   }

261   /**
262    * Validates that the given method's signature meets all of our assumptions.
263    */
264   private static void validateMethod(Method method) {
265     String desc = method.toString();
266 
267     assertTrue(desc, isAnyEnter(method) || isWaitFor(method));
268 
269     switch (method.getParameterTypes().length) {
270       case 0:
271         assertFalse(desc, isGuarded(method));
272         assertFalse(desc, isTimed(method));
273         break;
274       case 1:
275         assertTrue(desc, isGuarded(method));
276         assertFalse(desc, isTimed(method));
277         break;
278       case 2:
279         assertFalse(desc, isGuarded(method));
280         assertTrue(desc, isTimed(method));
281         break;
282       case 3:
283         assertTrue(desc, isGuarded(method));
284         assertTrue(desc, isTimed(method));
285         break;
286       default:
287         fail(desc);
288     }
289 
290     if (method.getReturnType() == void.class) {
291       assertFalse(desc, isBoolean(method));
292     } else {
293       assertTrue(desc, isBoolean(method));
294     }
295 
296     switch (method.getExceptionTypes().length) {
297       case 0:
298         assertFalse(desc, isInterruptible(method));
299         break;
300       case 1:
301         assertTrue(desc, isInterruptible(method));
302         break;
303       default:
304         fail(desc);
305     }
306 
307     if (isEnterIf(method)) {
308       assertTrue(desc, isGuarded(method));
309       assertTrue(desc, isBoolean(method));
310     } else if (isTryEnter(method)) {
311       assertFalse(desc, isTimed(method));
312       assertTrue(desc, isBoolean(method));
313       assertFalse(desc, isInterruptible(method));
314     } else if (isWaitFor(method)) {
315       assertTrue(desc, isGuarded(method));
316       assertEquals(desc, isTimed(method), isBoolean(method));
317     } else { // any other enterXxx method
318       assertEquals(desc, isTimed(method), isBoolean(method));
319     }
320   }

322   /**
323    * Generates all test cases appropriate for the given method.
324    */
325   private static void addTests(TestSuite suite, Method method) {
326     if (isGuarded(method)) {
327       for (boolean fair1 : new boolean[] { true, false }) {
328         for (boolean fair2 : new boolean[] { true, false }) {
329           suite.addTest(generateGuardWithWrongMonitorTestCase(method, fair1, fair2));
330         }
331       }
332     }
333     if (isAnyEnter(method)) {
334       addTests(suite, method,
335           Scenario.SATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
336           TimeoutsToUse.ANY,
337           Outcome.SUCCESS);
338       addTests(suite, method,
339           Scenario.UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
340           TimeoutsToUse.FINITE,
341           isGuarded(method)
342               ? (isBoolean(method) ? Outcome.FAILURE : Outcome.HANG)
343               : Outcome.SUCCESS);
344       addTests(suite, method,
345           Scenario.UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
346           TimeoutsToUse.INFINITE,
347           isGuarded(method)
348               ? (isTryEnter(method) || isEnterIf(method) ? Outcome.FAILURE : Outcome.HANG)
349               : Outcome.SUCCESS);
350       addTests(suite, method,
351           Scenario.SATISFIED_AND_OCCUPIED_BEFORE_ENTERING,
352           TimeoutsToUse.FINITE,
353           isBoolean(method) ? Outcome.FAILURE : Outcome.HANG);
354       addTests(suite, method,
355           Scenario.SATISFIED_AND_OCCUPIED_BEFORE_ENTERING,
356           TimeoutsToUse.INFINITE,
357           isGuarded(method)
358               ? Outcome.HANG
359               : (isTryEnter(method) ? Outcome.FAILURE : Outcome.HANG));
360       addTests(suite, method,
361           Scenario.SATISFIED_UNOCCUPIED_AND_INTERRUPTED_BEFORE_ENTERING,
362           TimeoutsToUse.ANY,
363           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.SUCCESS);
364     } else { // any waitForXxx method
365       suite.addTest(generateWaitForWhenNotOccupyingTestCase(method, true));
366       suite.addTest(generateWaitForWhenNotOccupyingTestCase(method, false));
367       addTests(suite, method,
368           Scenario.SATISFIED_BEFORE_WAITING,
369           TimeoutsToUse.ANY,
370           Outcome.SUCCESS);
371       addTests(suite, method,
372           Scenario.SATISFIED_WHILE_WAITING,
373           TimeoutsToUse.INFINITE,
374           Outcome.SUCCESS);
375       addTests(suite, method,
376           Scenario.SATISFIED_WHILE_WAITING,
377           TimeoutsToUse.PAST,
378           Outcome.FAILURE);
379       addTests(suite, method,
380           Scenario.SATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
381           TimeoutsToUse.ANY,
382           Outcome.SUCCESS);
383       addTests(suite, method,
384           Scenario.UNSATISFIED_BEFORE_AND_WHILE_WAITING,
385           TimeoutsToUse.FINITE,
386           Outcome.FAILURE);
387       addTests(suite, method,
388           Scenario.UNSATISFIED_BEFORE_AND_WHILE_WAITING,
389           TimeoutsToUse.INFINITE,
390           Outcome.HANG);
391       addTests(suite, method,
392           Scenario.UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
393           TimeoutsToUse.PAST,
394           // prefer responding to interrupt over timing out
395           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.FAILURE);
396       addTests(suite, method,
397           Scenario.UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
398           TimeoutsToUse.SMALL,
399           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.FAILURE);
400       addTests(suite, method,
401           Scenario.UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
402           TimeoutsToUse.INFINITE,
403           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.HANG);
404     }
405   }

407   /**
408    * Generates test cases for the given combination of scenario and timeouts. For methods that take
409    * an explicit timeout value, all of the given timeoutsToUse result in individual test cases. For
410    * methods that do not take an explicit timeout value, a single test case is generated only if the
411    * implicit timeout of that method matches the given timeoutsToUse. For example, enter() is
412    * treated like enter(MAX, MILLIS) and tryEnter() is treated like enter(0, MILLIS).
413    */
414   private static void addTests(TestSuite suite, Method method, Scenario scenario,
415       TimeoutsToUse timeoutsToUse, Outcome expectedOutcome) {
416     for (boolean fair : new boolean[] { true, false }) {
417       if (isTimed(method)) {
418         for (Timeout timeout : timeoutsToUse.timeouts) {
419           suite.addTest(new GeneratedMonitorTest(method, scenario, fair, timeout, expectedOutcome));
420         }
421       } else {
422         Timeout implicitTimeout = (isTryEnter(method) ? Timeout.ZERO : Timeout.MAX);
423         if (timeoutsToUse.timeouts.contains(implicitTimeout)) {
424           suite.addTest(new GeneratedMonitorTest(method, scenario, fair, null, expectedOutcome));
425         }
426       }
427     }
428   }

433   static class FlagGuard extends Monitor.Guard {
434 
435     private boolean satisfied;
436 
437     protected FlagGuard(Monitor monitor) {
438       super(monitor);
439     }
440 
441     @Override
442     public boolean isSatisfied() {
443       return satisfied;
444     }
445 
446     public void setSatisfied(boolean satisfied) {
447       this.satisfied = satisfied;
448     }

452   private final Method method;
453   private final Scenario scenario;
454   private final Timeout timeout;
455   private final Outcome expectedOutcome;
456   private final Monitor monitor;
457   private final FlagGuard guard;
458   private final CountDownLatch tearDownLatch;
459   private final CountDownLatch doingCallLatch;
460   private final CountDownLatch callCompletedLatch;
461 
462   private GeneratedMonitorTest(
463       Method method, Scenario scenario, boolean fair, Timeout timeout, Outcome expectedOutcome) {
464     super(nameFor(method, scenario, fair, timeout, expectedOutcome));
465     this.method = method;
466     this.scenario = scenario;
467     this.timeout = timeout;
468     this.expectedOutcome = expectedOutcome;
469     this.monitor = new Monitor(fair);
470     this.guard = new FlagGuard(monitor);
471     this.tearDownLatch = new CountDownLatch(1);
472     this.doingCallLatch = new CountDownLatch(1);
473     this.callCompletedLatch = new CountDownLatch(1);
474   }

476   private static String nameFor(
477       Method method, Scenario scenario, boolean fair, Timeout timeout, Outcome expectedOutcome) {
478     return String.format("%s%s(%s)/%s->%s",
479                          method.getName(),
480                          fair ? "(fair)" : "(nonfair)",
481                          (timeout == null) ? "untimed" : timeout,
482                          scenario,
483                          expectedOutcome);
484   }

486   @Override
487   protected void runTest() throws Throwable {
488     final Runnable runChosenTest = new Runnable() {
489       @Override public void run() { runChosenTest(); }
490     };
491     final FutureTask<Void> task = new FutureTask<Void>(runChosenTest, null);
492     startThread(new Runnable() {
493         @Override public void run() { task.run(); }
494       });
495     awaitUninterruptibly(doingCallLatch);
496     long hangDelayMillis = (expectedOutcome == Outcome.HANG)
497         ? EXPECTED_HANG_DELAY_MILLIS
498         : UNEXPECTED_HANG_DELAY_MILLIS;
499     boolean hung = !awaitUninterruptibly(
500         callCompletedLatch, hangDelayMillis, TimeUnit.MILLISECONDS);
501     if (hung) {
502       assertEquals(expectedOutcome, Outcome.HANG);
503     } else {
504       assertNull(task.get(UNEXPECTED_HANG_DELAY_MILLIS, TimeUnit.MILLISECONDS));
505     }
506   }

508   @Override
509   protected void tearDown() throws Exception {
510     // We don't want to leave stray threads running after each test. At this point, every thread
511     // launched by this test is either:
512     //
513     // (a) Blocked attempting to enter the monitor.
514     // (b) Waiting for the single guard to become satisfied.
515     // (c) Occupying the monitor and awaiting the tearDownLatch.
516     //
517     // Except for (c), every thread should occupy the monitor very briefly, and every thread leaves
518     // the monitor with the guard satisfied. Therefore as soon as tearDownLatch is triggered, we
519     // should be able to enter the monitor, and then we set the guard to satisfied for the benefit
520     // of any remaining waiting threads.
521 
522     tearDownLatch.countDown();
523     assertTrue("Monitor still occupied in tearDown()",
524         monitor.enter(UNEXPECTED_HANG_DELAY_MILLIS, TimeUnit.MILLISECONDS));
525     try {
526       guard.setSatisfied(true);
527     } finally {
528       monitor.leave();
529     }
530   }

532   private void runChosenTest() {
533     if (isAnyEnter(method)) {
534       runEnterTest();
535     } else {
536       runWaitTest();
537     }
538   }

540   private void runEnterTest() {
541     assertFalse(Thread.currentThread().isInterrupted());
542     assertFalse(monitor.isOccupiedByCurrentThread());
543 
544     doEnterScenarioSetUp();
545 
546     boolean interruptedBeforeCall = Thread.currentThread().isInterrupted();
547     Outcome actualOutcome = doCall();
548     boolean occupiedAfterCall = monitor.isOccupiedByCurrentThread();
549     boolean interruptedAfterCall = Thread.currentThread().isInterrupted();
550 
551     if (occupiedAfterCall) {
552       guard.setSatisfied(true);
553       monitor.leave();
554       assertFalse(monitor.isOccupiedByCurrentThread());
555     }
556 
557     assertEquals(expectedOutcome, actualOutcome);
558     assertEquals(expectedOutcome == Outcome.SUCCESS, occupiedAfterCall);
559     assertEquals(interruptedBeforeCall && expectedOutcome != Outcome.INTERRUPT,
560         interruptedAfterCall);
561   }
562 
563   private void doEnterScenarioSetUp() {
564     switch (scenario) {
565       case SATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING:
566         enterSatisfyGuardAndLeaveInCurrentThread();
567         break;
568       case UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING:
569         break;
570       case SATISFIED_AND_OCCUPIED_BEFORE_ENTERING:
571         enterSatisfyGuardAndLeaveInCurrentThread();
572         enterAndRemainOccupyingInAnotherThread();
573         break;
574       case SATISFIED_UNOCCUPIED_AND_INTERRUPTED_BEFORE_ENTERING:
575         enterSatisfyGuardAndLeaveInCurrentThread();
576         Thread.currentThread().interrupt();
577         break;
578       default:
579         throw new AssertionError("unsupported scenario: " + scenario);
580     }
581   }
582 
583   private void runWaitTest() {
584     assertFalse(Thread.currentThread().isInterrupted());
585     assertFalse(monitor.isOccupiedByCurrentThread());
586     monitor.enter();
587     try {
588       assertTrue(monitor.isOccupiedByCurrentThread());
589 
590       doWaitScenarioSetUp();
591 
592       boolean interruptedBeforeCall = Thread.currentThread().isInterrupted();
593       Outcome actualOutcome = doCall();
594       boolean occupiedAfterCall = monitor.isOccupiedByCurrentThread();
595       boolean interruptedAfterCall = Thread.currentThread().isInterrupted();
596 
597       assertEquals(expectedOutcome, actualOutcome);
598       assertTrue(occupiedAfterCall);
599       assertEquals(interruptedBeforeCall && expectedOutcome != Outcome.INTERRUPT,
600           interruptedAfterCall);
601     } finally {
602       guard.setSatisfied(true);
603       monitor.leave();
604       assertFalse(monitor.isOccupiedByCurrentThread());
605     }
606   }
607 
608   private void doWaitScenarioSetUp() {
609     switch (scenario) {
610       case SATISFIED_BEFORE_WAITING:
611         guard.setSatisfied(true);
612         break;
613       case SATISFIED_WHILE_WAITING:
614         guard.setSatisfied(false);
615         enterSatisfyGuardAndLeaveInAnotherThread(); // enter blocks until we call waitFor
616         break;
617       case UNSATISFIED_BEFORE_AND_WHILE_WAITING:
618         guard.setSatisfied(false);
619         break;
620       case SATISFIED_AND_INTERRUPTED_BEFORE_WAITING:
621         guard.setSatisfied(true);
622         Thread.currentThread().interrupt();
623         break;
624       case UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING:
625         guard.setSatisfied(false);
626         Thread.currentThread().interrupt();
627         break;
628       default:
629         throw new AssertionError("unsupported scenario: " + scenario);
630     }
631   }

633   private Outcome doCall() {
634     boolean guarded = isGuarded(method);
635     boolean timed = isTimed(method);
636     Object[] arguments = new Object[(guarded ? 1 : 0) + (timed ? 2 : 0)];
637     if (guarded) {
638       arguments[0] = guard;
639     }
640     if (timed) {
641       arguments[arguments.length - 2] = timeout.millis;
642       arguments[arguments.length - 1] = TimeUnit.MILLISECONDS;
643     }
644     try {
645       Object result;
646       doingCallLatch.countDown();
647       try {
648         result = method.invoke(monitor, arguments);
649       } finally {
650         callCompletedLatch.countDown();
651       }
652       if (result == null) {
653         return Outcome.SUCCESS;
654       } else if ((Boolean) result) {
655         return Outcome.SUCCESS;
656       } else {
657         return Outcome.FAILURE;
658       }
659     } catch (InvocationTargetException targetException) {
660       Throwable actualException = targetException.getTargetException();
661       if (actualException instanceof InterruptedException) {
662         return Outcome.INTERRUPT;
663       } else {
664         throw new AssertionError("unexpected exception", targetException);
665       }
666     } catch (IllegalAccessException e) {
667       throw new AssertionError("unexpected exception", e);
668     }
669   }
670 
671   private void enterSatisfyGuardAndLeaveInCurrentThread() {
672     monitor.enter();
673     try {
674       guard.setSatisfied(true);
675     } finally {
676       monitor.leave();
677     }
678   }
679 
680   private void enterSatisfyGuardAndLeaveInAnotherThread() {
681     final CountDownLatch startedLatch = new CountDownLatch(1);
682     startThread(new Runnable() {
683       @Override public void run() {
684         startedLatch.countDown();
685         enterSatisfyGuardAndLeaveInCurrentThread();
686       }
687     });
688     awaitUninterruptibly(startedLatch);

691   private void enterAndRemainOccupyingInAnotherThread() {
692     final CountDownLatch enteredLatch = new CountDownLatch(1);
693     startThread(new Runnable() {
694       @Override public void run() {
695         monitor.enter();
696         try {
697           enteredLatch.countDown();
698           awaitUninterruptibly(tearDownLatch);
699           guard.setSatisfied(true);
700         } finally {
701           monitor.leave();
702         }
703       }
704     });
705     awaitUninterruptibly(enteredLatch);
706   }
