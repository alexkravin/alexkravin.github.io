147 	/** Easymock <code>ArgumentMatcher</code> implementation that matches SAX arguments. */
148 	protected static class SaxArgumentMatcher extends AbstractMatcher {
149 
150 		@Override
151 		public boolean matches(Object[] expected, Object[] actual) {
152 			if (expected == actual) {
153 				return true;
154 			}
155 			if (expected == null || actual == null) {
156 				return false;
157 			}
158 			if (expected.length != actual.length) {
159 				throw new IllegalArgumentException("Expected and actual arguments must have the same size");
160 			}
161 			if (expected.length == 3 && expected[0] instanceof char[] && expected[1] instanceof Integer &&
162 					expected[2] instanceof Integer) {
163 				// handling of the character(char[], int, int) methods
164 				String expectedString = new String((char[]) expected[0], (Integer) expected[1], (Integer) expected[2]);
165 				String actualString = new String((char[]) actual[0], (Integer) actual[1], (Integer) actual[2]);
166 				return expectedString.equals(actualString);
167 			}
168 			else if (expected.length == 1 && (expected[0] instanceof Locator)) {
169 				return true;
170 			}
171 			else {
172 				return super.matches(expected, actual);
173 			}
174 		}
175 
176 		@Override
177 		protected boolean argumentMatches(Object expected, Object actual) {
178 			if (expected instanceof char[]) {
179 				return Arrays.equals((char[]) expected, (char[]) actual);
180 			}
181 			else if (expected instanceof Attributes) {
182 				Attributes expectedAttributes = (Attributes) expected;
183 				Attributes actualAttributes = (Attributes) actual;
184 				if (expectedAttributes.getLength() != actualAttributes.getLength()) {
185 					return false;
186 				}
187 				for (int i = 0; i < expectedAttributes.getLength(); i++) {
188 					boolean found = false;
189 					for (int j = 0; j < actualAttributes.getLength(); j++) {
190 						if (expectedAttributes.getURI(i).equals(actualAttributes.getURI(j)) &&
191 								expectedAttributes.getQName(i).equals(actualAttributes.getQName(j)) &&
192 								expectedAttributes.getType(i).equals(actualAttributes.getType(j)) &&
193 								expectedAttributes.getValue(i).equals(actualAttributes.getValue(j))) {
194 							found = true;
195 							break;
196 						}
197 					}
198 					if (!found) {
199 						return false;
200 					}
201 				}
202 				return true;
203 			}
204 			else {
205 				return super.argumentMatches(expected, actual);
206 			}
207 		}
208 
209 		@Override
210 		public String toString(Object[] arguments) {
211 			if (arguments != null && arguments.length == 3 && arguments[0] instanceof char[] &&
212 					arguments[1] instanceof Integer && arguments[2] instanceof Integer) {
213 				return new String((char[]) arguments[0], (Integer) arguments[1], (Integer) arguments[2]);
214 			}
215 			else {
216 				return super.toString(arguments);
217 			}
218 		}
219 
220 		@Override
221 		protected String argumentToString(Object argument) {
222 			if (argument instanceof char[]) {
223 				char[] array = (char[]) argument;
224 				StringBuilder buffer = new StringBuilder();
225 				for (char anArray : array) {
226 					buffer.append(anArray);
227 				}
228 				return buffer.toString();
229 			}
230 			else if (argument instanceof Attributes) {
231 				Attributes attributes = (Attributes) argument;
232 				StringBuilder buffer = new StringBuilder("[");
233 				for (int i = 0; i < attributes.getLength(); i++) {
234 					if (attributes.getURI(i).length() != 0) {
235 						buffer.append('{');
236 						buffer.append(attributes.getURI(i));
237 						buffer.append('}');
238 					}
239 					if (attributes.getQName(i).length() != 0) {
240 						buffer.append(attributes.getQName(i));
241 					}
242 					buffer.append('=');
243 					buffer.append(attributes.getValue(i));
244 					if (i < attributes.getLength() - 1) {
245 						buffer.append(", ");
246 					}
247 				}
248 				buffer.append(']');
249 				return buffer.toString();
250 			}
251 			else if (argument instanceof Locator) {
252 				Locator locator = (Locator) argument;
253 				StringBuilder buffer = new StringBuilder("[");
254 				buffer.append(locator.getLineNumber());
255 				buffer.append(',');
256 				buffer.append(locator.getColumnNumber());
257 				buffer.append(']');
258 				return buffer.toString();
259 			}
260 			else {
261 				return super.argumentToString(argument);
262 			}
263 		}
264 	}
