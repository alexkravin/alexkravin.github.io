49  /**
50   * SAX <code>XMLReader</code> that reads from a StAX <code>XMLEventReader</code>. Consumes <code>XMLEvents</code> from
51   * an <code>XMLEventReader</code>, and calls the corresponding methods on the SAX callback interfaces.
52   *
53   * @author Arjen Poutsma
54   * @see XMLEventReader
55   * @see #setContentHandler(org.xml.sax.ContentHandler)
56   * @see #setDTDHandler(org.xml.sax.DTDHandler)

73  	/**
74  	 * Constructs a new instance of the <code>StaxEventXmlReader</code> that reads from the given
75  	 * <code>XMLEventReader</code>. The supplied event reader must be in <code>XMLStreamConstants.START_DOCUMENT</code> or
76  	 * <code>XMLStreamConstants.START_ELEMENT</code> state.
77  	 *
78  	 * @param reader the <code>XMLEventReader</code> to read from
79  	 * @throws IllegalStateException if the reader is not at the start of a document or element
80  	 */
81  	StaxEventXMLReader(XMLEventReader reader) {
82  		Assert.notNull(reader, "'reader' must not be null");
83  		try {
84  			XMLEvent event = reader.peek();
85  			if (event != null && !(event.isStartDocument() || event.isStartElement())) {
86  				throw new IllegalStateException("XMLEventReader not at start of document or element");
87  			}
88  		}
89  		catch (XMLStreamException ex) {
90  			throw new IllegalStateException("Could not read first element: " + ex.getMessage());
91  		}
92  
93  		this.reader = reader;
94  	}

96  	@Override
97  	protected void parseInternal() throws SAXException, XMLStreamException {
98  		boolean documentStarted = false;
99  		boolean documentEnded = false;
100 		int elementDepth = 0;
101 		while (reader.hasNext() && elementDepth >= 0) {
102 			XMLEvent event = reader.nextEvent();
103 			if (!event.isStartDocument() && !event.isEndDocument() && !documentStarted) {
104 				handleStartDocument(event);
105 				documentStarted = true;
106 			}
107 			switch (event.getEventType()) {
108 				case XMLStreamConstants.START_DOCUMENT:
109 					handleStartDocument(event);
110 					documentStarted = true;
111 					break;
112 				case XMLStreamConstants.START_ELEMENT:
113 					elementDepth++;
114 					handleStartElement(event.asStartElement());
115 					break;
116 				case XMLStreamConstants.END_ELEMENT:
117 					elementDepth--;
118 					if (elementDepth >= 0) {
119 						handleEndElement(event.asEndElement());
120 					}
121 					break;
122 				case XMLStreamConstants.PROCESSING_INSTRUCTION:
123 					handleProcessingInstruction((ProcessingInstruction) event);
124 					break;
125 				case XMLStreamConstants.CHARACTERS:
126 				case XMLStreamConstants.SPACE:
127 				case XMLStreamConstants.CDATA:
128 					handleCharacters(event.asCharacters());
129 					break;
130 				case XMLStreamConstants.END_DOCUMENT:
131 					handleEndDocument();
132 					documentEnded = true;
133 					break;
134 				case XMLStreamConstants.NOTATION_DECLARATION:
135 					handleNotationDeclaration((NotationDeclaration) event);
136 					break;
137 				case XMLStreamConstants.ENTITY_DECLARATION:
138 					handleEntityDeclaration((EntityDeclaration) event);
139 					break;
140 				case XMLStreamConstants.COMMENT:
141 					handleComment((Comment) event);
142 					break;
143 				case XMLStreamConstants.DTD:
144 					handleDtd((DTD) event);
145 					break;
146 				case XMLStreamConstants.ENTITY_REFERENCE:
147 					handleEntityReference((EntityReference) event);
148 					break;
149 			}
150 		}
151 		if (documentStarted && !documentEnded) {
152 			handleEndDocument();
153 		}
154 
155 	}

157 	private void handleStartDocument(final XMLEvent event) throws SAXException {
158 		if (event.isStartDocument()) {
159 			StartDocument startDocument = (StartDocument) event;
160 			String xmlVersion = startDocument.getVersion();
161 			if (StringUtils.hasLength(xmlVersion)) {
162 				this.xmlVersion = xmlVersion;
163 			}
164 			if (startDocument.encodingSet()) {
165 				this.encoding = startDocument.getCharacterEncodingScheme();
166 			}
167 		}
168 
169 		if (getContentHandler() != null) {
170 			final Location location = event.getLocation();
171 			getContentHandler().setDocumentLocator(new Locator2() {
172 
173 				public int getColumnNumber() {
174 					return location != null ? location.getColumnNumber() : -1;
175 				}
176 
177 				public int getLineNumber() {
178 					return location != null ? location.getLineNumber() : -1;
179 				}
180 
181 				public String getPublicId() {
182 					return location != null ? location.getPublicId() : null;
183 				}
184 
185 				public String getSystemId() {
186 					return location != null ? location.getSystemId() : null;
187 				}
188 
189 				public String getXMLVersion() {
190 					return xmlVersion;
191 				}
192 
193 				public String getEncoding() {
194 					return encoding;
195 				}
196 			});
197 
198 			getContentHandler().startDocument();
199 		}
200 	}

202 	private void handleStartElement(StartElement startElement) throws SAXException {
203 		if (getContentHandler() != null) {
204 			QName qName = startElement.getName();
205 			if (hasNamespacesFeature()) {
206 				for (Iterator i = startElement.getNamespaces(); i.hasNext();) {
207 					Namespace namespace = (Namespace) i.next();
208 					startPrefixMapping(namespace.getPrefix(), namespace.getNamespaceURI());
209 				}
210 				for (Iterator i = startElement.getAttributes(); i.hasNext();){
211 					Attribute attribute = (Attribute) i.next();
212 					QName attributeName = attribute.getName();
213 					startPrefixMapping(attributeName.getPrefix(), attributeName.getNamespaceURI());
214 				}
215 
216 				getContentHandler().startElement(qName.getNamespaceURI(), qName.getLocalPart(), toQualifiedName(qName),
217 						getAttributes(startElement));
218 			}
219 			else {
220 				getContentHandler().startElement("", "", toQualifiedName(qName), getAttributes(startElement));
221 			}
222 		}
223 	}

225 	private void handleCharacters(Characters characters) throws SAXException {
226 		char[] data = characters.getData().toCharArray();
227 		if (getContentHandler() != null && characters.isIgnorableWhiteSpace()) {
228 			getContentHandler().ignorableWhitespace(data, 0, data.length);
229 			return;
230 		}
231 		if (characters.isCData() && getLexicalHandler() != null) {
232 			getLexicalHandler().startCDATA();
233 		}
234 		if (getContentHandler() != null) {
235 			getContentHandler().characters(data, 0, data.length);
236 		}
237 		if (characters.isCData() && getLexicalHandler() != null) {
238 			getLexicalHandler().endCDATA();
239 		}
240 	}

242 	private void handleEndElement(EndElement endElement) throws SAXException {
243 		if (getContentHandler() != null) {
244 			QName qName = endElement.getName();
245 			if (hasNamespacesFeature()) {
246 				getContentHandler().endElement(qName.getNamespaceURI(), qName.getLocalPart(), toQualifiedName(qName));
247 				for (Iterator i = endElement.getNamespaces(); i.hasNext();) {
248 					Namespace namespace = (Namespace) i.next();
249 					endPrefixMapping(namespace.getPrefix());
250 				}
251 			}
252 			else {
253 				getContentHandler().endElement("", "", toQualifiedName(qName));
254 			}
255 
256 		}
257 	}

312 	private Attributes getAttributes(StartElement event) {
313 		AttributesImpl attributes = new AttributesImpl();
314 
315 		for (Iterator i = event.getAttributes(); i.hasNext();) {
316 			Attribute attribute = (Attribute) i.next();
317 			QName qName = attribute.getName();
318 			String namespace = qName.getNamespaceURI();
319 			if (namespace == null || !hasNamespacesFeature()) {
320 				namespace = "";
321 			}
322 			String type = attribute.getDTDType();
323 			if (type == null) {
324 				type = "CDATA";
325 			}
326 			attributes
327 					.addAttribute(namespace, qName.getLocalPart(), toQualifiedName(qName), type, attribute.getValue());
328 		}
329 		if (hasNamespacePrefixesFeature()) {
330 			for (Iterator i = event.getNamespaces(); i.hasNext();) {
331 				Namespace namespace = (Namespace) i.next();
332 				String prefix = namespace.getPrefix();
333 				String namespaceUri = namespace.getNamespaceURI();
334 				String qName;
335 				if (StringUtils.hasLength(prefix)) {
336 					qName = "xmlns:" + prefix;
337 				}
338 				else {
339 					qName = "xmlns";
340 				}
341 				attributes.addAttribute("", "", qName, "CDATA", namespaceUri);
342 			}
343 		}
344 
345 		return attributes;
346 	}
