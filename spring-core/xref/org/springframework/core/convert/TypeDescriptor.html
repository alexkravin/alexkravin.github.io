103 	/**
104 	 * Create a new type descriptor from the given type.
105 	 * Use this to instruct the conversion system to convert an object to a specific target type, when no type location such as a method parameter or field is available to provide additional conversion context.
106 	 * Generally prefer use of {@link #forObject(Object)} for constructing type descriptors from source objects, as it handles the null object case.
107 	 * @param type the class
108 	 * @return the type descriptor
109 	 */
110 	public static TypeDescriptor valueOf(Class<?> type) {
111 		TypeDescriptor desc = typeDescriptorCache.get(type);
112 		return (desc != null ? desc : new TypeDescriptor(type));
113 	}

115 	/**
116 	 * Create a new type descriptor from a java.util.Collection type.
117 	 * Useful for converting to typed Collections.
118 	 * For example, a List&lt;String&gt; could be converted to a List&lt;EmailAddress&gt; by converting to a targetType built with this method.
119 	 * The method call to construct such a TypeDescriptor would look something like: collection(List.class, TypeDescriptor.valueOf(EmailAddress.class));
120 	 * @param collectionType the collection type, which must implement {@link Collection}.
121 	 * @param elementTypeDescriptor a descriptor for the collection's element type, used to convert collection elements
122 	 * @return the collection type descriptor
123 	 */
124 	public static TypeDescriptor collection(Class<?> collectionType, TypeDescriptor elementTypeDescriptor) {
125 		if (!Collection.class.isAssignableFrom(collectionType)) {
126 			throw new IllegalArgumentException("collectionType must be a java.util.Collection");
127 		}
128 		return new TypeDescriptor(collectionType, elementTypeDescriptor);
129 	}

132 	 * Create a new type descriptor from a java.util.Map type.
133 	 * Useful for Converting to typed Maps.
134 	 * For example, a Map&lt;String, String&gt; could be converted to a Map&lt;Id, EmailAddress&gt; by converting to a targetType built with this method:
135 	 * The method call to construct such a TypeDescriptor would look something like: map(Map.class, TypeDescriptor.valueOf(Id.class), TypeDescriptor.valueOf(EmailAddress.class));
136 	 * @param mapType the map type, which must implement {@link Map}.
137 	 * @param keyTypeDescriptor a descriptor for the map's key type, used to convert map keys
138 	 * @param valueTypeDescriptor the map's value type, used to convert map values
139 	 * @return the map type descriptor
140 	 */
141 	public static TypeDescriptor map(Class<?> mapType, TypeDescriptor keyTypeDescriptor, TypeDescriptor valueTypeDescriptor) {
142 		if (!Map.class.isAssignableFrom(mapType)) {
143 			throw new IllegalArgumentException("mapType must be a java.util.Map");
144 		}
145 		return new TypeDescriptor(mapType, keyTypeDescriptor, valueTypeDescriptor);
146 	}

148 	/**
149 	 * Creates a type descriptor for a nested type declared within the method parameter.
150 	 * For example, if the methodParameter is a List&lt;String&gt; and the nestingLevel is 1, the nested type descriptor will be String.class.
151 	 * If the methodParameter is a List<List<String>> and the nestingLevel is 2, the nested type descriptor will also be a String.class.
152 	 * If the methodParameter is a Map<Integer, String> and the nesting level is 1, the nested type descriptor will be String, derived from the map value.
153 	 * If the methodParameter is a List<Map<Integer, String>> and the nesting level is 2, the nested type descriptor will be String, derived from the map value.
154 	 * Returns null if a nested type cannot be obtained because it was not declared.
155 	 * For example, if the method parameter is a List&lt;?&gt;, the nested type descriptor returned will be null.
156 	 * @param methodParameter the method parameter with a nestingLevel of 1
157 	 * @param nestingLevel the nesting level of the collection/array element or map key/value declaration within the method parameter.
158 	 * @return the nested type descriptor at the specified nesting level, or null if it could not be obtained.
159 	 * @throws IllegalArgumentException if the nesting level of the input {@link MethodParameter} argument is not 1.
160 	 * @throws IllegalArgumentException if the types up to the specified nesting level are not of collection, array, or map types.
161 	 */
162 	public static TypeDescriptor nested(MethodParameter methodParameter, int nestingLevel) {
163 		if (methodParameter.getNestingLevel() != 1) {
164 			throw new IllegalArgumentException("methodParameter nesting level must be 1: use the nestingLevel parameter to specify the desired nestingLevel for nested type traversal");
165 		}
166 		return nested(new ParameterDescriptor(methodParameter), nestingLevel);
167 	}

234 	/**
235 	 * Narrows this {@link TypeDescriptor} by setting its type to the class of the provided value.
236 	 * If the value is null, no narrowing is performed and this TypeDescriptor is returned unchanged.
237 	 * Designed to be called by binding frameworks when they read property, field, or method return values.
238 	 * Allows such frameworks to narrow a TypeDescriptor built from a declared property, field, or method return value type.
239 	 * For example, a field declared as java.lang.Object would be narrowed to java.util.HashMap if it was set to a java.util.HashMap value.
240 	 * The narrowed TypeDescriptor can then be used to convert the HashMap to some other type.
241 	 * Annotation and nested type context is preserved by the narrowed copy.
242 	 * @param value the value to use for narrowing this type descriptor
243 	 * @return this TypeDescriptor narrowed (returns a copy with its type updated to the class of the provided value)
244 	 */
245 	public TypeDescriptor narrow(Object value) {
246 		if (value == null) {
247 			return this;
248 		}
249 		return new TypeDescriptor(value.getClass(), this.elementTypeDescriptor,
250 				this.mapKeyTypeDescriptor, this.mapValueTypeDescriptor, this.annotations);
251 	}

288 	/**
289 	 * Returns true if an object of this type descriptor can be assigned to the location described by the given type descriptor.
290 	 * For example, valueOf(String.class).isAssignableTo(valueOf(CharSequence.class)) returns true because a String value can be assigned to a CharSequence variable. 
291 	 * On the other hand, valueOf(Number.class).isAssignableTo(valueOf(Integer.class)) returns false because, while all Integers are Numbers, not all Numbers are Integers.
292 	 * <p>
293 	 * For arrays, collections, and maps, element and key/value types are checked if declared.
294 	 * For example, a List&lt;String&gt; field value is assignable to a Collection&lt;CharSequence&gt; field, but List&lt;Number&gt; is not assignable to List&lt;Integer&gt;.
295 	 * @return true if this type is assignable to the type represented by the provided type descriptor.
296 	 * @see #getObjectType()
297 	 */
298 	public boolean isAssignableTo(TypeDescriptor typeDescriptor) {
299 		boolean typesAssignable = typeDescriptor.getObjectType().isAssignableFrom(getObjectType());
300 		if (!typesAssignable) {
301 			return false;
302 		}
303 		if (isArray() && typeDescriptor.isArray()) {
304 			return getElementTypeDescriptor().isAssignableTo(typeDescriptor.getElementTypeDescriptor());
305 		}
306 		else if (isCollection() && typeDescriptor.isCollection()) {
307 			return isNestedAssignable(getElementTypeDescriptor(), typeDescriptor.getElementTypeDescriptor());
308 		}
309 		else if (isMap() && typeDescriptor.isMap()) {
310 			return isNestedAssignable(getMapKeyTypeDescriptor(), typeDescriptor.getMapKeyTypeDescriptor()) &&
311 				isNestedAssignable(getMapValueTypeDescriptor(), typeDescriptor.getMapValueTypeDescriptor());
312 		}
313 		else {
314 			return true;
315 		}
316 	}

493 	private static TypeDescriptor nested(AbstractDescriptor descriptor, int nestingLevel) {
494 		for (int i = 0; i < nestingLevel; i++) {
495 			descriptor = descriptor.nested();
496 			if (descriptor == null) {
497 				return null;
498 			}
499 		}
500 		return new TypeDescriptor(descriptor);
501 	}

507 		if (!isCollection() && !isArray()) {
508 			throw new IllegalStateException("Not a java.util.Collection or Array");

515 		}

518 	private TypeDescriptor narrow(Object value, TypeDescriptor typeDescriptor) {
519 		if (typeDescriptor != null) {
520 			return typeDescriptor.narrow(value);
521 		}
522 		else {
523 			return (value != null ? new TypeDescriptor(value.getClass(), null, null, null, this.annotations) : null);
524 		}		
525 	}

527 	private boolean isNestedAssignable(TypeDescriptor nestedTypeDescriptor, TypeDescriptor otherNestedTypeDescriptor) {
528 		if (nestedTypeDescriptor == null || otherNestedTypeDescriptor == null) {
529 			return true;
530 		}
531 		return nestedTypeDescriptor.isAssignableTo(otherNestedTypeDescriptor);
532 	}
