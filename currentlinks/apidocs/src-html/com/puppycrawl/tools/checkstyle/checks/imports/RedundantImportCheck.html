020package com.puppycrawl.tools.checkstyle.checks.imports;
021
022import com.google.common.collect.Sets;
023import com.puppycrawl.tools.checkstyle.api.Check;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.FullIdent;
026import com.puppycrawl.tools.checkstyle.api.TokenTypes;
027import java.util.Set;
028
029/**
030 * <p>
031 * Checks for imports that are redundant. An import statement is
032 * considered redundant if:
033 * </p>
034 *<ul>
035 *  <li>It is a duplicate of another import. This is, when a class is imported
036 *  more than once.</li>
037 *  <li>The class non-statically imported is from the <code>java.lang</code>
038 *  package. For example importing <code>java.lang.String</code>.</li>
039 *  <li>The class non-statically imported is from the same package as the
040 *  current package.</li>
041 *</ul>
042 * <p>
043 * An example of how to configure the check is:
044 * </p>
045 * <pre>
046 * &lt;module name="RedundantImport"/&gt;
047 * </pre>
048 *
049 * Compatible with Java 1.5 source.
050 *
051 * @author Oliver Burn
052 * @version 1.0
053 */
054public class RedundantImportCheck
055    extends Check
056{
057    /** name of package in file */
058    private String pkgName;
059    /** set of the imports */
060    private final Set<FullIdent> imports = Sets.newHashSet();
061    /** set of static imports */
062    private final Set<FullIdent> staticImports = Sets.newHashSet();
063
064    @Override
065    public void beginTree(DetailAST aRootAST)
066    {
067        pkgName = null;
068        imports.clear();
069        staticImports.clear();
070    }
071
072    @Override
073    public int[] getDefaultTokens()
074    {
075        return new int[]
076        {TokenTypes.IMPORT,
077         TokenTypes.STATIC_IMPORT,
078         TokenTypes.PACKAGE_DEF, };
079    }
080
081    @Override
082    public int[] getAcceptableTokens()
083    {
084        return new int[]
085        {TokenTypes.IMPORT,
086         TokenTypes.STATIC_IMPORT,
087         TokenTypes.PACKAGE_DEF, };
088    }
089
090    @Override
091    public void visitToken(DetailAST ast)
092    {
093        if (ast.getType() == TokenTypes.PACKAGE_DEF) {
094            pkgName = FullIdent.createFullIdent(
095                    ast.getLastChild().getPreviousSibling()).getText();
096        }
097        else if (ast.getType() == TokenTypes.IMPORT) {
098            final FullIdent imp = FullIdent.createFullIdentBelow(ast);
099            if (fromPackage(imp.getText(), "java.lang")) {
100                log(ast.getLineNo(), ast.getColumnNo(), "import.lang",
101                    imp.getText());
102            }
103            else if (fromPackage(imp.getText(), pkgName)) {
104                log(ast.getLineNo(), ast.getColumnNo(), "import.same",
105                    imp.getText());
106            }
107            // Check for a duplicate import
108            for (FullIdent full : imports) {
109                if (imp.getText().equals(full.getText())) {
110                    log(ast.getLineNo(), ast.getColumnNo(),
111                            "import.duplicate", full.getLineNo(),
112                            imp.getText());
113                }
114            }
115
116            imports.add(imp);
117        }
118        else {
119            // Check for a duplicate static import
120            final FullIdent imp =
121                FullIdent.createFullIdent(
122                    ast.getLastChild().getPreviousSibling());
123            for (FullIdent full : staticImports) {
124                if (imp.getText().equals(full.getText())) {
125                    log(ast.getLineNo(), ast.getColumnNo(),
126                        "import.duplicate", full.getLineNo(), imp.getText());
127                }
128            }
129
130            staticImports.add(imp);
131        }
132    }
133
134    /**
135     * Determines if an import statement is for types from a specified package.
136     * @param importName the import name
137     * @param pkg the package name
138     * @return whether from the package
139     */
140    private static boolean fromPackage(String importName, String pkg)
141    {
142        boolean retVal = false;
143        if (pkg == null) {
144            // If not package, then check for no package in the import.
145            retVal = (importName.indexOf('.') == -1);
146        }
147        else {
148            final int index = importName.lastIndexOf('.');
149            if (index != -1) {
150                final String front = importName.substring(0, index);
151                retVal = front.equals(pkg);
152            }
153        }
154        return retVal;
155    }
156}
