019package com.puppycrawl.tools.checkstyle.checks.coding;
020
021import java.util.regex.Matcher;
022import java.util.regex.Pattern;
023
024import com.puppycrawl.tools.checkstyle.api.Check;
025import com.puppycrawl.tools.checkstyle.api.DetailAST;
026import com.puppycrawl.tools.checkstyle.api.TokenTypes;
027import com.puppycrawl.tools.checkstyle.api.Utils;
028
029/**
030 * Checks for fall through in switch statements
031 * Finds locations where a case contains Java code -
032 * but lacks a break, return, throw or continue statement.
033 *
034 * <p>
035 * The check honors special comments to suppress warnings about
036 * the fall through. By default the comments "fallthru",
037 * "fall through", "falls through" and "fallthrough" are recognized.
038 * </p>
039 * <p>
040 * The following fragment of code will NOT trigger the check,
041 * because of the comment "fallthru".
042 * </p>
043 * <pre>
044 * case 3:
045 *     x = 2;
046 *     // fallthru
047 * case 4:
048 * </pre>
049 * <p>
050 * The recognized relief comment can be configured with the property
051 * <code>reliefPattern</code>. Default value of this regular expression
052 * is "fallthru|fall through|fallthrough|falls through".
053 * </p>
054 * <p>
055 * An example of how to configure the check is:
056 * </p>
057 * <pre>
058 * &lt;module name="FallThrough"&gt;
059 *     &lt;property name=&quot;reliefPattern&quot;
060 *                  value=&quot;Fall Through&quot;/&gt;
061 * &lt;/module&gt;
062 * </pre>
063 *
064 * @author o_sukhodolsky
065 */
066public class FallThroughCheck extends Check
067{
068    /** Do we need to check last case group. */
069    private boolean checkLastGroup;
070
071    /** Relief pattern to allow fall throught to the next case branch. */
072    private String reliefPattern = "fallthru|falls? ?through";
073
074    /** Relief regexp. */
075    private Pattern regExp;
076
077    /** Creates new instance of the check. */
078    public FallThroughCheck()
079    {
080        // do nothing
081    }
082
083    @Override
084    public int[] getDefaultTokens()
085    {
086        return new int[]{TokenTypes.CASE_GROUP};
087    }
088
089    @Override
090    public int[] getRequiredTokens()
091    {
092        return getDefaultTokens();
093    }
094
095    @Override
096    public int[] getAcceptableTokens()
097    {
098        return new int[]{TokenTypes.CASE_GROUP};
099    }
100
101    /**
102     * Set the relief pattern.
103     *
104     * @param pattern
105     *            The regular expression pattern.
106     */
107    public void setReliefPattern(String pattern)
108    {
109        reliefPattern = pattern;
110    }
111
112    /**
113     * Configures whether we need to check last case group or not.
114     * @param value new value of the property.
115     */
116    public void setCheckLastCaseGroup(boolean value)
117    {
118        checkLastGroup = value;
119    }
120
121    @Override
122    public void init()
123    {
124        super.init();
125        regExp = Utils.getPattern(reliefPattern);
126    }
127
128    @Override
129    public void visitToken(DetailAST ast)
130    {
131        final DetailAST nextGroup = ast.getNextSibling();
132        final boolean isLastGroup =
133            ((nextGroup == null)
134             || (nextGroup.getType() != TokenTypes.CASE_GROUP));
135        if (isLastGroup && !checkLastGroup) {
136            // we do not need to check last group
137            return;
138        }
139
140        final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
141
142        if (slist != null && !isTerminated(slist, true, true)
143            && !hasFallTruComment(ast, nextGroup))
144        {
145            if (!isLastGroup) {
146                log(nextGroup, "fall.through");
147            }
148            else {
149                log(ast, "fall.through.last");
150            }
151        }
152    }
153
154    /**
155     * Checks if a given subtree terminated by return, throw or,
156     * if allowed break, continue.
157     * @param ast root of given subtree
158     * @param useBreak should we consider break as terminator.
159     * @param useContinue should we consider continue as terminator.
160     * @return true if the subtree is terminated.
161     */
162    private boolean isTerminated(final DetailAST ast, boolean useBreak,
163                                 boolean useContinue)
164    {
165        switch (ast.getType()) {
166            case TokenTypes.LITERAL_RETURN:
167            case TokenTypes.LITERAL_THROW:
168                return true;
169            case TokenTypes.LITERAL_BREAK:
170                return useBreak;
171            case TokenTypes.LITERAL_CONTINUE:
172                return useContinue;
173            case TokenTypes.SLIST:
174                return checkSlist(ast, useBreak, useContinue);
175            case TokenTypes.LITERAL_IF:
176                return checkIf(ast, useBreak, useContinue);
177            case TokenTypes.LITERAL_FOR:
178            case TokenTypes.LITERAL_WHILE:
179            case TokenTypes.LITERAL_DO:
180                return checkLoop(ast);
181            case TokenTypes.LITERAL_TRY:
182                return checkTry(ast, useBreak, useContinue);
183            case TokenTypes.LITERAL_SWITCH:
184                return checkSwitch(ast, useContinue);
185            default:
186                return false;
187        }
188    }
189
190    /**
191     * Checks if a given SLIST terminated by return, throw or,
192     * if allowed break, continue.
193     * @param ast SLIST to check
194     * @param useBreak should we consider break as terminator.
195     * @param useContinue should we consider continue as terminator.
196     * @return true if SLIST is terminated.
197     */
198    private boolean checkSlist(final DetailAST ast, boolean useBreak,
199                               boolean useContinue)
200    {
201        DetailAST lastStmt = ast.getLastChild();
202        if (lastStmt == null) {
203            // if last case in switch is empty then slist is empty
204            // since this is a last case it is not a fall-through
205            return true;
206        }
207
208        if (lastStmt.getType() == TokenTypes.RCURLY) {
209            lastStmt = lastStmt.getPreviousSibling();
210        }
211
212        return (lastStmt != null)
213            && isTerminated(lastStmt, useBreak, useContinue);
214    }
215
216    /**
217     * Checks if a given IF terminated by return, throw or,
218     * if allowed break, continue.
219     * @param ast IF to check
220     * @param useBreak should we consider break as terminator.
221     * @param useContinue should we consider continue as terminator.
222     * @return true if IF is terminated.
223     */
224    private boolean checkIf(final DetailAST ast, boolean useBreak,
225                            boolean useContinue)
226    {
227        final DetailAST thenStmt = ast.findFirstToken(TokenTypes.RPAREN)
228                .getNextSibling();
229        final DetailAST elseStmt = thenStmt.getNextSibling();
230        boolean isTerminated = isTerminated(thenStmt, useBreak, useContinue);
231
232        if (isTerminated && (elseStmt != null)) {
233            isTerminated = isTerminated(elseStmt.getFirstChild(),
234                                        useBreak, useContinue);
235        }
236        return isTerminated;
237    }
238
239    /**
240     * Checks if a given loop terminated by return, throw or,
241     * if allowed break, continue.
242     * @param ast loop to check
243     * @return true if loop is terminated.
244     */
245    private boolean checkLoop(final DetailAST ast)
246    {
247        DetailAST loopBody = null;
248        if (ast.getType() == TokenTypes.LITERAL_DO) {
249            final DetailAST lparen = ast.findFirstToken(TokenTypes.DO_WHILE);
250            loopBody = lparen.getPreviousSibling();
251        }
252        else {
253            final DetailAST rparen = ast.findFirstToken(TokenTypes.RPAREN);
254            loopBody = rparen.getNextSibling();
255        }
256        return isTerminated(loopBody, false, false);
257    }
258
259    /**
260     * Checks if a given try/catch/finally block terminated by return, throw or,
261     * if allowed break, continue.
262     * @param ast loop to check
263     * @param useBreak should we consider break as terminator.
264     * @param useContinue should we consider continue as terminator.
265     * @return true if try/cath/finally block is terminated.
266     */
267    private boolean checkTry(final DetailAST ast, boolean useBreak,
268                             boolean useContinue)
269    {
270        final DetailAST finalStmt = ast.getLastChild();
271        if (finalStmt.getType() == TokenTypes.LITERAL_FINALLY) {
272            return isTerminated(finalStmt.findFirstToken(TokenTypes.SLIST),
273                                useBreak, useContinue);
274        }
275
276        boolean isTerminated = isTerminated(ast.getFirstChild(),
277                                            useBreak, useContinue);
278
279        DetailAST catchStmt = ast.findFirstToken(TokenTypes.LITERAL_CATCH);
280        while ((catchStmt != null) && isTerminated) {
281            final DetailAST catchBody =
282                catchStmt.findFirstToken(TokenTypes.SLIST);
283            isTerminated &= isTerminated(catchBody, useBreak, useContinue);
284            catchStmt = catchStmt.getNextSibling();
285        }
286        return isTerminated;
287    }
288
289    /**
290     * Checks if a given switch terminated by return, throw or,
291     * if allowed break, continue.
292     * @param ast loop to check
293     * @param useContinue should we consider continue as terminator.
294     * @return true if switch is terminated.
295     */
296    private boolean checkSwitch(final DetailAST ast, boolean useContinue)
297    {
298        DetailAST caseGroup = ast.findFirstToken(TokenTypes.CASE_GROUP);
299        boolean isTerminated = (caseGroup != null);
300        while (isTerminated && (caseGroup != null)
301               && (caseGroup.getType() != TokenTypes.RCURLY))
302        {
303            final DetailAST caseBody =
304                caseGroup.findFirstToken(TokenTypes.SLIST);
305            isTerminated &= isTerminated(caseBody, false, useContinue);
306            caseGroup = caseGroup.getNextSibling();
307        }
308        return isTerminated;
309    }
310
311    /**
312     * Determines if the fall through case between <code>currentCase</code> and
313     * <code>nextCase</code> is reliefed by a appropriate comment.
314     *
315     * @param currentCase AST of the case that falls through to the next case.
316     * @param nextCase AST of the next case.
317     * @return True if a relief comment was found
318     */
319    private boolean hasFallTruComment(DetailAST currentCase,
320            DetailAST nextCase)
321    {
322
323        final int startLineNo = currentCase.getLineNo();
324        final int endLineNo = nextCase.getLineNo();
325        final int endColNo = nextCase.getColumnNo();
326
327        /*
328         * Remember: The lines number returned from the AST is 1-based, but
329         * the lines number in this array are 0-based. So you will often
330         * see a "lineNo-1" etc.
331         */
332        final String[] lines = getLines();
333
334        /*
335         * Handle:
336         *    case 1:
337         *    /+ FALLTHRU +/ case 2:
338         *    ....
339         * and
340         *    switch(i) {
341         *    default:
342         *    /+ FALLTHRU +/}
343         */
344        final String linepart = lines[endLineNo - 1].substring(0, endColNo);
345        if (commentMatch(regExp, linepart, endLineNo)) {
346            return true;
347        }
348
349        /*
350         * Handle:
351         *    case 1:
352         *    .....
353         *    // FALLTHRU
354         *    case 2:
355         *    ....
356         * and
357         *    switch(i) {
358         *    default:
359         *    // FALLTHRU
360         *    }
361         */
362        for (int i = endLineNo - 2; i > startLineNo - 1; i--) {
363            if (lines[i].trim().length() != 0) {
364                return commentMatch(regExp, lines[i], i + 1);
365            }
366        }
367
368        // Well -- no relief comment found.
369        return false;
370    }
371
372    /**
373     * Does a regular expression match on the given line and checks that a
374     * possible match is within a comment.
375     * @param pattern The regular expression pattern to use.
376     * @param line The line of test to do the match on.
377     * @param lineNo The line number in the file.
378     * @return True if a match was found inside a comment.
379     */
380    private boolean commentMatch(Pattern pattern, String line, int lineNo
381    )
382    {
383        final Matcher matcher = pattern.matcher(line);
384
385        final boolean hit = matcher.find();
386
387        if (hit) {
388            final int startMatch = matcher.start();
389            // -1 because it returns the char position beyond the match
390            final int endMatch = matcher.end() - 1;
391            return getFileContents().hasIntersectionWithComment(lineNo,
392                    startMatch, lineNo, endMatch);
393        }
394        return false;
395    }
