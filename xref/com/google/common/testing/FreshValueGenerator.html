135         ImmutableMap.builder();

169   /**
170    * Returns a fresh instance for {@code type} if possible. The returned instance could be:
171    * <ul>
172    * <li>exactly of the given type, including generic type parameters, such as
173    *     {@code ImmutableList<String>};
174    * <li>of the raw type;
175    * <li>null if no value can be generated.
176    * </ul>
177    */
178   @Nullable final Object generateFresh(TypeToken<?> type) {
179     Object generated = generate(type);
180     if (generated != null) {
181       freshness.incrementAndGet();
182     }
183     return generated;
184   }
185 
186   @Nullable final <T> T generateFresh(Class<T> type) {
187     return Primitives.wrap(type).cast(generateFresh(TypeToken.of(type)));
188   }

191     T proxy = newProxy(interfaceType);
192     freshness.incrementAndGet();
193     return proxy;
194   }

196   /**
197    * Generates an instance for {@code type} using the current {@link #freshness}.
198    * The generated instance may or may not be unique across different calls.
199    */
200   private Object generate(TypeToken<?> type) {
201     Class<?> rawType = type.getRawType();
202     List<Object> samples = sampleInstances.get(rawType);
203     Object sample = pickInstance(samples, null);
204     if (sample != null) {
205       return sample;
206     }
207     if (rawType.isEnum()) {
208       return pickInstance(rawType.getEnumConstants(), null);
209     }
210     if (type.isArray()) {
211       TypeToken<?> componentType = type.getComponentType();
212       Object array = Array.newInstance(componentType.getRawType(), 1);
213       Array.set(array, 0, generate(componentType));
214       return array;
215     }
216     Method emptyGenerate = EMPTY_GENEREATORS.get(rawType);
217     if (emptyGenerate != null) {
218       if (emptyInstanceGenerated.containsKey(type.getType())) {
219         // empty instance already generated
220         if (emptyInstanceGenerated.get(type.getType()).intValue() == freshness.get()) {
221           // same freshness, generate again.
222           return invokeGeneratorMethod(emptyGenerate);
223         } else {
224           // Cannot use empty generator. Proceed with other generators.
225         }
226       } else {
227         // never generated empty instance for this type before.
228         Object emptyInstance = invokeGeneratorMethod(emptyGenerate);
229         emptyInstanceGenerated.put(type.getType(), freshness.get());
230         return emptyInstance;
231       }
232     }
233     Method generate = GENERATORS.get(rawType);
234     if (generate != null) {
235       ImmutableList<Parameter> params = Invokable.from(generate).getParameters();
236       List<Object> args = Lists.newArrayListWithCapacity(params.size());
237       TypeVariable<?>[] typeVars = rawType.getTypeParameters();
238       for (int i = 0; i < params.size(); i++) {
239         TypeToken<?> paramType = type.resolveType(typeVars[i]);
240         // We require all @Generates methods to either be parameter-less or accept non-null
241         // values for their generic parameter types.
242         Object argValue = generate(paramType);
243         if (argValue == null) {
244           // When a parameter of a @Generates method cannot be created,
245           // The type most likely is a collection.
246           // Our distinct proxy doesn't work for collections.
247           // So just refuse to generate.
248           return null;
249         }
250         args.add(argValue);
251       }
252       return invokeGeneratorMethod(generate, args.toArray());
253     }
254     return defaultGenerate(rawType);
255   }
256 
257   private <T> T defaultGenerate(Class<T> rawType) {
258     if (rawType.isInterface()) {
259       // always create a new proxy
260       return newProxy(rawType);
261     }
262     return ArbitraryInstances.get(rawType);
263   }

265   private <T> T newProxy(final Class<T> interfaceType) {
266     return Reflection.newProxy(interfaceType, new FreshInvocationHandler(interfaceType));
267   }

269   private Object invokeGeneratorMethod(Method generator, Object... args) {
270     try {
271       return generator.invoke(this, args);
272     } catch (InvocationTargetException e) {
273       throw Throwables.propagate(e.getCause());
274     } catch (Exception e) {
275       throw Throwables.propagate(e);
276     }
277   }

279   private final class FreshInvocationHandler extends AbstractInvocationHandler {
280     private final int identity = generateInt();
281     private final Class<?> interfaceType;
282 
283     FreshInvocationHandler(Class<?> interfaceType) {
284       this.interfaceType = interfaceType;
285     }
286 
287     @Override protected Object handleInvocation(Object proxy, Method method, Object[] args) {
288       return interfaceMethodCalled(interfaceType, method);
289     }
290 
291     @Override public int hashCode() {
292       return identity;
293     }
294 
295     @Override public boolean equals(@Nullable Object obj) {
296       if (obj instanceof FreshInvocationHandler) {
297         FreshInvocationHandler that = (FreshInvocationHandler) obj;
298         return identity == that.identity;
299       }
300       return false;
301     }
302 
303     @Override public String toString() {
304       return paramString(interfaceType, identity);
305     }
306   }

309   Object interfaceMethodCalled(
310       @SuppressWarnings("unused") Class<?> interfaceType,
311       @SuppressWarnings("unused") Method method) {
312     throw new UnsupportedOperationException();
313   }

315   private <T> T pickInstance(T[] instances, T defaultValue) {
316     return pickInstance(Arrays.asList(instances), defaultValue);
317   }

319   private <T> T pickInstance(Collection<T> instances, T defaultValue) {
320     if (instances.isEmpty()) {
321       return defaultValue;
322     }
323     // generateInt() is 1-based.
324     return Iterables.get(instances, (generateInt() - 1) % instances.size());
325   }

470   @Generates private Currency generateCurrency() {
471     try {
472       Method method = Currency.class.getMethod("getAvailableCurrencies");
473       @SuppressWarnings("unchecked") // getAvailableCurrencies() returns Set<Currency>.
474       Set<Currency> currencies = (Set<Currency>) method.invoke(null);
475       return pickInstance(currencies, Currency.getInstance(Locale.US));
476     } catch (NoSuchMethodException notJava7) {
477       return preJava7FreshCurrency();
478     } catch (InvocationTargetException cantCallGetAvailableCurrencies) {
479       return preJava7FreshCurrency();
480     } catch (IllegalAccessException impossible) {
481       throw new AssertionError(impossible);
482     }
483   }
484 
485   private Currency preJava7FreshCurrency() {
486     for (Set<Locale> uselessLocales = Sets.newHashSet(); ; ) {
487       Locale locale = generateLocale();
488       if (uselessLocales.contains(locale)) { // exhausted all locales
489         return Currency.getInstance(Locale.US);
490       }
491       try {
492         return Currency.getInstance(locale);
493       } catch (IllegalArgumentException e) {
494         uselessLocales.add(locale);
495       }
496     }
497   }
